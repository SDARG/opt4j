047/**
048 * Superclass for {@link GenericOperator}s.
049 * 
050 * @author lukasiewycz
051 * 
052 * @param <O>
053 *            The specified {@link Operator}.
054 * @param <Q>
055 *            The specified {@link Operator} with a wildcard (?).
056 */
057public abstract class AbstractGenericOperator<O extends Operator<?>, Q extends Operator<?>>
058                implements GenericOperator<O> {
059
060        protected SortedMap<Class<? extends Genotype>, O> classOperators = new TreeMap<>(new ClassComparator());
061        protected Map<OperatorPredicate, O> genericOperators = new HashMap<>();
062
063        protected List<Class<? extends Q>> cldef = new ArrayList<>();
064
065        /**
066         * Comparator for a specific order: Superclasses always are sorted after
067         * subclasses.
068         * 
069         * @author lukasiewycz
070         * 
071         */
072        protected static class ClassComparator implements Comparator<Class<? extends Genotype>> {
073
074                /*
075                 * (non-Javadoc)
076                 * 
077                 * @see java.util.Comparator#compare(java.lang.Object, java.lang.Object)
078                 */
079                @Override
080                public int compare(Class<? extends Genotype> arg0, Class<? extends Genotype> arg1) {
081                        if (arg0.equals(arg1)) {
082                                return 0;
083                        } else if (arg0.isAssignableFrom(arg1)) {
084                                // arg0 is superclass of arg1
085                                return 1;
086                        } else if (arg1.isAssignableFrom(arg0)) {
087                                // arg1 is superclass of arg0
088                                return -1;
089                        } else {
090                                return arg0.getCanonicalName().compareTo(arg1.getCanonicalName());
091                        }
092                }
093
094        }
095
096        /**
097         * Constructs an {@link AbstractGenericOperator} class with the given
098         * clazzes of default operators.
099         * 
100         * @param clazzes
101         *            the default operators
102         */
103        public AbstractGenericOperator(Class<? extends Q>... clazzes) {
104                for (Class<? extends Q> cl : clazzes) {
105                        cldef.add(cl);
106                }
107        }
108
109        /**
110         * Inject and organize the operators.
111         * 
112         * @param holder
113         *            the operator holder
114         */
115        @SuppressWarnings("unchecked")
116        @Inject
117        protected synchronized void inject(OperatorHolder<Q> holder) {
118                if (classOperators.isEmpty()) {
119                        classOperators.put(CompositeGenotype.class, null);
120                        holder.add(cldef);
121
122                        for (Entry<OperatorPredicate, Q> entry : holder.getMap().entrySet()) {
123                                addOperator(entry.getKey(), (O) entry.getValue());
124                        }
125
126                }
127        }
128
129        /*
130         * (non-Javadoc)
131         * 
132         * @see org.opt4j.operator.GenericOperator#addOperator(org.opt4j.operator.
133         * AbstractGenericOperator.OperatorPredicate,
134         * org.opt4j.core.optimizer.Operator)
135         */
136        @Override
137        public void addOperator(OperatorPredicate predicate, O operator) {
138                if (predicate instanceof OperatorClassPredicate) {
139                        Class<? extends Genotype> clazz = ((OperatorClassPredicate) predicate).getClazz();
140                        classOperators.put(clazz, operator);
141                } else {
142                        genericOperators.put(predicate, operator);
143                }
144        }
145
146        /*
147         * (non-Javadoc)
148         * 
149         * @see
150         * org.opt4j.operator.GenericOperator#getOperator(org.opt4j.core.problem
151         * .Genotype)
152         */
153        @Override
154        public O getOperator(Genotype genotype) {
155                if (classOperators.containsKey(genotype.getClass())) {
156                        return classOperators.get(genotype.getClass());
157                } else {
158                        // Search for a predicate that satisfies the genotype.
159                        for (Entry<OperatorPredicate, O> predicate : genericOperators.entrySet()) {
160                                if (predicate.getKey().evaluate(genotype)) {
161                                        return predicate.getValue();
162                                }
163                        }
164
165                        // Searches for a superclass that is registered as an operator.
166                        for (Entry<Class<? extends Genotype>, O> entry : classOperators.entrySet()) {
167                                Class<? extends Genotype> c = entry.getKey();
168                                if (c.isAssignableFrom(genotype.getClass())) {
169                                        O operator = entry.getValue();
170                                        addOperator(new OperatorClassPredicate(genotype.getClass()), operator);
171                                        return operator;
172                                }
173                        }
174                        throw new IncompatibilityException(
175                                        "No handler found for " + genotype.getClass() + " in " + this.getClass());
176                }
177
178        }
179
180        /*
181         * (non-Javadoc)
182         * 
183         * @see org.opt4j.operator.common.GenericOperator#getHandlers()
184         */
185        @Override
186        public Collection<O> getOperators() {
187                Set<O> set = new HashSet<>();
188                set.addAll(classOperators.values());
189                set.addAll(genericOperators.values());
190                return set;
191        }
192
193        /**
194         * Returns the target {@link Genotype} for an operator based on the
195         * {@link Apply} annotation.
196         * 
197         * @param <O>
198         *            the type of operator
199         * @param operator
200         *            the operator
201         * @return the target genotype
202         */
203        protected static <O> Class<? extends Genotype> getTarget(O operator) {
204                Apply apply = operator.getClass().getAnnotation(Apply.class);
205
206                if (apply != null) {
207                        return apply.value();
208                }
209
210                Type type = Parameters.getType(Operator.class, operator, "G");
211                if (type != null) {
212                        Class<? extends Genotype> target = Parameters.getClass(type).asSubclass(Genotype.class);
213                        return target;
214                }
215
216                throw new IllegalArgumentException("No target specified for the operator " + operator.getClass().getName()
217                                + ". Either parameterize the Operator or use the " + Apply.class.getName()
218                                + " annotation to specify a target.");
219        }
220
221        protected static class OperatorHolder<P> {
222
223                @Inject(optional = true)
224                protected Map<OperatorPredicate, P> map = new HashMap<>();
225
226                @Inject
227                protected Injector injector;
228
229                protected Collection<Class<? extends P>> clazzes = new ArrayList<>();
230
231                public void add(Collection<Class<? extends P>> clazzes) {
232                        this.clazzes.addAll(clazzes);
233                }
234
235                public Map<OperatorPredicate, P> getMap() {
236                        Map<OperatorPredicate, P> map = new HashMap<>();
237
238                        for (Class<? extends P> clazz : clazzes) {
239                                P p = injector.getInstance(clazz);
240                                map.put(new OperatorClassPredicate(getTarget((Operator<?>) p)), p);
241                        }
242
243                        for (Entry<OperatorPredicate, P> entry : this.map.entrySet()) {
244                                OperatorPredicate predicate = entry.getKey();
245                                if (predicate instanceof OperatorVoidPredicate) {
246                                        predicate = new OperatorClassPredicate(getTarget((Operator<?>) entry.getValue()));
247                                }
248                                map.put(predicate, entry.getValue());
249                        }
250
251                        return map;
252                }
253        }
254
255        /**
256         * The {@link OperatorPredicate} interface.
257         * 
258         * @author lukasiewycz
259         * 
260         */
261        public interface OperatorPredicate {
262
263                /**
264                 * Checks whether a {@link Genotype} satisfies the predicate.
265                 * 
266                 * @param genotype
267                 *            the genotype
268                 * @return {@code true} if the predicate is satisfied
269                 */
270                public boolean evaluate(Genotype genotype);
271        }
272
273        /**
274         * The {@link OperatorVoidPredicate} interface is used as marker for
275         * {@link Operator}s for which the predicate is not explicitly defined.
276         * 
277         * @author lukasiewycz
278         * 
279         */
280        public static class OperatorVoidPredicate implements OperatorPredicate {
281                /*
282                 * (non-Javadoc)
283                 * 
284                 * @see
285                 * org.opt4j.operator.AbstractGenericOperator.OperatorPredicate#evaluate
286                 * (org.opt4j.core.problem.Genotype)
287                 */
288                @Override
289                public boolean evaluate(Genotype genotype) {
290                        return false;
291                }
292
293                /*
294                 * (non-Javadoc)
295                 * 
296                 * @see java.lang.Object#toString()
297                 */
298                @Override
299                public String toString() {
300                        return "Predicate Void";
301                }
302
303        }
304
305        /**
306         * The {@link OperatorClassPredicate} returns {@code true} for a given
307         * specific class.
308         * 
309         * @author lukasiewycz
310         * 
311         */
312        public static class OperatorClassPredicate implements OperatorPredicate {
313
314                protected final Class<? extends Genotype> clazz;
315
316                /**
317                 * Creates a new {@link OperatorClassPredicate} for the given
318                 * {@link Genotype} class.
319                 * 
320                 * @param clazz
321                 *            the class of the genotype
322                 */
323                public OperatorClassPredicate(Class<? extends Genotype> clazz) {
324                        this.clazz = clazz;
325                }
326
327                /*
328                 * (non-Javadoc)
329                 * 
330                 * @see
331                 * org.opt4j.operator.AbstractGenericOperator.OperatorPredicate#evaluate
332                 * (org.opt4j.core.problem.Genotype)
333                 */
334                @Override
335                public boolean evaluate(Genotype genotype) {
336                        return clazz.equals(genotype.getClass());
337                }
338
339                /**
340                 * Returns the genotype class for the operator.
341                 * 
342                 * @return the genotype class
343                 */
344                public Class<? extends Genotype> getClazz() {
345                        return clazz;
346                }
347
348                /*
349                 * (non-Javadoc)
350                 * 
351                 * @see java.lang.Object#toString()
352                 */
353                @Override
354                public String toString() {
355                        return "Predicate [clazz=" + clazz.getSimpleName() + "]";
356                }
357
358                /*
359                 * (non-Javadoc)
360                 * 
361                 * @see java.lang.Object#hashCode()
362                 */
363                @Override
364                public int hashCode() {
365                        final int prime = 31;
366                        int result = 1;
367                        result = prime * result + ((clazz == null) ? 0 : clazz.hashCode());
368                        return result;
369                }
370
371                /*
372                 * (non-Javadoc)
373                 * 
374                 * @see java.lang.Object#equals(java.lang.Object)
375                 */
376                @Override
377                public boolean equals(Object obj) {
378                        if (this == obj) {
379                                return true;
380                        }
381                        if (obj == null) {
382                                return false;
383                        }
384                        if (getClass() != obj.getClass()) {
385                                return false;
386                        }
387                        OperatorClassPredicate other = (OperatorClassPredicate) obj;
388                        if (clazz == null) {
389                                if (other.clazz != null) {
390                                        return false;
391                                }
392                        } else if (!clazz.equals(other.clazz)) {
393                                return false;
394                        }
395                        return true;
396                }
397        }
