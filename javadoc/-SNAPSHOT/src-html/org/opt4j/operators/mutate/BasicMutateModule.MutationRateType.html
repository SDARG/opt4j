023package org.opt4j.operators.mutate;
024
025import org.opt4j.core.config.annotations.Info;
026import org.opt4j.core.config.annotations.Required;
027import org.opt4j.core.genotype.DoubleGenotype;
028import org.opt4j.core.genotype.PermutationGenotype;
029import org.opt4j.core.start.Constant;
030import org.opt4j.operators.copy.CopyModule;
031
032/**
033 * The basic {@link CopyModule}.
034 * 
035 * @author lukasiewycz
036 * 
037 */
038@Info("Setting for the basic mutate classOperators for genotype variation.")
039public class BasicMutateModule extends MutateModule {
040
041        @Info("The type of mutation rate.")
042        protected MutationRateType mutationRateType = MutationRateType.ADAPTIVE;
043
044        @Info("The type of the mutate operator for the Permutation genotype.")
045        protected PermutationType permutationType = PermutationType.MIXED;
046
047        @Info("The type of the mutate operator for the Double genotype.")
048        protected DoubleType doubleType = DoubleType.POLYNOMIAL;
049
050        @Required(property = "doubleType", elements = { "POLYNOMIAL" })
051        @Constant(value = "eta", namespace = MutateDoublePolynomial.class)
052        protected double eta = 20;
053
054        @Required(property = "doubleType", elements = { "GAUSS" })
055        @Constant(value = "sigma", namespace = MutateDoubleGauss.class)
056        protected double sigma = 0.1;
057
058        @Required(property = "mutationRateType", elements = { "CONSTANT" })
059        @Constant(value = "rate", namespace = ConstantMutationRate.class)
060        protected double mutationRate = 0.01;
061
062        /**
063         * The type of the used mutation rate.
064         * 
065         * @author lukasiewycz
066         * 
067         */
068        public enum MutationRateType {
069                /**
070                 * Use a constant mutation rate.
071                 */
072                CONSTANT,
073                /**
074                 * Use a variable (adaptive) mutation rate.
075                 */
076                ADAPTIVE;
077        }
078
079        /**
080         * Type of {@link Mutate} operator for the {@link DoubleGenotype}.
081         * 
082         * @author lukasiewycz
083         * 
084         */
085        public enum DoubleType {
086                /**
087                 * Use a constant mutation rate.
088                 */
089                GAUSS,
090                /**
091                 * Use a variable (adaptive) mutation rate.
092                 */
093                POLYNOMIAL;
094        }
095
096        /**
097         * Type of {@link Mutate} operator for the {@link PermutationGenotype}.
098         * 
099         * @author lukasiewycz
100         * 
101         */
102        public enum PermutationType {
103                /**
104                 * Use the {@link MutatePermutationMixed}.
105                 */
106                @Info("Use randomly SWAP,INSERT, or REVERT")
107                MIXED,
108                /**
109                 * Use the {@link MutatePermutationSwap}.
110                 */
111                @Info("Swaps two elements")
112                SWAP,
113                /**
114                 * Use the {@link MutatePermutationInsert}.
115                 */
116                @Info("Moves one element to another position")
117                INSERT,
118                /**
119                 * Use the {@link MutatePermutationRevert}.
120                 */
121                @Info("Reverts a part for the genotype")
122                REVERT;
123        }
124
125        /**
126         * Returns the {@link MutationRateType}.
127         * 
128         * @return the type of mutation rate
129         */
130        public MutationRateType getMutationRateType() {
131                return mutationRateType;
132        }
133
134        /**
135         * Sets the {@link MutationRateType}.
136         * 
137         * @param mutationRateType
138         *            the type of mutation rate
139         */
140        public void setMutationRateType(MutationRateType mutationRateType) {
141                this.mutationRateType = mutationRateType;
142        }
143
144        /**
145         * Returns the {@link MutationRate} as a double value.
146         * 
147         * @return the mutation rate
148         */
149        public double getMutationRate() {
150                return mutationRate;
151        }
152
153        /**
154         * Sets the {@link MutationRate}.
155         * 
156         * @param mutationRate
157         *            the mutation rate
158         */
159        public void setMutationRate(double mutationRate) {
160                this.mutationRate = mutationRate;
161        }
162
163        /**
164         * Returns the permutation mode.
165         * 
166         * @return the permutation mode
167         */
168        public PermutationType getPermutationType() {
169                return permutationType;
170        }
171
172        /**
173         * Sets the permutation mode.
174         * 
175         * @param permutationMode
176         *            the permutation mode
177         */
178        public void setPermutationType(PermutationType permutationMode) {
179                this.permutationType = permutationMode;
180        }
181
182        /**
183         * Returns the double type.
184         * 
185         * @return the doubleType
186         */
187        public DoubleType getDoubleType() {
188                return doubleType;
189        }
190
191        /**
192         * Sets the double type.
193         * 
194         * @param doubleType
195         *            the doubleType to set
196         */
197        public void setDoubleType(DoubleType doubleType) {
198                this.doubleType = doubleType;
199        }
200
201        /**
202         * Returns the eta.
203         * 
204         * @return the eta
205         */
206        public double getEta() {
207                return eta;
208        }
209
210        /**
211         * Sets the eta.
212         * 
213         * @param eta
214         *            the eta to set
215         */
216        public void setEta(double eta) {
217                this.eta = eta;
218        }
219
220        /**
221         * Returns the sigma value.
222         * 
223         * @return the sigma
224         */
225
226        public double getSigma() {
227                return sigma;
228        }
229
230        /**
231         * Sets the sigma value.
232         * 
233         * @param sigma
234         *            the sigma to set
235         */
236        public void setSigma(double sigma) {
237                this.sigma = sigma;
238        }
239
240        /*
241         * (non-Javadoc)
242         * 
243         * @see org.opt4j.start.Opt4JModule#config()
244         */
245        @Override
246        public void config() {
247                switch (mutationRateType) {
248                case ADAPTIVE:
249                        bind(MutationRate.class).to(AdaptiveMutationRate.class).in(SINGLETON);
250                        break;
251                case CONSTANT:
252                        bind(MutationRate.class).to(ConstantMutationRate.class).in(SINGLETON);
253                        break;
254                default:
255                        bind(MutationRate.class).to(AdaptiveMutationRate.class).in(SINGLETON);
256                        break;
257                }
258
259                Class<? extends MutatePermutation> permutation = MutatePermutationMixed.class;
260
261                switch (permutationType) {
262                case MIXED:
263                        permutation = MutatePermutationMixed.class;
264                        break;
265                case SWAP:
266                        permutation = MutatePermutationSwap.class;
267                        break;
268                case INSERT:
269                        permutation = MutatePermutationInsert.class;
270                        break;
271                case REVERT:
272                        permutation = MutatePermutationRevert.class;
273                        break;
274                default:
275                        break;
276                }
277                bind(MutatePermutation.class).to(permutation).in(SINGLETON);
278
279                switch (doubleType) {
280                case GAUSS:
281                        bind(MutateDouble.class).to(MutateDoubleGauss.class).in(SINGLETON);
282                        break;
283                case POLYNOMIAL:
284                        bind(MutateDouble.class).to(MutateDoublePolynomial.class).in(SINGLETON);
285                        break;
286                default:
287                        bind(MutateDouble.class).to(MutateDoublePolynomial.class).in(SINGLETON);
288                        break;
289                }
290        }
291}
