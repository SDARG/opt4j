023package org.opt4j.operators.crossover;
024
025import java.util.List;
026
027import org.opt4j.core.common.random.Rand;
028import org.opt4j.core.start.Constant;
029import org.opt4j.operators.normalize.NormalizeDouble;
030
031import com.google.inject.Inject;
032
033/**
034 * The {@link CrossoverDoubleUnfairAverage} implements the unfair average
035 * crossover proposed by Nomura and Miyhoshi, 1996.
036 * 
037 * <p>
038 * The unfair average crossover is applied to the whole genotype, i.e., the used
039 * vector of double values. In the original work, two offspring created using
040 * </p>
041 * <p>
042 * {@code z_1 = (1 + alpha)x - alpha * y} for {@code i = 1, ..., j} and<br>
043 * {@code z_1 = -alpha * x + (1 + alpha)y} for {@code i = j + 1, ..., n}<br>
044 * {@code z_2 = (1 - alpha)x + alpha * y} for {@code i = 1, ..., j} and<br>
045 * {@code z_2 = alpha * x + (1 - alpha)y} for {@code i = j + 1, ..., n}
046 * </p>
047 * <p>
048 * for each double value in the double vector. In this implementation, one
049 * offspring is created following {@code z_1} with a probability of {@code 0.5}
050 * and following {@code z_2} with a probability of {@code 0.5}, respectively. At
051 * this juncture, {@code n} is the number of variables in the vector and
052 * {@code j} a randomly chosen integer between {@code 1} and {@code n}. The
053 * value {@code alpha} is within the range (0, 0.5).
054 * </p>
055 * <p>
056 * Unlike the BLX and SBX operator, the unfair average will create offspring
057 * towards one of the parent solutions.
058 * </p>
059 * 
060 * @author glass
061 * 
062 */
063public class CrossoverDoubleUnfairAverage extends CrossoverDouble {
064
065        protected final double alpha;
066
067        /**
068         * Constructs an {@link CrossoverDoubleUnfairAverage} with an alpha value
069         * and a random number generator.
070         * 
071         * @param alpha
072         *            the alpha value
073         * @param normalize
074         *            a normalize operator
075         * @param random
076         *            the random number generator
077         */
078        @Inject
079        public CrossoverDoubleUnfairAverage(
080                        @Constant(value = "alpha", namespace = CrossoverDoubleUnfairAverage.class) double alpha,
081                        NormalizeDouble normalize, Rand random) {
082                super(normalize, random);
083                this.alpha = alpha;
084        }
085
086        /*
087         * (non-Javadoc)
088         * 
089         * @see
090         * org.opt4j.operator.crossover.CrossoverDouble#crossover(java.util.List,
091         * java.util.List, java.util.List)
092         */
093        @Override
094        protected void crossover(List<Double> p1, List<Double> p2, List<Double> o1, List<Double> o2) {
095                double z1;
096                double z2;
097                double x;
098                double y;
099
100                int size = p1.size();
101                int j = random.nextInt(size);
102
103                for (int i = 0; i < size; i++) {
104                        x = p1.get(i);
105                        y = p2.get(i);
106
107                        if (i <= j) {
108                                z1 = (1 + alpha) * x - alpha * y;
109                                z2 = (1 - alpha) * x + alpha * y;
110                        } else {
111                                z1 = -alpha * x + (1 + alpha) * y;
112                                z2 = alpha * x + (1 - alpha) * y;
113                        }
114                        o1.add(z1);
115                        o2.add(z2);
116                }
117        }
118
119}
