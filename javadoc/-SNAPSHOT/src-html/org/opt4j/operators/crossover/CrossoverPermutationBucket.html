023package org.opt4j.operators.crossover;
024
025import java.util.HashSet;
026import java.util.Random;
027import java.util.Set;
028
029import org.opt4j.core.common.random.Rand;
030import org.opt4j.core.genotype.PermutationGenotype;
031
032import com.google.inject.Inject;
033
034/**
035 * <p>
036 * Crossover for the {@link PermutationGenotype}.
037 * </p>
038 * 
039 * <p>
040 * The bucket operator randomly runs through to permutation from the beginning
041 * to the end, and fill the current element into the new permutation if not
042 * existent.
043 * </p>
044 * 
045 * <p>
046 * Given two permutations
047 * </p>
048 * <p>
049 * {@code 1 2 3 4 5 6 7 8} and {@code 8 7 6 5 4 3 2 1}
050 * </p>
051 * <p>
052 * this results, for instance, in
053 * </p>
054 * <p>
055 * {@code 1 8 2 7 3 6 4 5} or {@code 8 7 1 2 3 6 5 4}.
056 * </p>
057 * 
058 * @author lukasiewycz
059 * 
060 */
061public class CrossoverPermutationBucket implements CrossoverPermutation {
062
063        protected final Random random;
064
065        /**
066         * Constructs a new {@link CrossoverPermutationBucket}.
067         * 
068         * @param random
069         *            the random number generator
070         */
071        @Inject
072        public CrossoverPermutationBucket(Rand random) {
073                this.random = random;
074        }
075
076        /*
077         * (non-Javadoc)
078         * 
079         * @see
080         * org.opt4j.operator.crossover.Crossover#crossover(org.opt4j.core.Genotype,
081         * org.opt4j.core.Genotype)
082         */
083        @Override
084        public Pair<PermutationGenotype<?>> crossover(PermutationGenotype<?> p1, PermutationGenotype<?> p2) {
085
086                PermutationGenotype<Object> o1 = p1.newInstance();
087                PermutationGenotype<Object> o2 = p1.newInstance();
088
089                int size = p1.size();
090
091                assert (size == p2.size()) : "Permutation is undefined for genotypes with different lengths.";
092                assert p1.containsAll(p2) : "Permutation is undefined for different domains.";
093
094                Set<Object> elements = new HashSet<>();
095
096                int i = 0;
097                int j = 0;
098
099                while (o1.size() != size || o2.size() != size) {
100                        if (j == size || (random.nextBoolean() && i < size)) {
101                                Object e = p1.get(i);
102                                i++;
103                                if (elements.add(e)) {
104                                        o1.add(e);
105                                } else {
106                                        o2.add(e);
107                                }
108                        } else {
109                                Object e = p2.get(j);
110                                j++;
111
112                                if (elements.add(e)) {
113                                        o1.add(e);
114                                } else {
115                                        o2.add(e);
116                                }
117                        }
118                }
119
120                Pair<PermutationGenotype<?>> offspring = new Pair<>(o1, o2);
121                return offspring;
122        }
123}
