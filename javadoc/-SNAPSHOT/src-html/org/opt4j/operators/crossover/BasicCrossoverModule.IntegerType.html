023package org.opt4j.operators.crossover;
024
025import org.opt4j.core.config.annotations.Info;
026import org.opt4j.core.config.annotations.Required;
027import org.opt4j.core.genotype.BooleanGenotype;
028import org.opt4j.core.genotype.DoubleGenotype;
029import org.opt4j.core.genotype.IntegerGenotype;
030import org.opt4j.core.genotype.PermutationGenotype;
031import org.opt4j.core.optimizer.Operator;
032import org.opt4j.core.start.Constant;
033import org.opt4j.operators.crossover.CrossoverDoubleSBX.Nu;
034
035/**
036 * The {@link BasicCrossoverModule}.
037 * 
038 * @author lukasiewycz
039 * 
040 */
041@Info("Setting for the basic crossover classOperators for genotype variation.")
042public class BasicCrossoverModule extends CrossoverModule {
043
044        @Info("The type of the crossover operator for the Boolean genotype.")
045        protected BooleanType booleanType = BooleanType.RATE;
046
047        @Required(property = "booleanType", elements = { "RATE" })
048        @Info("The probability for a crossover point.")
049        @Constant(value = "rate", namespace = CrossoverBooleanRate.class)
050        protected double booleanRate = 0.5;
051
052        @Required(property = "booleanType", elements = { "XPOINT" })
053        @Info("The number of crossover points.")
054        @Constant(value = "x", namespace = CrossoverBooleanXPoint.class)
055        protected int booleanXPoints = 1;
056
057        @Info("The type of the crossover operator for the Double genotype.")
058        protected DoubleType doubleType = DoubleType.SBX;
059
060        @Required(property = "doubleType", elements = { "BLX", "UNFAIR_AVERAGE" })
061        protected double alpha = 0.5;
062
063        @Required(property = "doubleType", elements = { "SBX" })
064        @Constant(value = "nu", namespace = CrossoverDoubleSBX.class)
065        protected double nu = 15;
066
067        @Info("The type of the crossover operator for the Permutation genotype.")
068        protected PermutationType permutationType = PermutationType.ONEPOINT;
069
070        @Required(property = "permutationType", elements = { "ONEPOINT" })
071        @Constant(value = "rotation", namespace = CrossoverPermutationOnePoint.class)
072        protected boolean rotation = false;
073
074        @Info("The type of the crossover operator for the Integer genotype.")
075        protected IntegerType integerType = IntegerType.RATE;
076
077        @Required(property = "integerType", elements = { "RATE" })
078        @Info("The probability for a crossover point.")
079        @Constant(value = "rate", namespace = CrossoverIntegerRate.class)
080        protected double integerRate = 0.5;
081
082        @Required(property = "integerType", elements = { "XPOINT" })
083        @Info("The number of crossover points.")
084        @Constant(value = "x", namespace = CrossoverIntegerXPoint.class)
085        protected int integerXPoints = 1;
086
087        /**
088         * Type of {@link Crossover} operator for the {@link BooleanGenotype}.
089         * 
090         * @author lukasiewycz
091         */
092        public enum BooleanType {
093                /**
094                 * Use the {@link CrossoverBooleanRate}.
095                 */
096                RATE,
097                /**
098                 * Use the {@link CrossoverBooleanXPoint}.
099                 */
100                XPOINT;
101        }
102
103        /**
104         * Type of {@link Crossover} operator for the {@link DoubleGenotype}.
105         * 
106         * @author glass
107         */
108        public enum DoubleType {
109                /**
110                 * Use the {@link CrossoverDoubleBLX}.
111                 */
112                BLX,
113                /**
114                 * Use the {@link CrossoverDoubleSBX}.
115                 */
116                SBX,
117                /**
118                 * Use the {@link CrossoverDoubleUnfairAverage}.
119                 */
120                UNFAIR_AVERAGE;
121        }
122
123        /**
124         * Type of {@link Crossover} operator for the {@link PermutationGenotype}.
125         * 
126         * @author lukasiewycz
127         * 
128         */
129        public enum PermutationType {
130                /**
131                 * Use the {@link CrossoverPermutationOnePoint}.
132                 */
133                ONEPOINT,
134                /**
135                 * Use the {@link CrossoverPermutationBucket}.
136                 */
137                BUCKET;
138        }
139
140        /**
141         * Type of {@link Crossover} operator for the {@link IntegerGenotype}.
142         * 
143         * @author lukasiewycz
144         */
145        public enum IntegerType {
146                /**
147                 * Use the {@link CrossoverBooleanRate}.
148                 */
149                RATE,
150                /**
151                 * Use the {@link CrossoverBooleanXPoint}.
152                 */
153                XPOINT;
154        }
155
156        /**
157         * Returns the rate of the {@link CrossoverIntegerRate}.
158         * 
159         * @return the rate
160         */
161        public double getIntegerRate() {
162                return integerRate;
163        }
164
165        /**
166         * Sets the rate of the {@link CrossoverIntegerRate}.
167         * 
168         * @param integerRate
169         *            the rate
170         */
171        public void setIntegerRate(double integerRate) {
172                this.integerRate = integerRate;
173        }
174
175        /**
176         * Returns the number of crossover points of the
177         * {@link CrossoverIntegerXPoint}.
178         * 
179         * @return the number of crossover points
180         */
181        public int getIntegerXPoints() {
182                return integerXPoints;
183        }
184
185        /**
186         * Sets the number of crossover points of the {@link CrossoverIntegerXPoint}
187         * .
188         * 
189         * @param integerXPoint
190         *            the number of crossover points
191         */
192        public void setIntegerXPoints(int integerXPoint) {
193                this.integerXPoints = integerXPoint;
194        }
195
196        /**
197         * Return the {@link Operator} for {@link IntegerGenotype}.
198         * 
199         * @return the operator
200         */
201        public IntegerType getIntegerType() {
202                return integerType;
203        }
204
205        /**
206         * Sets the {@link Operator} for {@link IntegerGenotype}.
207         * 
208         * @param integerType
209         *            the operator
210         */
211        public void setIntegerType(IntegerType integerType) {
212                this.integerType = integerType;
213        }
214
215        /**
216         * Returns the rate of the {@link CrossoverBooleanRate}.
217         * 
218         * @return the rate
219         */
220        public double getBooleanRate() {
221                return booleanRate;
222        }
223
224        /**
225         * Sets the rate of the {@link CrossoverBooleanRate}.
226         * 
227         * @param booleanRate
228         *            the rate
229         */
230        public void setBooleanRate(double booleanRate) {
231                this.booleanRate = booleanRate;
232        }
233
234        /**
235         * Returns the number of crossover points of the
236         * {@link CrossoverBooleanXPoint}.
237         * 
238         * @return the number of crossover points
239         */
240        public int getBooleanXPoints() {
241                return booleanXPoints;
242        }
243
244        /**
245         * Sets the number of crossover points of the {@link CrossoverBooleanXPoint}
246         * .
247         * 
248         * @param booleanXPoints
249         *            the number of crossover points
250         */
251        public void setBooleanXPoints(int booleanXPoints) {
252                this.booleanXPoints = booleanXPoints;
253        }
254
255        /**
256         * Return the {@link Operator} for {@link BooleanGenotype}.
257         * 
258         * @return the operator
259         */
260        public BooleanType getBooleanType() {
261                return booleanType;
262        }
263
264        /**
265         * Sets the {@link Operator} for {@link BooleanGenotype}.
266         * 
267         * @param booleanType
268         *            the operator
269         */
270        public void setBooleanType(BooleanType booleanType) {
271                this.booleanType = booleanType;
272        }
273
274        /**
275         * Returns the alpha value of the {@link CrossoverDoubleBLX} or
276         * {@link CrossoverDoubleUnfairAverage}, respectively.
277         * 
278         * @return the alpha value
279         */
280        public double getAlpha() {
281                return alpha;
282        }
283
284        /**
285         * Sets the alpha value of the {@link CrossoverDoubleBLX} or
286         * {@link CrossoverDoubleUnfairAverage}, respectively.
287         * 
288         * @param alpha
289         *            the alpha value
290         */
291        public void setAlpha(double alpha) {
292                this.alpha = alpha;
293        }
294
295        /**
296         * Returns the nu value of the {@link CrossoverDoubleSBX}.
297         * 
298         * @return the nu value
299         */
300        public double getNu() {
301                return nu;
302        }
303
304        /**
305         * Sets the nu value of the {@link CrossoverDoubleSBX}.
306         * 
307         * @param nu
308         *            the nu value
309         */
310        public void setNu(double nu) {
311                this.nu = nu;
312        }
313
314        /**
315         * Return the {@link Operator} for {@link DoubleGenotype}.
316         * 
317         * @return the operator
318         */
319        public DoubleType getDoubleType() {
320                return doubleType;
321        }
322
323        /**
324         * Sets the {@link Operator} for {@link DoubleGenotype}.
325         * 
326         * @param doubleType
327         *            the operator
328         */
329        public void setDoubleType(DoubleType doubleType) {
330                this.doubleType = doubleType;
331        }
332
333        /**
334         * Returns the {@link PermutationType}.
335         * 
336         * @return the permutation type
337         */
338        public PermutationType getPermutationType() {
339                return permutationType;
340        }
341
342        /**
343         * Sets the {@link PermutationType}.
344         * 
345         * @param permutationType
346         *            the permutation type
347         */
348        public void setPermutationType(PermutationType permutationType) {
349                this.permutationType = permutationType;
350        }
351
352        /**
353         * Returns {@code true} if rotation is used for the {@link Crossover}
354         * operator for the {@link PermutationGenotype}.
355         * 
356         * @return {@code true} if rotation is used
357         */
358        public boolean isRotation() {
359                return rotation;
360        }
361
362        /**
363         * Sets the rotation value for the {@link Crossover} operator for the
364         * {@link PermutationGenotype}.
365         * 
366         * @param rotation
367         *            the rotation value
368         */
369        public void setRotation(boolean rotation) {
370                this.rotation = rotation;
371        }
372
373        /*
374         * (non-Javadoc)
375         * 
376         * @see org.opt4j.start.Opt4JModule#config()
377         */
378        @Override
379        public void config() {
380                switch (booleanType) {
381                case XPOINT:
382                        bind(CrossoverBoolean.class).to(CrossoverBooleanXPoint.class).in(SINGLETON);
383                        break;
384                default: // RATE
385                        bind(CrossoverBoolean.class).to(CrossoverBooleanRate.class).in(SINGLETON);
386                        break;
387                }
388
389                switch (integerType) {
390                case XPOINT:
391                        bind(CrossoverInteger.class).to(CrossoverIntegerXPoint.class).in(SINGLETON);
392                        break;
393                default: // RATE
394                        bind(CrossoverInteger.class).to(CrossoverIntegerRate.class).in(SINGLETON);
395                        break;
396                }
397
398                switch (doubleType) {
399                case BLX:
400                        bind(CrossoverDouble.class).to(CrossoverDoubleBLX.class).in(SINGLETON);
401                        break;
402                case UNFAIR_AVERAGE:
403                        bind(CrossoverDouble.class).to(CrossoverDoubleUnfairAverage.class).in(SINGLETON);
404
405                        break;
406                default: // SBX
407                        bind(CrossoverDouble.class).to(CrossoverDoubleSBX.class).in(SINGLETON);
408                        break;
409                }
410
411                switch (permutationType) {
412                case BUCKET:
413                        bind(CrossoverPermutation.class).to(CrossoverPermutationBucket.class).in(SINGLETON);
414                        break;
415                default: // ONEPOINT
416                        bind(CrossoverPermutation.class).to(CrossoverPermutationOnePoint.class).in(SINGLETON);
417                        break;
418                }
419
420                bindConstant("alpha", CrossoverDoubleBLX.class).to(alpha);
421                bindConstant("alpha", CrossoverDoubleUnfairAverage.class).to(alpha);
422                bindConstant(Nu.class).to(nu);
423        }
