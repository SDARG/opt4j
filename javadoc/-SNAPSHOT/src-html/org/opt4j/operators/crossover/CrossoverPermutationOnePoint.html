023package org.opt4j.operators.crossover;
024
025import java.util.HashSet;
026import java.util.Random;
027import java.util.Set;
028
029import org.opt4j.core.common.random.Rand;
030import org.opt4j.core.genotype.PermutationGenotype;
031import org.opt4j.core.start.Constant;
032
033import com.google.inject.Inject;
034
035/**
036 * <p>
037 * Crossover for the {@link PermutationGenotype}.
038 * </p>
039 * 
040 * <p>
041 * This operator takes sublist of the first permutation from the beginning to a
042 * random cut point and fills the remaining elements from the second
043 * permutation.
044 * </p>
045 * 
046 * <p>
047 * Given two permutations<br />
048 * {@code 1 2 3 4 5 6 7 8} and {@code 8 7 6 5 4 3 2 1}.<br />
049 * This results, for instance, in<br/>
050 * {@code 1 2 3 4 5 8 7 6} or {@code 1 2 8 7 6 5 4 3}.
051 * </p>
052 * 
053 * <p>
054 * Additionally, a {@code rotation} value defines if the initial permutation are
055 * randomly rotated. With the {@code rotation} possible results would be<br />
056 * {@code 4 5 6 8 7 3 2 1} and {@code 7 8 1 2 4 3 6 5}.
057 * </p>
058 * 
059 * @author lukasiewycz
060 * 
061 */
062public class CrossoverPermutationOnePoint implements CrossoverPermutation {
063
064        protected final Random random;
065
066        protected final boolean rotation;
067
068        /**
069         * Constructs a new {@link CrossoverPermutationOnePoint}.
070         * 
071         * @param random
072         *            the random number generator
073         * 
074         * @param rotation
075         *            use rotation
076         */
077        @Inject
078        public CrossoverPermutationOnePoint(Rand random,
079                        @Constant(value = "rotation", namespace = CrossoverPermutationOnePoint.class) boolean rotation) {
080                this.random = random;
081                this.rotation = rotation;
082        }
083
084        /*
085         * (non-Javadoc)
086         * 
087         * @see
088         * org.opt4j.operator.crossover.Crossover#crossover(org.opt4j.core.Genotype,
089         * org.opt4j.core.Genotype)
090         */
091        @Override
092        public Pair<PermutationGenotype<?>> crossover(PermutationGenotype<?> p1, PermutationGenotype<?> p2) {
093
094                PermutationGenotype<Object> o1 = p1.newInstance();
095                PermutationGenotype<Object> o2 = p1.newInstance();
096
097                int size = p1.size();
098
099                assert (size == p2.size()) : "Permutation is undefined for genotypes with different lengths.";
100                assert p1.containsAll(p2) : "Permutation is undefined for different domains.";
101
102                if (size > 0) {
103                        Set<Object> elements1 = new HashSet<>();
104                        Set<Object> elements2 = new HashSet<>();
105
106                        crossoverToCutpoint(p1, p2, o1, o2, elements1, elements2);
107                        crossoverFill(p1, p2, o1, o2, elements1, elements2);
108
109                }
110
111                Pair<PermutationGenotype<?>> offspring = new Pair<>(o1, o2);
112                return offspring;
113        }
114
115        /**
116         * Helper function for crossover() to crossover and rotate up to the coined
117         * cut point.
118         * 
119         * @param p1
120         *            parent one
121         * @param p2
122         *            parent two
123         * @param o1
124         *            child one
125         * @param o2
126         *            child two
127         * @param elements1
128         *            elements of genotype child one
129         * @param elements2
130         *            elements of genotype child two
131         */
132        protected void crossoverToCutpoint(PermutationGenotype<?> p1, PermutationGenotype<?> p2,
133                        PermutationGenotype<Object> o1, PermutationGenotype<Object> o2, Set<Object> elements1,
134                        Set<Object> elements2) {
135                int size = p1.size();
136                int offset = rotation ? random.nextInt(size) : 0;
137
138                int cutpoint = random.nextInt(size);
139
140                for (int i = 0; i < cutpoint; i++) {
141                        final int pos = (offset + i) % size;
142                        Object e1 = p1.get(pos);
143                        Object e2 = p2.get(pos);
144                        o1.add(e1);
145                        o2.add(e2);
146                        elements1.add(e1);
147                        elements2.add(e2);
148                }
149
150        }
151
152        /**
153         * Helper function for crossover() to fill and rotate after the cut point.
154         * 
155         * @param p1
156         *            parent one
157         * @param p2
158         *            parent two
159         * @param o1
160         *            child one
161         * @param o2
162         *            child two
163         * @param elements1
164         *            elements of genotype child one
165         * @param elements2
166         *            elements of genotype child two
167         */
168        protected void crossoverFill(PermutationGenotype<?> p1, PermutationGenotype<?> p2, PermutationGenotype<Object> o1,
169                        PermutationGenotype<Object> o2, Set<Object> elements1, Set<Object> elements2) {
170                int size = p1.size();
171                int offset = rotation ? random.nextInt(size) : 0;
172
173                for (int i = 0; i < size; i++) {
174                        final int pos = (offset + i) % size;
175                        Object e1 = p1.get(pos);
176                        Object e2 = p2.get(pos);
177                        if (!elements1.contains(e2)) {
178                                o1.add(e2);
179                        }
180                        if (!elements2.contains(e1)) {
181                                o2.add(e1);
182                        }
183                }
184
185        }
186
187}
