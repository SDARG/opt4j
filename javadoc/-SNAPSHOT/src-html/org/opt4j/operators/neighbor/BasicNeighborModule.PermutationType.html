023package org.opt4j.operators.neighbor;
024
025import org.opt4j.core.config.annotations.Info;
026import org.opt4j.core.genotype.PermutationGenotype;
027
028/**
029 * The basic {@link NeighborModule}.
030 * 
031 * @author lukasiewycz
032 * 
033 */
034@Info("Setting for the basic neighbor classOperators for genotype variation.")
035public class BasicNeighborModule extends NeighborModule {
036
037        @Info("The type of the neighbor operator for the Permutation genotype.")
038        protected PermutationType permutationType = PermutationType.MIXED;
039
040        /**
041         * Type of {@link Neighbor} operator for the {@link PermutationGenotype}.
042         * 
043         * @author lukasiewycz
044         * 
045         */
046        public enum PermutationType {
047                /**
048                 * Use the {@link NeighborPermutationMixed}.
049                 */
050                @Info("Use randomly SWAP,INSERT, or REVERT")
051                MIXED,
052                /**
053                 * Use the {@link NeighborPermutationSwap}.
054                 */
055                @Info("Swaps two elements")
056                SWAP,
057                /**
058                 * Use the {@link NeighborPermutationInsert}.
059                 */
060                @Info("Moves one element to another position")
061                INSERT,
062                /**
063                 * Use the {@link NeighborPermutationRevert}.
064                 */
065                @Info("Reverts a part for the genotype")
066                REVERT;
067        }
068
069        /**
070         * Returns the permutation mode.
071         * 
072         * @return the permutation mode
073         */
074        public PermutationType getPermutationType() {
075                return permutationType;
076        }
077
078        /**
079         * Sets the permutation mode.
080         * 
081         * @param permutationMode
082         *            the permutation mode
083         */
084        public void setPermutationType(PermutationType permutationMode) {
085                this.permutationType = permutationMode;
086        }
087
088        /*
089         * (non-Javadoc)
090         * 
091         * @see org.opt4j.start.Opt4JModule#config()
092         */
093        @Override
094        public void config() {
095
096                Class<? extends NeighborPermutation> permutation = NeighborPermutationMixed.class;
097                switch (permutationType) {
098                case MIXED:
099                        permutation = NeighborPermutationMixed.class;
100                        break;
101                case SWAP:
102                        permutation = NeighborPermutationSwap.class;
103                        break;
104                case INSERT:
105                        permutation = NeighborPermutationInsert.class;
106                        break;
107                case REVERT:
108                        permutation = NeighborPermutationRevert.class;
109                        break;
110                default:
111                        permutation = NeighborPermutationMixed.class;
112                        break;
113                }
114                bind(NeighborPermutation.class).to(permutation).in(SINGLETON);
115        }
116}
