024package org.opt4j.core;
025
026import static org.opt4j.core.Objective.Sign.MIN;
027
028/**
029 * <p>
030 * The {@link Objective} is the identifier for a single objective in the
031 * {@link Objectives}. It is specified by the following properties:
032 * </p>
033 * <ul>
034 * <li>Name</li>
035 * <li>Minimization or Maximization</li>
036 * </ul>
037 * <p>
038 * Each {@link org.opt4j.core.problem.Evaluator} sets a specific amount of
039 * {@link Objective}-{@link Value} pairs. Moreover, each
040 * {@link org.opt4j.core.problem.Evaluator} has to set the same
041 * {@link Objective}s for all phenotypes.
042 * </p>
043 * 
044 * @see Objectives
045 * @author lukasiewycz
046 * 
047 */
048public class Objective implements Comparable<Objective> {
049
050        /**
051         * The sign of the {@link Objective}.
052         */
053        public enum Sign {
054                /**
055                 * Minimize the {@link Objective}.
056                 */
057                MIN,
058                /**
059                 * Maximize the {@link Objective}.
060                 */
061                MAX;
062        }
063
064        /**
065         * Identifier for infeasible results ({@code null}).
066         */
067        public static final Value<?> INFEASIBLE = null;
068
069        protected final Sign sign;
070
071        protected final String name;
072
073        /**
074         * Constructs an {@link Objective} with a given name, sign=MIN, and
075         * rank=RANK_OBJECTIVE(0).
076         * 
077         * @param name
078         *            the name
079         */
080        public Objective(String name) {
081                this(name, MIN);
082        }
083
084        /**
085         * Constructs an {@link Objective} with a given name, sign, and
086         * rank=RANK_OBJECTIVE(0).
087         * 
088         * @param name
089         *            the name
090         * @param sign
091         *            the sign of the objective
092         */
093        public Objective(String name, Sign sign) {
094                this.name = name;
095                this.sign = sign;
096        }
097
098        /**
099         * Returns the sign.
100         * 
101         * @return the sign
102         */
103        public Sign getSign() {
104                return sign;
105        }
106
107        /**
108         * Returns the name.
109         * 
110         * @return the name
111         */
112        public String getName() {
113                return name;
114        }
115
116        /*
117         * (non-Javadoc)
118         * 
119         * @see java.lang.Comparable#compareTo(java.lang.Object)
120         */
121        @Override
122        public int compareTo(Objective other) {
123                if (this.equals(other)) {
124                        return 0;
125                } else if (other == null) {
126                        return 1;
127                } else {
128                        return this.getName().compareTo(other.getName());
129                }
130        }
131
132        /*
133         * (non-Javadoc)
134         * 
135         * @see java.lang.Object#toString()
136         */
137        @Override
138        public String toString() {
139                return getName() + "(" + getSign() + ")";
140        }
141
142        /*
143         * (non-Javadoc)
144         * 
145         * @see java.lang.Object#hashCode()
146         */
147        @Override
148        public int hashCode() {
149                final int prime = 31;
150                int result = 1;
151                result = prime * result + ((name == null) ? 0 : name.hashCode());
152                result = prime * result + ((sign == null) ? 0 : sign.hashCode());
153                return result;
154        }
155
156        /*
157         * (non-Javadoc)
158         * 
159         * @see java.lang.Object#equals(java.lang.Object)
160         */
161        @Override
162        public boolean equals(Object obj) {
163                if (this == obj)
164                        return true;
165                if (obj == null)
166                        return false;
167                if (getClass() != obj.getClass())
168                        return false;
169                Objective other = (Objective) obj;
170                if (name == null) {
171                        if (other.name != null)
172                                return false;
173                } else if (!name.equals(other.name))
174                        return false;
175                return sign == other.sign;
176        }
177
178}
