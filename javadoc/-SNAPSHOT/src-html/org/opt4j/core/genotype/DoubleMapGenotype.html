023package org.opt4j.core.genotype;
024
025import java.lang.reflect.Constructor;
026import java.util.Collection;
027import java.util.Collections;
028import java.util.HashSet;
029import java.util.List;
030import java.util.Random;
031import java.util.Set;
032
033import org.opt4j.core.Genotype;
034
035/**
036 * <p>
037 * The {@link DoubleMapGenotype} is a {@link DoubleGenotype} with the
038 * {@link MapGenotype} functionality.
039 * </p>
040 * <p>
041 * Example problem: Select filling level of five bottles bottle<sub>1</sub>,
042 * bottle<sub>2</sub>, bottle<sub>3</sub>, bottle<sub>4</sub>,
043 * bottle<sub>5</sub><br/>
044 * Example usage: <blockquote>
045 * 
046 * <pre>
047 * DoubleMapGenotype&lt;Bottle&gt; genotype = new DoubleMapGenotype&lt;Bottle&gt;(
048 *              Arrays.asList(bottle1, bottle2, bottle3, bottle4, bottle5));
049 * genotype.init(new Random());
050 * </pre>
051 * 
052 * </blockquote> Example instance:
053 * [bottle1=0.5035947840006195,bottle2=0.9693492473483428
054 * ,bottle3=0.12786372316728167
055 * ,bottle4=0.5299369900029843,bottle5=0.8055193291478467]<br/>
056 * Example search space size: [0;1]<sup>5</sup>
057 * </p>
058 * 
059 * @author lukasiewycz
060 * @param <K>
061 *            the type of keys
062 */
063public class DoubleMapGenotype<K> extends DoubleGenotype implements MapGenotype<K, Double> {
064
065        protected final List<K> keys;
066
067        private static final long serialVersionUID = 1L;
068
069        /**
070         * Constructs a {@link DoubleMapGenotype}.
071         * 
072         * @param keys
073         *            the list of keys
074         * @param bounds
075         *            the lower and upper bounds
076         */
077        public DoubleMapGenotype(List<K> keys, Bounds<Double> bounds) {
078                super(bounds);
079                Set<K> uniqueKeys = new HashSet<>(keys);
080                if (uniqueKeys.size() < keys.size()) {
081                        throw new IllegalArgumentException(MapGenotype.ERROR_MESSAGE_NON_UNIQUE_KEYS);
082                }
083                this.keys = keys;
084        }
085
086        /**
087         * Initialize this genotype with random values based on the size of the
088         * {@code keys}.
089         * 
090         * @param random
091         *            the random number generator
092         */
093        public void init(Random random) {
094                super.init(random, keys.size());
095        }
096
097        /**
098         * Not supported. Use {@link DoubleMapGenotype#init(Random)} instead.
099         * 
100         * @see org.opt4j.core.genotype.DoubleGenotype#init(java.util.Random, int)
101         */
102        @Override
103        public void init(Random random, int n) {
104                throw new UnsupportedOperationException(MapGenotype.ERROR_MESSAGE_UNSUPPORTED_INIT);
105        }
106
107        /*
108         * (non-Javadoc)
109         * 
110         * @see org.opt4j.genotype.MapGenotype#containsKey(java.lang.Object)
111         */
112        @Override
113        public boolean containsKey(K key) {
114                return keys.contains(key);
115        }
116
117        /*
118         * (non-Javadoc)
119         * 
120         * @see org.opt4j.genotype.MapGenotype#getValue(java.lang.Object)
121         */
122        @Override
123        public Double getValue(K key) {
124                int i = keys.indexOf(key);
125                return get(i);
126        }
127
128        /*
129         * (non-Javadoc)
130         * 
131         * @see org.opt4j.genotype.MapGenotype#setValue(java.lang.Object,
132         * java.lang.Object)
133         */
134        @Override
135        public void setValue(K key, Double value) {
136                if (!containsKey(key)) {
137                        throw new IllegalArgumentException(MapGenotype.ERROR_MESSAGE_INVALID_KEY);
138                }
139                int i = keys.indexOf(key);
140                if (value < bounds.getLowerBound(i) || value > bounds.getUpperBound(i)) {
141                        throw new IllegalArgumentException(MapGenotype.ERROR_MESSAGE_OUT_OF_BOUNDS);
142                }
143                while (size() <= i) {
144                        add(bounds.getLowerBound(i));
145                }
146                set(i, value);
147        }
148
149        /*
150         * (non-Javadoc)
151         * 
152         * @see org.opt4j.core.Genotype#newInstance()
153         */
154        @Override
155        @SuppressWarnings({ "unchecked", "rawtypes" })
156        public <G extends Genotype> G newInstance() {
157                try {
158                        Constructor<? extends DoubleMapGenotype> cstr = this.getClass().getConstructor(List.class, Bounds.class);
159                        return (G) cstr.newInstance(keys, bounds);
160                } catch (Exception e) {
161                        throw new RuntimeException(e);
162                }
163        }
164
165        /*
166         * (non-Javadoc)
167         * 
168         * @see java.util.AbstractCollection#toString()
169         */
170        @Override
171        public String toString() {
172                StringBuilder stringBuilder = new StringBuilder();
173                stringBuilder.append("[");
174                for (int i = 0; i < size(); i++) {
175                        K key = keys.get(i);
176                        double value = this.get(i);
177                        stringBuilder.append(key + "=" + value + ";");
178                }
179                stringBuilder.append("]");
180                return stringBuilder.toString();
181        }
182
183        /*
184         * (non-Javadoc)
185         * 
186         * @see org.opt4j.genotype.MapGenotype#getIndexOf(java.lang.Object)
187         */
188        @Override
189        public int getIndexOf(K key) {
190                if (!containsKey(key)) {
191                        throw new IllegalArgumentException(MapGenotype.ERROR_MESSAGE_INVALID_KEY);
192                }
193                return keys.indexOf(key);
194        }
195
196        /*
197         * (non-Javadoc)
198         * 
199         * @see org.opt4j.genotype.MapGenotype#getKeys()
200         */
201        @Override
202        public Collection<K> getKeys() {
203                return Collections.unmodifiableList(keys);
204        }
205
206}
