023package org.opt4j.core.genotype;
024
025import java.lang.reflect.Constructor;
026import java.util.Collection;
027import java.util.Collections;
028import java.util.HashSet;
029import java.util.List;
030import java.util.Random;
031import java.util.Set;
032
033import org.opt4j.core.Genotype;
034
035/**
036 * <p>
037 * The {@link BooleanMapGenotype} is a {@link BooleanGenotype} with the
038 * {@link MapGenotype} functionality.
039 * </p>
040 * <p>
041 * Example problem: Select on/off state of five switches switch<sub>1</sub>,
042 * switch<sub>2</sub>, switch<sub>3</sub>, switch<sub>4</sub>,
043 * switch<sub>5</sub><br/>
044 * Example usage: <blockquote>
045 * 
046 * <pre>
047 * BooleanMapGenotype&lt;Switch&gt; genotype = new BooleanMapGenotype&lt;Switch&gt;(
048 *              Arrays.asList(switch1, switch2, switch3, switch4, switch5));
049 * genotype.init(new Random());
050 * </pre>
051 * 
052 * </blockquote> Example instance:
053 * [switch1=true;switch2=true;switch3=false;switch4=false;switch5=true;] <br/>
054 * Example search space size: 2<sup>5</sup>
055 * </p>
056 * 
057 * @author lukasiewycz
058 * 
059 * @param <K>
060 *            the type of keys
061 */
062public class BooleanMapGenotype<K> extends BooleanGenotype implements MapGenotype<K, Boolean> {
063
064        protected final List<K> list;
065
066        private static final long serialVersionUID = 1L;
067
068        /**
069         * Constructs a {@link BooleanMapGenotype}.
070         * 
071         * @param list
072         *            the list of keys
073         */
074        public BooleanMapGenotype(List<K> list) {
075                super();
076                Set<K> uniqueKeys = new HashSet<>(list);
077                if (uniqueKeys.size() < list.size()) {
078                        throw new IllegalArgumentException(MapGenotype.ERROR_MESSAGE_NON_UNIQUE_KEYS);
079                }
080                this.list = list;
081        }
082
083        /**
084         * Initialize this genotype with random values based on the number of keys.
085         * 
086         * @param random
087         *            the random number generator
088         */
089        public void init(Random random) {
090                super.init(random, list.size());
091        }
092
093        /**
094         * Not supported. Use {@link BooleanMapGenotype#init(Random)} instead.
095         * 
096         * @see org.opt4j.core.genotype.BooleanGenotype#init(java.util.Random, int)
097         */
098        @Override
099        public void init(Random random, int n) {
100                throw new UnsupportedOperationException(MapGenotype.ERROR_MESSAGE_UNSUPPORTED_INIT);
101        }
102
103        /*
104         * (non-Javadoc)
105         * 
106         * @see org.opt4j.genotype.MapGenotype#getValue(java.lang.Object)
107         */
108        @Override
109        public Boolean getValue(K key) {
110                if (!containsKey(key)) {
111                        throw new IllegalArgumentException(MapGenotype.ERROR_MESSAGE_INVALID_KEY);
112                }
113                int i = list.indexOf(key);
114                return get(i);
115        }
116
117        /*
118         * (non-Javadoc)
119         * 
120         * @see org.opt4j.genotype.MapGenotype#setValue(java.lang.Object,
121         * java.lang.Object)
122         */
123        @Override
124        public void setValue(K key, Boolean value) {
125                if (!containsKey(key)) {
126                        throw new IllegalArgumentException(MapGenotype.ERROR_MESSAGE_INVALID_KEY);
127                }
128                int i = list.indexOf(key);
129                while (size() <= i) {
130                        add(false);
131                }
132                set(i, value);
133        }
134
135        /*
136         * (non-Javadoc)
137         * 
138         * @see org.opt4j.genotype.MapGenotype#containsKey(java.lang.Object)
139         */
140        @Override
141        public boolean containsKey(K key) {
142                return list.contains(key);
143        }
144
145        /*
146         * (non-Javadoc)
147         * 
148         * @see org.opt4j.core.Genotype#newInstance()
149         */
150        @Override
151        @SuppressWarnings({ "unchecked", "rawtypes" })
152        public <G extends Genotype> G newInstance() {
153                try {
154                        Constructor<? extends BooleanMapGenotype> cstr = this.getClass().getConstructor(List.class);
155                        return (G) cstr.newInstance(list);
156                } catch (Exception e) {
157                        throw new RuntimeException(e);
158                }
159        }
160
161        /*
162         * (non-Javadoc)
163         * 
164         * @see java.util.AbstractCollection#toString()
165         */
166        @Override
167        public String toString() {
168                StringBuilder stringBuilder = new StringBuilder();
169                stringBuilder.append("[");
170                for (int i = 0; i < size(); i++) {
171                        K key = list.get(i);
172                        boolean value = this.get(i);
173                        stringBuilder.append(key + "=" + value + ";");
174                }
175                stringBuilder.append("]");
176                return stringBuilder.toString();
177        }
178
179        /*
180         * (non-Javadoc)
181         * 
182         * @see org.opt4j.genotype.MapGenotype#getIndexOf(java.lang.Object)
183         */
184        @Override
185        public int getIndexOf(K key) {
186                if (!containsKey(key)) {
187                        throw new IllegalArgumentException(ERROR_MESSAGE_INVALID_KEY);
188                }
189                return list.indexOf(key);
190        }
191
192        /*
193         * (non-Javadoc)
194         * 
195         * @see org.opt4j.genotype.MapGenotype#getKeys()
196         */
197        @Override
198        public Collection<K> getKeys() {
199                return Collections.unmodifiableList(list);
200        }
201
202}
