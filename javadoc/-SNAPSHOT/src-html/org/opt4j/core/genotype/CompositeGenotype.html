023package org.opt4j.core.genotype;
024
025import java.util.Collection;
026import java.util.HashMap;
027import java.util.Iterator;
028import java.util.Map;
029import java.util.Map.Entry;
030import java.util.Set;
031
032import org.opt4j.core.Genotype;
033
034/**
035 * The {@link CompositeGenotype} is a base class for {@link Genotype} classes
036 * that consist of multiple {@link Genotype}s. The method
037 * {@link CompositeGenotype#size()} returns the sum of the sizes of the
038 * contained {@link Genotype}s.
039 * <p>
040 * A specific {@link CompositeGenotype} has to add each contained
041 * {@link Genotype} by calling the method {@link #put(Object, Genotype)} where
042 * {@link Object} is an arbitrary identifier.
043 * <p>
044 * Example:
045 * 
046 * <pre>
047 *      SpecificGenotype extends CompositeGenotype&lt;Integer, Genotype&gt; {
048 *              public void setDoubleVector(DoubleGenotype genotype) {
049 *                      put(0, genotype);
050 *              }
051 * 
052 *              public DoubleGenotype getDoubleVector() { 
053 *                      return get(0);
054 *              }
055 * 
056 *              public void setBinaryVector(BooleanGenotype genotype) {
057 *                      put(1, genotype);
058 *              }
059 * 
060 *              public BooleanGenotype getBinaryVector() {
061 *                      return get(1); 
062 *              }
063 *      }
064 * </pre>
065 * 
066 * @param <K>
067 *            the type of key for the mapping
068 * @param <V>
069 *            the type of {@link Genotype}
070 * 
071 * @author lukasiewycz
072 */
073public class CompositeGenotype<K, V extends Genotype> implements Genotype, Iterable<Entry<K, V>> {
074
075        protected final Map<K, V> map = new HashMap<>();
076
077        /**
078         * Constructs a {@link CompositeGenotype}.
079         */
080        public CompositeGenotype() {
081                super();
082        }
083
084        /**
085         * Constructs a {@link CompositeGenotype} with values from a given map.
086         * 
087         * @param map
088         *            initial values
089         */
090        public CompositeGenotype(Map<K, V> map) {
091                this();
092                this.map.putAll(map);
093        }
094
095        /*
096         * (non-Javadoc)
097         * 
098         * @see org.opt4j.core.Genotype#size()
099         */
100        @Override
101        public int size() {
102                int sum = 0;
103                for (Genotype element : map.values()) {
104                        sum += element.size();
105                }
106                return sum;
107        }
108
109        /**
110         * Returns the {@link Genotype} for the given key with an implicit cast to
111         * the specific {@link Genotype} type.
112         * 
113         * @param <G>
114         *            the type of genotype
115         * @param key
116         *            the key
117         * @return the addressed genotype
118         */
119        @SuppressWarnings("unchecked")
120        public <G> G get(Object key) {
121                V value = map.get(key);
122                return (G) value;
123        }
124
125        /**
126         * Adds a {@code key}, {@code value} pair.
127         * 
128         * @param key
129         *            the key
130         * @param value
131         *            the value ({@code Genotype})
132         */
133        public void put(K key, V value) {
134                map.put(key, value);
135        }
136
137        /**
138         * Removes all {@code key}, {@code value} pairs.
139         */
140        public void clear() {
141                map.clear();
142        }
143
144        /**
145         * Returns all {@code keys}.
146         * 
147         * @return all {@code keys}
148         */
149        public Set<K> keySet() {
150                return map.keySet();
151        }
152
153        /**
154         * Returns all {@code values} which are the contained {@link Genotype}
155         * objects.
156         * 
157         * @return all {@code values}
158         */
159        public Collection<V> values() {
160                return map.values();
161        }
162
163        /**
164         * Returns the {@link Iterator} over the {@link Entry} pairs.
165         * 
166         * @return the iterator over the entry pairs
167         */
168        @Override
169        public Iterator<Entry<K, V>> iterator() {
170                return map.entrySet().iterator();
171        }
172
173        /*
174         * (non-Javadoc)
175         * 
176         * @see org.opt4j.core.Genotype#newInstance()
177         */
178        @Override
179        @SuppressWarnings("unchecked")
180        public <G extends Genotype> G newInstance() {
181                try {
182                        return (G) this.getClass().newInstance();
183                } catch (InstantiationException e) {
184                        throw new RuntimeException(e);
185                } catch (IllegalAccessException e) {
186                        throw new RuntimeException(e);
187                }
188        }
189
190        /*
191         * (non-Javadoc)
192         * 
193         * @see java.lang.Object#toString()
194         */
195        @Override
196        public String toString() {
197                StringBuilder s = new StringBuilder("[");
198                for (Entry<K, V> entry : this) {
199                        K key = entry.getKey();
200                        V value = entry.getValue();
201                        s.append(key).append("=").append(value).append(";");
202                }
203                return s.append("]").toString();
204        }
205}
