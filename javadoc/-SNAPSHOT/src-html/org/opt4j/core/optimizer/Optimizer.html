024package org.opt4j.core.optimizer;
025
026import com.google.inject.ImplementedBy;
027
028/**
029 * <p>
030 * This {@link Optimizer} interface has to be implemented by all
031 * population-based optimizers.
032 * </p>
033 * <p>
034 * To implement an iteration-based {@link org.opt4j.core.optimizer.Optimizer}s,
035 * use the simpler {@link IterativeOptimizer} interface.
036 * </p>
037 * <p>
038 * To bind an {@link Optimizer}, use {@link OptimizerModule#bindOptimizer}.
039 * </p>
040 * 
041 * @author glass, lukasiewycz
042 * 
043 */
044@ImplementedBy(OptimizationMediator.class)
045public interface Optimizer {
046
047        /**
048         * Starts the optimization process.
049         * 
050         * @throws StopException
051         *             if the optimization is stopped
052         * @throws TerminationException
053         *             if the optimization is terminated
054         */
055        public abstract void optimize() throws StopException, TerminationException;
056
057        /**
058         * Adds an {@link OptimizerIterationListener} to this optimizer.
059         * 
060         * @see #removeOptimizerIterationListener
061         * @param listener
062         *            the OptimizerIterationListener to add
063         */
064        public void addOptimizerIterationListener(OptimizerIterationListener listener);
065
066        /**
067         * Removes an {@link OptimizerIterationListener} from this optimizer.
068         * 
069         * @see #addOptimizerIterationListener
070         * @param listener
071         *            the OptimizerIterationListener to remove
072         */
073        public void removeOptimizerIterationListener(OptimizerIterationListener listener);
074
075        /**
076         * Adds an {@link OptimizerStateListener} to this optimizer.
077         * 
078         * @see #removeOptimizerStateListener
079         * @param listener
080         *            the OptimizerStateListener to add
081         */
082        public void addOptimizerStateListener(OptimizerStateListener listener);
083
084        /**
085         * Removes an {@link OptimizerStateListener} from this optimizer.
086         * 
087         * @see #addOptimizerStateListener
088         * @param listener
089         *            the OptimizerStateListener to remove
090         */
091        public void removeOptimizerStateListener(OptimizerStateListener listener);
092
093        /**
094         * Return the current iteration.
095         * 
096         * @return the current iteration
097         */
098        public int getIteration();
099
100        /**
101         * Returns {@code true} if the {@link Optimizer} is running.
102         * 
103         * @return {@code true} if the optimizer is running
104         */
105        public boolean isRunning();
106
107        /**
108         * This method is called to start the optimization process.
109         */
110        public void startOptimization();
111
112        /**
113         * This method is called once the optimization process has stopped.
114         */
115        public void stopOptimization();
116
117}
