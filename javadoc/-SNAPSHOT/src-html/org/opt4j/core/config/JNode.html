023package org.opt4j.core.config;
024
025import java.util.HashMap;
026import java.util.LinkedList;
027import java.util.List;
028import java.util.Map;
029
030import org.w3c.dom.Document;
031import org.w3c.dom.Element;
032import org.w3c.dom.NamedNodeMap;
033import org.w3c.dom.Node;
034import org.w3c.dom.NodeList;
035
036/**
037 * Wrapper for {@link org.w3c.dom.Node} elements. This class integrates Java 5
038 * functionality with generics.
039 * 
040 * @author lukasiewycz
041 * 
042 */
043public class JNode {
044
045        protected final Node node;
046
047        /**
048         * Creates a Wrapper {@link JNode} for a {@link Node} element.
049         * 
050         * @param node
051         *            the node to wrap
052         */
053        public JNode(Node node) {
054                this.node = node;
055        }
056
057        /**
058         * Constructs a new {@link JNode} for a {@link Document}.
059         * 
060         * @param document
061         *            the XML document
062         * @param name
063         *            the specified name of the node
064         */
065        public JNode(Document document, String name) {
066                node = document.createElement(name);
067        }
068
069        /**
070         * Retrieves the {@link Node} element.
071         * 
072         * @return the XML node
073         */
074        public Node getNode() {
075                return node;
076        }
077
078        /**
079         * Returns the name.
080         * 
081         * @return the name
082         */
083        public String getName() {
084                return node.getNodeName();
085        }
086
087        /**
088         * Returns a {@link Map} of all attributes.
089         * 
090         * @return a map of all attributes
091         */
092        public Map<String, String> getAttributes() {
093
094                Map<String, String> map = new HashMap<>();
095
096                try {
097
098                        NamedNodeMap attributes = node.getAttributes();
099                        for (int i = 0; i < attributes.getLength(); i++) {
100                                Node attribute = attributes.item(i);
101                                map.put(attribute.getNodeName(), attribute.getNodeValue());
102                        }
103
104                } catch (ClassCastException e) {
105                }
106
107                return map;
108        }
109
110        /**
111         * Returns the value of an attribute.
112         * 
113         * @see #setAttribute
114         * @param attribute
115         *            the name of the attribute
116         * @return the value of the attribute
117         */
118        public String getAttribute(String attribute) {
119
120                try {
121                        Element element = (Element) node;
122                        return element.getAttribute(attribute);
123                } catch (ClassCastException e) {
124                }
125
126                return null;
127
128        }
129
130        /**
131         * Returns {@code true} if the attribute is existent.
132         * 
133         * @param attribute
134         *            the name of the attribute
135         * @return the value of the attribute
136         */
137        public boolean hasAttribute(String attribute) {
138
139                try {
140                        Element element = (Element) node;
141                        if (element.hasAttribute(attribute)) {
142                                return true;
143                        }
144                } catch (ClassCastException e) {
145                }
146                return false;
147
148        }
149
150        /**
151         * Returns the first child.
152         * 
153         * @return the first child
154         */
155        public JNode getChild() {
156                return new JNode(node.getFirstChild());
157        }
158
159        /**
160         * Returns the first child with a specified name.
161         * 
162         * @param name
163         *            the name of the child
164         * @return the first child with the specified name
165         */
166        public JNode getChild(String name) {
167                List<JNode> list = getChildren();
168
169                for (JNode node : list) {
170                        if (node.getName().equalsIgnoreCase(name)) {
171                                return node;
172                        }
173                }
174                return null;
175        }
176
177        /**
178         * Returns all children.
179         * 
180         * @return a list of all children.
181         */
182        public List<JNode> getChildren() {
183
184                List<JNode> list = new LinkedList<>();
185                NodeList children = node.getChildNodes();
186
187                for (int i = 0; i < children.getLength(); i++) {
188                        Node child = children.item(i);
189                        list.add(new JNode(child));
190                }
191
192                return list;
193        }
194
195        /**
196         * Returns all children with a specified name.
197         * 
198         * @param name
199         *            the name of the children
200         * @return a list of all children with the specified name
201         */
202        public List<JNode> getChildren(String name) {
203
204                List<JNode> list = getChildren();
205                List<JNode> namedList = new LinkedList<>();
206
207                for (JNode node : list) {
208                        if (node.getName().equalsIgnoreCase(name)) {
209                                namedList.add(node);
210                        }
211                }
212                return namedList;
213        }
214
215        /**
216         * Returns the strings content.
217         * 
218         * @see #setText
219         * @return the strings content
220         */
221        public String getText() {
222                return node.getTextContent();
223        }
224
225        /**
226         * Returns {@code true} if this node has a strings content.
227         * 
228         * @return {@code true} if this node has a strings content
229         */
230        public boolean hasText() {
231                return (node.getTextContent() != null && !node.getTextContent().equals(""));
232        }
233
234        /**
235         * Sets the strings content.
236         * 
237         * @see #getText
238         * @param text
239         *            the strings content to set
240         */
241        public void setText(String text) {
242                node.setTextContent(text);
243        }
244
245        /**
246         * Sets an attribute to a value.
247         * 
248         * @see #getAttribute
249         * @param attribute
250         *            the name of the attribute
251         * @param value
252         *            the value
253         */
254        public void setAttribute(String attribute, String value) {
255                ((Element) node).setAttribute(attribute, value);
256        }
257
258        /**
259         * Append a new {@link JNode} with a given name.
260         * 
261         * @param name
262         *            the specified name
263         * @return the appended JNode
264         */
265        public JNode appendChild(String name) {
266                Document document = node.getOwnerDocument();
267                Node n = document.createElement(name);
268                return appendChild(n);
269        }
270
271        /**
272         * Append an XML {@link Node}.
273         * 
274         * @param node
275         *            the XML node
276         * @return the appended JNode
277         */
278        public JNode appendChild(Node node) {
279                return appendChild(new JNode(node));
280        }
281
282        /**
283         * Append a {@link JNode}.
284         * 
285         * @param jNode
286         *            the node to append
287         * @return the appended JNode
288         */
289        public JNode appendChild(JNode jNode) {
290                this.node.appendChild(jNode.getNode());
291                return jNode;
292        }
293
294        /**
295         * Returns the owner {@link Document}.
296         * 
297         * @return the owner XML document
298         */
299        public Document getDocument() {
300                return node.getOwnerDocument();
301        }
302
303        /*
304         * (non-Javadoc)
305         * 
306         * @see java.lang.Object#toString()
307         */
308        @Override
309        public String toString() {
310                return node.toString();
311        }
312}
