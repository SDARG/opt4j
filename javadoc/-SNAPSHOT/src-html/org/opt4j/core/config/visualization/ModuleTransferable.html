024package org.opt4j.core.config.visualization;
025
026import java.awt.datatransfer.DataFlavor;
027import java.awt.datatransfer.Transferable;
028import java.awt.datatransfer.UnsupportedFlavorException;
029import java.io.IOException;
030import java.util.Arrays;
031import java.util.Collection;
032
033import org.opt4j.core.config.PropertyModule;
034
035/**
036 * The {@link ModuleTransferable} is a (Drag and Drop) {@link Transferable} for
037 * {@link PropertyModule}s.
038 * 
039 * @author lukasiewycz
040 * 
041 */
042public class ModuleTransferable implements Transferable {
043
044        protected PropertyModule module;
045
046        public static DataFlavor localModuleFlavor = null;
047
048        static {
049                try {
050                        localModuleFlavor = new DataFlavor(DataFlavor.javaJVMLocalObjectMimeType + "; class="
051                                        + PropertyModule.class.getCanonicalName(), "Local PropertyModule");
052                } catch (Exception e) {
053                        e.printStackTrace();
054                }
055        }
056
057        /**
058         * Constructs a {@link ModuleTransferable}.
059         * 
060         * @param module
061         *            the property module
062         */
063        public ModuleTransferable(PropertyModule module) {
064                this.module = module;
065        }
066
067        /*
068         * (non-Javadoc)
069         * 
070         * @see
071         * java.awt.datatransfer.Transferable#getTransferData(java.awt.datatransfer
072         * .DataFlavor)
073         */
074        @Override
075        public Object getTransferData(DataFlavor flavor) throws UnsupportedFlavorException, IOException {
076                if (flavor.equals(DataFlavor.stringFlavor)) {
077                        return module.getModule().getClass().toString();
078                }
079                return module;
080        }
081
082        /*
083         * (non-Javadoc)
084         * 
085         * @see java.awt.datatransfer.Transferable#getTransferDataFlavors()
086         */
087        @Override
088        public DataFlavor[] getTransferDataFlavors() {
089                DataFlavor[] flavors = new DataFlavor[2];
090                flavors[0] = localModuleFlavor;
091                flavors[1] = DataFlavor.stringFlavor;
092                return flavors;
093        }
094
095        /*
096         * (non-Javadoc)
097         * 
098         * @see java.awt.datatransfer.Transferable#isDataFlavorSupported(java.awt.
099         * datatransfer.DataFlavor)
100         */
101        @Override
102        public boolean isDataFlavorSupported(DataFlavor flavor) {
103                Collection<DataFlavor> list = Arrays.asList(getTransferDataFlavors());
104                return list.contains(flavor);
105        }
106
107}
