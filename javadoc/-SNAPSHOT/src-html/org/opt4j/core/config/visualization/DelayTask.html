023package org.opt4j.core.config.visualization;
024
025/**
026 * The {@link DelayTask} executes tasks immediately or delayed if they arrive
027 * too close (delay). If they arrive way too close, older tasks will be dropped.
028 * 
029 * @author lukasiewycz, reimann
030 * 
031 */
032public class DelayTask {
033
034        private final DelayThread thread;
035
036        /**
037         * The {@link DelayThread} executes a task and sleeps {@code delay}
038         * milliseconds. Then it executes the next task.
039         * 
040         * If no task is available, it waits until it is notified of the arrival of
041         * a new task.
042         * 
043         * @author reimann, lukasiewycz
044         * 
045         */
046        private static class DelayThread extends Thread {
047                private Runnable task = null;
048                private boolean run = true;
049                private final long delay;
050
051                /**
052                 * Creates a new {@link DelayTask}.
053                 * 
054                 * @param delay
055                 *            the time between the execution of two tasks
056                 */
057                public DelayThread(long delay) {
058                        this.delay = delay;
059                }
060
061                /**
062                 * Stops this thread.
063                 */
064                public synchronized void doStop() {
065                        run = false;
066                        notify();
067                }
068
069                /*
070                 * (non-Javadoc)
071                 * 
072                 * @see java.lang.Thread#run()
073                 */
074                @Override
075                public void run() {
076                        while (run) {
077                                Runnable task = getTaskAndSetToNull();
078                                if (task != null) {
079                                        task.run();
080                                        try {
081                                                sleep(delay);
082                                        } catch (InterruptedException e) {
083                                        }
084                                } else {
085                                        synchronized (this) {
086                                                try {
087                                                        wait();
088                                                } catch (InterruptedException e) {
089                                                }
090                                        }
091                                }
092                        }
093                }
094
095                /**
096                 * Synchronized getter for the task that sets the task to {@code null}.
097                 * 
098                 * @return the task
099                 */
100                protected synchronized Runnable getTaskAndSetToNull() {
101                        Runnable t = task;
102                        task = null;
103                        return t;
104                }
105
106                /**
107                 * Synchronized update of the task.
108                 * 
109                 * @param task
110                 *            the task to execute
111                 */
112                public synchronized void updateTask(Runnable task) {
113                        this.task = task;
114                        notify();
115                }
116        }
117
118        /**
119         * Constructs a {@link DelayTask}.
120         * 
121         * @param delay
122         *            the minimum delay between two tasks
123         */
124        public DelayTask(long delay) {
125                this(delay, Thread.NORM_PRIORITY);
126        }
127
128        /**
129         * Constructs a {@link DelayTask} with a given priority.
130         * 
131         * 
132         * @param delay
133         *            the minimum delay between two tasks
134         * @param priority
135         *            the priority of the executing thread
136         */
137        public DelayTask(long delay, int priority) {
138                super();
139                if (delay < 0) {
140                        throw new IllegalArgumentException("Delay is negative:" + delay);
141                }
142                thread = new DelayThread(delay);
143                if (priority < Thread.MIN_PRIORITY || priority > Thread.MAX_PRIORITY) {
144                        throw new IllegalArgumentException("Invalid priority: " + priority);
145                }
146                thread.setPriority(priority);
147                thread.setDaemon(true);
148                thread.start();
149        }
150
151        /**
152         * Executes the task. If tasks arrive to close, some of them are dropped.
153         * 
154         * @param task
155         *            the task to be executed
156         */
157        public void execute(final Runnable task) {
158                thread.updateTask(task);
159        }
160
161        /*
162         * (non-Javadoc)
163         * 
164         * @see java.lang.Object#finalize()
165         */
166        @Override
167        protected void finalize() throws Throwable {
168                thread.doStop();
169                super.finalize();
170        }
171}
