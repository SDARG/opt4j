023package org.opt4j.core.config.visualization;
024
025import java.awt.BorderLayout;
026import java.awt.Color;
027import java.awt.Component;
028
029import javax.swing.BorderFactory;
030import javax.swing.JComponent;
031import javax.swing.JScrollPane;
032import javax.swing.JTable;
033import javax.swing.ScrollPaneConstants;
034import javax.swing.SwingUtilities;
035import javax.swing.table.AbstractTableModel;
036import javax.swing.table.TableCellRenderer;
037
038import org.opt4j.core.config.ExecutionEnvironment;
039import org.opt4j.core.config.Task;
040import org.opt4j.core.config.TaskListener;
041
042import com.google.inject.Inject;
043
044/**
045 * The {@link DefaultTasksPanel} extends the {@link TasksPanel} and shows all
046 * {@link Task}s in a table.
047 * 
048 * @author lukasiewycz
049 * 
050 */
051@SuppressWarnings("serial")
052public class DefaultTasksPanel extends TasksPanel implements TaskListener {
053
054        protected final ExecutionEnvironment executionEnvironment;
055
056        protected final Format format;
057
058        protected JTable table;
059
060        protected JScrollPane scroll;
061
062        protected final DelayTask delay = new DelayTask(40);
063
064        /**
065         * The {@link Table}.
066         */
067        protected class Table extends JTable {
068
069                /*
070                 * (non-Javadoc)
071                 * 
072                 * @see javax.swing.JTable#prepareRenderer(javax.swing.table.
073                 * TableCellRenderer , int, int)
074                 */
075                @Override
076                public Component prepareRenderer(TableCellRenderer renderer, int row, int column) {
077                        Component c = super.prepareRenderer(renderer, row, column);
078                        Task task = executionEnvironment.getTasks().get(row);
079                        if (task.getException() == null) {
080                                c.setForeground(Color.BLACK);
081                        } else {
082                                c.setForeground(Color.RED);
083                        }
084
085                        if (c instanceof JComponent) {
086                                JComponent jc = (JComponent) c;
087                                jc.setBorder(BorderFactory.createEmptyBorder(0, 3, 0, 0));
088
089                                char[] chars = this.getValueAt(row, column).toString().toCharArray();
090                                int length = jc.getFontMetrics(jc.getFont()).charsWidth(chars, 0, chars.length);
091                                if (this.getColumnModel().getColumn(column).getWidth() < length) {
092                                        jc.setToolTipText(format.formatTooltip(this.getValueAt(row, column).toString()));
093                                } else {
094                                        jc.setToolTipText(null);
095                                }
096                        }
097
098                        return c;
099                }
100
101        }
102
103        /**
104         * The {@link Model}.
105         */
106        protected class Model extends AbstractTableModel {
107
108                protected String[] columnNames = { "Task", "State" };
109
110                @Override
111                public int getColumnCount() {
112                        return 2;
113                }
114
115                @Override
116                public int getRowCount() {
117                        return executionEnvironment.getTasks().size();
118                }
119
120                @Override
121                public Object getValueAt(int row, int column) {
122                        Task task = executionEnvironment.getTasks().get(row);
123                        if (column == 0) {
124                                return row + " " + task.toString();
125                        } else if (column == 1) {
126                                if (task.getException() != null) {
127                                        return "EXCEPTION: \n" + task.getException();
128                                }
129                                return task.getState();
130                        }
131                        return null;
132                }
133
134                @Override
135                public String getColumnName(int i) {
136                        return columnNames[i];
137                }
138        }
139
140        /**
141         * Constructs a {@link DefaultTasksPanel}.
142         * 
143         * @param executionEnvironment
144         *            the execution environment
145         * @param format
146         *            the format informations
147         */
148        @Inject
149        public DefaultTasksPanel(ExecutionEnvironment executionEnvironment, Format format) {
150                super();
151                this.executionEnvironment = executionEnvironment;
152                this.format = format;
153        }
154
155        /**
156         * Registers a listener at the execution environment.
157         */
158        @Inject
159        public void init() {
160                executionEnvironment.addListener(this);
161        }
162
163        /*
164         * (non-Javadoc)
165         * 
166         * @see org.opt4j.conf.gui.Startupable#startup()
167         */
168        @Override
169        public void startup() {
170                this.setLayout(new BorderLayout());
171
172                AbstractTableModel model = getModel();
173                table = getTable();
174                table.setModel(model);
175                table.getColumnModel().getColumn(0).setPreferredWidth(50);
176                table.getColumnModel().getColumn(1).setPreferredWidth(500);
177
178                scroll = new JScrollPane(table, ScrollPaneConstants.VERTICAL_SCROLLBAR_AS_NEEDED,
179                                ScrollPaneConstants.HORIZONTAL_SCROLLBAR_NEVER);
180
181                this.add(scroll, BorderLayout.CENTER);
182        }
183
184        /**
185         * Returns the instance of the table.
186         * 
187         * @return the table
188         */
189        protected JTable getTable() {
190                return new Table();
191        }
192
193        /**
194         * Returns the instance of the model.
195         * 
196         * @return the model
197         */
198        protected AbstractTableModel getModel() {
199                return new Model();
200        }
201
202        /*
203         * (non-Javadoc)
204         * 
205         * @see org.opt4j.config.TaskListener#added(org.opt4j.config.Task)
206         */
207        @Override
208        public void added(Task task) {
209                table.revalidate();
210                table.repaint();
211                SwingUtilities.invokeLater(() -> scroll.getVerticalScrollBar().setValue(table.getRowCount() * 120));
212        }
213
214        /*
215         * (non-Javadoc)
216         * 
217         * @see
218         * org.opt4j.config.TaskStateListener#stateChanged(org.opt4j.config.Task)
219         */
220        @Override
221        public void stateChanged(Task task) {
222                delay.execute(() -> {
223                        table.revalidate();
224                        table.repaint();
225                });
226        }
227}
