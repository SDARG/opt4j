023package org.opt4j.core.config.visualization;
024
025import java.awt.event.ActionEvent;
026import java.awt.event.KeyEvent;
027import java.io.File;
028
029import javax.swing.JDialog;
030import javax.swing.JMenu;
031import javax.swing.JMenuBar;
032import javax.swing.JMenuItem;
033
034import org.opt4j.core.config.Icons;
035import org.opt4j.core.config.visualization.FileOperations.FileOperationsListener;
036
037import com.google.inject.Inject;
038import com.google.inject.Provider;
039
040/**
041 * The {@link Menu}.
042 * 
043 * @author lukasiewycz
044 * 
045 */
046@SuppressWarnings("serial")
047public class Menu extends JMenuBar implements FileOperationsListener, Startupable {
048
049        protected final JMenuItem load = new JMenuItem("Load ...", Icons.getIcon(Icons.FOLDER));
050
051        protected final JMenuItem save = new JMenuItem("Save", Icons.getIcon(Icons.DISK));
052
053        protected final JMenuItem saveAs = new JMenuItem("Save As ...", Icons.getIcon(Icons.DISK));
054
055        protected final JMenuItem about = new JMenuItem("About");
056
057        protected final FileOperations fileOperations;
058
059        protected final Provider<ApplicationFrame> frame;
060
061        protected final About aboutInfo;
062
063        /**
064         * Constructs a {@link Menu}.
065         * 
066         * @param fileOperations
067         *            the file operations model
068         * @param about
069         *            the about information
070         * @param frame
071         *            the provider of the application frame
072         */
073        @Inject
074        public Menu(final FileOperations fileOperations, final About about, final Provider<ApplicationFrame> frame) {
075                super();
076                this.fileOperations = fileOperations;
077                this.aboutInfo = about;
078                this.frame = frame;
079        }
080
081        /*
082         * (non-Javadoc)
083         * 
084         * @see org.opt4j.config.visualization.Startupable#startup()
085         */
086        @Override
087        public void startup() {
088
089                JMenu file = new JMenu("File");
090                JMenu etc = new JMenu("?");
091
092                file.setMnemonic(KeyEvent.VK_F);
093
094                add(file);
095                add(etc);
096
097                load.setMnemonic(KeyEvent.VK_L);
098                load.addActionListener((ActionEvent e) -> fileOperations.load());
099
100                save.setMnemonic(KeyEvent.VK_S);
101                save.addActionListener((ActionEvent e) -> fileOperations.save());
102                save.setEnabled(false);
103
104                saveAs.setMnemonic(KeyEvent.VK_A);
105                saveAs.addActionListener((ActionEvent e) -> fileOperations.saveAs());
106
107                about.setMnemonic(KeyEvent.VK_A);
108                about.addActionListener((ActionEvent e) -> {
109                        JDialog dialog = aboutInfo.getDialog(frame.get());
110                        dialog.pack();
111                        dialog.setVisible(true);
112                });
113
114                file.add(load);
115                file.add(save);
116                file.add(saveAs);
117
118                etc.add(about);
119        }
120
121        /**
122         * Initializes the listeners.
123         */
124        @Inject
125        public void init() {
126                fileOperations.addListener(this);
127        }
128
129        /*
130         * (non-Javadoc)
131         * 
132         * @see
133         * org.opt4j.config.visualization.FileOperations.FileOperationsListener#
134         * setCurrentFile(java.io.File)
135         */
136        @Override
137        public void setCurrentFile(File file) {
138                if (file != null) {
139                        save.setEnabled(true);
140                } else {
141                        save.setEnabled(false);
142                }
143        }
144
145}
