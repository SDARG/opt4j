023package org.opt4j.core.config.visualization;
024
025import java.util.ArrayList;
026import java.util.Collection;
027import java.util.Set;
028import java.util.TreeSet;
029import java.util.concurrent.CopyOnWriteArraySet;
030
031import org.opt4j.core.config.PropertyModule;
032
033import com.google.inject.Singleton;
034
035/**
036 * The {@link SelectedModules}. A set of all selected modules that is extended
037 * by a listener functionality.
038 * 
039 * @author lukasiewycz
040 * 
041 */
042@SuppressWarnings("serial")
043@Singleton
044public class SelectedModules extends TreeSet<PropertyModule> {
045
046        protected Set<SetListener> listeners = new CopyOnWriteArraySet<>();
047
048        /*
049         * (non-Javadoc)
050         * 
051         * @see java.util.HashSet#add(java.lang.Object)
052         */
053        @Override
054        public boolean add(PropertyModule module) {
055                boolean b = super.add(module);
056                for (SetListener listener : listeners) {
057                        listener.moduleAdded(this, module);
058                }
059                return b;
060        }
061
062        /*
063         * (non-Javadoc)
064         * 
065         * @see java.util.AbstractCollection#addAll(java.util.Collection)
066         */
067        @Override
068        public boolean addAll(Collection<? extends PropertyModule> modules) {
069                boolean r = false;
070                for (PropertyModule module : modules) {
071                        r = add(module) || r;
072                }
073                return r;
074        }
075
076        /*
077         * (non-Javadoc)
078         * 
079         * @see java.util.HashSet#remove(java.lang.Object)
080         */
081        @Override
082        public boolean remove(Object o) {
083                boolean b = super.remove(o);
084                if (b) {
085                        for (SetListener listener : listeners) {
086                                listener.moduleRemoved(this, (PropertyModule) o);
087                        }
088                }
089                return b;
090        }
091
092        /*
093         * (non-Javadoc)
094         * 
095         * @see java.util.HashSet#clear()
096         */
097        @Override
098        public void clear() {
099                Collection<PropertyModule> modules = new ArrayList<>();
100                modules.addAll(this);
101
102                for (PropertyModule module : modules) {
103                        remove(module);
104                }
105        }
106
107        /**
108         * Adds a {@link SetListener}.
109         * 
110         * @see #removeListener
111         * @param listener
112         *            the listener to be added
113         */
114        public void addListener(SetListener listener) {
115                listeners.add(listener);
116        }
117
118        /**
119         * Removes a {@link SetListener}.
120         * 
121         * @see #addListener
122         * @param listener
123         *            the listener to be removed
124         */
125        public void removeListener(SetListener listener) {
126                listeners.remove(listener);
127        }
128
129}
