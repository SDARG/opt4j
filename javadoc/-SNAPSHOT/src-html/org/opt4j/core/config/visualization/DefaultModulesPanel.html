097        /**
098         * The {@link UserNode} is a basic node of the tree.
099         */
100        protected abstract class UserNode extends DefaultMutableTreeNode {
101
102                public UserNode(Object object) {
103                        super(object);
104                }
105
106                @Override
107                public String toString() {
108                        return format.getName(getType());
109                }
110
111                public ImageIcon getIcon() {
112                        return format.getIcon(getType());
113                }
114
115                public String getTooltip() {
116                        return format.getTooltip(getType());
117                }
118
119                public abstract Class<?> getType();
120        }

122        /**
123         * The {@link RootTreeNode} is the root node of the tree.
124         */
125        protected class RootTreeNode extends UserNode {
126
127                public RootTreeNode(boolean b) {
128                        super(b);
129                }
130
131                public void setLoading(boolean b) {
132                        userObject = b;
133                }
134
135                public boolean isLoading() {
136                        return (Boolean) userObject;
137                }
138
139                @Override
140                public ImageIcon getIcon() {
141                        if (isLoading()) {
142                                ImageIcon icon = Icons.getIcon(Icons.LOADING);
143                                icon.setImageObserver(tree);
144                                return icon;
145                        }
146                        return format.asFolder(null);
147                }
148
149                @Override
150                public String toString() {
151                        if (isLoading()) {
152                                return "Loading";
153                        }
154                        return "Modules";
155                }
156
157                @Override
158                public String getTooltip() {
159                        return null;
160                }
161
162                @Override
163                public Class<?> getType() {
164                        return null;
165                }
166        }

223        /**
224         * The {@link TreeCellRenderer} for the modules tree.
225         * 
226         */
227        protected static class TreeCellRenderer extends DefaultTreeCellRenderer {
228                @Override
229                public Component getTreeCellRendererComponent(JTree tree, Object value, boolean sel, boolean expanded,
230                                boolean leaf, int row, boolean hasFocus) {
231                        Component comp = super.getTreeCellRendererComponent(tree, value, sel, expanded, leaf, row, hasFocus);
232
233                        ImageIcon icon = null;
234                        String tooltip = null;
235
236                        if (value instanceof UserNode) {
237                                UserNode userNode = (UserNode) value;
238                                icon = userNode.getIcon();
239                                tooltip = userNode.getTooltip();
240                        }
241                        if (icon == null) {
242                                icon = Icons.getDefault();
243                        }
244
245                        setIcon(icon);
246                        setToolTipText(tooltip);
247
248                        return comp;
249                }
250        }

252        /**
253         * The {@link MyTree} that extends a {@link JTree} by a drag method.
254         * 
255         */
256        protected static class MyTree extends JTree implements DragGestureListener {
257
258                protected DragSource dragSource = null;
259
260                protected DragSourceListener dragSourceListener = new DragSourceAdapter() {
261                };
262
263                public MyTree(TreeNode root) {
264                        super(root);
265
266                        dragSource = new DragSource();
267                        dragSource.createDefaultDragGestureRecognizer(this, DnDConstants.ACTION_MOVE, this);
268                }
269
270                @Override
271                public void dragGestureRecognized(DragGestureEvent dge) {
272                        TreeNode node = (TreeNode) this.getLastSelectedPathComponent();
273
274                        if (node instanceof ModuleTreeNode) {
275                                ModuleTreeNode moduleNode = (ModuleTreeNode) node;
276                                PropertyModule module = moduleNode.getModule();
277
278                                Transferable transferable = new ModuleTransferable(module);
279
280                                dragSource.startDrag(dge, DragSource.DefaultMoveDrop, transferable, dragSourceListener);
281                        }
282                }
283
284        }

328                        public void run() {
329                                try {

339        private Class<?> getCategory(Class<?> clazz, boolean include) {
340                if (include && isCategory(clazz)) {
341                        return clazz;
342                }
343                if (clazz.equals(Object.class)) {
344                        return null;
345                } else if (clazz.getAnnotation(Parent.class) != null) {
346                        Class<?> pClass = clazz.getAnnotation(Parent.class).value();
347                        return pClass;
348                } else {
349                        assert (clazz.getSuperclass() != null) : clazz;
350                        return getCategory(clazz.getSuperclass(), true);
351                }
352        }
353
354        private boolean isCategory(Class<?> clazz) {
355                Category category = clazz.getAnnotation(Category.class);
356                return (category != null);
357        }

359        /**
360         * Build and order the tree
361         */
362        protected void populateTree() {
363                // get all relevant categories
364
365                // map from the modules (categories) to their categories
366                // (supercategories)
367                Map<Class<?>, Class<?>> map = new HashMap<>();
368
369                for (Class<? extends Module> module : allModules) {
370                        Class<?> category = getCategory(module, false);
371                        assert (!module.equals(category)) : module;
372                        map.put(module, category);
373                }
374
375                int size;
376                do {
377                        size = map.size();
378                        Set<Class<?>> values = new HashSet<>(map.values());
379                        for (Class<?> cat : values) {
380                                if (cat != null) {
381                                        Class<?> category = getCategory(cat, false);
382                                        map.put(cat, category);
383                                }
384                        }
385                } while (map.size() != size);
386
387                // create all nodes
388                Map<Class<?>, CategoryTreeNode> ctn = new HashMap<>();
389                Map<Class<?>, ModuleTreeNode> mtn = new HashMap<>();
390                Map<Class<?>, DefaultMutableTreeNode> atn = new HashMap<>();
391
392                for (Class<?> clazz : map.keySet()) {
393                        if (isCategory(clazz)) {
394                                CategoryTreeNode node = new CategoryTreeNode(clazz);
395                                ctn.put(clazz, node);
396                        } else {
397                                @SuppressWarnings("unchecked")
398                                ModuleTreeNode node = new ModuleTreeNode((Class<? extends Module>) clazz);
399                                mtn.put(clazz, node);
400                        }
401                }
402                atn.putAll(ctn);
403                atn.putAll(mtn);
404
405                // create hierarchy by parent annotation
406                CategoryTreeNode def = new CategoryTreeNode(Default.class);
407                root.add(def);
408                for (Entry<Class<?>, DefaultMutableTreeNode> entry : atn.entrySet()) {
409                        Class<?> clazz = entry.getKey();
410                        DefaultMutableTreeNode node = entry.getValue();
411
412                        Class<?> category = map.get(clazz);
413
414                        if (category == null) {
415                                if (isCategory(clazz)) {
416                                        root.add(node);
417                                } else {
418                                        def.add(node);
419                                }
420                        } else {
421                                DefaultMutableTreeNode n = atn.get(category);
422                                assert (n != null && n != node) : clazz + " " + category;
423                                n.add(node);
424                        }
425
426                }
427
428                // clean empty categories
429                boolean removed;
430                do {
431                        removed = removeEmptyCategories(root);
432                } while (removed);
433
434                SwingUtilities.invokeLater(new Runnable() {
435                        @Override
436                        public void run() {
437                                tree.setSelectionModel(new DefaultTreeSelectionModel());
438                                tree.getSelectionModel().setSelectionMode(TreeSelectionModel.SINGLE_TREE_SELECTION);
439                                root.setLoading(false);
440                                sort(tree);
441                                tree.expandRow(0);
442                        }
443                });
444                revalidate();
445                repaint();
446        }

448        /**
449         * Removes empty categories. This method can create some new empty
450         * categories!
451         * 
452         * @param current
453         *            the current node
454         * @return true if an empty category was removed
455         */
456        private boolean removeEmptyCategories(DefaultMutableTreeNode current) {
457                if (!current.children().hasMoreElements()) {
458                        return false;
459                }
460
461                TreeNode node = current.getFirstChild();
462                boolean removed = false;
463                while (node != null) {
464                        DefaultMutableTreeNode n = (DefaultMutableTreeNode) node;
465                        node = current.getChildAfter(n);
466
467                        if (n.isLeaf()) {
468                                if (n instanceof CategoryTreeNode && n.getChildCount() == 0) {
469                                        DefaultMutableTreeNode parent = (DefaultMutableTreeNode) n.getParent();
470                                        parent.remove(n);
471                                        removed = true;
472                                }
473                        } else {
474                                removed = removed || removeEmptyCategories(n);
475                        }
476                }
477                return removed;
478        }

480        protected MouseListener mouseListener = new MouseAdapter() {
481
482                @Override
483                public void mousePressed(MouseEvent e) {
484                        if (e.isPopupTrigger()) {
485                                displayMenu(e);
486                        }
487                }
488
489                @Override
490                public void mouseReleased(MouseEvent e) {
491                        int count = e.getClickCount();
492
493                        if (e.isPopupTrigger()) {
494                                displayMenu(e);
495                        } else if (count > 1) {
496                                Object object = tree.getLastSelectedPathComponent();
497                                if (object instanceof ModuleTreeNode) {
498                                        ModuleTreeNode node = (ModuleTreeNode) object;
499                                        PropertyModule pmodule = node.getModule();
500                                        if (selectedModules.contains(pmodule)) {
501                                                remove(pmodule);
502                                        } else {
503                                                add(pmodule);
504                                        }
505                                }
506                        }
507                }
508        };

510        /**
511         * Adds a new {@link PropertyModule}.
512         * 
513         * @param pm
514         *            the property module to be added
515         */
516        protected void add(PropertyModule pm) {
517                Module module = pm.getModule();
518                if (module.getClass().getAnnotation(Multi.class) != null) {
519                        try {
520                                Module m2 = module.getClass().newInstance();
521                                PropertyModule pm2 = new PropertyModule(m2);
522                                selectedModules.add(pm2);
523                        } catch (InstantiationException e1) {
524                                e1.printStackTrace();
525                        } catch (IllegalAccessException e1) {
526                                e1.printStackTrace();
527                        }
528                } else {
529                        selectedModules.add(pm);
530                }
531        }

535         * 
536         * @param pm
537         *            the property module to be removed
538         */
539        protected void remove(PropertyModule pm) {
540                selectedModules.remove(pm);
541        }

543        /**
544         * Displays a popup menu for the tree elements.
545         * 
546         * @param e
547         *            the mouse event
548         */
549        protected void displayMenu(MouseEvent e) {
550                if (e.isPopupTrigger()) {
551                        Point pt = e.getPoint();
552                        tree.setSelectionPath(tree.getPathForLocation(pt.x, pt.y));
553
554                        Object object = tree.getLastSelectedPathComponent();
555                        if (object instanceof ModuleTreeNode) {
556
557                                ModuleTreeNode node = (ModuleTreeNode) object;
558                                final PropertyModule module = node.getModule();
559
560                                moduleMenu.removeAll();
561
562                                JMenuItem add = new JMenuItem("Add Module", Icons.getIcon(Icons.ADD));
563                                add.addActionListener((ActionEvent event) -> add(module));
564
565                                JMenuItem remove = new JMenuItem("Remove Module", Icons.getIcon(Icons.DELETE));
566                                remove.addActionListener((ActionEvent event) -> remove(module));
567
568                                if (selectedModules.contains(module)) {
569                                        add.setEnabled(false);
570                                } else {
571                                        remove.setEnabled(false);
572                                }
573
574                                moduleMenu.add(add);
575                                moduleMenu.add(remove);
576                                moduleMenu.show(e.getComponent(), e.getX(), e.getY());
577                        }
578
579                }
580        }

582        /**
583         * Sorts the {@link JTree} alphabetically.
584         * 
585         * @param tree
586         *            the tree to be sorted
587         */
588        protected void sort(JTree tree) {
589                DefaultMutableTreeNode root = (DefaultMutableTreeNode) tree.getModel().getRoot();
590                sort(root);
591
592        }

594        /**
595         * Sorts the child elements of one tree node alphabetically.
596         * 
597         * @param node
598         *            the node which children are sorted
599         */
600        protected void sort(DefaultMutableTreeNode node) {
601                if (!node.children().hasMoreElements()) {
602                        return;
603                }
604
605                List<UserNode> nodes = new ArrayList<>();
606
607                Enumeration<?> children = node.children();
608
609                while (children.hasMoreElements()) {
610                        UserNode n = (UserNode) children.nextElement();
611                        nodes.add(n);
612                }
613
614                Collections.sort(nodes, new Comparator<UserNode>() {
615                        @Override
616                        public int compare(UserNode o1, UserNode o2) {
617                                boolean cat1 = o1 instanceof CategoryTreeNode;
618                                boolean cat2 = o2 instanceof CategoryTreeNode;
619
620                                if (cat1 && !cat2) {
621                                        return -1;
622                                } else if (!cat1 && cat2) {
623                                        return 1;
624                                } else {
625                                        return o1.toString().compareTo(o2.toString());
626                                }
627                        }
628                });
629
630                node.removeAllChildren();
631
632                for (UserNode n : nodes) {
633                        node.add(n);
634                        sort(n);
635                }
636        }
