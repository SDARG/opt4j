075/**
076 * The {@link DefaultSelectedPanel}. The selected modules are organized as tabs.

081@SuppressWarnings("serial")
082public class DefaultSelectedPanel extends SelectedPanel implements SetListener {
083
084        protected final TabbedPane tabs = new TabbedPane();
085
086        protected final Format format;
087
088        protected final SelectedModules selectedModules;
089
090        protected final FileChooser fileChooser;
091
092        protected final Map<PropertyModule, PropertyPanel> map = new HashMap<>();
093
094        protected final DropTarget dropTarget;
095
096        /**
097         * The drop listener for the drag-and-drop functionality.
098         */
099        protected final DropTargetListener dropListener = new DropTargetAdapter() {
100                @Override
101                public void drop(DropTargetDropEvent dtde) {
102                        Transferable transferable = dtde.getTransferable();
103                        DataFlavor flaver = ModuleTransferable.localModuleFlavor;
104                        try {
105
106                                if (transferable.isDataFlavorSupported(flaver)) {
107                                        Object o = transferable.getTransferData(flaver);
108                                        PropertyModule module = (PropertyModule) o;
109                                        selectedModules.add(module);
110
111                                        dtde.acceptDrop(DnDConstants.ACTION_MOVE);
112                                        dtde.getDropTargetContext().dropComplete(true);
113                                } else {
114                                        dtde.rejectDrop();
115                                }
116
117                        } catch (UnsupportedFlavorException e) {
118                                e.printStackTrace();
119                        } catch (IOException e) {
120                                e.printStackTrace();
121                        }
122                }
123        };

131                        this.panel = panel;
132                }

135                        return panel;
136                }

140        /**
141         * The {@link TabbedPane} is a custom {@link JTabbedPane}.
142         */
143        protected class TabbedPane extends JTabbedPane {
144                @Override
145                public void remove(int index) {
146                        Component component = tabs.getComponentAt(index);
147                        super.remove(index);
148
149                        if (component instanceof MyScrollPane) {
150                                MyScrollPane scroll = (MyScrollPane) component;
151                                map.remove(scroll.getPanel().getModule());
152                        }
153                }
154
155                protected int getIndex(PropertyModule module) {
156                        if (map.keySet().contains(module)) {
157                                for (int i = 0; i < tabs.getTabCount(); i++) {
158                                        Component component = tabs.getComponentAt(i);
159                                        if (component instanceof MyScrollPane) {
160                                                MyScrollPane scroll = (MyScrollPane) component;
161                                                if (scroll.getPanel().getModule().equals(module)) {
162                                                        return i;
163                                                }
164                                        }
165                                }
166                        }
167                        return -1;
168                }
169        }

171        /**
172         * The custom tab of the {@link JTabbedPane}.
173         */
174        protected class TabComponent extends JPanel {
175
176                private JLabel label;
177                private final JButton xmlButton;
178                private final JButton closeButton;
179
180                private final PropertyModule module;
181
182                public TabComponent(PropertyModule module) {
183                        super(new FlowLayout(FlowLayout.LEFT, 0, 0));
184                        this.module = module;
185                        setOpaque(false);
186
187                        String name = format.getName(module);
188                        ImageIcon icon = format.getIcon(module);
189
190                        label = new JLabel(name, icon, SwingConstants.LEFT);
191                        xmlButton = new XMLButton();
192                        xmlButton.setToolTipText("Show module xml");
193                        closeButton = new CloseButton();
194                        closeButton.setToolTipText("Remove module");
195
196                        setLabelName(name);
197                }
198
199                protected void setLabelName(String name) {
200                        if (name.length() > 22) {
201                                label.setName(name.substring(0, 18) + "...");
202                        } else {
203                                label.setName(name);
204                        }
205                        removeAll();
206
207                        label = new JLabel(label.getName(), label.getIcon(), SwingConstants.LEFT);
208
209                        add(label);
210                        add(new JLabel(" "));
211                        label.setBorder(BorderFactory.createEmptyBorder(0, label.getIcon() == null ? 5 : 0, 0, 5));
212
213                        add(xmlButton);
214                        add(closeButton);
215                        setBorder(BorderFactory.createEmptyBorder(2, 0, 0, 0));
216                }
217
218                protected PropertyModule getModule() {
219                        return module;
220                }
221
222                private abstract class TabButton extends JButton implements ActionListener {
223
224                        protected final static int SIZE = 16;
225
226                        TabButton() {
227                                setPreferredSize(new Dimension(SIZE, SIZE));
228                                setContentAreaFilled(false);
229                                setFocusable(false);
230                                setBorder(BorderFactory.createEtchedBorder());
231                                setBorderPainted(false);
232                                addMouseListener(buttonMouseListener);
233                                setRolloverEnabled(true);
234                                addActionListener(this);
235                        }
236
237                        // paint the cross
238                        @Override
239                        protected void paintComponent(Graphics g) {
240                                super.paintComponent(g);
241                                Graphics2D g2 = (Graphics2D) g.create();
242                                g2.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
243                                // shift the image for pressed buttons
244                                if (getModel().isPressed()) {
245                                        g2.translate(1, 1);
246                                }
247                                drawIcon(g2);
248                                g2.dispose();
249                        }
250
251                        protected abstract void drawIcon(Graphics2D g);
252
253                }
254
255                private class CloseButton extends TabButton {
256
257                        @Override
258                        public void actionPerformed(ActionEvent e) {
259                                int i = tabs.indexOfTabComponent(TabComponent.this);
260                                if (i != -1) {
261                                        selectedModules.remove(module);
262                                }
263                        }
264
265                        @Override
266                        protected void drawIcon(Graphics2D g) {
267                                g.setStroke(new BasicStroke(2));
268                                g.setColor(Color.RED);
269
270                                int delta = 5;
271                                g.drawLine(delta, delta, getWidth() - delta - 2, getHeight() - delta - 2);
272                                g.drawLine(getWidth() - delta - 2, delta, delta, getHeight() - delta - 2);
273                        }
274                }
275
276                private class XMLButton extends TabButton {
277                        @Override
278                        public void actionPerformed(ActionEvent e) {
279                                ModuleSaver saver = new ModuleSaver();
280                                final ClipboardFrame frame = new ClipboardFrame(saver.toXMLString(module));
281                                frame.pack();
282                                frame.setVisible(true);
283                        }
284
285                        @Override
286                        protected void drawIcon(Graphics2D g) {
287                                g.setStroke(new BasicStroke(1));
288                                Image image = Icons.getIcon(Icons.XMLTAG).getImage();
289                                g.drawImage(image, 0, -1, 15, 16, this);
290                        }
291                }
292
293                private final MouseListener buttonMouseListener = new MouseAdapter() {
294                        @Override
295                        public void mouseEntered(MouseEvent e) {
296                                Component component = e.getComponent();
297                                if (component instanceof AbstractButton) {
298                                        AbstractButton button = (AbstractButton) component;
299                                        button.setBorderPainted(true);
300                                }
301                        }
302
303                        @Override
304                        public void mouseExited(MouseEvent e) {
305                                Component component = e.getComponent();
306                                if (component instanceof AbstractButton) {
307                                        AbstractButton button = (AbstractButton) component;
308                                        button.setBorderPainted(false);
309                                }
310                        }
311                };
312        }

352        /**
353         * Adds a module to the panel.
354         * 
355         * @param module
356         *            the module to add
357         */
358        protected void addModule(PropertyModule module) {
359                if (!map.keySet().contains(module)) {
360
361                        PropertyPanel panel = null;
362                        Panel p = module.getModule().getClass().getAnnotation(Panel.class);
363                        if (p != null) {
364                                Class<? extends PropertyPanel> clazz = p.value();
365
366                                try {
367                                        Constructor<? extends PropertyPanel> cstr = clazz.getConstructor(PropertyModule.class,
368                                                        FileChooser.class, Format.class);
369                                        panel = cstr.newInstance(module, fileChooser, format);
370
371                                } catch (Exception e) {
372                                        throw new RuntimeException(e);
373                                }
374
375                        } else {
376                                panel = new PropertyPanel(module, fileChooser, format);
377                        }
378
379                        MyScrollPane scroll = new MyScrollPane(panel);
380                        scroll.setVerticalScrollBarPolicy(ScrollPaneConstants.VERTICAL_SCROLLBAR_AS_NEEDED);
381                        scroll.setPreferredSize(new Dimension(300, 300));
382
383                        String name = format.getName(module);
384                        String tooltip = format.getTooltip(module);
385
386                        int count = selectedModules.subSet(selectedModules.first(), module).size();
387                        tabs.insertTab(name, null, scroll, tooltip, count);
388                        tabs.setSelectedIndex(count);
389                        tabs.setTabComponentAt(count, new TabComponent(module));
390
391                        map.put(module, panel);
392
393                } else {
394                        int index = tabs.getIndex(module);
395                        if (index != -1) {
396                                tabs.setSelectedIndex(index);
397                        }
398                }
399                updateTabNames();
400        }
401
402        /**
403         * Removes a module from the panel.
404         * 
405         * @param module
406         *            the module to remove
407         */
408        protected void removeModule(PropertyModule module) {
409                int index = tabs.getIndex(module);
410                if (index != -1) {
411                        map.remove(module);
412                        tabs.remove(index);
413                }
414                updateTabNames();
415        }
416
417        /**
418         * Update the tab names. Consider multiple tabs of the same module.
419         */
420        protected void updateTabNames() {
421                Set<Class<? extends Module>> exist = new HashSet<>();
422                Set<Class<? extends Module>> multi = new HashSet<>();
423                for (PropertyModule pm : map.keySet()) {
424                        Class<? extends Module> module = pm.getModule().getClass();
425                        if (exist.contains(module)) {
426                                multi.add(module);
427                        } else {
428                                exist.add(module);
429                        }
430                }
431
432                Map<Class<? extends Module>, Integer> counter = new HashMap<>();
433                for (Class<? extends Module> module : multi) {
434                        counter.put(module, 1);
435                }
436
437                for (int i = 0; i < tabs.getTabCount(); i++) {
438                        TabComponent comp = (TabComponent) tabs.getTabComponentAt(i);
439
440                        Class<? extends Module> module = comp.getModule().getModule().getClass();
441                        if (counter.containsKey(module)) {
442                                int c = counter.get(module);
443                                String name = "(" + c + ") " + format.getName(comp.getModule());
444                                comp.setLabelName(name);
445                                counter.put(module, c + 1);
446                        }
447                }
448                tabs.repaint();
449        }
