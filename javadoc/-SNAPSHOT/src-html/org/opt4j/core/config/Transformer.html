024package org.opt4j.core.config;
025
026/**
027 * The {@link Transformer} design pattern.
028 * 
029 * @author reimann
030 * 
031 * @param <I>
032 *            the type of the source object
033 * @param <O>
034 *            the type of the target object
035 */
036public interface Transformer<I, O> {
037        /**
038         * The given input {@code in} is not changed by transforming it to a new
039         * object of type <O>. Subsequent calls with equal instances of {@code in}
040         * should result in equal outputs.
041         * 
042         * @param in
043         *            the input to convert
044         * @return the corresponding output
045         */
046        public O transform(I in);
047}
