023package org.opt4j.core.config;
024
025import java.io.File;
026import java.io.IOException;
027import java.util.ArrayList;
028import java.util.Collection;
029import java.util.List;
030
031import javax.xml.parsers.DocumentBuilder;
032import javax.xml.parsers.DocumentBuilderFactory;
033import javax.xml.parsers.ParserConfigurationException;
034
035import org.w3c.dom.Document;
036import org.w3c.dom.Node;
037import org.xml.sax.SAXException;
038
039import com.google.inject.Module;
040
041/**
042 * Helper class for loading {@link PropertyModule} configurations from files or
043 * retrieving these from XML {@link Node}s.
044 * 
045 * @author lukasiewycz
046 * 
047 */
048public class ModuleLoader {
049
050        protected final ModuleRegister moduleRegister;
051
052        /**
053         * Constructs a {@link ModuleLoader}.
054         * 
055         * @param moduleRegister
056         *            the register of all found modules
057         */
058        public ModuleLoader(ModuleRegister moduleRegister) {
059                super();
060                this.moduleRegister = moduleRegister;
061        }
062
063        /**
064         * Loads all modules from a file (as filename).
065         * 
066         * @param filename
067         *            the file (as filename)
068         * @return the modules
069         */
070        public Collection<? extends Module> load(String filename) {
071                File file = new File(filename);
072                return load(file);
073        }
074
075        /**
076         * Loads all modules from a {@link File}.
077         * 
078         * @param file
079         *            the input file
080         * @return a list of the modules
081         */
082        public Collection<? extends Module> load(File file) {
083
084                List<Module> modules = new ArrayList<>();
085
086                try {
087                        DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
088                        DocumentBuilder builder = factory.newDocumentBuilder();
089                        Document document = builder.parse(file);
090
091                        modules.addAll(get(document.getFirstChild()));
092
093                } catch (ParserConfigurationException e) {
094                        e.printStackTrace();
095                } catch (SAXException e) {
096                        e.printStackTrace();
097                } catch (IOException e) {
098                        e.printStackTrace();
099                }
100
101                return modules;
102        }
103
104        /**
105         * Loads all modules from an XML {@link Node}.
106         * 
107         * @param node
108         *            the XML node
109         * @return a list of the modules
110         */
111        public Collection<? extends Module> get(Node node) {
112                List<Module> modules = new ArrayList<>();
113
114                JNode application = new JNode(node);
115
116                for (JNode child : application.getChildren("module")) {
117                        String name = child.getAttribute("class");
118
119                        PropertyModule module = null;
120                        try {
121                                Class<? extends Module> clazz = Class.forName(name).asSubclass(Module.class);
122                                module = moduleRegister.get(clazz);
123                        } catch (ClassNotFoundException e) {
124                                System.err.println("Class " + name + " not found.");
125                        }
126
127                        if (module != null) {
128                                module.setConfiguration(child.getNode());
129                                modules.add(module.getModule());
130                        }
131                }
132
133                return modules;
134        }
135
136}
