024package org.opt4j.core.config;
025
026import java.io.ByteArrayOutputStream;
027import java.io.File;
028import java.io.FileNotFoundException;
029import java.io.FileOutputStream;
030import java.io.IOException;
031import java.io.OutputStream;
032import java.io.OutputStreamWriter;
033import java.io.UnsupportedEncodingException;
034import java.util.Collection;
035
036import javax.xml.parsers.DocumentBuilder;
037import javax.xml.parsers.DocumentBuilderFactory;
038import javax.xml.parsers.ParserConfigurationException;
039import javax.xml.transform.OutputKeys;
040import javax.xml.transform.Transformer;
041import javax.xml.transform.TransformerConfigurationException;
042import javax.xml.transform.TransformerException;
043import javax.xml.transform.TransformerFactory;
044import javax.xml.transform.dom.DOMSource;
045import javax.xml.transform.stream.StreamResult;
046
047import org.w3c.dom.Document;
048import org.w3c.dom.Node;
049
050import com.google.inject.Module;
051
052/**
053 * Helper class for saving the {@link PropertyModule} configurations or adding
054 * these to XML {@link Node}s.
055 * 
056 * @author lukasiewycz
057 * 
058 */
059public class ModuleSaver {
060
061        /**
062         * Save the module configurations to a file (as filename).
063         * 
064         * @param filename
065         *            the name of the file
066         * @param modules
067         *            the modules
068         * @return {@code true} if successful
069         */
070        public boolean save(String filename, Collection<? extends Module> modules) {
071                File file = new File(filename);
072                return save(file, modules);
073        }
074
075        /**
076         * Save the module configurations to a {@link File}.
077         * 
078         * @param file
079         *            the file
080         * @param modules
081         *            the modules
082         * @return {@code true} if successful
083         */
084        public boolean save(File file, Collection<? extends Module> modules) {
085                if (!file.exists()) {
086                        try {
087                                file.createNewFile();
088                        } catch (IOException e) {
089                                e.printStackTrace();
090                        }
091                }
092                try {
093                        return write(new FileOutputStream(file), modules);
094                } catch (FileNotFoundException e) {
095                        e.printStackTrace();
096                        return false;
097                }
098        }
099
100        /**
101         * Save the modules configuration to an {@link OutputStream}.
102         * 
103         * @param out
104         *            the output stream
105         * @param modules
106         *            the modules
107         * @return {@code true} if successful
108         */
109        public boolean write(OutputStream out, Collection<? extends Module> modules) {
110                try {
111                        Document document = getEmptyDocument();
112
113                        Node configuration = document.createElement("configuration");
114                        document.appendChild(configuration);
115                        for (Module module : modules) {
116                                configuration.appendChild(getNode(module, document));
117                        }
118                        StreamResult result = new StreamResult(new OutputStreamWriter(out, "utf-8"));
119                        DOMSource source = new DOMSource(document);
120                        getStandardTransformer().transform(source, result);
121
122                        return true;
123
124                } catch (ParserConfigurationException e) {
125                        e.printStackTrace();
126                } catch (TransformerConfigurationException e) {
127                        e.printStackTrace();
128                } catch (TransformerException e) {
129                        e.printStackTrace();
130                } catch (UnsupportedEncodingException e) {
131                        e.printStackTrace();
132                }
133
134                return false;
135        }
136
137        /**
138         * Save the module configuration to an {@link OutputStream}.
139         * 
140         * @param out
141         *            the output stream}
142         * @param module
143         *            the module
144         * @return {@code true} if successful
145         */
146        public boolean write(OutputStream out, Module module) {
147                try {
148                        Document document = getEmptyDocument();
149                        document.appendChild(getNode(module, document));
150                        StreamResult result = new StreamResult(new OutputStreamWriter(out, "utf-8"));
151                        DOMSource source = new DOMSource(document);
152                        getStandardTransformer().transform(source, result);
153                        return true;
154                } catch (ParserConfigurationException e) {
155                        e.printStackTrace();
156                } catch (TransformerConfigurationException e) {
157                        e.printStackTrace();
158                } catch (TransformerException e) {
159                        e.printStackTrace();
160                } catch (UnsupportedEncodingException e) {
161                        e.printStackTrace();
162                }
163                return false;
164        }
165
166        /**
167         * Returns the xml representation of the {@link Module}.
168         * 
169         * @param module
170         *            the module
171         * @return the xml representation
172         */
173        public String toXMLString(Module module) {
174                ByteArrayOutputStream out = new ByteArrayOutputStream();
175                write(out, module);
176                return out.toString();
177        }
178
179        /**
180         * Returns the xml representation of the {@link Module}s.
181         * 
182         * @param modules
183         *            the modules
184         * @return the xml representation
185         */
186        public String toXMLString(Collection<? extends Module> modules) {
187                ByteArrayOutputStream out = new ByteArrayOutputStream();
188                write(out, modules);
189                return out.toString();
190        }
191
192        protected Document getEmptyDocument() throws ParserConfigurationException {
193                DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
194                DocumentBuilder builder = factory.newDocumentBuilder();
195                Document document = builder.newDocument();
196                return document;
197        }
198
199        protected Transformer getStandardTransformer() throws TransformerConfigurationException {
200                TransformerFactory tf = TransformerFactory.newInstance();
201                Transformer transformer = tf.newTransformer();
202                transformer.setOutputProperty(OutputKeys.OMIT_XML_DECLARATION, "yes");
203                transformer.setOutputProperty(OutputKeys.INDENT, "yes");
204                transformer.setOutputProperty(OutputKeys.METHOD, "xml");
205                transformer.setOutputProperty("{http://xml.apache.org/xslt}indent-amount", "2");
206                return transformer;
207        }
208
209        protected Node getNode(Module module, Document document) {
210                PropertyModule pModule;
211                if (module instanceof PropertyModule) {
212                        pModule = (PropertyModule) module;
213                } else {
214                        pModule = new PropertyModule(module);
215                }
216                Node node = pModule.getConfiguration(document);
217                return node;
218        }
219
220}
