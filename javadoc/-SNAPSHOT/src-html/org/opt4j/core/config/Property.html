039/**
040 * The {@link Property} contains information about a single property of a
041 * {@link PropertyModule}.
042 * 
043 * @author lukasiewycz
044 * 
045 */
046public class Property {
047
048        protected final Module module;
049
050        protected final String name;
051
052        protected final Class<?> type;
053
054        protected final Method getter;
055
056        protected final Method setter;
057
058        protected String info = "";
059
060        protected int order = 10000;
061
062        protected final Collection<Requirement> requirements = new ArrayList<>();
063
064        protected final Map<Class<? extends Annotation>, Annotation> annotations = new HashMap<>();
065
066        /**
067         * Constructs a {@link Property} .
068         * 
069         * @param module
070         *            the module
071         * @param name
072         *            the name
073         * @param type
074         *            the type
075         * @param getter
076         *            the getter method
077         * @param setter
078         *            the setter method
079         * 
080         * @param annotations
081         *            the annotations
082         */
083        public Property(Module module, String name, Class<?> type, Method getter, Method setter,
084                        Iterable<Annotation> annotations) {
085                super();
086                this.module = module;
087                this.name = name;
088                this.type = type;
089                this.getter = getter;
090                this.setter = setter;
091
092                for (Annotation annotation : annotations) {
093                        Class<? extends Annotation> clazz = annotation.annotationType();
094                        this.annotations.put(clazz, annotation);
095                }
096        }

186        /**
187         * Returns {@code true} if the property is active, i.e., all
188         * {@link Requirement}s must be fulfilled.
189         * 
190         * @return {@code true} if the property is active
191         */
192        public boolean isActive() {
193
194                for (Requirement requirement : requirements) {
195                        if (!requirement.isFulfilled()) {
196                                return false;
197                        }
198                }
199                return true;
200
201        }

212        /**
213         * Returns the value of the property.
214         * 
215         * @see #setValue
216         * @return the value of the property
217         */
218        public Object getValue() {
219                try {
220                        return getter.invoke(module);
221                } catch (IllegalAccessException e) {
222                } catch (IllegalArgumentException e) {
223                } catch (InvocationTargetException e) {
224                }
225                return null;
226        }

228        /**
229         * Sets the value of the property.
230         * 
231         * @see #getValue
232         * @param value
233         *            the value to set
234         * @throws InvocationTargetException
235         *             thrown if the value cannot be assigned
236         */
237        public void setValue(Object value) throws InvocationTargetException {
238                setValueObject(value);
239        }
240
241        /**
242         * Sets the value of the property. The property has first to be converted to
243         * the corresponding type.
244         * 
245         * @param value
246         *            the value to set
247         * @throws InvocationTargetException
248         *             thrown if the value cannot be assigned
249         */
250        public void setValue(String value) throws InvocationTargetException {
251                Class<?> type = this.type;
252                String valueToSet = value;
253
254                if (type.equals(Integer.TYPE)) {
255                        type = Integer.class;
256                } else if (type.equals(Long.TYPE)) {
257                        type = Long.class;
258                } else if (type.equals(Double.TYPE)) {
259                        type = Double.class;
260                } else if (type.equals(Float.TYPE)) {
261                        type = Float.class;
262                } else if (type.equals(Boolean.TYPE)) {
263                        type = Boolean.class;
264                } else if (type.equals(Byte.TYPE)) {
265                        type = Byte.class;
266                } else if (type.equals(Short.TYPE)) {
267                        type = Short.class;
268                } else if (type.equals(String.class) && valueToSet == null) {
269                        valueToSet = "";
270                }
271
272                Object object = null;
273                try {
274                        if (Character.TYPE.equals(type) || Character.class.equals(type)) {
275                                object = valueToSet.toCharArray()[0];
276                        } else if (type.isEnum()) {
277                                object = PropertyModule.toEnumElement(valueToSet, type.asSubclass(Enum.class));
278                        } else if (type.equals(Class.class)) {
279                                if (valueToSet != null && !valueToSet.equals("")) {
280                                        Class<?> c = Class.forName(valueToSet);
281
282                                        Type gtype = getter.getGenericReturnType();
283                                        Type actual = getEnclosingType(gtype);
284
285                                        if (isAssignable(actual, c)) {
286                                                object = c;
287                                        } else {
288                                                System.err.println("Can not assign " + c + " to Class<" + actual + ">");
289                                        }
290                                }
291                        } else {
292                                // for Byte/Integer/Long/Float/Double/String/Boolean
293                                // and all objects that have a constructor that accepts one
294                                // string
295                                Constructor<?> constructor = type.getConstructor(String.class);
296                                object = constructor.newInstance(valueToSet.trim());
297                        }
298                } catch (Exception e) {
299                        Throwable t = e;
300                        while (t.getCause() != null) {
301                                t = t.getCause();
302                        }
303                        String message = t.getLocalizedMessage();
304
305                        throw new InvocationTargetException(e,
306                                        "Failed assignment: module=" + module.getClass().getName() + " property=\"" + getName()
307                                                        + "\" value=\"" + valueToSet + "\" (" + t.getClass().getName()
308                                                        + (message != null ? ": " + message : "") + ")");
309                }
310                setValueObject(object);
311        }

313        /**
314         * Returns {@code true} if the property represents a number.
315         * 
316         * @return {@code true} if the property represents a number
317         */
318        public boolean isNumber() {
319                boolean isNumber = false;
320                if (type.equals(Integer.TYPE)) {
321                        isNumber = true;
322                } else if (type.equals(Long.TYPE)) {
323                        isNumber = true;
324                } else if (type.equals(Double.TYPE)) {
325                        isNumber = true;
326                } else if (type.equals(Float.TYPE)) {
327                        isNumber = true;
328                } else if (type.equals(Byte.TYPE)) {
329                        isNumber = true;
330                } else if (type.equals(Short.TYPE)) {
331                        isNumber = true;
332                }
333                return isNumber;
334        }
335
336        /**
337         * Returns the enclosing actual type of a class.
338         * 
339         * @param gtype
340         *            the type of the class
341         * @return the enclosing actual type
342         */
343        private Type getEnclosingType(Type gtype) {
344                if (gtype instanceof ParameterizedType) {
345                        ParameterizedType ptype = (ParameterizedType) gtype;
346                        Type rtype = ptype.getRawType();
347                        if (rtype.equals(Class.class)) {
348                                Type actual = ptype.getActualTypeArguments()[0];
349                                return actual;
350                        }
351                }
352                return null;
353        }
354
355        /**
356         * Verifies if a class is assignable to a type.
357         * 
358         * @param type
359         *            the type
360         * @param clazz
361         *            the class
362         * @return {@code true} if assignable
363         */
364        private boolean isAssignable(Type type, Class<?> clazz) {
365                if (type instanceof WildcardType) {
366                        WildcardType wildcard = (WildcardType) type;
367                        for (Type lower : wildcard.getLowerBounds()) {
368                                if (lower instanceof Class<?>) {
369                                        Class<?> lowerClass = (Class<?>) lower;
370                                        if (!clazz.isAssignableFrom(lowerClass)) {
371                                                return false;
372                                        }
373                                }
374                        }
375                        for (Type upper : wildcard.getUpperBounds()) {
376                                if (upper instanceof Class<?>) {
377                                        Class<?> upperClass = (Class<?>) upper;
378                                        if (!upperClass.isAssignableFrom(clazz)) {
379                                                return false;
380                                        }
381                                }
382                        }
383                }
384                return true;
385        }

387        /**
388         * Sets the value of the property.
389         * 
390         * @param value
391         *            the value to set
392         * @throws InvocationTargetException
393         *             thrown if the value cannot be assigned
394         */
395        protected void setValueObject(Object value) throws InvocationTargetException {
396                try {
397                        setter.invoke(module, value);
398                } catch (Exception e) {
399                        throw new InvocationTargetException(e);
400                }
401        }

409                return annotations.values();

412        /**
413         * Returns the annotation of the specified class or null if not existent.
414         * 
415         * @param <A>
416         *            the annotation type
417         * @param clazz
418         *            the specific annotation class
419         * @return the annotation
420         */
421        @SuppressWarnings("unchecked")
422        public <A extends Annotation> A getAnnotation(Class<? extends A> clazz) {
423                return (A) annotations.get(clazz);
