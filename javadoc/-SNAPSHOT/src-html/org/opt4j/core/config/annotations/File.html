018package org.opt4j.core.config.annotations;
019
020import static java.lang.annotation.ElementType.FIELD;
021import static java.lang.annotation.ElementType.METHOD;
022import static java.lang.annotation.RetentionPolicy.RUNTIME;
023
024import java.lang.annotation.Retention;
025import java.lang.annotation.Target;
026
027/**
028 * The {@link File} annotation marks a {@link String} property as a file such that the {@link String} can be changed by
029 * a browser.
030 * 
031 * @author lukasiewycz
032 * 
033 */
034@Retention(RUNTIME)
035@Target({ METHOD, FIELD })
036public @interface File {
037        /**
038         * The file extension string.
039         * 
040         * @return the file extension string
041         */
042        String value() default "";
043
044        /**
045         * The annotated property is a file. Defaults to {@code true}.
046         * 
047         * @return {@code true}, if files are selectable
048         */
049        boolean file() default true;
050
051        /**
052         * The annotated property is a folder. Defaults to {@code false}.
053         * 
054         * @return {@code true}, if folders are selectable
055         */
056        boolean folder() default false;
057}
