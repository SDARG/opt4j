024package org.opt4j.core.config.annotations;
025
026import static java.lang.annotation.ElementType.TYPE;
027import static java.lang.annotation.RetentionPolicy.RUNTIME;
028
029import java.lang.annotation.Retention;
030import java.lang.annotation.Target;
031
032import org.opt4j.core.config.visualization.PropertyPanel;
033
034/**
035 * The {@link Panel} annotation allows the definition of a custom panel for a
036 * module.
037 * 
038 * @author lukasiewycz
039 * 
040 */
041@Retention(RUNTIME)
042@Target({ TYPE })
043public @interface Panel {
044
045        /**
046         * Returns the parent class.
047         * 
048         * @return the parent class
049         */
050        Class<? extends PropertyPanel> value();
051}
