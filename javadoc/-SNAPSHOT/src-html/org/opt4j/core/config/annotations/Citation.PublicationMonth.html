001package org.opt4j.core.config.annotations;
002
003import static java.lang.annotation.ElementType.FIELD;
004import static java.lang.annotation.ElementType.METHOD;
005import static java.lang.annotation.ElementType.TYPE;
006import static java.lang.annotation.RetentionPolicy.RUNTIME;
007
008import java.lang.annotation.Documented;
009import java.lang.annotation.Retention;
010import java.lang.annotation.Target;
011import java.time.DateTimeException;
012import java.time.Month;
013
014import org.opt4j.core.start.Opt4JModule;
015
016/**
017 * The {@link Citation} can be used in {@link Opt4JModule}s to add citations to
018 * scientific works.
019 * 
020 * Example:
021 * 
022 * <pre>
023 * &#64;Citation(title     = "Opt4J: A Modular Framework for Meta-heuristic Optimization",
024 *           authors   = "Martin Lukasiewycz, Michael Gla&szlig;, Felix Reimann, and J&uuml;rgen Teich",
025 *           journal   = "Proceedings of the 13th Annual Conference on Genetic and Evolutionary Computation",
026 *           pageFirst = 1723
027 *           pageLast  = 1730
028 *           volume    = 13
029 *           number    = 1
030 *           month     = PublicationMonth.JULY,
031 *           year      = 2011,
032 *           doi       = "10.1145/2001576.2001808")
033 * </pre>
034 * 
035 * @author Felix Reimann
036 *
037 */
038@Retention(RUNTIME)
039@Target({ METHOD, TYPE, FIELD })
040@Documented
041public @interface Citation {
042        enum PublicationMonth {
043                UNKNOWN, JANUARY, FEBRUARY, MARCH, APRIL, MAY, JUNE, JULY, AUGUST, SEPTEMBER, OCTOBER, NOVEMBER, DECEMBER;
044
045                public Month toMonth() {
046                        switch (this) {
047                        case JANUARY:
048                                return Month.JANUARY;
049                        case FEBRUARY:
050                                return Month.FEBRUARY;
051                        case MARCH:
052                                return Month.MARCH;
053                        case APRIL:
054                                return Month.APRIL;
055                        case MAY:
056                                return Month.MAY;
057                        case JUNE:
058                                return Month.JUNE;
059                        case JULY:
060                                return Month.JULY;
061                        case AUGUST:
062                                return Month.AUGUST;
063                        case SEPTEMBER:
064                                return Month.SEPTEMBER;
065                        case OCTOBER:
066                                return Month.OCTOBER;
067                        case NOVEMBER:
068                                return Month.NOVEMBER;
069                        case DECEMBER:
070                                return Month.DECEMBER;
071                        default:
072                                throw new DateTimeException("publication month is unkown");
073                        }
074                }
075        }
076
077        /**
078         * The title.
079         * 
080         * @return the title of the citation
081         */
082        String title();
083
084        /**
085         * The authors of the work.
086         * 
087         * @return the authors
088         */
089        String authors() default "";
090
091        /**
092         * The volume of the journal.
093         * 
094         * @return the volume
095         */
096        int volume() default -1;
097
098        /**
099         * The number of the journal issue.
100         * 
101         * @return the number
102         */
103        int number() default -1;
104
105        /**
106         * The month, in which the paper has been published.
107         * 
108         * @return the month
109         */
110        PublicationMonth month();
111
112        /**
113         * The year, in which the paper has been published.
114         * 
115         * @return the year
116         */
117        int year();
118
119        /**
120         * The journal, in which the paper has been published.
121         * 
122         * @return the title of the journal
123         */
124        String journal() default "";
125
126        /**
127         * The number of the page, where the paper starts.
128         * 
129         * @return the first page
130         */
131        int pageFirst() default -1;
132
133        /**
134         * The number of the page, where the paper ends.
135         * 
136         * @return the last page
137         */
138        int pageLast() default -1;
139
140        /**
141         * The Digital Object Identifier of the work.
142         * 
143         * @return the doi
144         */
145        String doi() default "";
146}
