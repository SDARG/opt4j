054/**
055 * The {@link PropertyModule} is a decorator for a {@link Module} that enables
056 * property methods.
057 * 
058 * @author lukasiewycz
059 * 
060 */
061@Ignore
062@SuppressWarnings("serial")
063public final class PropertyModule implements Module, Serializable, Comparable<PropertyModule> {
064
065        protected final Module module;
066
067        protected final List<Property> properties = new ArrayList<>();
068
069        protected static int c = 0;
070
071        public final int id;
072
073        /**
074         * Converts a String to an element of a given enumeration.
075         * 
076         * @param name
077         *            the name of the element
078         * @param type
079         *            the enumeration type
080         * @return the element of the enumeration
081         */
082        @SuppressWarnings("rawtypes")
083        public static Object toEnumElement(String name, Class<? extends Enum> type) {
084                for (Enum e : type.getEnumConstants()) {
085                        if (e.name().equalsIgnoreCase(name)) {
086                                return e;
087                        }
088                }
089                return null;
090        }

092        /**
093         * Constructs a {@link PropertyModule} that decorates a {@link Module}.
094         * 
095         * @param module
096         *            the decorated module
097         */
098        public PropertyModule(Module module) {
099                super();
100
101                synchronized (this) {
102                        id = c++;
103                }
104
105                this.module = module;
106
107                if (module instanceof PropertyModule) {
108                        throw new IllegalArgumentException("PropertyModule can not decorate a PropertyModule");
109                }
110
111                // search all bean methods and their annotations
112                Map<String, Method> getters = new HashMap<>();
113                Map<String, Method> setters = new HashMap<>();
114                Map<String, Class<?>> types = new HashMap<>();
115                Map<String, Collection<Annotation>> annotations = new HashMap<>();
116
117                for (Method method : module.getClass().getMethods()) {
118                        String name = method.getName();
119
120                        boolean get = name.startsWith("get");
121                        boolean is = name.startsWith("is");
122
123                        try {
124
125                                if (get || is) {
126
127                                        if (get) {
128                                                name = name.substring(3);
129                                        } else if (is) {
130                                                name = name.substring(2);
131                                        }
132
133                                        // this is a get method, search for a corresponding setter
134                                        Class<?> type = method.getReturnType();
135                                        Method setter = module.getClass().getMethod("set" + name, type);
136
137                                        // first letter to lower case
138                                        char[] letters = name.toCharArray();
139                                        letters[0] = Character.toLowerCase(letters[0]);
140                                        name = new String(letters);
141
142                                        Collection<Annotation> a = new HashSet<>();
143
144                                        Class<?> clazz = module.getClass();
145                                        while (!clazz.equals(Object.class)) {
146                                                try {
147                                                        Field field = clazz.getDeclaredField(name);
148                                                        for (Annotation annotation : field.getAnnotations()) {
149                                                                a.add(annotation);
150                                                        }
151                                                        break;
152                                                } catch (NoSuchFieldException e) {
153                                                }
154                                                clazz = clazz.getSuperclass();
155                                        }
156
157                                        for (Annotation annotation : method.getAnnotations()) {
158                                                a.add(annotation);
159                                        }
160                                        for (Annotation annotation : setter.getAnnotations()) {
161                                                a.add(annotation);
162                                        }
163
164                                        getters.put(name, method);
165                                        setters.put(name, setter);
166                                        types.put(name, type);
167                                        annotations.put(name, a);
168                                }
169
170                        } catch (SecurityException e) {
171                                // if the setter is not public
172                        } catch (NoSuchMethodException e) {
173                                // if the setter does not exist
174                        }
175                }
176
177                // construct the properties
178                for (String name : getters.keySet()) {
179                        Method getter = getters.get(name);
180                        Method setter = setters.get(name);
181                        Class<?> type = types.get(name);
182                        Collection<Annotation> anno = annotations.get(name);
183
184                        Property property = new Property(module, name, type, getter, setter, anno);
185                        properties.add(property);
186                }
187
188                // interpret the annotations of the properties
189                for (String name : getters.keySet()) {
190                        Collection<Annotation> a = annotations.get(name);
191                        Property property = getProperty(name);
192
193                        for (Annotation annotation : a) {
194
195                                if (annotation instanceof Info) {
196
197                                        Info info = (Info) annotation;
198                                        property.setInfo(info.value());
199
200                                } else if (annotation instanceof Order) {
201
202                                        Order order = (Order) annotation;
203                                        property.setOrder(order.value());
204
205                                } else if (annotation instanceof Required) {
206
207                                        Required required = (Required) annotation;
208
209                                        Property p = getProperty(required.property());
210                                        if (p == null) {
211                                                throw new IllegalArgumentException("Unknown property " + required.property() + " in annotation "
212                                                                + required + " in module " + this.getClass());
213                                        }
214
215                                        Class<?> type = p.getType();
216
217                                        if (type.isEnum()) {
218
219                                                String[] values = required.elements();
220                                                Collection<Object> elements = new HashSet<>();
221
222                                                for (String value : values) {
223                                                        elements.add(toEnumElement(value, type.asSubclass(Enum.class)));
224                                                }
225
226                                                Requirement requirement = new EnumRequirement(p, elements);
227                                                property.addRequirement(requirement);
228
229                                        } else if (type.equals(Boolean.TYPE)) {
230
231                                                boolean value = required.value();
232
233                                                Requirement requirement = new BooleanRequirement(p, value);
234                                                property.addRequirement(requirement);
235
236                                        }
237
238                                }
239                        }
240                }
241
242                // sort the properties
243                final Map<Property, List<Property>> hierarchy = new HashMap<>();
244
245                for (Property property : properties) {
246                        Property parent = null;
247                        for (Requirement requirement : property.getRequirements()) {
248                                parent = requirement.getProperty();
249                        }
250                        assert (parent != property);
251
252                        List<Property> level = hierarchy.get(parent);
253                        if (level == null) {
254                                level = new ArrayList<>();
255                                hierarchy.put(parent, level);
256                        }
257                        level.add(property);
258                }
259
260                final class PropertyComparator implements Comparator<Property> {
261                        @Override
262                        public int compare(final Property o1, final Property o2) {
263                                Integer value1 = o1.getOrder();
264                                Integer value2 = o2.getOrder();
265
266                                if (!value1.equals(value2)) {
267                                        return value1.compareTo(value2);
268                                }
269
270                                String s1 = o1.getName();
271                                String s2 = o2.getName();
272                                return s1.compareTo(s2);
273                        }
274                }
275
276                for (List<Property> level : hierarchy.values()) {
277                        Collections.sort(level, new PropertyComparator());
278                }
279
280                properties.clear();
281                LinkedList<Property> added = new LinkedList<>();
282                added.add(null);
283
284                while (!added.isEmpty()) {
285                        Property property = added.pop();
286                        if (hierarchy.containsKey(property)) {
287                                List<Property> level = hierarchy.get(property);
288                                if (property == null) {
289                                        properties.addAll(level);
290                                } else {
291                                        int index = properties.indexOf(property);
292                                        properties.addAll(index + 1, level);
293                                }
294                                added.addAll(level);
295                        }
296                }
297
298                // remove ignored properties
299                for (Iterator<Property> it = properties.iterator(); it.hasNext();) {
300                        Property property = it.next();
301                        Ignore ignore = property.getAnnotation(Ignore.class);
302                        if (ignore != null) {
303                                it.remove();
304                        }
305                }
306
307        }

328        /**
329         * Returns the {@link Property} with the specified {@code name}.
330         * 
331         * @param name
332         *            the name of the property
333         * @return the property with the specified {@code name}
334         */
335        public Property getProperty(String name) {
336                for (Property property : properties) {
337                        if (property.getName().equals(name)) {
338                                return property;
339                        }
340                }
341                return null;
342        }

353        /*
354         * (non-Javadoc)
355         * 
356         * @see java.lang.Object#toString()
357         */
358        @Override
359        public String toString() {
360                String s = "Module " + module.getClass().getName() + " ";
361                for (Property property : properties) {
362                        s += property.getName() + "=\"" + property.getValue() + "\" ";
363                }
364
365                return s;
366        }

368        /**
369         * Configures a {@link PropertyModule} with an XML {@link Node}.
370         * 
371         * @see #getConfiguration
372         * @param node
373         *            the configuration as XML node
374         */
375        public void setConfiguration(Node node) {
376                JNode xode = new JNode(node);
377
378                Map<String, String> values = new HashMap<>();
379
380                List<JNode> propertyNodes = xode.getChildren("property");
381                for (JNode propertyNode : propertyNodes) {
382                        String key = propertyNode.getAttribute("name");
383                        String value = null;
384                        if (propertyNode.hasText()) {
385                                value = propertyNode.getText();
386                        } else if (propertyNode.hasAttribute("value")) {
387                                value = propertyNode.getAttribute("value");
388                        }
389                        values.put(key, value);
390                }
391
392                for (Property property : properties) {
393                        String key = property.getName();
394                        if (values.containsKey(key)) {
395                                String value = values.get(key);
396                                try {
397                                        property.setValue(value);
398                                } catch (InvocationTargetException e) {
399                                        System.err.println(e.getMessage());
400                                }
401                        }
402                }
403        }

405        /**
406         * Returns the configuration of the {@link PropertyModule} as XML
407         * {@link Node}.
408         * 
409         * @see #setConfiguration
410         * @param document
411         *            the XML document
412         * @return the configuration as XML node
413         */
414        public Node getConfiguration(Document document) {
415                JNode node = new JNode(document, "module");
416                node.setAttribute("class", module.getClass().getName());
417
418                for (Property property : properties) {
419                        try {
420                                JNode p = node.appendChild("property");
421
422                                String name = property.getName();
423                                Object value = property.getValue();
424
425                                p.setAttribute("name", name);
426                                if (value != null) {
427                                        if (value instanceof Class<?>) {
428                                                p.setText(((Class<?>) value).getCanonicalName());
429                                        } else {
430                                                p.setText(value.toString());
431                                        }
432                                }
433
434                        } catch (IllegalArgumentException e) {
435                                e.printStackTrace();
436                        }
437
438                }
439                return node.getNode();
440        }

442        /*
443         * (non-Javadoc)
444         * 
445         * @see java.lang.Object#clone()
446         */
447        @Override
448        public PropertyModule clone() {
449                try {
450                        Module module = this.module.getClass().newInstance();
451                        PropertyModule propertyModule = new PropertyModule(module);
452                        for (Property p : getProperties()) {
453                                Property property = propertyModule.getProperty(p.getName());
454                                property.setValue(p.getValue());
455                        }
456                        return propertyModule;
457                } catch (IllegalAccessException e) {
458                        throw new IllegalArgumentException(e);
459                } catch (InstantiationException e) {
460                        throw new IllegalArgumentException(e);
461                } catch (InvocationTargetException e) {
462                        throw new IllegalArgumentException(e);
463                }
464        }

477        /*
478         * (non-Javadoc)
479         * 
480         * @see java.lang.Comparable#compareTo(java.lang.Object)
481         */
482        @Override
483        public int compareTo(PropertyModule other) {
484                Class<? extends Module> clazz1 = this.getModule().getClass();
485                Class<? extends Module> clazz2 = other.getModule().getClass();
486
487                String classname1 = clazz1.getName();
488                String classname2 = clazz2.getName();
489
490                if (!clazz1.equals(clazz2)) {
491                        return classname1.compareTo(classname2);
492                } else {
493                        Multi multi = clazz1.getAnnotation(Multi.class);
494                        if (multi == null) {
495                                return 0;
496                        } else {
497                                return this.id - other.id;
498                        }
499                }
500        }

512        /*
513         * (non-Javadoc)
514         * 
515         * @see java.lang.Object#equals(java.lang.Object)
516         */
517        @Override
518        public boolean equals(Object obj) {
519                if (this == obj) {
520                        return true;
521                }
522                if (obj == null) {
523                        return false;
524                }
525                if (getClass() != obj.getClass()) {
526                        return false;
527                }
528                PropertyModule other = (PropertyModule) obj;
529                return this.compareTo(other) == 0;
530        }
