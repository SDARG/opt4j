080        /*
081         * (non-Javadoc)
082         * 
083         * @see org.opt4j.config.gui.Startupable#startup()
084         */
085        @Override
086        public void startup() {
087                Container content = this;
088
089                content.setLayout(new BorderLayout());
090
091                JLabel logoLabel = new JLabel(Icons.getIcon("img/top_logo.png"));
092                logoLabel.setBorder(BorderFactory.createEmptyBorder(5, 5, 5, 5));
093                JPanel logo = new JPanel(new BorderLayout());
094                logo.setBackground(Color.WHITE);
095                logo.add(logoLabel);
096
097                content.add(logo, BorderLayout.PAGE_START);
098
099                JTextPane license = new JTextPane();
100                license.setEditable(false);
101                final JScrollPane licenseScroll = new JScrollPane(license);
102                licenseScroll.setVerticalScrollBarPolicy(ScrollPaneConstants.VERTICAL_SCROLLBAR_ALWAYS);
103                StyledDocument doc = license.getStyledDocument();
104
105                Style regular = StyleContext.getDefaultStyleContext().getStyle(StyleContext.DEFAULT_STYLE);
106
107                try {
108                        doc.insertString(doc.getLength(), LICENSE_TEXT, regular);
109                } catch (BadLocationException e) {
110                        e.printStackTrace();
111                }
112                license.setPreferredSize(new Dimension(360, 100));
113                content.add(licenseScroll, BorderLayout.CENTER);
114
115                SwingUtilities.invokeLater(() -> licenseScroll.getVerticalScrollBar().setValue(0));
116                JPanel footer = new JPanel(new BorderLayout());
117                footer.setBackground(Color.WHITE);
118
119                // Add Copyright & Credits
120                String copyright = "<html>Build " + Opt4J.getDateISO() + " <br /> Version " + Opt4J.getVersion()
121                                + "   \u00a9 Opt4J.org 2007</html>";
122                JLabel copyrightLabel = new JLabel(copyright);
123                copyrightLabel.setHorizontalAlignment(SwingConstants.CENTER);
124                copyrightLabel.setVerticalAlignment(SwingConstants.BOTTOM);
125                copyrightLabel.setBorder(BorderFactory.createEmptyBorder(3, 3, 3, 3));
126                footer.add(copyrightLabel, BorderLayout.EAST);
127
128                String credits = "<html><p>Credits:<br />";
129                for (String author : AUTHORS) {
130                        credits += author + "<br/>";
131                }
132                credits += "</p></html>";
133
134                JLabel creditsLabel = new JLabel(credits);
135                creditsLabel.setHorizontalAlignment(SwingConstants.CENTER);
136                creditsLabel.setBorder(BorderFactory.createEmptyBorder(3, 3, 3, 3));
137                footer.add(creditsLabel, BorderLayout.WEST);
138
139                content.add(footer, BorderLayout.PAGE_END);
140        }
141
142        /*
143         * (non-Javadoc)
144         * 
145         * @see org.opt4j.config.visualization.About#getDialog(org.opt4j.config.
146         * visualization .ApplicationFrame)
147         */
148        @Override
149        public JDialog getDialog(ApplicationFrame frame) {
150                JDialog dialog = new JDialog(frame, "About Opt4J", true);
151                dialog.setBackground(Color.WHITE);
152                dialog.setDefaultCloseOperation(WindowConstants.DISPOSE_ON_CLOSE);
153                dialog.setResizable(false);
154
155                Opt4JAbout content = new Opt4JAbout();
156                content.startup();
157                dialog.add(content);
158
159                Dimension screen = Toolkit.getDefaultToolkit().getScreenSize();
160                Dimension window = dialog.getPreferredSize();
161                dialog.setLocation((screen.width - window.width) / 2, (screen.height - window.height) / 2);
162
163                return dialog;
164        }
