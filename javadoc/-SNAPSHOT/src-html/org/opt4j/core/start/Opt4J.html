034import org.opt4j.core.config.ModuleAutoFinder;

047/**
048 * The {@link Opt4J} configuration GUI.
049 * 
050 * @author lukasiewycz
051 * 
052 */
053public class Opt4J extends Configurator {
054
055        protected static final ModuleListUser moduleList = new ModuleListUser();
056
057        protected static Properties props = new Properties();
058
059        static {
060                try {
061                        props.load(Opt4J.class.getResourceAsStream("/main.properties"));
062                } catch (IOException e) {
063                        e.printStackTrace();
064                }
065        }
066
067        /**
068         * Returns the version of the current build.
069         * 
070         * @return the version
071         */
072        public static String getVersion() {
073                return props.getProperty("version");
074        }
075
076        /**
077         * Returns the date of the current build.
078         * 
079         * @return the date
080         */
081        public static String getDateISO() {
082                return props.getProperty("date");
083        }
084
085        static class SplashDecorator {
086
087                protected DelayTask delay = new DelayTask(1);
088
089                protected SplashScreen splash;
090
091                public SplashDecorator(SplashScreen splash) {
092                        this.splash = splash;
093                }
094
095                public SplashScreen getSplash() {
096                        return splash;
097                }
098
099                public void print(final String message, final Color color) {
100                        delay.execute(new Runnable() {
101                                @Override
102                                public void run() {
103                                        Graphics2D g = splash.createGraphics();
104                                        g.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
105                                        g.setColor(Color.WHITE);
106                                        g.setBackground(new Color(252, 230, 212));
107                                        g.clearRect(10, 175, 280, 20);
108                                        g.setClip(10, 175, 280, 20);
109                                        g.setColor(color);
110                                        g.setFont(new Font("SansSerif", Font.BOLD, 9));
111                                        g.drawString(message, 12, 188);
112                                        splash.update();
113                                }
114                        });
115                }
116        }
117
118        /**
119         * Starts the {@link Opt4J} configuration GUI.
120         * 
121         * @param args
122         *            accepts a configuration file
123         * @throws Exception
124         */
125        public static void main(String[] args) throws Exception {
126                System.out.println("Starting Opt4J " + getVersion() + " (Build " + getDateISO() + ")");
127                if (args.length > 0 && args[0].equalsIgnoreCase("-s")) {
128                        SplashScreen splash = null;
129                        try {
130                                splash = SplashScreen.getSplashScreen();
131                        } catch (HeadlessException e) {
132                                // ignore
133                        }
134                        if (splash != null) {
135                                splash.close();
136                        }
137                        String[] a = new String[args.length - 1];
138                        System.arraycopy(args, 1, a, 0, a.length);
139                        Opt4JStarter.main(a);
140                } else {
141                        SplashScreen splash = SplashScreen.getSplashScreen();
142                        SplashDecorator decorator = null;
143                        if (splash != null) {
144                                decorateVersionDate(splash);
145                                decorator = new SplashDecorator(splash);
146                        }
147                        searchModules(decorator);
148
149                        Configurator configurator = new Opt4J();
150                        configurator.start(args);
151                }
152        }
153
154        /**
155         * Decorate the splash screen with the version and date.
156         * 
157         * @param splash
158         *            the splash screen
159         */
160        protected static void decorateVersionDate(SplashScreen splash) {
161                if (splash != null) {
162                        Graphics2D g = splash.createGraphics();
163                        g.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
164                        g.setColor(new Color(242, 130, 38));
165                        g.setFont(new Font("SansSerif", Font.BOLD, 11));
166                        g.drawString("version " + getVersion(), 170, 76);
167                        g.drawString(getDateISO(), 170, 91);
168                        splash.update();
169                }
170        }
171
172        /**
173         * Search for modules and forward the output to the splash screen.
174         * 
175         * @param splash
176         *            the splash screen
177         */
178        private static void searchModules(SplashDecorator splash) {
179
180                class SplashSearchDecorator implements ModuleAutoFinderListener {
181
182                        protected final SplashDecorator splash;
183
184                        SplashSearchDecorator(SplashDecorator splash) {
185                                this.splash = splash;
186                        }
187
188                        @Override
189                        public void err(String message) {
190                                splash.print(message, Color.RED);
191                        }
192
193                        @Override
194                        public void out(final String message) {
195                                splash.print(message, Color.GRAY.darker());
196                        }
197                }
198
199                ModuleAutoFinder finder = new ModuleAutoFinder();
200                if (splash != null) {
201                        SplashSearchDecorator deco = new SplashSearchDecorator(splash);
202                        finder.addListener(deco);
203                }
204
205                for (Class<? extends Module> module : finder.getModules()) {
206                        moduleList.add(module);
207                }
208
209                if (splash != null) {
210                        try {
211                                Thread.sleep(50);
212                        } catch (InterruptedException e) {
213                                e.printStackTrace();
214                        }
215                }
216        }
217
218        /*
219         * (non-Javadoc)
220         * 
221         * @see org.opt4j.config.gui.Configurator#start(java.lang.String[])
222         */
223        @Override
224        public void start(String[] args) {
225                String filename = null;
226                if (args.length > 0) {
227                        filename = args[0];
228                }
229                main(Opt4JTask.class, filename);
230        }
231
232        /*
233         * (non-Javadoc)
234         * 
235         * @see org.opt4j.config.gui.Configurator#getModule(java.lang.Class)
236         */
237        @Override
238        public Module getModule(final Class<? extends Task> taskClass) {
239                Module module = new Module() {
240                        @Override
241                        public void configure(Binder b) {
242                                b.bind(Task.class).to(taskClass);
243                                b.bind(About.class).to(Opt4JAbout.class);
244                                b.bind(TasksPanel.class).to(Opt4JTasksPanel.class);
245                                b.bind(ModuleList.class).toInstance(moduleList);
246                        }
247                };
248                return module;
249        }
