024package org.opt4j.core;
025
026import com.google.inject.ImplementedBy;
027
028/**
029 * The {@link IndividualFactory} is a creator for {@link Individual}s.
030 * 
031 * @author lukasiewycz, glass
032 * 
033 */
034@ImplementedBy(DefaultIndividualFactory.class)
035public interface IndividualFactory {
036
037        /**
038         * Creates a new {@link Individual}.
039         * 
040         * @return the built individual
041         */
042        public Individual create();
043
044        /**
045         * Creates a new {@link Individual} with a specified {@link Genotype}.
046         * 
047         * @param genotype
048         *            the genotype of the individual
049         * @return the built individual
050         */
051        public Individual create(Genotype genotype);
052
053        /**
054         * Adds an {@link IndividualStateListener} to each {@link Individual} that
055         * is created by this class. This listener is invoked if the state of the
056         * {@link Individual} changes.
057         * 
058         * @see #removeIndividualStateListener
059         * @param listener
060         *            the listener that is invoked if the state of any individual
061         *            changes
062         */
063        public void addIndividualStateListener(IndividualStateListener listener);
064
065        /**
066         * Removes an {@link IndividualStateListener} that is invoked if the state
067         * of any {@link Individual} changes. All {@link Individual}s created after
068         * this method call by this {@link IndividualFactory} do no longer use the
069         * listener.
070         * 
071         * @see #addIndividualStateListener
072         * @param listener
073         *            the listener that is invoked if the state of any individual
074         *            changes
075         */
076        public void removeIndividualStateListener(IndividualStateListener listener);
077
078}
