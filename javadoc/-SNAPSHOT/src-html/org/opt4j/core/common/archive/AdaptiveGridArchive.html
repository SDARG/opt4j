023package org.opt4j.core.common.archive;
024
025import java.util.ArrayList;
026import java.util.Collection;
027import java.util.LinkedHashMap;
028import java.util.LinkedHashSet;
029import java.util.List;
030import java.util.Map;
031import java.util.Set;
032
033import org.opt4j.core.Individual;
034import org.opt4j.core.common.random.Rand;
035import org.opt4j.core.optimizer.Archive;
036import org.opt4j.core.start.Constant;
037
038import com.google.inject.Inject;
039
040/**
041 * The {@link AdaptiveGridArchive} uses an adaptive grid in order to bound the
042 * size of the {@link Archive}, see "Properties of an Adaptive Archiving
043 * Algorithm for Storing Nondominated Vectors, J. Knowles, D. Corne,
044 * Transactions of Evolutionary Computation, Vol. 7, No. 2, April 2003".
045 * 
046 * @author lukasiewycz, helwig
047 * 
048 */
049public class AdaptiveGridArchive extends BoundedArchive {
050
051        protected boolean isInit = false;
052
053        protected final int div;
054
055        protected final Rand random;
056
057        protected final Map<Long, Cell> cells = new LinkedHashMap<>();
058
059        protected final List<Individual> extrema = new ArrayList<>();
060
061        protected double[] lb;
062
063        protected double[] ub;
064
065        protected int dim;
066
067        /**
068         * A single {@link Cell} of the grid.
069         * 
070         * @author lukasiewycz
071         * 
072         */
073        protected static class Cell extends LinkedHashSet<Individual> {
074
075                private static final long serialVersionUID = 1L;
076
077                protected final long identifier;
078
079                /**
080                 * Constructs the {@link Cell} with the {@code identifier}.
081                 * 
082                 * @param identifier
083                 *            the indentifier
084                 */
085                public Cell(long identifier) {
086                        super();
087                        this.identifier = identifier;
088                }
089
090                /**
091                 * Returns the {@code identifier}.
092                 * 
093                 * @return the identifier
094                 */
095                public long getIdentifier() {
096                        return identifier;
097                }
098        }
099
100        /**
101         * Constructs an {@link AdaptiveGridArchive}.
102         * 
103         * @param capacity
104         *            the capacity
105         * @param div
106         *            the divisions
107         * @param random
108         *            the random number generator
109         */
110        @Inject
111        public AdaptiveGridArchive(@Constant(value = "capacity", namespace = BoundedArchive.class) int capacity,
112                        @Constant(value = "div", namespace = AdaptiveGridArchive.class) int div, Rand random) {
113                super(capacity);
114                this.random = random;
115                this.div = div;
116
117        }
118
119        /*
120         * (non-Javadoc)
121         * 
122         * @see
123         * org.opt4j.common.archive.AbstractArchive#updateWithNondominated(java.
124         * util.Collection)
125         */
126        @Override
127        protected boolean updateWithNondominated(Collection<Individual> candidates) {
128
129                if (!isInit && candidates.size() > 0) {
130                        init(candidates.iterator().next());
131                }
132
133                if (determineBounds(candidates)) {
134                        cells.clear();
135                        addToCell(this);
136                }
137                addToCell(candidates);
138
139                boolean changed = false;
140                if (this.size() + candidates.size() <= capacity) {
141                        changed = addCheckedIndividuals(candidates);
142                } else {
143                        candidates.addAll(this);
144
145                        while (candidates.size() > capacity) {
146                                Cell cell = getMostCrowdedCell();
147
148                                List<Individual> list = new ArrayList<>(cell);
149                                list.removeAll(extrema);
150
151                                Individual individual = list.get(random.nextInt(list.size()));
152                                candidates.remove(individual);
153                                removeFromCell(individual);
154                        }
155
156                        this.retainAll(candidates);
157                        for (Individual individual : candidates) {
158                                if (!this.contains(individual)) {
159                                        changed |= addCheckedIndividual(individual);
160                                }
161                        }
162                }
163
164                return changed;
165        }
166
167        /**
168         * Initialize the {@link org.opt4j.core.optimizer.Archive}.
169         * 
170         * @param individual
171         *            one random individual
172         */
173        protected void init(Individual individual) {
174                if (!isInit) {
175                        dim = individual.getObjectives().size();
176
177                        lb = new double[dim];
178                        ub = new double[dim];
179
180                        for (int i = 0; i < dim; i++) {
181                                lb[i] = Double.MAX_VALUE;
182                                ub[i] = Double.MIN_VALUE;
183                                extrema.add(null);
184                                extrema.add(null);
185                        }
186
187                        isInit = true;
188                }
189        }
190
191        /**
192         * Returns the appropriate {@link Cell} for the given {@link Individual}.
193         * 
194         * @param individual
195         *            the individual
196         * @return the cell
197         */
198        protected Cell getCell(Individual individual) {
199                final double[] obj = individual.getObjectives().array();
200
201                int scale = 1;
202                long identifier = 0;
203                for (int i = 0; i < obj.length; i++) {
204                        double half = (1.0 / (2.0 * div)) * (ub[i] - lb[i]);
205                        double lbi = lb[i] - half;
206                        double ubi = ub[i] + half;
207
208                        double step = (ubi - lbi) / div;
209                        int position = (int) Math.floor((obj[i] - lbi) / step);
210
211                        identifier += position * scale;
212                        scale *= div;
213                }
214
215                Cell cell = cells.get(identifier);
216                if (cell != null) {
217                        return cell;
218                } else {
219                        return new Cell(identifier);
220                }
221        }
222
223        /**
224         * Returns the most crowded {@link Cell}.
225         * 
226         * @return the most crowded cell
227         */
228        protected Cell getMostCrowdedCell() {
229                Cell mc = null;
230                int size = 0;
231
232                Set<Individual> extrema = new LinkedHashSet<>(this.extrema);
233
234                for (Cell cell : cells.values()) {
235                        int s = 0;
236                        assert (!cell.isEmpty());
237                        for (Individual individual : cell) {
238                                if (!extrema.contains(individual)) {
239                                        s++;
240                                }
241                        }
242                        if (s > size) {
243                                size = s;
244                                mc = cell;
245                        }
246                }
247
248                return mc;
249        }
250
251        /*
252         * (non-Javadoc)
253         * 
254         * @see org.opt4j.core.IndividualCollection#remove(java.lang.Object)
255         */
256        @Override
257        public boolean remove(Object o) {
258                Individual individual = (Individual) o;
259
260                removeFromCell(individual);
261
262                if (extrema.contains(individual)) {
263                        for (int i = 0; i < extrema.size(); i++) {
264                                if (individual.equals(extrema.get(i))) {
265                                        extrema.set(i, null);
266                                        int j = (int) Math.floor(i / 2.0);
267                                        if (i % 2 == 0) {
268                                                lb[j] = Double.MAX_VALUE;
269                                        } else {
270                                                ub[j] = Double.MIN_VALUE;
271                                        }
272                                }
273                        }
274                }
275
276                return super.remove(individual);
277        }
278
279        /**
280         * Adds all {@link Individual}s to theirs {@link Cell}s and the {@link Cell}
281         * s to the {@link Archive}.
282         * 
283         * @param individuals
284         *            the individuals to be added
285         */
286        protected void addToCell(Iterable<Individual> individuals) {
287                for (Individual individual : individuals) {
288                        Cell cell = getCell(individual);
289                        cell.add(individual);
290                        cells.put(cell.identifier, cell);
291                }
292        }
293
294        /**
295         * Removes this {@link Individual} from its {@link Cell} and removes the
296         * {@link Cell} from the {@link org.opt4j.core.optimizer.Archive} if it is
297         * empty.
298         * 
299         * @param individual
300         *            the individual to be removed
301         */
302        protected void removeFromCell(Individual individual) {
303                Cell cell = getCell(individual);
304                cell.remove(individual);
305                if (cell.isEmpty()) {
306                        cells.remove(cell.getIdentifier());
307                }
308        }
309
310        /**
311         * Determines the new bounds for the individuals and returns {@code true} if
312         * the bounds have changed.
313         * 
314         * @param individuals
315         *            the individuals
316         * @return {@code true} if the bounds have changed
317         */
318        protected boolean determineBounds(Collection<Individual> individuals) {
319                boolean boundsChanged = false;
320
321                for (Individual individual : individuals) {
322                        final double[] obj = individual.getObjectives().array();
323
324                        for (int i = 0; i < dim; i++) {
325                                double value = obj[i];
326
327                                if (value < lb[i]) {
328                                        boundsChanged = true;
329                                        extrema.set(2 * i, individual);
330                                        lb[i] = obj[i];
331                                } else if (value > ub[i]) {
332                                        boundsChanged = true;
333                                        extrema.set(2 * i + 1, individual);
334                                        ub[i] = obj[i];
335                                }
336                        }
337                }
338
339                return boundsChanged;
340        }
341}
