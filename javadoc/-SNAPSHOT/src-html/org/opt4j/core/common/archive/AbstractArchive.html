023package org.opt4j.core.common.archive;
024
025import java.util.ArrayList;
026import java.util.Collection;
027import java.util.Iterator;
028import java.util.List;
029import java.util.Set;
030
031import org.opt4j.core.Individual;
032import org.opt4j.core.Objectives;
033import org.opt4j.core.optimizer.Archive;
034
035/**
036 * This {@link AbstractArchive} provides some common methods for {@link Archive}
037 * s. If one or more new {@link Individual}s are added to this {@link Archive},
038 * it is ensured that all {@link Individual}s in this {@link Archive} are not
039 * Pareto-dominated. Actual implementations of this class may still refuse or
040 * drop some {@link Individual}s. An {@link Archive} can be a
041 * {@link BoundedArchive} if it has a bounded size or an
042 * {@link UnboundedArchive}, otherwise.
043 * 
044 * @author helwig, glass, lukasiewycz
045 * 
046 */
047public abstract class AbstractArchive extends Archive {
048
049        /*
050         * (non-Javadoc)
051         * 
052         * @see org.opt4j.core.optimizer.Archive#update(java.util.Set)
053         */
054        @Override
055        public boolean update(Set<? extends Individual> individuals) {
056                List<Individual> candidates = new ArrayList<>(individuals);
057                /*
058                 * Remove all individuals that are already in the archive.
059                 */
060                candidates.removeAll(this);
061                removeDominatedCandidates(candidates);
062
063                removeArchiveDominated(candidates);
064
065                return updateWithNondominated(candidates);
066        }
067
068        /**
069         * Remove candidates, which are weakly dominated by another candidate.
070         * 
071         * Thus, the list of candidates which need to be tested against the whole
072         * archive is reduced.
073         * 
074         * @param candidates
075         *            the list of candidates to sanitize
076         */
077        protected void removeDominatedCandidates(List<Individual> candidates) {
078                for (int i = 0; i < candidates.size() - 1; i++) {
079                        Objectives o1 = candidates.get(i).getObjectives();
080                        for (int j = i + 1; j < candidates.size(); j++) {
081                                Objectives o2 = candidates.get(j).getObjectives();
082                                if (o2.weaklyDominates(o1)) {
083                                        candidates.remove(i);
084                                        i--;
085                                        break;
086                                } else if (o1.weaklyDominates(o2)) {
087                                        candidates.remove(j);
088                                        j--;
089                                }
090                        }
091                }
092        }
093
094        /**
095         * Remove those individuals from the candidates which are weakly dominated
096         * by the archive. Remove those individuals from the archive which are
097         * dominated by the candidates. In case of equal objectives, this gives
098         * priority to the individuals in the archive and avoids unnecessary archive
099         * updates.
100         * 
101         * @param candidates
102         *            the list of candidates to sanitize
103         */
104        protected void removeArchiveDominated(List<Individual> candidates) {
105                Iterator<Individual> i1;
106                Iterator<Individual> i2;
107                for (i1 = candidates.iterator(); i1.hasNext();) {
108                        Objectives o1 = i1.next().getObjectives();
109                        for (i2 = this.iterator(); i2.hasNext();) {
110                                Objectives o2 = i2.next().getObjectives();
111                                if (o2.weaklyDominates(o1)) {
112                                        i1.remove();
113                                        break;
114                                } else if (o1.dominates(o2)) {
115                                        i2.remove();
116                                }
117                        }
118                }
119        }
120
121        /**
122         * Adds new {@code candidates} which are already checked to be not
123         * Pareto-dominated by any other individual in this {@link Archive}. All
124         * {@link Individual}s in the {@link Archive} which were dominated by the
125         * candidates have already been removed.
126         * 
127         * @param candidates
128         *            the non-dominated individuals which can be added
129         * @return true if one or more candidates are added to the archive
130         */
131        protected abstract boolean updateWithNondominated(Collection<Individual> candidates);
132
133}
