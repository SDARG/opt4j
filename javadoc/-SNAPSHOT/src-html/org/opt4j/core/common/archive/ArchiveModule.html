018package org.opt4j.core.common.archive;
019
020import static org.opt4j.core.config.annotations.Citation.PublicationMonth.APRIL;
021import static org.opt4j.core.config.annotations.Citation.PublicationMonth.AUGUST;
022
023import org.opt4j.core.config.Icons;
024import org.opt4j.core.config.annotations.Citation;
025import org.opt4j.core.config.annotations.Icon;
026import org.opt4j.core.config.annotations.Info;
027import org.opt4j.core.config.annotations.Required;
028import org.opt4j.core.optimizer.Archive;
029import org.opt4j.core.start.Constant;
030import org.opt4j.core.start.Opt4JModule;
031
032/**
033 * The {@link ArchiveModule} determines an implementation for the {@link Archive} interface.
034 * 
035 * @see UnboundedArchive
036 * @see PopulationArchive
037 * @see CrowdingArchive
038 * @see AdaptiveGridArchive
039 * @see Archive
040 * @author helwig, lukasiewycz
041 * 
042 */
043@Icon(Icons.PUZZLE_BLUE)
044@Info("The archive of non-dominated solutions found during the optimization process.")
045public class ArchiveModule extends Opt4JModule {
046
047        @Info("Archive type")
048        protected Type type = Type.CROWDING;
049
050        @Info("Maximal archive capacity if a bounded archive is selected")
051        @Required(property = "type", elements = { "ADAPTIVE_GRID", "CROWDING" })
052        @Constant(value = "capacity", namespace = BoundedArchive.class)
053        protected int capacity = 100;
054
055        @Info("Divisions for the adaptive grid archive. A convergence is guaranteed if 'capacity>1+div^m+(div-1)^m+2*m' with m being the number of objectives holds")
056        @Citation(title = "Properties of an Adaptive Archiving Algorithm for Storing Nondominated Vectors", authors = "Joshua Knowles and David Corne", journal = "Transactions of Evolutionary Computation", volume = 7, number = 2, month = APRIL, year = 2003)
057        @Required(property = "type", elements = { "ADAPTIVE_GRID" })
058        @Constant(value = "div", namespace = AdaptiveGridArchive.class)
059        protected int divisions = 7;
060
061        /** Archive type. */
062        public enum Type {
063                /**
064                 * Archive of unlimited size.
065                 * 
066                 * @see UnboundedArchive
067                 */
068                @Info("Archive of unlimited size")
069                UNBOUNDED,
070
071                /**
072                 * Archive that keeps the non-dominated individual of the population.
073                 * 
074                 * @see PopulationArchive
075                 */
076                @Info("Archive that keeps the non-dominated individual of the population")
077                POPULATION,
078
079                /**
080                 * Adaptive grid archive.
081                 * 
082                 * @see AdaptiveGridArchive
083                 */
084                @Info("Adaptive grid archive")
085                @Citation(title = "Properties of an Adaptive Archiving Algorithm for Storing Nondominated Vectors", authors = "Joshua Knowles and David Corne", journal = "Transactions of Evolutionary Computation", volume = 7, number = 2, month = APRIL, year = 2003)
086                ADAPTIVE_GRID,
087
088                /**
089                 * Bounded archive based on the NSGA2 crowding distance.
090                 * 
091                 * @see CrowdingArchive
092                 */
093                @Info("Bounded archive based on the crowding distance (NSGA2)")
094                @Citation(title = "A fast and elitist multiobjective genetic algorithm : NSGA-II", authors = "Kalyanmoy Deb, Amrit Pratap, Samir Agarwal, and Tanaka Meyarivan", journal = "IEEETransactions on Evolutionary Computation", pageFirst = 182, pageLast = 197, month = AUGUST, year = 2002)
095                CROWDING;
096        }
097
098        /**
099         * Returns the number of divisions for the {@link AdaptiveGridArchive}.
100         * 
101         * @see #setDivisions
102         * @return the number of divisions
103         */
104        public int getDivisions() {
105                return divisions;
106        }
107
108        /**
109         * Set the number of divisions for the {@link AdaptiveGridArchive}.
110         * 
111         * @see #getDivisions
112         * @param divisions
113         *            the number of divisions
114         */
115        public void setDivisions(int divisions) {
116                this.divisions = divisions;
117        }
118
119        /**
120         * Sets the archive type to the specified value.
121         * 
122         * @see #getType
123         * @param type
124         *            the new archive type
125         */
126        public void setType(Type type) {
127                this.type = type;
128        }
129
130        /**
131         * Returns the archive type.
132         * 
133         * @see #setType
134         * @return the archive type
135         */
136        public Type getType() {
137                return type;
138        }
139
140        /**
141         * Sets the archive's capacity to the specified value.
142         * 
143         * @see #getCapacity
144         * @param capacity
145         *            the new capacity (using namespace {@link BoundedArchive})
146         */
147        public void setCapacity(int capacity) {
148                this.capacity = capacity;
149        }
150
151        /**
152         * Returns the archive's capacity.
153         * 
154         * @see #setCapacity
155         * @return the archive's capacity
156         */
157        public int getCapacity() {
158                return capacity;
159        }
160
161        /*
162         * (non-Javadoc)
163         * 
164         * @see org.opt4j.start.Opt4JModule#config()
165         */
166        @Override
167        public void config() {
168                Class<? extends Archive> archiveClass = null;
169
170                switch (type) {
171                case ADAPTIVE_GRID:
172                        archiveClass = AdaptiveGridArchive.class;
173                        break;
174                case POPULATION:
175                        archiveClass = PopulationArchive.class;
176                        break;
177                case CROWDING:
178                        archiveClass = CrowdingArchive.class;
179                        break;
180                default: // UNBOUNDED
181                        archiveClass = UnboundedArchive.class;
182                        break;
183                }
184
185                bind(Archive.class).to(archiveClass).in(SINGLETON);
186        }
187}
