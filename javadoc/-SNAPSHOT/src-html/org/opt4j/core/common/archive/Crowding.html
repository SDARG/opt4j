023package org.opt4j.core.common.archive;
024
025import java.util.ArrayList;
026import java.util.Collection;
027import java.util.Collections;
028import java.util.Comparator;
029import java.util.HashMap;
030import java.util.List;
031import java.util.Map;
032
033import org.opt4j.core.Individual;
034
035/**
036 * The {@link Crowding}, see "A fast and elitist multiobjective genetic
037 * algorithm : NSGA-II, K. Deb, A. Pratap, S. Agarwal, and T. Meyarivan
038 * Evolutionary Computation, IEEETransactions on, vol. 6, no. 2, pp. 182-197,
039 * August 2002.", calculates the crowding distance of {@link Individual}s as
040 * used for instance in the the {@code Nsga2} algorithm. A bounding box around
041 * each individual is defined and restricted by the neighboring individuals in
042 * each dimension. The crowding distance is the sum of the side-lengths of the
043 * bounding box. If an individual does not have neighbors on all sides, the
044 * crowding distance is maximal. A normalization in each dimension is applied.
045 * 
046 * @author lukasiewycz
047 * 
048 */
049
050public class Crowding implements FrontDensityIndicator {
051
052        /**
053         * Calculates the crowding distance for a collection of {@link Individual}s.
054         * 
055         * @param individuals
056         *            the individuals
057         * @return a map of the individuals to the corresponding crowding distance
058         */
059        private Map<Individual, Double> getValues(Collection<Individual> individuals) {
060                Map<Individual, Double> map = new HashMap<>();
061                List<Individual> list = new ArrayList<>(individuals);
062
063                for (Individual individual : list) {
064                        map.put(individual, 0.0);
065                }
066
067                int m = list.get(0).getObjectives().array().length;
068
069                for (int i = 0; i < m; i++) {
070                        final int dim = i;
071                        Comparator<Individual> foo = (Individual o1, Individual o2) -> (int) Math
072                                        .signum(o1.getObjectives().array()[dim] - o2.getObjectives().array()[dim]);
073                        Collections.sort(list, foo);
074
075                        Individual minIndividual = list.get(0);
076                        Individual maxIndividual = list.get(list.size() - 1);
077
078                        double diff = maxIndividual.getObjectives().array()[dim] - minIndividual.getObjectives().array()[dim];
079                        if (diff > 0) {
080                                map.put(minIndividual, Double.MAX_VALUE);
081                                map.put(maxIndividual, Double.MAX_VALUE);
082
083                                for (int j = 1; j < list.size() - 1; j++) {
084                                        double p = list.get(j - 1).getObjectives().array()[dim];
085                                        double n = list.get(j + 1).getObjectives().array()[dim];
086                                        Individual ind = list.get(j);
087                                        map.put(ind, map.get(ind) + ((n - p) / diff));
088                                }
089                        }
090                }
091
092                return map;
093        }
094
095        /**
096         * Returns an ordered list of the {@link Individual}s corresponding to their
097         * crowding distance. {@link Individual}s with a high crowding distance are
098         * sorted to the front.
099         * 
100         * @param values
101         *            the map of individuals to their crowding distance values
102         * @return the sorted list of individuals based on their crowding distance
103         */
104        public List<Individual> order(final Map<Individual, Double> values) {
105                List<Individual> list = new ArrayList<>(values.keySet());
106                Comparator<Individual> valueCompare = (Individual o1, Individual o2) -> values.get(o2)
107                                .compareTo(values.get(o1));
108                Collections.sort(list, valueCompare);
109                return list;
110        }
111
112        /*
113         * (non-Javadoc)
114         * 
115         * @see
116         * org.opt4j.optimizer.ea.FrontDensityIndicator#getDensityValues(java.util
117         * .Collection)
118         */
119        @Override
120        public Map<Individual, Double> getDensityValues(Collection<Individual> individuals) {
121                return getValues(individuals);
122        }
123
124}
