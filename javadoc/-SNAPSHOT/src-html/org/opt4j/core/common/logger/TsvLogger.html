018package org.opt4j.core.common.logger;
019
020import static org.opt4j.core.Objective.INFEASIBLE;
021
022import java.io.BufferedWriter;
023import java.io.File;
024import java.io.FileWriter;
025import java.io.IOException;
026import java.io.PrintWriter;
027import java.util.Collection;
028
029import org.opt4j.core.Individual;
030import org.opt4j.core.Objective;
031import org.opt4j.core.Objectives;
032import org.opt4j.core.Value;
033import org.opt4j.core.optimizer.Archive;
034import org.opt4j.core.start.Constant;
035
036import com.google.inject.Inject;
037
038/**
039 * The {@link TsvLogger} writes all {@link Individual}s from the {@link Archive} to the specified file. It can be
040 * configured to write the data each {@code iterationStep} iterations or each {@code evaluationStep} evaluations.
041 * 
042 * The file format is TSV (tab separated values), according to the
043 * <a href= "http://www.iana.org/assignments/media-types/text/tab-separated-values" >Definition of tab-separated-values
044 * (tsv)</a> by IANA.
045 * 
046 * An infeasible Objective is printed as <it>INFEASIBLE</it>.
047 * 
048 * @see Objective
049 * @author reimann, lukasiewycz
050 */
051public class TsvLogger extends AbstractLogger implements Logger {
052
053        protected final Archive archive;
054
055        private final PrintWriter out;
056        private long startTime = -1;
057
058        /**
059         * Creates an {@link TsvLogger}.
060         * 
061         * @param archive
062         *            the archive
063         * @param filename
064         *            the filename (using namespace {@link TsvLogger})
065         * @param evaluationStep
066         *            the number of evaluations between two logging events (using namespace {@link TsvLogger})
067         * @param iterationStep
068         *            the number of iterations between two logging events (using namespace {@link TsvLogger})
069         */
070        @Inject
071        public TsvLogger(Archive archive, @Constant(value = "filename", namespace = TsvLogger.class) String filename,
072                        @Constant(value = "evaluationStep", namespace = TsvLogger.class) int evaluationStep,
073                        @Constant(value = "iterationStep", namespace = TsvLogger.class) int iterationStep) {
074                super(iterationStep, evaluationStep);
075                this.archive = archive;
076                this.out = initWriter(filename);
077        }
078
079        /**
080         * Initialize the print write from a filename
081         * 
082         * @param filename
083         *            the filename
084         * @return the print writer
085         */
086        protected PrintWriter initWriter(String filename) {
087                if (filename == null || filename.equals("")) {
088                        throw new IllegalArgumentException("File name must not be the empty string.");
089                }
090
091                File file = new File(filename);
092                File dir = file.getParentFile();
093                if (dir != null) {
094                        dir.mkdirs();
095                }
096                try {
097                        return new PrintWriter(new BufferedWriter(new FileWriter(file)));
098                } catch (IOException e) {
099                        throw new IllegalArgumentException(e);
100                }
101        }
102
103        /*
104         * (non-Javadoc)
105         * 
106         * @see org.opt4j.common.logger.AbstractLogger#optimizationStarted()
107         */
108        @Override
109        public void optimizationStarted() {
110                startTime = System.currentTimeMillis();
111        }
112
113        /*
114         * (non-Javadoc)
115         * 
116         * @see org.opt4j.common.logger.AbstractLogger#logEvent(int, int)
117         */
118        @Override
119        public void logEvent(int iteration, int evaluation) {
120                assert startTime != -1 : "not initialized";
121                double time = ((double) System.currentTimeMillis() - startTime) / 1000.0;
122                for (Individual individual : archive) {
123                        String statistics = getStatistics(iteration, evaluation, time);
124                        String objectives = getIndividual(individual);
125                        if (objectives != null) {
126                                out.println(statistics + objectives);
127                        }
128                }
129                out.flush();
130        }
131
132        /*
133         * (non-Javadoc)
134         * 
135         * @see org.opt4j.common.logger.AbstractLogger#optimizationStopped()
136         */
137        @Override
138        public void optimizationStopped() {
139                out.close();
140        }
141
142        /*
143         * (non-Javadoc)
144         * 
145         * @see org.opt4j.common.logger.AbstractLogger#logHeader(java.util.Collection)
146         */
147        @Override
148        public void logHeader(Collection<Objective> objectives) {
149                String header = getCommentDelimiter() + "iteration" + getColumnDelimiter() + "evaluations"
150                                + getColumnDelimiter() + "runtime[s]";
151                for (Objective objective : objectives) {
152                        String name = objective.getName().replaceAll("[ \n\t\r]", "_");
153                        header += getColumnDelimiter() + name + "[" + objective.getSign() + "]";
154                }
155                out.println(header);
156        }
157
158        /**
159         * The {@link String} separating two columns.
160         * 
161         * The tab character ("\t") is the default, which leads to a tab separated values file format (TSV).
162         * 
163         * @return the delimiter
164         */
165        protected String getColumnDelimiter() {
166                return "\t";
167        }
168
169        /**
170         * Creates a {@link String} representation of the given {@link Individual}.
171         * 
172         * Per default, the {@link Value}s of all {@link Objectives} of the individual, separated by
173         * {@link #getColumnDelimiter()}, are returned. The {@link String} representation of a {@link Value} must not
174         * contain the tab character.
175         * 
176         * If one of the values of the {@link Objectives} is INFEASIBLE, {@code null} is returned.
177         * 
178         * @see Objective
179         * @param individual
180         *            the individual
181         * @return the corresponding string or {@code null}
182         */
183        protected String getIndividual(Individual individual) {
184                String output = "";
185                Objectives objectives = individual.getObjectives();
186
187                for (Objective objective : objectives.getKeys()) {
188                        Value<?> value = objectives.get(objective);
189                        assert value != null : "Objective " + objective.getName() + " not set for individual " + individual;
190
191                        String valueString;
192                        if (value == INFEASIBLE || value == null || value.getValue() == null) {
193                                valueString = "INFEASIBLE";
194                        } else {
195                                String v = value.getValue().toString();
196                                if (v.contains("\t")) {
197                                        System.err.println(this + ": Value must not contain the tab character:" + v);
198                                        v = v.replace("\t", "_");
199                                }
200                                valueString = v;
201                        }
202                        output += getColumnDelimiter() + valueString;
203                }
204                return output;
205        }
206
207        /**
208         * Statistics for the given {@code iteration} and {@code evaluation}.
209         * 
210         * Per default, the iteration, the number of evaluations and the optimization run time in milliseconds, separated by
211         * {@link #getColumnDelimiter()}, are returned.
212         * 
213         * @param iteration
214         *            the current iteration
215         * @param evaluation
216         *            the current number of evaluations
217         * @param time
218         *            the current runtime of the optimization
219         * @return the corresponding string
220         */
221        protected String getStatistics(int iteration, int evaluation, double time) {
222                return iteration + getColumnDelimiter() + evaluation + getColumnDelimiter() + time;
223        }
224
225        /**
226         * The {@link String} starting a line comment.
227         * 
228         * Default is the empty String ("") according to the TSV specification.
229         * 
230         * @return the comment delimiter
231         */
232        protected String getCommentDelimiter() {
233                return "";
234        }
235
236}
