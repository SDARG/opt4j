023package org.opt4j.satdecoding;
024
025import java.util.ArrayList;
026import java.util.Iterator;
027
028/**
029 * A linear pseudo-Boolean {@link Constraint} as used in an Integer Linear
030 * Program (ILP). A set of {@link Constraint}s specify a search space.
031 * <p>
032 * 
033 * This {@link Constraint} consists of a list of {@link Term}s, an
034 * {@link Operator} and a right hand side. E.g., {@code 4a - 3b + not(c) <= 7}
035 * is a valid {@link Constraint} consisting of three {@link Literal}s
036 * {@code a, b, not(c)} with coefficients {@code 4, 3, 1} , the less or equal
037 * operator from {@link Operator}, and the right hand side {@code 7}.
038 * <p>
039 * 
040 * This implementation is a plain data structure, i.e., {@code 4a - 3a + c <= 1}
041 * is not simplified to {@code a + c <= 1}. Moreover, two {@link Constraint}s
042 * are equal if they have an equal {@link Operator}, right hand side, and equal
043 * {@link Term}s in the same order.
044 * 
045 * @author lukasiewycz
046 * 
047 */
048public class Constraint extends ArrayList<Term> {
049
050        protected Operator operator = Operator.GE;
051        protected int rhs = 1;
052
053        private static final long serialVersionUID = 1L;
054
055        /**
056         * The {@link Operator}s correspond to the Boolean operators &lt;=,=,&gt;=.
057         */
058        public enum Operator {
059                /**
060                 * Boolean operation &lt;=.
061                 */
062                LE,
063                /**
064                 * Boolean operation =.
065                 */
066                EQ,
067                /**
068                 * Boolean operation &gt;=.
069                 */
070                GE;
071
072                /**
073                 * Returns a {@link String} representation of this {@link Operator}.
074                 * 
075                 * @return a string representation of this operator
076                 */
077                @Override
078                public String toString() {
079                        switch (this) {
080                        case LE:
081                                return "<=";
082                        case EQ:
083                                return "=";
084                        default: // GE
085                                return ">=";
086                        }
087                }
088
089                /**
090                 * Returns {@code true} if {@code lhs} {@link Operator} {@code rhs} is
091                 * true.
092                 * 
093                 * @param lhs
094                 *            the left hand side value
095                 * @param rhs
096                 *            the right hand side value
097                 * @return {@code true} if the operation is true for the two values
098                 */
099                public boolean isTrue(int lhs, int rhs) {
100                        switch (this) {
101                        case LE:
102                                return (lhs <= rhs);
103                        case EQ:
104                                return (lhs == rhs);
105                        default: // GE
106                                return (lhs >= rhs);
107                        }
108                }
109
110                /**
111                 * Converts a {@link String} to an operator.
112                 * 
113                 * @param string
114                 *            the string (&lt;=,=,&gt;=).
115                 * @return the operator
116                 */
117                public static Operator getOperator(String string) {
118                        if ("<=".equals(string)) {
119                                return LE;
120                        } else if ("=".equals(string)) {
121                                return EQ;
122                        } else if (">=".equals(string)) {
123                                return GE;
124                        } else {
125                                throw new IllegalArgumentException("Unknown operator " + string + ". Allowed operators are: <=,=,>=");
126                        }
127                }
128        }
129
130        /**
131         * Constructs a linear {@link Constraint} with {@code >=1}.
132         */
133        public Constraint() {
134                this(Operator.GE, 1);
135        }
136
137        /**
138         * Constructs a linear {@link Constraint}.
139         * 
140         * @param operator
141         *            the operator (&lt;=,=,&gt;=) as a String
142         * @param rhs
143         *            the right-hand-side value
144         */
145        public Constraint(String operator, int rhs) {
146                this(Operator.getOperator(operator), rhs);
147        }
148
149        /**
150         * Constructs a linear {@link Constraint}.
151         * 
152         * @param operator
153         *            the operator
154         * @param rhs
155         *            the right-hand-side value
156         */
157        public Constraint(Operator operator, int rhs) {
158                this.operator = operator;
159                this.rhs = rhs;
160        }
161
162        /**
163         * Returns an iterable of the coefficients.
164         * 
165         * @return the coefficients
166         */
167        public Iterable<Integer> getCoefficients() {
168                return new Iterable<Integer>() {
169                        @Override
170                        public Iterator<Integer> iterator() {
171                                return new Iterator<Integer>() {
172
173                                        int i = 0;
174
175                                        @Override
176                                        public boolean hasNext() {
177                                                return (i < size());
178                                        }
179
180                                        @Override
181                                        public Integer next() {
182                                                return get(i++).getCoefficient();
183                                        }
184
185                                        @Override
186                                        public void remove() {
187                                                throw new UnsupportedOperationException("Do not remove constraint coefficients.");
188                                        }
189                                };
190                        }
191                };
192        }
193
194        /**
195         * Returns an iterable of the {@link Literal}s.
196         * 
197         * @return the literals.
198         */
199        public Iterable<Literal> getLiterals() {
200                return new Iterable<Literal>() {
201                        @Override
202                        public Iterator<Literal> iterator() {
203                                return new Iterator<Literal>() {
204
205                                        int i = 0;
206
207                                        @Override
208                                        public boolean hasNext() {
209                                                return (i < size());
210                                        }
211
212                                        @Override
213                                        public Literal next() {
214                                                return get(i++).getLiteral();
215                                        }
216
217                                        @Override
218                                        public void remove() {
219                                                throw new UnsupportedOperationException("Do not remove constraint literals.");
220                                        }
221                                };
222                        }
223                };
224        }
225
226        /**
227         * Returns the {@link Operator}.
228         * 
229         * @see #setOperator
230         * @return the operator
231         */
232        public Operator getOperator() {
233                return operator;
234        }
235
236        /**
237         * Sets the {@link Operator}.
238         * 
239         * @see #getOperator
240         * @param operator
241         *            the operator to be set
242         */
243        public void setOperator(Operator operator) {
244                this.operator = operator;
245        }
246
247        /**
248         * Return the right hand side value.
249         * 
250         * @see #setRhs
251         * @return the right hand side value
252         */
253        public int getRhs() {
254                return rhs;
255        }
256
257        /**
258         * Sets the right hand side value.
259         * 
260         * @see #getRhs
261         * @param rhs
262         *            the right hand side value to set
263         */
264        public void setRhs(int rhs) {
265                this.rhs = rhs;
266        }
267
268        /*
269         * (non-Javadoc)
270         * 
271         * @see java.util.AbstractCollection#toString()
272         */
273        @Override
274        public String toString() {
275                String s = "";
276
277                for (int i = 0; i < size(); i++) {
278                        s += get(i);
279                        if (i < size() - 1) {
280                                s += " + ";
281                        }
282                }
283
284                s += " " + operator + " ";
285                s += rhs;
286
287                return s;
288        }
289
290        /**
291         * Checks, if this constraint is satisfied for a given model.
292         * 
293         * @param model
294         *            the model
295         * @return true, iff this constraint is satisfied
296         */
297        public boolean isSatisfied(Model model) {
298                int count = getCount(model);
299                return operator.isTrue(count, rhs);
300
301        }
302
303        /**
304         * Specifies if the {@link Constraint} is satisfied for a given
305         * {@link Model}.
306         * 
307         * @param model
308         *            the model
309         * @return {@code true} if this constraint is satisfied for the model
310         */
311        public int getViolationCount(Model model) {
312                int count = getCount(model);
313
314                switch (operator) {
315                case LE:
316                        return (count <= rhs) ? 0 : (count - rhs);
317                case EQ:
318                        return Math.abs(count - rhs);
319                default: // GE
320                        return (count >= rhs) ? 0 : (rhs - count);
321                }
322        }
323
324        /**
325         * Add a {@link Literal} with the coefficient {@code 1}.
326         * 
327         * @param lit
328         *            the literal
329         */
330        public void add(Literal lit) {
331                add(1, lit);
332        }
333
334        /**
335         * Add a {@link Literal} with specified coefficient.
336         * 
337         * @param coeff
338         *            the coefficient
339         * @param lit
340         *            the literal
341         */
342        public void add(int coeff, Literal lit) {
343                Term term = new Term(coeff, lit);
344                add(term);
345        }
346
347        /**
348         * Returns the sum of the left hand side of the constraint for a given
349         * model.
350         * 
351         * @param model
352         *            the model
353         * @return the sum
354         */
355        private int getCount(Model model) {
356                int count = 0;
357                for (Term term : this) {
358                        Literal literal = term.getLiteral();
359                        int coefficient = term.getCoefficient();
360
361                        Object var = literal.variable();
362                        Boolean phase = literal.phase();
363
364                        if (phase.equals(model.get(var))) {
365                                count += coefficient;
366                        }
367                }
368                return count;
369        }
370
371        /**
372         * Copies the {@link Constraint}.
373         * 
374         * @return a copy of the constraint
375         */
376        public Constraint copy() {
377                Constraint pb = new Constraint(operator, rhs);
378                for (Term term : this) {
379                        pb.add(term.copy());
380                }
381                return pb;
382        }
383
384        /**
385         * Returns {@code true} if the {@link Constraint} contains the
386         * {@link Literal}.
387         * 
388         * @param literal
389         *            the literal
390         * 
391         * @return {@code true} if the constrain} contains the literal
392         */
393        public boolean contains(Literal literal) {
394                for (Term term : this) {
395                        if (literal.equals(term.getLiteral())) {
396                                return true;
397                        }
398                }
399                return false;
400        }
401
402        /*
403         * (non-Javadoc)
404         * 
405         * @see java.util.AbstractList#hashCode()
406         */
407        @Override
408        public int hashCode() {
409                final int prime = 31;
410                int result = super.hashCode();
411                result = prime * result + ((operator == null) ? 0 : operator.hashCode());
412                result = prime * result + rhs;
413                return result;
414        }
415
416        /*
417         * (non-Javadoc)
418         * 
419         * @see java.util.AbstractList#equals(java.lang.Object)
420         */
421        @Override
422        public boolean equals(Object obj) {
423                if (this == obj) {
424                        return true;
425                }
426                if (getClass() != obj.getClass()) {
427                        return false;
428                }
429                Constraint other = (Constraint) obj;
430                return (super.equals(other) && operator == other.operator && rhs == other.rhs);
431        }
