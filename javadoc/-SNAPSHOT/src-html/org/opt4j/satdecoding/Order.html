024package org.opt4j.satdecoding;
025
026/**
027 * The abstract {@link Order} is the base class for a decision strategy.
028 * 
029 * Each time a variable or {@link Literal} is involved in a conflict, the offset
030 * value {@code varInc} is added to its activity. Moreover, each time a conflict
031 * occurs the offset value {@code varInc} is multiplied with the scaling value
032 * {@code varDecay}.
033 * 
034 * @author lukasiewycz
035 * 
036 */
037public abstract class Order {
038
039        protected double varInc = 0;
040
041        protected double varDecay = 1;
042
043        /**
044         * Returns the scaling factor for the offset value.
045         * 
046         * @see #setVarDecay
047         * @return the scaling factor for the offset value
048         */
049        public double getVarDecay() {
050                return varDecay;
051        }
052
053        /**
054         * Sets the scaling factor for the offset value.
055         * 
056         * @see #getVarDecay
057         * @param varDecay
058         *            the scaling factor for the offset value to set
059         */
060        public void setVarDecay(double varDecay) {
061                this.varDecay = varDecay;
062        }
063
064        /**
065         * Returns the initial offset value.
066         * 
067         * @see #setVarInc
068         * @return the initial offset value
069         */
070        public double getVarInc() {
071                return varInc;
072        }
073
074        /**
075         * Sets the initial offset value.
076         * 
077         * @see #getVarInc
078         * @param varInc
079         *            the initial offset value to set
080         */
081        public void setVarInc(double varInc) {
082                this.varInc = varInc;
083        }
084
085}
