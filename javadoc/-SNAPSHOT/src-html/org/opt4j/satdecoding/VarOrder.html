023package org.opt4j.satdecoding;
024
025import java.util.HashMap;
026import java.util.Map;
027import java.util.Map.Entry;
028import java.util.Set;
029
030/**
031 * The {@link VarOrder} implements an {@link Order} like it is used in MiniSAT.
032 * Each variable has an activity and a prioritized decision phase.
033 * 
034 * @author lukasiewycz
035 * 
036 */
037public class VarOrder extends Order {
038
039        protected final Map<Object, Double> activity = new HashMap<>();
040
041        protected final Map<Object, Boolean> phase = new HashMap<>();
042
043        /**
044         * Sets the activity of the specified variable.
045         * 
046         * @see #getActivity
047         * @param variable
048         *            the specific variable
049         * @param value
050         *            the activity value
051         */
052        public void setActivity(Object variable, double value) {
053                activity.put(variable, value);
054        }
055
056        /**
057         * Sets the prioritized phase of the specified variable.
058         * 
059         * @see #getPhase
060         * @param variable
061         *            the specific variable
062         * @param value
063         *            the prioritized phase
064         */
065        public void setPhase(Object variable, boolean value) {
066                phase.put(variable, value);
067        }
068
069        /**
070         * Returns the activity of a variable.
071         * 
072         * @see #setActivity
073         * @param variable
074         *            the variable
075         * @return the activity
076         */
077        public double getActivity(Object variable) {
078                return activity.get(variable);
079        }
080
081        /**
082         * Returns the prioritized phase of a variable.
083         * 
084         * @see #setPhase
085         * @param variable
086         *            the variable
087         * @return the prioritized phase
088         */
089        public boolean getPhase(Object variable) {
090                return phase.get(variable);
091        }
092
093        /**
094         * Returns the entry set of the activities.
095         * 
096         * @return the entry set of the activities
097         */
098        public Set<Entry<Object, Double>> getActivityEntrySet() {
099                return activity.entrySet();
100        }
101
102        /**
103         * Returns the entry set of the phases.
104         * 
105         * @return the entry set of the phases
106         */
107        public Set<Entry<Object, Boolean>> getPhaseEntrySet() {
108                return phase.entrySet();
109        }
110
111}
