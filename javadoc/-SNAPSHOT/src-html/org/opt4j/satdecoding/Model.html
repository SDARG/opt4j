023package org.opt4j.satdecoding;
024
025import java.util.HashMap;
026import java.util.Map;
027import java.util.Map.Entry;
028import java.util.Set;
029
030/**
031 * The {@link Model} is a data structure that represents a solution of the given
032 * problem, i.e., all {@link Literal}s are set such that all {@link Constraint}s
033 * specified by the {@link AbstractSATDecoder} are feasible.
034 * 
035 * @author lukasiewycz
036 * 
037 */
038public class Model {
039
040        protected final Map<Object, Boolean> map = new HashMap<>();
041
042        /**
043         * Returns the value of the variable, or null if this variable is not
044         * defined.
045         * 
046         * @param var
047         *            the variable
048         * @return {@code true} if the variable in this model is true or
049         *         {@code null} if it is not defined
050         */
051        public Boolean get(Object var) {
052                return map.get(var);
053        }
054
055        /**
056         * Sets the variable to a phase.
057         * 
058         * @param var
059         *            the variable
060         * @param phase
061         *            the phase
062         */
063        public void set(Object var, boolean phase) {
064                map.put(var, phase);
065        }
066
067        /**
068         * Returns all variables of the model.
069         * 
070         * @return all variables
071         */
072        public Set<Object> getVars() {
073                return map.keySet();
074        }
075
076        /**
077         * Returns the entry set: The entry pairs of all variables to the boolean
078         * values.
079         * 
080         * @return the entry set
081         */
082        public Set<Entry<Object, Boolean>> pairs() {
083                return map.entrySet();
084        }
085
086}
