023package org.opt4j.satdecoding;
024
025import java.util.List;
026import java.util.Map;
027
028import org.opt4j.core.Genotype;
029import org.opt4j.core.genotype.BooleanGenotype;
030import org.opt4j.core.genotype.BooleanMapGenotype;
031import org.opt4j.core.genotype.DoubleBounds;
032import org.opt4j.core.genotype.DoubleGenotype;
033import org.opt4j.core.genotype.DoubleMapGenotype;
034
035import com.google.inject.Inject;
036import com.google.inject.Singleton;
037
038/**
039 * The {@link MixedSATManager} encodes the decision strategy into two vectors:
040 * One binary vector for the phase and one double vector for the priority.
041 * 
042 * @author lukasiewycz
043 * 
044 */
045@Singleton
046public class MixedSATManager implements SATManager {
047
048        protected final Solver solver;
049
050        /**
051         * Constructs a {@link MixedSATManager}.
052         * 
053         * @param solver
054         *            the solver
055         */
056        @Inject
057        public MixedSATManager(Solver solver) {
058                this.solver = solver;
059        }
060
061        /*
062         * (non-Javadoc)
063         * 
064         * @see org.opt4j.sat.SATManager#createSATGenotype(java.util.List,
065         * java.util.Map, java.util.Map)
066         */
067        @Override
068        public Genotype createSATGenotype(List<Object> variables, Map<Object, Double> lowerBounds,
069                        Map<Object, Double> upperBounds, Map<Object, Double> priorities, Map<Object, Boolean> phases) {
070
071                int size = variables.size();
072
073                double[] lower = new double[size];
074                double[] upper = new double[size];
075
076                for (int i = 0; i < size; i++) {
077                        Object variable = variables.get(i);
078                        Double lb = lowerBounds.get(variable);
079                        Double ub = upperBounds.get(variable);
080
081                        if (lb == null) {
082                                lb = 0.0;
083                        }
084                        if (ub == null) {
085                                ub = 1.0;
086                        }
087                        lower[i] = lb;
088                        upper[i] = ub;
089                }
090
091                DoubleBounds doubleBounds = new DoubleBounds(lower, upper);
092                DoubleGenotype doubleVector = new DoubleMapGenotype<>(variables, doubleBounds);
093                BooleanGenotype booleanVector = new BooleanMapGenotype<>(variables);
094
095                SATGenotype satGenotype = new SATGenotype(booleanVector, doubleVector);
096
097                for (Object variable : variables) {
098                        double priority = priorities.get(variable);
099                        boolean phase = phases.get(variable);
100
101                        doubleVector.add(priority);
102                        booleanVector.add(phase);
103                }
104
105                return satGenotype;
106        }
107
108        /*
109         * (non-Javadoc)
110         * 
111         * @see org.opt4j.sat.SATManager#decodeSATGenotype(java.util.List,
112         * org.opt4j.core.Genotype)
113         */
114        @Override
115        public Model decodeSATGenotype(List<Object> variables, Genotype genotype) {
116                SATGenotype satGenotype = (SATGenotype) genotype;
117
118                BooleanGenotype booleanVector = satGenotype.getBooleanVector();
119                DoubleGenotype doubleVector = satGenotype.getDoubleVector();
120
121                VarOrder varorder = new VarOrder();
122
123                for (int i = 0; i < variables.size(); i++) {
124                        Object var = variables.get(i);
125                        varorder.setActivity(var, doubleVector.get(i));
126                        varorder.setPhase(var, booleanVector.get(i));
127                }
128                varorder.setVarInc(1.0 / (2.0 * variables.size()));
129                varorder.setVarDecay(1.0 / 0.95);
130
131                Model model = null;
132                try {
133                        model = solver.solve(varorder);
134                        if (model == null) {
135                                throw new ContradictionException("no satisfying solution left");
136                        }
137                } catch (TimeoutException e) {
138                        System.err.println("timeout");
139                }
140                return model;
141        }
142
143        /*
144         * (non-Javadoc)
145         * 
146         * @see org.opt4j.sat.SATManager#getSolver()
147         */
148        @Override
149        public Solver getSolver() {
150                return solver;
151        }
152
153}
