023package org.opt4j.satdecoding;
024
025import java.util.ArrayList;
026import java.util.Collection;
027import java.util.HashMap;
028import java.util.HashSet;
029import java.util.List;
030import java.util.Map;
031import java.util.Random;
032import java.util.Set;
033
034import org.opt4j.core.Genotype;
035import org.opt4j.core.common.random.Rand;
036import org.opt4j.core.optimizer.Optimizer;
037import org.opt4j.core.problem.Creator;
038import org.opt4j.core.problem.Decoder;
039
040import com.google.inject.Inject;
041
042/**
043 * The {@link AbstractSATDecoder} can be used for the construction of custom SAT
044 * {@link Decoder}s.
045 * <p>
046 * The {@link Creator} part takes the problem dependent {@link Constraint}s and
047 * creates a {@link SATGenotype} according to the variables which is the input
048 * for the specified {@link Optimizer}.
049 * <p>
050 * The {@link Decoder} takes a {@link SATGenotype}, converts it to an
051 * {@link Order} of the variables, and solves the problem in the {@link Order}
052 * using a SAT {@link Solver} . The resulting {@link Model}, which respects the
053 * {@link Constraint}s, must then be converted to the problem specific
054 * phenotype.
055 * 
056 * @author lukasiewycz
057 * 
058 * @param <G>
059 *            The genotype (use {@link Genotype} if you do not override the
060 *            decode method)
061 * @param <P>
062 *            The phenotype
063 */
064public abstract class AbstractSATDecoder<G extends Genotype, P extends Object> implements Decoder<G, P>, Creator<G> {
065
066        private final List<Constraint> constraints = new ArrayList<>();
067        private final List<Object> variables = new ArrayList<>();
068
069        protected Map<Object, Double> lowerBounds;
070
071        protected Map<Object, Double> upperBounds;
072
073        protected final Random random;
074
075        protected final SATManager manager;
076
077        protected boolean isInit = false;
078
079        /**
080         * Constructs an {@link AbstractSATDecoder} with the given
081         * {@link SATManager}.
082         * 
083         * @param manager
084         *            the specified sat manager
085         * @param random
086         *            the random number generator
087         */
088        @Inject
089        public AbstractSATDecoder(SATManager manager, Rand random) {
090                this.random = random;
091                this.manager = manager;
092        }
093
094        /**
095         * Initializes the {@link Constraint}s and variables.
096         */
097        protected synchronized void init() {
098
099                if (!isInit) {
100                        try {
101                                Solver solver = manager.getSolver();
102
103                                Set<Constraint> constraints = createConstraints();
104
105                                Set<Object> variables = new HashSet<>();
106
107                                for (Constraint constraint : constraints) {
108                                        solver.addConstraint(constraint);
109                                        this.constraints.add(constraint);
110
111                                        for (Literal literal : constraint.getLiterals()) {
112                                                variables.add(literal.variable());
113                                        }
114                                }
115
116                                Set<Object> ignores = ignoreVariables(variables);
117
118                                variables.removeAll(ignores);
119                                this.variables.addAll(variables);
120
121                                lowerBounds = getLowerBounds(variables);
122                                upperBounds = getUpperBounds(variables);
123
124                        } catch (Throwable e) {
125                                System.err.println("Failed initialization of " + getClass().getName() + " with " + e);
126                                throw new IllegalArgumentException(e);
127                        } finally {
128                                isInit = true;
129                        }
130
131                }
132        }
133
134        /**
135         * Creates all {@link Constraint}s.
136         * <p>
137         * Override this method to return the problem specific {@link Constraint}s
138         * that have to be respected by the resulting {@link Model}.
139         * 
140         * @return the set of all constraints of the given problem
141         * 
142         */
143        public abstract Set<Constraint> createConstraints();
144
145        /**
146         * Creates a random initialization of the priorities and phases for the
147         * variables. Override this method for a custom initialization.
148         * 
149         * @param variables
150         *            all variables
151         * @param lowerBounds
152         *            the lower bounds for the priority for the variables
153         * @param upperBounds
154         *            the upper bounds for the priority for the variables
155         * @param priorities
156         *            the priorities of the variables
157         * @param phases
158         *            the phases of the variables
159         */
160        public void randomize(Collection<Object> variables, Map<Object, Double> lowerBounds,
161                        Map<Object, Double> upperBounds, Map<Object, Double> priorities, Map<Object, Boolean> phases) {
162                for (Object variable : variables) {
163                        Double lb = lowerBounds.get(variable);
164                        Double ub = upperBounds.get(variable);
165                        if (lb == null) {
166                                lb = 0.0;
167                        }
168                        if (ub == null) {
169                                ub = 1.0;
170                        }
171
172                        double priority = lb + (ub - lb) * random.nextDouble();
173
174                        priorities.put(variable, priority);
175                        phases.put(variable, random.nextBoolean());
176                }
177        }
178
179        /**
180         * Set the variables to be ignored (not part of the search process and
181         * genotype). Override this method if variables are ignored.
182         * 
183         * @param variables
184         *            all variables of the problem
185         * @return the variables to be ignored
186         */
187        public Set<Object> ignoreVariables(Set<Object> variables) {
188                return new HashSet<>();
189        }
190
191        /**
192         * Returns the lower bounds for the variables.
193         * 
194         * @param variables
195         *            all variables of the problem
196         * @return the lower bounds
197         */
198        public Map<Object, Double> getLowerBounds(Set<Object> variables) {
199                return new HashMap<>();
200        }
201
202        /**
203         * Returns the upper bounds for the variables.
204         * 
205         * @param variables
206         *            all variables of the problem
207         * @return the upper bounds
208         */
209        public Map<Object, Double> getUpperBounds(Set<Object> variables) {
210                return new HashMap<>();
211        }
212
213        /**
214         * Converts a {@link Model} into a phenotype. Override this with your custom
215         * method.
216         * 
217         * @param model
218         *            the found model
219         * @return the converted phenotype
220         */
221        public abstract P convertModel(Model model);
222
223        /*
224         * (non-Javadoc)
225         * 
226         * @see org.opt4j.core.problem.Creator#create()
227         */
228        @Override
229        @SuppressWarnings("unchecked")
230        public G create() {
231                return (G) createSATGenotype();
232        }
233
234        /**
235         * Returns a random SAT genotype of the constrained problem.
236         * 
237         * @return a random SAT genotype
238         */
239        protected Genotype createSATGenotype() {
240                if (!isInit) {
241                        init();
242                }
243
244                Map<Object, Double> priorities = new HashMap<>();
245                Map<Object, Boolean> phases = new HashMap<>();
246
247                randomize(variables, lowerBounds, upperBounds, priorities, phases);
248
249                return manager.createSATGenotype(variables, lowerBounds, upperBounds, priorities, phases);
250        }
251
252        /*
253         * (non-Javadoc)
254         * 
255         * @see org.opt4j.core.problem.Decoder#decode(org.opt4j.core.Genotype)
256         */
257        @Override
258        public P decode(G genotype) {
259                Model model = decodeSATGenotype(genotype);
260                assert model != null;
261
262                return convertModel(model);
263        }
264
265        /**
266         * Decodes the {@link Genotype} to a phenotype by using a SAT/PB solver.
267         * 
268         * @param genotype
269         *            the genotype
270         * @return the phenotype
271         */
272        protected Model decodeSATGenotype(Genotype genotype) {
273                if (!isInit) {
274                        init();
275                }
276
277                return manager.decodeSATGenotype(variables, genotype);
278        }
279
280        /**
281         * Returns the constraints.
282         * 
283         * @return the constraints
284         */
285        public List<Constraint> getConstraints() {
286                if (!isInit) {
287                        init();
288                }
289                return constraints;
290        }
291
292        /**
293         * Returns the variables.
294         * 
295         * @return the variables
296         */
297        public List<Object> getVariables() {
298                if (!isInit) {
299                        init();
300                }
301                return variables;
302        }
303
