024package org.opt4j.satdecoding.sat4j;
025
026import static org.sat4j.core.LiteralsUtils.negLit;
027import static org.sat4j.core.LiteralsUtils.posLit;
028
029import org.sat4j.minisat.core.Heap;
030import org.sat4j.minisat.core.ILits;
031import org.sat4j.minisat.orders.UserFixedPhaseSelectionStrategy;
032import org.sat4j.minisat.orders.VarOrderHeap;
033
034/**
035 * A {@link VariableOrder} implementation for the SAT4J interface. <br>
036 * (not documented, see <a href="http://www.sat4j.org">Sat4J.org</a>)
037 * 
038 * @author lukasiewycz
039 * 
040 */
041public class VariableOrder extends VarOrderHeap {
042
043        protected static final double VAR_RESCALE_FACTOR = 1e-100;
044
045        protected static final double VAR_RESCALE_BOUND = 1 / VAR_RESCALE_FACTOR;
046
047        private static final long serialVersionUID = 1L;
048
049        protected double varInc = 0;
050
051        protected boolean[] phase;
052
053        /**
054         * Constructs a {@link VariableOrder}.
055         */
056        public VariableOrder() {
057                super(new UserFixedPhaseSelectionStrategy());
058        }
059
060        /*
061         * (non-Javadoc)
062         * 
063         * @see
064         * org.sat4j.minisat.orders.VarOrderHeap#setLits(org.sat4j.minisat.core.
065         * ILits)
066         */
067        @Override
068        public void setLits(ILits lits) {
069                this.lits = lits;
070
071                int nlength = lits.nVars() + 1;
072                activity = new double[nlength];
073                phase = new boolean[nlength];
074                activity[0] = -1;
075                heap = new Heap(activity);
076                heap.setBounds(nlength);
077                for (int i = 1; i < nlength; i++) {
078                        assert i > 0;
079                        assert i <= lits.nVars() : lits.nVars() + "/" + i;
080                        activity[i] = 0.0;
081                        if (lits.belongsToPool(i)) {
082                                heap.insert(i);
083                                phase[i] = false;
084                        } else {
085                                phase[i] = false;
086                        }
087                }
088
089        }
090
091        /*
092         * (non-Javadoc)
093         * 
094         * @see org.sat4j.minisat.orders.VarOrderHeap#init()
095         */
096        @Override
097        public void init() {
098                // nothing to be done
099        }
100
101        /*
102         * (non-Javadoc)
103         * 
104         * @see org.sat4j.minisat.orders.VarOrderHeap#updateActivity(int)
105         */
106        @Override
107        protected void updateActivity(final int var) {
108                if ((activity[var] += varInc) > VAR_RESCALE_BOUND) {
109                        varRescaleActivity();
110                }
111        }
112
113        /**
114         * Increments the activity of a variable {@code var} by the specified
115         * {@code value}.
116         * 
117         * @param var
118         *            the variable
119         * @param value
120         *            the value to increment the variable
121         */
122        protected void updateActivity(final int var, final double value) {
123                if ((activity[var] += value) > VAR_RESCALE_BOUND) {
124                        varRescaleActivity();
125                }
126        }
127
128        /*
129         * (non-Javadoc)
130         * 
131         * @see org.sat4j.minisat.orders.VarOrderHeap#updateVar(int)
132         */
133        @Override
134        public void updateVar(int p) {
135                int var = p >> 1;
136                updateActivity(var);
137                if (heap.inHeap(var)) {
138                        heap.increase(var);
139                }
140        }
141
142        /**
143         * Sets the value {@code varInc} to increase the activity of the variables.
144         * 
145         * @param value
146         *            the value to increase the activity of the variables
147         */
148        public void setVarInc(double value) {
149                varInc = value;
150        }
151
152        /**
153         * Rescales the activities of the variables.
154         */
155        protected void varRescaleActivity() {
156                for (int i = 1; i < activity.length; i++) {
157                        activity[i] *= VAR_RESCALE_FACTOR;
158                }
159                varInc *= VAR_RESCALE_FACTOR;
160        }
161
162        /**
163         * Sets the activity of a variable {@code var} to the specified
164         * {@code value}.
165         * 
166         * @param var
167         *            the variable
168         * @param value
169         *            the activity to set
170         */
171        public void setVarActivity(int var, double value) {
172                updateActivity(var, value);
173                if (heap.inHeap(var)) {
174                        heap.increase(var);
175                }
176        }
177
178        /**
179         * Sets the {@code phase} of a variable {@code var}.
180         * 
181         * @param var
182         *            the variable
183         * @param phase
184         *            the phase
185         */
186        public void setVarPhase(int var, boolean phase) {
187                this.phase[var] = phase;
188        }
189
190        /*
191         * (non-Javadoc)
192         * 
193         * @see org.sat4j.minisat.orders.VarOrderHeap#select()
194         */
195        @Override
196        public int select() {
197                while (!heap.empty()) {
198                        int var = heap.getmin();
199                        int next;
200                        if (phase[var]) {
201                                next = posLit(var);
202                        } else {
203                                next = negLit(var);
204                        }
205                        if (lits.isUnassigned(next)) {
206                                return next;
207                        }
208                }
209                return ILits.UNDEFINED;
210        }
211}
