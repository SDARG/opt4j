024package org.opt4j.benchmarks.lotz;
025
026import org.opt4j.core.config.annotations.Info;
027import org.opt4j.core.problem.ProblemModule;
028import org.opt4j.core.start.Constant;
029
030/**
031 * <p>
032 * The {@link LOTZModule} for the "Leading Ones Trailing Zeros" optimization
033 * problem. This is a 2-dimensional optimization problem:
034 * </p>
035 * 
036 * <p>
037 * <ol>
038 * <li>maximize the consecutive ones from the beginning of a binary string</li>
039 * <li>maximize the consecutive ones from the end of a binary string</li>
040 * </ol>
041 * </p>
042 * 
043 * @author lukasiewycz
044 * 
045 */
046@Info("The 2-dimensional \"Leading Ones Trailing Zeros\" optimization problem.")
047public class LOTZModule extends ProblemModule {
048
049        @Info("The length of the binary string.")
050        @Constant(value = "size", namespace = LOTZCreator.class)
051        protected int size = 30;
052
053        /**
054         * Returns the size of the binary string.
055         * 
056         * @return the size of the binary string
057         */
058        public int getSize() {
059                return size;
060        }
061
062        /**
063         * Sets the size of the binary string.
064         * 
065         * @param size
066         *            the size of the binary string to set
067         */
068        public void setSize(int size) {
069                this.size = size;
070        }
071
072        /*
073         * (non-Javadoc)
074         * 
075         * @see org.opt4j.start.Opt4JModule#config()
076         */
077        @Override
078        public void config() {
079                bindProblem(LOTZCreator.class, LOTZDecoder.class, LOTZEvaluator.class);
080        }
081
082}
