023package org.opt4j.benchmarks.wfg;
024
025import static java.lang.Math.max;
026
027import java.util.ArrayList;
028import java.util.List;
029
030import org.opt4j.benchmarks.DoubleString;
031import org.opt4j.benchmarks.K;
032import org.opt4j.benchmarks.M;
033import org.opt4j.core.Objective;
034import org.opt4j.core.Objectives;
035import org.opt4j.core.problem.Evaluator;
036
037import com.google.inject.Inject;
038
039/**
040 * The {@link WFGEvaluator}. All WFG test functions are derived from this class.
041 * 
042 * @author lukasiewycz
043 * 
044 */
045public abstract class WFGEvaluator implements Evaluator<DoubleString> {
046
047        protected final int M;
048        protected final int k;
049
050        protected final List<Objective> keys = new ArrayList<>();
051
052        /**
053         * Constructs a {@link WFGEvaluator}.
054         * 
055         * @param k
056         *            the position parameters
057         * @param M
058         *            the number of objectives
059         */
060        @Inject
061        public WFGEvaluator(@K int k, @M int M) {
062                this.M = M;
063                this.k = k;
064
065                for (int i = 0; i < M; i++) {
066                        keys.add(new Objective("f" + (i + 1)));
067                }
068        }
069
070        public abstract List<Double> f(List<Double> y);
071
072        /*
073         * (non-Javadoc)
074         * 
075         * @see org.opt4j.core.problem.Evaluator#evaluate(org.opt4j.core.Phenotype)
076         */
077        @Override
078        public Objectives evaluate(DoubleString phenotype) {
079                List<Double> f = f(phenotype);
080
081                assert (f.size() == keys.size());
082
083                Objectives objectives = new Objectives();
084                for (int i = 0; i < M; i++) {
085                        objectives.add(keys.get(i), f.get(i));
086                }
087                return objectives;
088        }
089
090        protected static List<Boolean> createA(final int M, final boolean degenerate) {
091                assert (M >= 2);
092
093                List<Boolean> A = new ArrayList<>();
094                for (int i = 0; i < M - 1; i++) {
095                        A.add(!degenerate || i == 0);
096                }
097                return A;
098        }
099
100        protected static List<Double> calculateF(final List<Double> x, final List<Double> h) {
101
102                assert (x.size() == h.size());
103
104                final int M = h.size();
105
106                List<Double> S = new ArrayList<>();
107                for (int m = 1; m <= M; m++) {
108                        S.add(m * 2.0);
109                }
110
111                return calculateF(x, h, S);
112        }
113
114        protected static List<Double> normalizeZ(final List<Double> z, final List<Double> z_max) {
115                final List<Double> result = new ArrayList<>();
116
117                for (int i = 0; i < z.size(); i++) {
118                        assert (z.get(i) >= 0.0);
119                        assert (z.get(i) <= z_max.get(i));
120                        assert (z_max.get(i) > 0.0);
121
122                        result.add(z.get(i) / z_max.get(i));
123                }
124
125                return result;
126        }
127
128        protected static List<Double> calculateX(final List<Double> t_p, final List<Boolean> A) {
129                final int size = t_p.size();
130
131                assert (size != 0);
132                assert (A.size() == t_p.size() - 1);
133
134                final List<Double> result = new ArrayList<>();
135
136                for (int i = 0; i < t_p.size() - 1; i++) {
137                        int A_i = A.get(i) ? 1 : 0;
138
139                        final double tmp1 = max(t_p.get(size - 1), A_i);
140                        result.add(tmp1 * (t_p.get(i) - 0.5) + 0.5);
141                }
142
143                result.add(t_p.get(size - 1));
144
145                return result;
146        }
147
148        protected static List<Double> calculateF(final List<Double> x, final List<Double> h, final List<Double> S) {
149                assert (x.size() == h.size());
150                assert (h.size() == S.size());
151
152                final List<Double> result = new ArrayList<>();
153
154                for (int i = 0; i < h.size(); i++) {
155                        assert (S.get(i) > 0.0);
156
157                        result.add(x.get(x.size() - 1) + S.get(i) * h.get(i));
158                }
159
160                return result;
161        }
162
163}
