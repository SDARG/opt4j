023package org.opt4j.benchmarks.wfg;
024
025import java.util.ArrayList;
026import java.util.List;
027
028import org.opt4j.benchmarks.K;
029import org.opt4j.benchmarks.M;
030
031import com.google.inject.Inject;
032
033/**
034 * The {@link WFG1} benchmark function.
035 * 
036 * @author lukasiewycz
037 * 
038 */
039public class WFG1 extends WFGEvaluator {
040
041        /**
042         * Constructs a {@link WFG1} benchmark function.
043         * 
044         * @param k
045         *            the position parameters
046         * @param M
047         *            the number of objectives
048         */
049        @Inject
050        public WFG1(@K int k, @M int M) {
051                super(k, M);
052        }
053
054        protected static List<Double> t1(final List<Double> y, final int k) {
055
056                final int n = y.size();
057
058                assert (k >= 1);
059                assert (k < n);
060
061                List<Double> t = new ArrayList<>();
062
063                for (int i = 0; i < k; i++) {
064                        t.add(y.get(i));
065                }
066
067                for (int i = k; i < n; i++) {
068                        t.add(WFGTransFunctions.sLinear(y.get(i), 0.35));
069                }
070
071                return t;
072        }
073
074        protected static List<Double> t2(final List<Double> y, final int k) {
075                final int n = y.size();
076
077                assert (k >= 1);
078                assert (k < n);
079
080                List<Double> t = new ArrayList<>();
081
082                for (int i = 0; i < k; i++) {
083                        t.add(y.get(i));
084                }
085
086                for (int i = k; i < n; i++) {
087                        t.add(WFGTransFunctions.bFlat(y.get(i), 0.8, 0.75, 0.85));
088                }
089
090                return t;
091        }
092
093        protected static List<Double> t3(final List<Double> y) {
094                final int n = y.size();
095                List<Double> t = new ArrayList<>();
096
097                for (int i = 0; i < n; i++) {
098                        t.add(WFGTransFunctions.bPoly(y.get(i), 0.02));
099                }
100
101                return t;
102        }
103
104        protected static List<Double> t4(final List<Double> y, final int k, final int M) {
105                final int n = y.size();
106
107                assert (k >= 1);
108                assert (k < n);
109                assert (M >= 2);
110                assert (k % (M - 1) == 0);
111
112                List<Double> w = new ArrayList<>();
113
114                for (int i = 1; i <= n; i++) {
115                        w.add(2.0 * i);
116                }
117
118                List<Double> t = new ArrayList<>();
119
120                for (int i = 1; i <= M - 1; i++) {
121                        final int head = (i - 1) * k / (M - 1);
122                        final int tail = i * k / (M - 1);
123
124                        final List<Double> y_sub = y.subList(head, tail);
125                        final List<Double> w_sub = w.subList(head, tail);
126
127                        t.add(WFGTransFunctions.rSum(y_sub, w_sub));
128                }
129
130                final List<Double> y_sub = y.subList(k, n);
131                final List<Double> w_sub = w.subList(k, n);
132
133                t.add(WFGTransFunctions.rSum(y_sub, w_sub));
134
135                return t;
136        }
137
138        protected static List<Double> shape(final List<Double> t_p) {
139
140                assert (t_p.size() >= 2);
141
142                final int M = t_p.size();
143
144                final List<Boolean> A = createA(M, false);
145                final List<Double> x = calculateX(t_p, A);
146
147                List<Double> h = new ArrayList<>();
148
149                for (int m = 1; m <= M - 1; m++) {
150                        h.add(WFGShapeFunctions.convex(x, m));
151                }
152                h.add(WFGShapeFunctions.mixed(x, 5, 1.0));
153
154                return calculateF(x, h);
155        }
156
157        /*
158         * (non-Javadoc)
159         * 
160         * @see org.opt4j.benchmark.wfg.WFGEvaluator#f(java.util.List)
161         */
162        @Override
163        public List<Double> f(List<Double> y) {
164                y = WFG1.t1(y, k);
165                y = WFG1.t2(y, k);
166                y = WFG1.t3(y);
167                y = WFG1.t4(y, k, M);
168
169                return WFG1.shape(y);
170        }
171}
