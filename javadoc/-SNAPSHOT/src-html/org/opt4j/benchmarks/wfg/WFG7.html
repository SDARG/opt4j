023package org.opt4j.benchmarks.wfg;
024
025import java.util.ArrayList;
026import java.util.List;
027
028import org.opt4j.benchmarks.K;
029import org.opt4j.benchmarks.M;
030
031import com.google.inject.Inject;
032
033/**
034 * The {@link WFG7} benchmark function.
035 * 
036 * @author lukasiewycz
037 * 
038 */
039public class WFG7 extends WFGEvaluator {
040
041        /**
042         * Constructs a {@link WFG7} benchmark function.
043         * 
044         * @param k
045         *            the position parameters
046         * @param M
047         *            the number of objectives
048         */
049        @Inject
050        public WFG7(@K int k, @M int M) {
051                super(k, M);
052        }
053
054        public static List<Double> t1(final List<Double> y, final int k) {
055                final int n = y.size();
056
057                assert (k >= 1);
058                assert (k < n);
059
060                List<Double> w = new ArrayList<>();
061                for (int i = 0; i < n; i++) {
062                        w.add(1.0);
063                }
064
065                List<Double> t = new ArrayList<>();
066
067                for (int i = 0; i < k; i++) {
068                        final List<Double> y_sub = y.subList(i + 1, n);
069                        final List<Double> w_sub = w.subList(i + 1, n);
070
071                        final double u = WFGTransFunctions.rSum(y_sub, w_sub);
072
073                        t.add(WFGTransFunctions.bParam(y.get(i), u, 0.98 / 49.98, 0.02, 50));
074                }
075
076                for (int i = k; i < n; i++) {
077                        t.add(y.get(i));
078                }
079
080                return t;
081        }
082
083        /*
084         * (non-Javadoc)
085         * 
086         * @see org.opt4j.benchmark.wfg.WFGEvaluator#f(java.util.List)
087         */
088        @Override
089        public List<Double> f(List<Double> y) {
090                y = WFG7.t1(y, k);
091                y = WFG1.t1(y, k);
092                y = WFG2.t3(y, k, M);
093
094                return WFG4.shape(y);
095        }
096}
