023package org.opt4j.benchmarks.wfg;
024
025import java.util.ArrayList;
026import java.util.List;
027
028import org.opt4j.benchmarks.K;
029import org.opt4j.benchmarks.M;
030
031import com.google.inject.Inject;
032
033/**
034 * The {@link WFG2} benchmark function.
035 * 
036 * @author lukasiewycz
037 * 
038 */
039public class WFG2 extends WFGEvaluator {
040
041        /**
042         * Constructs a {@link WFG2} benchmark function.
043         * 
044         * @param k
045         *            the position parameters
046         * @param M
047         *            the number of objectives
048         */
049        @Inject
050        public WFG2(@K int k, @M int M) {
051                super(k, M);
052        }
053
054        protected static List<Double> t2(final List<Double> y, final int k) {
055                final int n = y.size();
056                final int l = n - k;
057
058                assert (k >= 1);
059                assert (k < n);
060                assert (l % 2 == 0);
061
062                List<Double> t = new ArrayList<>();
063
064                for (int i = 0; i < k; i++) {
065                        t.add(y.get(i));
066                }
067
068                for (int i = k + 1; i <= k + l / 2; i++) {
069                        final int head = k + 2 * (i - k) - 2;
070                        final int tail = k + 2 * (i - k);
071
072                        t.add(WFGTransFunctions.rNonsep(y.subList(head, tail), 2));
073                }
074
075                return t;
076        }
077
078        protected static List<Double> t3(final List<Double> y, final int k, final int M) {
079                final int n = y.size();
080
081                assert (k >= 1);
082                assert (k < n);
083                assert (M >= 2);
084                assert (k % (M - 1) == 0);
085
086                List<Double> w = new ArrayList<>();
087                for (int i = 0; i < n; i++) {
088                        w.add(1.0);
089                }
090
091                List<Double> t = new ArrayList<>();
092
093                for (int i = 1; i <= M - 1; i++) {
094                        final int head = (i - 1) * k / (M - 1);
095                        final int tail = i * k / (M - 1);
096
097                        final List<Double> y_sub = y.subList(head, tail);
098                        final List<Double> w_sub = w.subList(head, tail);
099
100                        t.add(WFGTransFunctions.rSum(y_sub, w_sub));
101                }
102
103                final List<Double> y_sub = y.subList(k, n);
104                final List<Double> w_sub = w.subList(k, n);
105
106                t.add(WFGTransFunctions.rSum(y_sub, w_sub));
107
108                return t;
109        }
110
111        protected static List<Double> shape(final List<Double> t_p) {
112
113                assert (t_p.size() >= 2);
114
115                final int M = t_p.size();
116
117                final List<Boolean> A = createA(M, false);
118                final List<Double> x = calculateX(t_p, A);
119
120                List<Double> h = new ArrayList<>();
121
122                for (int m = 1; m <= M - 1; m++) {
123                        h.add(WFGShapeFunctions.convex(x, m));
124                }
125                h.add(WFGShapeFunctions.disc(x, 5, 1.0, 1.0));
126
127                return calculateF(x, h);
128        }
129
130        /*
131         * (non-Javadoc)
132         * 
133         * @see org.opt4j.benchmark.wfg.WFGEvaluator#f(java.util.List)
134         */
135        @Override
136        public List<Double> f(List<Double> y) {
137                y = WFG1.t1(y, k);
138                y = WFG2.t2(y, k);
139                y = WFG2.t3(y, k, M);
140
141                return WFG2.shape(y);
142        }
143}
