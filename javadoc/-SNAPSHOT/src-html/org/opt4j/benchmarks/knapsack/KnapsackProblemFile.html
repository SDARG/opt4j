023package org.opt4j.benchmarks.knapsack;
024
025import java.io.BufferedReader;
026import java.io.FileNotFoundException;
027import java.io.FileReader;
028import java.io.IOException;
029import java.util.ArrayList;
030import java.util.Collection;
031import java.util.List;
032import java.util.regex.Matcher;
033import java.util.regex.Pattern;
034
035import org.opt4j.core.start.Constant;
036
037import com.google.inject.Inject;
038
039/**
040 * 
041 * The {@link KnapsackProblemFile} is a {@link KnapsackProblem} that is
042 * initialized by a file. The format can be found on the referenced website.
043 * 
044 * @see <a href=
045 *      "http://www.tik.ee.ethz.ch/sop/download/supplementary/testProblemSuite/">http://www.tik.ee.ethz.ch/sop/download/supplementary/testProblemSuite/</a>
046 * 
047 * @author lukasiewycz
048 * 
049 */
050public class KnapsackProblemFile implements KnapsackProblem {
051
052        protected static final String INT = "[+-]?\\d+";
053        protected static final String DOUBLE = "[+-]?\\d+(\\.\\d*)?";
054        protected static final String BLANK = "[\\p{Blank}]*";
055        protected static final String POSINT = "[-]?\\d+"; // skip +
056        protected static final String POSDOUBLE = "[-]?\\d+(\\.\\d*)?"; // skip +
057
058        protected final List<Knapsack> knapsacks = new ArrayList<>();
059        protected final List<Item> items = new ArrayList<>();
060
061        /**
062         * Constructs a {@link KnapsackProblemFile} with a given file name.
063         * 
064         * @param filename
065         *            the file name
066         */
067        @Inject
068        public KnapsackProblemFile(@Constant(value = "filename", namespace = KnapsackProblemFile.class) String filename) {
069                try (BufferedReader reader = new BufferedReader(new FileReader(filename))) {
070                        String line;
071
072                        Knapsack knapsack = null;
073                        Item item = null;
074
075                        while ((line = reader.readLine()) != null) {
076
077                                if (line.matches(BLANK + "knapsack" + BLANK + INT + BLANK + ":" + BLANK)) {
078                                        knapsack = new Knapsack(getInt(line));
079                                        knapsacks.add(knapsack);
080                                } else if (knapsack == null) {
081                                        throw new IllegalStateException("knapsack not initialized." + filename);
082                                } else if (line.matches(BLANK + "capacity:" + BLANK + DOUBLE + BLANK)) {
083                                        double capacity = getDouble(line);
084                                        knapsack.setCapacity(capacity);
085                                } else if (line.matches(BLANK + "item" + BLANK + INT + BLANK + ":" + BLANK)) {
086                                        int i = getInt(line);
087                                        if (items.size() < i) {
088                                                item = new Item(getInt(line));
089                                                items.add(i - 1, item);
090                                        } else {
091                                                item = items.get(i - 1);
092                                        }
093                                } else if (line.matches(BLANK + "weight:" + BLANK + DOUBLE + BLANK)) {
094                                        int weight = getInt(line);
095                                        knapsack.setWeight(item, weight);
096                                } else if (line.matches(BLANK + "profit:" + BLANK + DOUBLE + BLANK)) {
097                                        int profit = getInt(line);
098                                        knapsack.setProfit(item, profit);
099                                }
100                        }
101                } catch (FileNotFoundException e1) {
102                        throw new IllegalArgumentException("Knapsack problem file not found " + filename, e1);
103                } catch (IOException e) {
104                        throw new IllegalArgumentException(e);
105                }
106        }
107
108        /**
109         * Get the integer value in the line.
110         * 
111         * @param line
112         *            the line
113         * @return the integer value
114         */
115        protected int getInt(String line) {
116                Matcher matcher = Pattern.compile(POSINT).matcher(line);
117                matcher.find();
118                return Integer.parseInt(matcher.group());
119        }
120
121        /**
122         * Get the double value in the line.
123         * 
124         * @param line
125         *            the line
126         * @return the double value
127         */
128        protected double getDouble(String line) {
129                Matcher matcher = Pattern.compile(POSDOUBLE).matcher(line);
130                matcher.find();
131                return Double.parseDouble(matcher.group());
132        }
133
134        /*
135         * (non-Javadoc)
136         * 
137         * @see org.opt4j.benchmark.knapsack.KnapsackProblem#getKnapsacks()
138         */
139        @Override
140        public Collection<Knapsack> getKnapsacks() {
141                return knapsacks;
142        }
143
144        /*
145         * (non-Javadoc)
146         * 
147         * @see org.opt4j.benchmark.knapsack.KnapsackProblem#getItems()
148         */
149        @Override
150        public Collection<Item> getItems() {
151                return items;
152        }
153
154}
