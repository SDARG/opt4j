023package org.opt4j.benchmarks.dtlz;
024
025import static java.lang.Math.PI;
026import static java.lang.Math.cos;
027import static java.lang.Math.sin;
028
029import java.util.ArrayList;
030import java.util.Collections;
031import java.util.List;
032
033import org.opt4j.benchmarks.M;
034import org.opt4j.benchmarks.N;
035
036import com.google.inject.Inject;
037
038/**
039 * The {@link DTLZ5} function.
040 * 
041 * @author lukasiewycz
042 * 
043 */
044public class DTLZ5 extends DTLZ2 {
045
046        /**
047         * Constructs a {@link DTLZ5} function.
048         * 
049         * @param m
050         *            the number of objective functions.
051         * @param n
052         *            the size of the search space (length of the x vector).
053         */
054        @Inject
055        public DTLZ5(@M int m, @N int n) {
056                super(m, n);
057        }
058
059        /*
060         * (non-Javadoc)
061         * 
062         * @see org.opt4j.benchmark.dtlz.DTLZEvaluator#f(java.util.List, double)
063         */
064        @Override
065        protected List<Double> f(List<Double> x, double g) {
066                List<Double> f = new ArrayList<>();
067                List<Double> theta = theta(x, g);
068
069                double a = (1.0 + g);
070
071                for (int i = 0; i < m; i++) {
072                        if (i > 0) {
073                                a *= cos(theta.get(i - 1));
074                        }
075                        double value = a;
076                        if (i < x.size()) {
077                                value *= sin(theta.get(i));
078                        }
079                        f.add(value);
080                }
081
082                Collections.reverse(f);
083                return f;
084        }
085
086        /**
087         * The theta function.
088         * 
089         * @param x
090         *            the sublist
091         * @param g
092         *            the result of g(x)
093         * @return theta_i
094         */
095        protected List<Double> theta(List<Double> x, double g) {
096                List<Double> theta = new ArrayList<>();
097
098                theta.add(x.get(0) * PI / 2);
099
100                double t = PI / (4 * (1 + g));
101                for (int i = 1; i < x.size(); i++) {
102                        theta.add(t * (1 + 2 * g * x.get(i)));
103                }
104                return theta;
105        }
106}
