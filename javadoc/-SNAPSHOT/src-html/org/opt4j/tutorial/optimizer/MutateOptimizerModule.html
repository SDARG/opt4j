001package org.opt4j.tutorial.optimizer;
002
003import org.opt4j.core.config.annotations.Parent;
004import org.opt4j.core.optimizer.MaxIterations;
005import org.opt4j.core.optimizer.OptimizerModule;
006import org.opt4j.tutorial.TutorialModule;
007
008@Parent(TutorialModule.class)
009public class MutateOptimizerModule extends OptimizerModule {
010
011        @MaxIterations
012        protected int iterations = 1000;
013
014        public int getIterations() {
015                return iterations;
016        }
017
018        public void setIterations(int iterations) {
019                this.iterations = iterations;
020        }
021
022        @Override
023        public void config() {
024                bindIterativeOptimizer(MutateOptimizer.class);
025        }
026
027}
