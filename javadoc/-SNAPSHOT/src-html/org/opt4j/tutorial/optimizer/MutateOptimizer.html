001package org.opt4j.tutorial.optimizer;
002
003import java.util.Collection;
004
005import org.opt4j.core.Genotype;
006import org.opt4j.core.Individual;
007import org.opt4j.core.IndividualFactory;
008import org.opt4j.core.optimizer.IterativeOptimizer;
009import org.opt4j.core.optimizer.Population;
010import org.opt4j.core.optimizer.TerminationException;
011import org.opt4j.operators.copy.Copy;
012import org.opt4j.operators.mutate.Mutate;
013import org.opt4j.optimizers.ea.Selector;
014
015import com.google.inject.Inject;
016
017public class MutateOptimizer implements IterativeOptimizer {
018
019        protected final IndividualFactory individualFactory;
020
021        protected final Mutate<Genotype> mutate;
022
023        protected final Copy<Genotype> copy;
024
025        protected final Selector selector;
026
027        private final Population population;
028
029        public static final int POPSIZE = 100;
030
031        public static final int OFFSIZE = 25;
032
033        @Inject
034        public MutateOptimizer(Population population, IndividualFactory individualFactory,
035                        Selector selector, Mutate<Genotype> mutate, Copy<Genotype> copy) {
036                this.individualFactory = individualFactory;
037                this.mutate = mutate;
038                this.copy = copy;
039                this.selector = selector;
040                this.population = population;
041        }
042
043        public void initialize() throws TerminationException {
044                selector.init(OFFSIZE + POPSIZE);
045        }
046
047        public void next() throws TerminationException {
048                if (population.isEmpty()) {
049                        for (int i = 0; i < POPSIZE; i++) {
050                                population.add(individualFactory.create());
051                        }
052                } else {
053                        if (population.size() > POPSIZE) {
054                                Collection<Individual> lames = selector.getLames(population.size() - POPSIZE,
055                                                population);
056                                population.removeAll(lames);
057                        }
058
059                        Collection<Individual> parents = selector.getParents(OFFSIZE, population);
060
061                        for (Individual parent : parents) {
062                                Genotype genotype = copy.copy(parent.getGenotype());
063                                mutate.mutate(genotype, 0.1);
064
065                                Individual child = individualFactory.create(genotype);
066                                population.add(child);
067                        }
068                }
069        }
070}
