001package org.opt4j.tutorial.helloworld;
002
003import java.util.Random;
004
005import org.opt4j.core.genotype.SelectGenotype;
006import org.opt4j.core.problem.Creator;
007
008public class HelloWorldCreator implements Creator<SelectGenotype<Character>> {
009
010        protected Character[] ALPHABET = { 'A', 'B', 'C', 'D', 'E', 'F', 'G', 'H', 'I', 'J', 'K', 'L', 'M', 'N', 'O', 'P',
011                        'Q', 'R', 'S', 'T', 'U', 'V', 'W', 'X', 'Y', 'Z', ' ' };
012        protected Random random = new Random();
013
014        @Override
015        public SelectGenotype<Character> create() {
016                SelectGenotype<Character> genotype = new SelectGenotype<>(ALPHABET);
017                genotype.init(random, 11);
018                return genotype;
019        }
020
021}
