001package org.opt4j.tutorial.salesman;
002
003import org.opt4j.core.genotype.PermutationGenotype;
004import org.opt4j.core.problem.Decoder;
005import org.opt4j.tutorial.salesman.SalesmanProblem.City;
006
007public class SalesmanDecoder implements Decoder<PermutationGenotype<City>, SalesmanRoute> {
008
009        public SalesmanRoute decode(PermutationGenotype<City> genotype) {
010                SalesmanRoute salesmanRoute = new SalesmanRoute();
011                for (City city : genotype) {
012                        salesmanRoute.add(city);
013                }
014
015                return salesmanRoute;
016        }
017
018}
