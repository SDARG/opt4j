001package org.opt4j.tutorial.salesman;
002
003import java.awt.BasicStroke;
004import java.awt.Color;
005import java.awt.Component;
006import java.awt.Dimension;
007import java.awt.Graphics;
008import java.awt.Graphics2D;
009import java.awt.Point;
010import java.awt.RenderingHints;
011import java.awt.event.ActionEvent;
012import java.awt.event.ActionListener;
013
014import javax.swing.JMenuItem;
015import javax.swing.JPanel;
016import javax.swing.JPopupMenu;
017
018import org.opt4j.core.Individual;
019import org.opt4j.tutorial.salesman.SalesmanProblem.City;
020import org.opt4j.viewer.IndividualMouseListener;
021import org.opt4j.viewer.Viewport;
022import org.opt4j.viewer.Widget;
023import org.opt4j.viewer.WidgetParameters;
024
025import com.google.inject.Inject;
026
027// The SalesmanWidgetService is an additional feature of this tutorial. It
028// enables the visualization
029// of a single route in the viewer.
030public class SalesmanWidgetService implements IndividualMouseListener {
031
032        protected final Viewport viewport;
033        
034        // Panel that paints a single SalesmanRoute (which is the phenotype of an
035        // Individual)
036        public static class MyPanel extends JPanel {
037
038                private static final long serialVersionUID = 1L;
039
040                protected final Individual individual;
041
042                public MyPanel(Individual individual) {
043                        super();
044                        this.individual = individual;
045                        setPreferredSize(new Dimension(208, 208));
046                }
047
048                @Override
049                protected void paintComponent(Graphics g) {
050                        Graphics2D g2d = (Graphics2D) g;
051                        g2d.setBackground(Color.WHITE);
052                        g2d.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
053                        g2d.setStroke(new BasicStroke(2f));
054                        g2d.clearRect(0, 0, 208, 212);
055
056                        SalesmanRoute salesmanRoute = (SalesmanRoute) individual.getPhenotype();
057
058                        for (int i = 0; i < salesmanRoute.size(); i++) {
059                                final int j = (i + 1) % salesmanRoute.size();
060                                City one = salesmanRoute.get(i);
061                                City two = salesmanRoute.get(j);
062
063                                int x1 = (int) (one.getX() * 2) + 4;
064                                int y1 = (int) (one.getY() * 2) + 4;
065                                int x2 = (int) (two.getX() * 2) + 4;
066                                int y2 = (int) (two.getY() * 2) + 4;
067
068                                g2d.drawLine(x1, y1, x2, y2);
069                                g2d.drawOval(x1 - 2, y1 - 2, 4, 4);
070
071                        }
072                }
073        }
074
075        // Use a custom widget
076        @WidgetParameters(title = "Route", resizable = false, maximizable = false)
077        protected static class SalesmanWidget implements Widget {
078
079                protected final Individual individual;
080
081                public SalesmanWidget(Individual individual) {
082                        super();
083                        this.individual = individual;
084                }
085
086                @Override
087                public JPanel getPanel() {
088                        JPanel panel = new MyPanel(individual);
089                        return panel;
090                }
091
092                @Override
093                public void init(Viewport viewport) {
094                        // nothing to be done
095                }
096
097        }
098
099        // The route is shown by a double click of a individual in the archive
100        // monitor panel. Thus we need the ArchiveMonitorPanel and the main
101        // GUIFrame.
102        @Inject
103        public SalesmanWidgetService(Viewport viewport) {
104                this.viewport = viewport;
105        }
106
107        // If an individual is double clicked, paint the route.
108        @Override
109        public void onDoubleClick(Individual individual, Component table, Point p) {
110                paintRoute(individual);
111        }
112
113        // If an individual is clicked with the right mouse button, open a popup
114        // menu that contains the option to paint the route.
115        @Override
116        public void onPopup(final Individual individual, Component table, Point p, JPopupMenu menu) {
117                JMenuItem paint = new JMenuItem("show route");
118                menu.add(paint);
119
120                paint.addActionListener(new ActionListener() {
121                        @Override
122                        public void actionPerformed(ActionEvent e) {
123                                paintRoute(individual);
124                        }
125                });
126
127        }
128
129        // Paint the route: Construct a JInternalFrame, add the MyPanel and add the
130        // frame to the desktop of the main GUIFrame.
131        protected void paintRoute(Individual individual) {
132                Widget widget = new SalesmanWidget(individual);
133                viewport.addWidget(widget);
134        }
135
136}
