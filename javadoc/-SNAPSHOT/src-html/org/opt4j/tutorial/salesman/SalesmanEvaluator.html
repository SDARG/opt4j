001package org.opt4j.tutorial.salesman;
002
003import org.opt4j.core.Objective.Sign;
004import org.opt4j.core.Objectives;
005import org.opt4j.core.problem.Evaluator;
006import org.opt4j.tutorial.salesman.SalesmanProblem.City;
007
008public class SalesmanEvaluator implements Evaluator<SalesmanRoute> {
009
010        public Objectives evaluate(SalesmanRoute salesmanRoute) {
011                double dist = 0;
012                for (int i = 0; i < salesmanRoute.size(); i++) {
013                        City one = salesmanRoute.get(i);
014                        City two = salesmanRoute.get((i + 1) % salesmanRoute.size());
015                        dist += getEuclideanDistance(one, two);
016                }
017
018                Objectives objectives = new Objectives();
019                objectives.add("distance", Sign.MIN, dist);
020                return objectives;
021        }
022
023        private double getEuclideanDistance(City one, City two) {
024                final double x = one.getX() - two.getX();
025                final double y = one.getY() - two.getY();
026                return Math.sqrt(x * x + y * y);
027        }
028
029}
