001package org.opt4j.tutorial.salesman;
002
003import java.util.HashSet;
004import java.util.Random;
005import java.util.Set;
006
007import org.opt4j.core.start.Constant;
008
009import com.google.inject.Inject;
010
011public class SalesmanProblem {
012
013        protected Set<City> cities = new HashSet<>();
014
015        public class City {
016                protected final double x;
017                protected final double y;
018
019                public City(double x, double y) {
020                        this.x = x;
021                        this.y = y;
022                }
023
024                public double getX() {
025                        return x;
026                }
027
028                public double getY() {
029                        return y;
030                }
031        }
032
033        @Inject
034        public SalesmanProblem(@Constant(value = "size") int size) {
035                Random random = new Random(0);
036
037                for (int i = 0; i < size; i++) {
038                        final double x = random.nextDouble() * 100;
039                        final double y = random.nextDouble() * 100;
040                        final City city = new City(x, y);
041
042                        cities.add(city);
043                }
044        }
045
046        public Set<City> getCities() {
047                return cities;
048        }
049
050}
