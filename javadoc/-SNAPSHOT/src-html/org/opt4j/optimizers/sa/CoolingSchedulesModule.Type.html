024package org.opt4j.optimizers.sa;
025
026import org.opt4j.core.config.annotations.Info;
027import org.opt4j.core.config.annotations.Order;
028import org.opt4j.core.config.annotations.Parent;
029import org.opt4j.core.config.annotations.Required;
030import org.opt4j.core.start.Constant;
031
032/**
033 * This module provides several common {@link CoolingSchedule}s for the
034 * {@link SimulatedAnnealing}.
035 * 
036 * @author lukasiewycz
037 * 
038 */
039@Info("Common Cooling Schedules")
040@Parent(SimulatedAnnealingModule.class)
041public class CoolingSchedulesModule extends CoolingScheduleModule {
042
043        @Constant(value = "t0", namespace = CoolingSchedule.class)
044        protected double initialTemperature = 10.0;
045
046        @Constant(value = "tn", namespace = CoolingSchedule.class)
047        protected double finalTemperature = 1.0;
048
049        @Required(property = "type", elements = { "EXPONENTIAL" })
050        @Constant(value = "alpha", namespace = CoolingScheduleExponential.class)
051        protected double alpha = 0.995;
052
053        @Order(0)
054        protected Type type = Type.LINEAR;
055
056        /**
057         * Type of {@link CoolingSchedule} to use.
058         * 
059         * @author lukasiewycz
060         * 
061         */
062        public enum Type {
063
064                /**
065                 * Use the {@link CoolingScheduleLinear}.
066                 */
067                LINEAR,
068                /**
069                 * Use the {@link CoolingScheduleHyperbolic}.
070                 */
071                HYPERBOLIC,
072                /**
073                 * Use the {@link CoolingScheduleExponential}.
074                 */
075                EXPONENTIAL;
076        }
077
078        /**
079         * Returns the final temperature.
080         * 
081         * @see #setFinalTemperature
082         * @return the final temperature
083         */
084        @Info("The temperature at the end of the optimization")
085        public double getFinalTemperature() {
086                return finalTemperature;
087        }
088
089        /**
090         * Sets the final temperature.
091         * 
092         * @see #getFinalTemperature
093         * @param finalTemperature
094         *            the final temperature
095         */
096        public void setFinalTemperature(double finalTemperature) {
097                this.finalTemperature = finalTemperature;
098        }
099
100        /**
101         * Returns the initial temperature.
102         * 
103         * @see #setInitialTemperature
104         * @return the initial temperature
105         */
106        @Info("The temperature at the beginning of the optimization")
107        public double getInitialTemperature() {
108                return initialTemperature;
109        }
110
111        /**
112         * Sets the initial temperature.
113         * 
114         * @see #getInitialTemperature
115         * @param initialTemperature
116         *            the initial temperature
117         */
118        public void setInitialTemperature(double initialTemperature) {
119                this.initialTemperature = initialTemperature;
120        }
121
122        /**
123         * Returns the type of cooling schedule.
124         * 
125         * @see #setType
126         * @return the type of cooling schedule
127         */
128        @Info("The type of the cooling schedule")
129        public Type getType() {
130                return type;
131        }
132
133        /**
134         * Sets the type of cooling schedule.
135         * 
136         * @see #getType
137         * @param type
138         *            the type of cooling schedule
139         */
140        public void setType(Type type) {
141                this.type = type;
142        }
143
144        /**
145         * Returns the alpha value.
146         * 
147         * @see #setAlpha
148         * @return the alpha value
149         */
150        public double getAlpha() {
151                return alpha;
152        }
153
154        /**
155         * Sets the alpha value.
156         * 
157         * @see #getAlpha
158         * @param alpha
159         *            the alpha value
160         */
161        public void setAlpha(double alpha) {
162                this.alpha = alpha;
163        }
164
165        /*
166         * (non-Javadoc)
167         * 
168         * @see org.opt4j.start.Opt4JModule#config()
169         */
170        @Override
171        public void config() {
172
173                switch (type) {
174                case LINEAR:
175                        bindCoolingSchedule(CoolingScheduleLinear.class);
176                        break;
177                case HYPERBOLIC:
178                        bindCoolingSchedule(CoolingScheduleHyperbolic.class);
179                        break;
180                default: // EXPONENTIAL
181                        bindCoolingSchedule(CoolingScheduleExponential.class);
182                        break;
183                }
184        }
185
186}
