024package org.opt4j.optimizers.mopso;
025
026import java.util.Random;
027
028import org.opt4j.core.common.random.Rand;
029import org.opt4j.operators.algebra.Add;
030import org.opt4j.operators.algebra.Index;
031import org.opt4j.operators.algebra.Mult;
032import org.opt4j.operators.algebra.Sub;
033import org.opt4j.operators.algebra.Term;
034import org.opt4j.operators.algebra.Var;
035
036/**
037 * The {@link VelocityTerm} is used to determine the new velocity of a
038 * {@link Particle}.
039 * 
040 * @author lukasiewycz
041 * 
042 */
043public class VelocityTerm implements Term {
044
045        protected final Term term;
046        protected final Random random;
047
048        protected final Var W = new Var();
049        protected final Var C1 = new Var();
050        protected final Var C2 = new Var();
051        protected final Var r1 = new Var();
052        protected final Var r2 = new Var();
053
054        /**
055         * Constructs a {@link VelocityTerm}.
056         * 
057         * @param random
058         *            the random number generator
059         */
060        public VelocityTerm(Rand random) {
061                this.random = random;
062
063                Index x = new Index(0);
064                Index v = new Index(1);
065                Index xbest = new Index(2);
066                Index xleader = new Index(3);
067
068                Term m1 = new Mult(W, v);
069                Term m2 = new Mult(C1, r1, new Sub(xbest, x));
070                Term m3 = new Mult(C2, r2, new Sub(xleader, x));
071
072                term = new Add(m1, m2, m3);
073
074                randomize();
075        }
076
077        /*
078         * (non-Javadoc)
079         * 
080         * @see org.opt4j.operator.algebra.Term#calculate(double[])
081         */
082        @Override
083        public double calculate(double... values) {
084                return term.calculate(values);
085        }
086
087        /**
088         * Randomizes the constants of the term.
089         */
090        public void randomize() {
091                W.setValue(random.nextDouble() * 0.4 + 0.1);
092                C1.setValue(random.nextDouble() * 0.5 + 1.5);
093                C2.setValue(random.nextDouble() * 0.5 + 1.5);
094                r1.setValue(random.nextDouble());
095                r2.setValue(random.nextDouble());
096        }
097
098}
