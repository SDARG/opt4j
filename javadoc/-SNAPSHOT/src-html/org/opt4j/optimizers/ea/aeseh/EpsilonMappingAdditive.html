001package org.opt4j.optimizers.ea.aeseh;
002
003import java.util.HashMap;
004import java.util.Iterator;
005import java.util.Map;
006import java.util.Map.Entry;
007import java.util.Set;
008
009import org.opt4j.core.Individual;
010import org.opt4j.core.Objective;
011import org.opt4j.core.Objective.Sign;
012import org.opt4j.core.Objectives;
013
014/**
015 * The {@link EpsilonMappingAdditive} implements the evenly spaced adaptive ε
016 * function.
017 * 
018 * @author Fedor Smirnov
019 *
020 */
021public class EpsilonMappingAdditive implements EpsilonMapping {
022
023        /**
024         * Applies ε mapping by enhancing all of the given {@link Objective}s by the
025         * ε fraction of the objective amplitude.
026         * 
027         * @param original
028         *            the {@link Objectives} that are enhanced by this method
029         * @param epsilon
030         *            the fraction used for the enhancement
031         * @param objectiveAmplitudes
032         *            the map mapping its objective onto its amplitude
033         * @return enhanced {@link Objectives} where each {@link Objective} is
034         *         improved by the ε fraction of the objective's amplitude
035         */
036        @Override
037        public Objectives mapObjectives(Objectives original, double epsilon, Map<Objective, Double> objectiveAmplitudes) {
038                Objectives result = new Objectives();
039                Iterator<Objective> iterator = original.getKeys().iterator();
040                double[] values = original.array();
041                for (int i = 0; i < original.size(); i++) {
042                        Objective obj = iterator.next();
043                        double value = values[i] * (obj.getSign().equals(Sign.MIN) ? 1 : -1);
044                        if (objectiveAmplitudes.containsKey(obj)) {
045                                // the ε mapping is only applied if the objective is feasible
046                                // for at least one
047                                // individual
048                                value += epsilon * objectiveAmplitudes.get(obj) * (obj.getSign().equals(Sign.MAX) ? 1 : -1);
049                        }
050                        result.add(obj, value);
051                }
052                return result;
053        }
054
055        @Override
056        public Map<Objective, Double> findObjectiveAmplitudes(Set<Individual> individuals) {
057                Map<Objective, Double> maximumMap = new HashMap<>();
058                Map<Objective, Double> minimumMap = new HashMap<>();
059                Map<Objective, Double> amplitudeMap = new HashMap<>();
060                for (Individual indi : individuals) {
061                        Objectives objectives = indi.getObjectives();
062                        Iterator<Objective> iterator = objectives.getKeys().iterator();
063                        double[] values = objectives.array();
064                        for (int i = 0; i < objectives.size(); i++) {
065                                Objective obj = iterator.next();
066                                if (objectives.get(obj).getValue() == Objective.INFEASIBLE) {
067                                        continue;
068                                }
069                                double value = values[i];
070                                if (!maximumMap.containsKey(obj) || maximumMap.get(obj) < value) {
071                                        maximumMap.put(obj, value);
072                                }
073                                if (!minimumMap.containsKey(obj) || minimumMap.get(obj) > value) {
074                                        minimumMap.put(obj, value);
075                                }
076                        }
077                }
078                for (Entry<Objective, Double> maxEntry : maximumMap.entrySet()) {
079                        Objective obj = maxEntry.getKey();
080                        Double maximum = maxEntry.getValue();
081                        Double minimum = minimumMap.get(obj);
082                        amplitudeMap.put(obj, maximum - minimum);
083                }
084                return amplitudeMap;
085        }
086}
