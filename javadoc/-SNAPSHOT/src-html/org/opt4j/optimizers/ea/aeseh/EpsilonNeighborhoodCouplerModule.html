001package org.opt4j.optimizers.ea.aeseh;
002
003import static org.opt4j.core.config.annotations.Citation.PublicationMonth.UNKNOWN;
004
005import org.opt4j.core.config.annotations.Citation;
006import org.opt4j.core.config.annotations.Info;
007import org.opt4j.core.config.annotations.Name;
008import org.opt4j.core.start.Constant;
009import org.opt4j.optimizers.ea.Coupler;
010import org.opt4j.optimizers.ea.CouplerModule;
011
012/**
013 * Binds the {@link EpsilonNeighborhoodCoupler} as the {@link Coupler}.
014 * 
015 * @author Fedor Smirnov
016 *
017 */
018@Name("Epsilon Neighborhood")
019@Citation(authors = "Hernán Aguirre, Akira Oyama, and Kiyoshi Tanaka", title = "Adaptive ε-sampling and ε-hood for evolutionary many-objective optimization.", journal = "Evolutionary Multi-Criterion Optimization (EMO)", pageFirst = 322, pageLast = 336, year = 2013, month = UNKNOWN)
020public class EpsilonNeighborhoodCouplerModule extends CouplerModule {
021
022        @Info("The reference value for the number of neighborhoods. The epsilon_h value is adapted to create a similar number of neighborhoods.")
023        @Constant(value = "neighborhoodNumber", namespace = EpsilonNeighborhoodCoupler.class)
024        protected int plannedNeighborhoodNumber = 5;
025        @Info("The start value used for the epsilon value which is applied during the neighborhood creation.")
026        @Constant(value = "epsilonNeighborhood", namespace = EpsilonNeighborhoodCoupler.class)
027        protected double epsilonNeighborhood = 0.0;
028        @Info("The start value used for the adaptation of the epsilon value which is applied during the neighborhood creation.")
029        @Constant(value = "epsilonNeighborhoodDelta", namespace = EpsilonNeighborhoodCoupler.class)
030        protected double epsilonNeighborhoodDelta = 0.005;
031        @Info("The maximal value used for the epsilon value which is applied during the neighborhood creation.")
032        @Constant(value = "epsilonNeighborhoodDeltaMax", namespace = EpsilonNeighborhoodCoupler.class)
033        protected double epsilonNeighborhoodDeltaMax = 0.005;
034        @Info("The minimal value used for the epsilon value which is applied during the neighborhood creation.")
035        @Constant(value = "epsilonNeighborhoodDeltaMin", namespace = EpsilonNeighborhoodCoupler.class)
036        protected double epsilonNeighborhoodDeltaMin = 0.0001;
037
038        @Override
039        protected void config() {
040                bindCoupler(EpsilonNeighborhoodCoupler.class);
041        }
042
043        public int getPlannedNeighborhoodNumber() {
044                return plannedNeighborhoodNumber;
045        }
046
047        public void setPlannedNeighborhoodNumber(int plannedNeighborhoodNumber) {
048                this.plannedNeighborhoodNumber = plannedNeighborhoodNumber;
049        }
050
051        public double getEpsilonNeighborhood() {
052                return epsilonNeighborhood;
053        }
054
055        public void setEpsilonNeighborhood(double epsilonNeighborhood) {
056                this.epsilonNeighborhood = epsilonNeighborhood;
057        }
058
059        public double getEpsilonNeighborhoodDelta() {
060                return epsilonNeighborhoodDelta;
061        }
062
063        public void setEpsilonNeighborhoodDelta(double epsilonNeighborhoodDelta) {
064                this.epsilonNeighborhoodDelta = epsilonNeighborhoodDelta;
065        }
066
067        public double getEpsilonNeighborhoodDeltaMax() {
068                return epsilonNeighborhoodDeltaMax;
069        }
070
071        public void setEpsilonNeighborhoodDeltaMax(double epsilonNeighborhoodDeltaMax) {
072                this.epsilonNeighborhoodDeltaMax = epsilonNeighborhoodDeltaMax;
073        }
074
075        public double getEpsilonNeighborhoodDeltaMin() {
076                return epsilonNeighborhoodDeltaMin;
077        }
078
079        public void setEpsilonNeighborhoodDeltaMin(double epsilonNeighborhoodDeltaMin) {
080                this.epsilonNeighborhoodDeltaMin = epsilonNeighborhoodDeltaMin;
081        }
082
083}
