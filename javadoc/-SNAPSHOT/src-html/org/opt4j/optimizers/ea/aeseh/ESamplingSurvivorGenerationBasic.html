003import java.util.ArrayList;
004import java.util.Collection;
005import java.util.HashMap;
006import java.util.HashSet;
007import java.util.List;
008import java.util.Map;
009import java.util.Random;
010import java.util.Set;
011
012import org.opt4j.core.Individual;
013import org.opt4j.core.Objective;
014import org.opt4j.core.Objectives;
015import org.opt4j.core.start.Constant;
016import org.opt4j.optimizers.ea.NonDominatedFronts;
017
018import com.google.inject.Inject;
019
020/**
021 * The {@link ESamplingSurvivorGenerationBasic} implements the basic survivor
022 * selection used by the Adaptive ε-sampling and ε-hood for evolutionary
023 * many-objective optimization.
024 * 
025 * @author Fedor Smirnov
026 *
027 */
028public class ESamplingSurvivorGenerationBasic implements ESamplingSurvivorGeneration {
029
030        protected final Random random;
031        protected final EpsilonMapping epsilonMapping;
032        protected final EpsilonAdaptation epsilonAdaption;
033        protected final AdaptiveEpsilon adaptiveEpsilonSampling;
034
035        /**
036         * Basic constructor.
037         * 
038         * @param epsilonMapping
039         *            an {@link EpsilonMapping} that is used to enhance the
040         *            {@link Objectives} during the choice of the survivors
041         * @param epsilonAdaption
042         *            an {@link EpsilonAdaptation} that adjusts the ε valued used
043         *            for the choice of the survivors
044         * @param random
045         *            the {@link Random} used for the sampling
046         */
047        @Inject
048        public ESamplingSurvivorGenerationBasic(Random random, EpsilonMapping epsilonMapping,
049                        EpsilonAdaptation epsilonAdaption,
050                        @Constant(value = "epsilonSample", namespace = ESamplingSurvivorGeneration.class) double epsilonSample,
051                        @Constant(value = "epsilonSampleDelta", namespace = ESamplingSurvivorGeneration.class) double epsilonSampleDelta,
052                        @Constant(value = "epsilonSampleDeltaMax", namespace = ESamplingSurvivorGeneration.class) double epsilonSampleDeltaMax,
053                        @Constant(value = "epsilonSampleDeltaMin", namespace = ESamplingSurvivorGeneration.class) double epsilonSampleDeltaMin) {
054                this.random = random;
055                this.epsilonMapping = epsilonMapping;
056                this.epsilonAdaption = epsilonAdaption;
057                this.adaptiveEpsilonSampling = new AdaptiveEpsilon(epsilonSample, epsilonSampleDelta, epsilonSampleDeltaMax,
058                                epsilonSampleDeltaMin);
059        }
060
061        @Override
062        public Set<Individual> getSurvivors(Collection<Individual> population, int survivorNumber) {
063                Set<Individual> survivors;
064                // get the non-dominated front and the extreme solutions
065                NonDominatedFronts fronts = new NonDominatedFronts(population);
066                Collection<Individual> paretoSolutions = fronts.getFrontAtIndex(0);
067                Set<Individual> extremeIndividuals = getExtremeIndividuals(paretoSolutions);
068
069                if (paretoSolutions.size() > survivorNumber) {
070                        // more non-dominated solutions than survivors => apply ε-sampling
071                        survivors = addNonDominatedSurvivors(extremeIndividuals, paretoSolutions, survivorNumber);
072                } else {
073                        survivors = addDominatedSurvivors(survivorNumber, fronts);
074                }
075                return survivors;
076        }
077
078        /**
079         * Creates the survivor pool by adding the ε-sampled individuals to the
080         * extreme individuals.
081         * 
082         * @param extremeIndividuals
083         *            the {@link Individual}s with (positively) extreme values for
084         *            their {@link Objective}s
085         * @param firstFront
086         *            the {@link Individual} that are not dominated at all
087         * @param survivorNumber
088         *            the desired number of survivors
089         */
090        protected Set<Individual> addNonDominatedSurvivors(Collection<Individual> extremeIndividuals,
091                        Collection<Individual> firstFront, int survivorNumber) {
092                Set<Individual> survivors = new HashSet<>();
093                List<Individual> nonDominatedIndividuals = new ArrayList<>(firstFront);
094                // the extreme values always survive
095                survivors.addAll(extremeIndividuals);
096                nonDominatedIndividuals.removeAll(extremeIndividuals);
097                Set<Individual> epsilonDominantIndividuals = new HashSet<>();
098                Set<Individual> epsilonDominatedIndividuals = new HashSet<>();
099                applyEpsilonSampling(nonDominatedIndividuals, epsilonDominantIndividuals, epsilonDominatedIndividuals,
100                                adaptiveEpsilonSampling.getEpsilon());
101                boolean tooManyEpsilonDominantIndividuals = (extremeIndividuals.size()
102                                + epsilonDominantIndividuals.size()) > survivorNumber;
103                // adapt the sampling epsilon
104                boolean epsilonTooBig = !tooManyEpsilonDominantIndividuals;
105                epsilonAdaption.adaptEpsilon(adaptiveEpsilonSampling, epsilonTooBig);
106                if (tooManyEpsilonDominantIndividuals) {
107                        // add a random subset of the epsilon dominant individuals
108                        List<Individual> survivalCandidates = new ArrayList<>(epsilonDominantIndividuals);
109                        while (survivors.size() < survivorNumber) {
110                                int idx = random.nextInt(survivalCandidates.size());
111                                Individual survivor = survivalCandidates.get(idx);
112                                survivors.add(survivor);
113                                survivalCandidates.remove(idx);
114                        }
115                } else {
116                        // add a random subset from the epsilon dominated individuals
117                        survivors.addAll(epsilonDominantIndividuals);
118                        List<Individual> survivorCandidates = new ArrayList<>(epsilonDominatedIndividuals);
119                        while (survivors.size() < survivorNumber) {
120                                int idx = random.nextInt(survivorCandidates.size());
121                                Individual survivor = survivorCandidates.get(idx);
122                                survivorCandidates.remove(idx);
123                                survivors.add(survivor);
124                        }
125                }
126                return survivors;
127        }
128
129        /**
130         * Applies ε-sampling by dividing the given individuals into the two sets of
131         * ε-dominant and ε-dominated individuals.
132         * 
133         * @param firstFront
134         *            the input individuals which constitute the first non-dominated
135         *            front of the current population
136         * @param epsilonDominantIndividuals
137         *            the set that will be filled with the epsilon-dominant
138         *            individuals
139         * @param epsilonDominatedIndividuals
140         *            the set that will be filled with epsilon-dominated individuals
141         * @param samplingEpsilon
142         *            the value used for the epsilon sampling
143         */
144        protected void applyEpsilonSampling(List<Individual> firstFront, Set<Individual> epsilonDominantIndividuals,
145                        Set<Individual> epsilonDominatedIndividuals, double samplingEpsilon) {
146                // apply epsilon sampling until the individual list is empty
147                List<Individual> nonDominatedIndividuals = new ArrayList<>(firstFront);
148                Map<Objective, Double> objectiveAmplitudes = epsilonMapping
149                                .findObjectiveAmplitudes(new HashSet<>(nonDominatedIndividuals));
150                while (!nonDominatedIndividuals.isEmpty()) {
151                        // pick a random individual
152                        Individual epsilonDominant = nonDominatedIndividuals.get(random.nextInt(nonDominatedIndividuals.size()));
153                        Set<Individual> epsilonDominated = new HashSet<>();
154                        nonDominatedIndividuals.remove(epsilonDominant);
155                        Objectives epsilonEnhancedObjectives = epsilonMapping.mapObjectives(epsilonDominant.getObjectives(),
156                                        samplingEpsilon, objectiveAmplitudes);
157                        // gather all individuals epsilon dominated by the picked individual
158                        for (int i = 0; i < nonDominatedIndividuals.size(); i++) {
159                                Individual comparisonIndividual = nonDominatedIndividuals.get(i);
160                                if (epsilonEnhancedObjectives.dominates(comparisonIndividual.getObjectives())) {
161                                        epsilonDominated.add(comparisonIndividual);
162                                }
163                        }
164                        nonDominatedIndividuals.removeAll(epsilonDominated);
165                        epsilonDominantIndividuals.add(epsilonDominant);
166                        epsilonDominatedIndividuals.addAll(epsilonDominated);
167                }
168        }
169
170        /**
171         * In the case where the first non-dominated front does not suffice to
172         * create enough survivors, dominated solutions are added to the survivor
173         * pool.
174         * 
175         * @param survivorNumber
176         *            the desired number of survivors
177         * @param fronts
178         *            the non-dominated fronts
179         */
180        protected Set<Individual> addDominatedSurvivors(int survivorNumber, NonDominatedFronts fronts) {
181                Set<Individual> survivors = new HashSet<>();
182                // non-dominated solutions do not suffice to generate the number of
183                // survivors => add dominated solutions
184                survivors.addAll(fronts.getFrontAtIndex(0));
185                int frontIndex = 1;
186                // Fill the survivors by iteratively adding the dominated fronts.
187                while (survivorNumber > (fronts.getFrontAtIndex(frontIndex).size()) + survivors.size()) {
188                        survivors.addAll(fronts.getFrontAtIndex(frontIndex));
189                        frontIndex++;
190                }
191                List<Individual> currentFront = new ArrayList<>(fronts.getFrontAtIndex(frontIndex));
192                while (survivorNumber > survivors.size()) {
193                        // choose a random survivor from the current front
194                        Individual survivor = currentFront.get(random.nextInt(currentFront.size()));
195                        survivors.add(survivor);
196                        currentFront.remove(survivor);
197                }
198                return survivors;
199        }
200
201        /**
202         * Returns the {@link Individual}s with the best values for the individual
203         * {@link Objective}.
204         * 
205         * @param firstFront
206         *            the list of {@link Individual}s constituting the first
207         *            non-dominated front
208         * 
209         * @return the set of the extreme individuals
210         */
211        protected Set<Individual> getExtremeIndividuals(Collection<Individual> firstFront) {
212                Map<Objective, Individual> bestIndis = new HashMap<>();
213                Map<Objective, Double> extremeValues = new HashMap<>();
214                Individual firstIndi = firstFront.iterator().next();
215                List<Objective> objList = new ArrayList<>(firstIndi.getObjectives().getKeys());
216                // iterate the individuals
217                for (Individual indi : firstFront) {
218                        // iterate the objectives and their values
219                        double[] values = indi.getObjectives().array();
220                        for (int i = 0; i < objList.size(); i++) {
221                                Objective obj = objList.get(i);
222                                double value = values[i];
223                                if (!bestIndis.containsKey(obj) || extremeValues.get(obj) > value) {
224                                        bestIndis.put(obj, indi);
225                                        extremeValues.put(obj, value);
226                                }
227                        }
228                }
229                return new HashSet<>(bestIndis.values());
230        }
231}
