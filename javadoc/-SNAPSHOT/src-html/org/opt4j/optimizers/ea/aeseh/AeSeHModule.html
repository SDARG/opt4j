001package org.opt4j.optimizers.ea.aeseh;
002
003import static org.opt4j.core.config.annotations.Citation.PublicationMonth.UNKNOWN;
004
005import org.opt4j.core.config.annotations.Citation;
006import org.opt4j.core.config.annotations.Info;
007import org.opt4j.core.config.annotations.Parent;
008import org.opt4j.core.optimizer.OptimizerModule;
009import org.opt4j.optimizers.ea.Coupler;
010import org.opt4j.optimizers.ea.EvolutionaryAlgorithmModule;
011import org.opt4j.optimizers.ea.Selector;
012
013/**
014 * The {@link AeSeHModule} binds the {@link EpsilonSamplingSelector} as
015 * {@link Selector} and the {@link EpsilonNeighborhoodCoupler} as
016 * {@link Coupler} and configures them with the parameters used in the AeSeH
017 * paper (see the citation for details). Together with the
018 * {@link EvolutionaryAlgorithmModule}, the {@link AeSeHModule}, hence, provides
019 * a default configuration for the Epsilon Sampling Epsilon Neighborhood
020 * Evolutionary Algorithm.
021 * 
022 * @author Fedor Smirnov
023 */
024@Info("Multi-objective evolutionary algorithm where the survival selection and the creation of neighborhoods is based on epsilon-dominance. The selection of parents is done within the created neighborhoods.")
025@Citation(authors = "Hernán Aguirre, Akira Oyama, and Kiyoshi Tanaka", title = "Adaptive ε-sampling and ε-hood for evolutionary many-objective optimization.", journal = "Evolutionary Multi-Criterion Optimization (EMO)", pageFirst = 322, pageLast = 336, year = 2013, month = UNKNOWN)
026@Parent(EvolutionaryAlgorithmModule.class)
027public class AeSeHModule extends OptimizerModule {
028
029        @Override
030        protected void config() {
031
032                bindConstant("epsilonSample", ESamplingSurvivorGeneration.class).to(0.0);
033                bindConstant("epsilonSampleDelta", ESamplingSurvivorGeneration.class).to(0.005);
034                bindConstant("epsilonSampleDeltaMax", ESamplingSurvivorGeneration.class).to(0.005);
035                bindConstant("epsilonSampleDeltaMin", ESamplingSurvivorGeneration.class).to(0.0001);
036
037                bindConstant("epsilonNeighborhood", EpsilonNeighborhoodCoupler.class).to(0.0);
038                bindConstant("epsilonNeighborhoodDelta", EpsilonNeighborhoodCoupler.class).to(0.005);
039                bindConstant("epsilonNeighborhoodDeltaMax", EpsilonNeighborhoodCoupler.class).to(0.005);
040                bindConstant("epsilonNeighborhoodDeltaMin", EpsilonNeighborhoodCoupler.class).to(0.0001);
041                bindConstant("neighborhoodNumber", EpsilonNeighborhoodCoupler.class).to(5);
042
043                bind(Selector.class).to(EpsilonSamplingSelector.class);
044                bind(Coupler.class).to(EpsilonNeighborhoodCoupler.class);
045        }
046}
