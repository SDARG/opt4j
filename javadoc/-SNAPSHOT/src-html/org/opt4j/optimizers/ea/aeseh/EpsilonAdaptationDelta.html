001package org.opt4j.optimizers.ea.aeseh;
002
003/**
004 * The {@link EpsilonAdaptationDelta} adapts the Îµ values exactly as described
005 * in the paper cited in the {@link AeSeHModule}.
006 * 
007 * @author Fedor Smirnov
008 *
009 */
010public class EpsilonAdaptationDelta implements EpsilonAdaptation {
011
012        @Override
013        public void adaptEpsilon(AdaptiveEpsilon adaptiveEpsilon, boolean epsilonTooBig) {
014                if (epsilonTooBig) {
015                        adaptiveEpsilon.setEpsilon(adaptiveEpsilon.getEpsilon() - adaptiveEpsilon.getEpsilonDelta());
016                        adaptiveEpsilon.setEpsilonDelta(adaptiveEpsilon.getEpsilonDelta() / 2);
017                } else {
018                        adaptiveEpsilon.setEpsilon(adaptiveEpsilon.getEpsilon() + adaptiveEpsilon.getEpsilonDelta());
019                        adaptiveEpsilon.setEpsilonDelta(adaptiveEpsilon.getEpsilonDelta() * 2);
020                }
021        }
022}
