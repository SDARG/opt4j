023package org.opt4j.optimizers.ea;
024
025import org.opt4j.core.config.annotations.Ignore;
026import org.opt4j.core.config.annotations.Info;
027import org.opt4j.core.config.annotations.Order;
028import org.opt4j.core.optimizer.MaxIterations;
029import org.opt4j.core.optimizer.OptimizerModule;
030import org.opt4j.core.start.Constant;
031
032/**
033 * The {@link EvolutionaryAlgorithmModule} configures the
034 * {@link EvolutionaryAlgorithm}.
035 * 
036 * @author lukasiewycz
037 * 
038 */
039
040@Info("Multi-Objective Evolutionary Algorithm that performs a Crossover and Mutate for variation and uses a Selector for the environmental selection.")
041public class EvolutionaryAlgorithmModule extends OptimizerModule {
042
043        @Info("The number of generations.")
044        @Order(0)
045        @MaxIterations
046        protected int generations = 1000;
047
048        @Constant(value = "alpha", namespace = EvolutionaryAlgorithm.class)
049        @Info("The size of the population α.")
050        @Order(1)
051        protected int populationSize = 100;
052
053        @Constant(value = "mu", namespace = EvolutionaryAlgorithm.class)
054        @Info("The number of parents per generation μ.")
055        @Order(2)
056        protected int parentsPerGeneration = 25;
057
058        @Constant(value = "lambda", namespace = EvolutionaryAlgorithm.class)
059        @Info("The number of offsprings per generation λ.")
060        @Order(3)
061        protected int offspringsPerGeneration = 25;
062
063        @Info("Performs a crossover operation with this given rate.")
064        @Order(4)
065        @Constant(value = "rate", namespace = ConstantCrossoverRate.class)
066        protected double crossoverRate = 0.95;
067
068        @Ignore
069        protected CrossoverRateType crossoverRateType = CrossoverRateType.CONSTANT;
070
071        /**
072         * The {@link CrossoverRateType} allows to choose between different types of
073         * crossover rates.
074         * 
075         * @author glass
076         * 
077         */
078        public enum CrossoverRateType {
079                /**
080                 * Use a constant crossover rate.
081                 */
082                CONSTANT;
083        }
084
085        /**
086         * Returns the population size {@code alpha}.
087         * 
088         * @return the population size
089         */
090        public int getPopulationSize() {
091                return populationSize;
092        }
093
094        /**
095         * Sets the population size {@code alpha}.
096         * 
097         * @param alpha
098         *            the population size to set
099         */
100        public void setPopulationSize(int alpha) {
101                this.populationSize = alpha;
102        }
103
104        /**
105         * Returns the number of generations.
106         * 
107         * @return the number of generations
108         */
109        public int getGenerations() {
110                return generations;
111        }
112
113        /**
114         * Sets the number of generations.
115         * 
116         * @see #getGenerations
117         * @param generations
118         *            the number of generations
119         */
120        public void setGenerations(int generations) {
121                this.generations = generations;
122        }
123
124        /**
125         * Returns the number of children {@code lambda}.
126         * 
127         * @return the number of children
128         */
129        public int getOffspringsPerGeneration() {
130                return offspringsPerGeneration;
131        }
132
133        /**
134         * Sets the number of children {@code lambda}.
135         * 
136         * @param lambda
137         *            the number of children
138         */
139        public void setOffspringsPerGeneration(int lambda) {
140                this.offspringsPerGeneration = lambda;
141        }
142
143        /**
144         * Returns the number of parents {@code mu}.
145         * 
146         * @return the number of parents
147         */
148        public int getParentsPerGeneration() {
149                return parentsPerGeneration;
150        }
151
152        /**
153         * Sets the number of parents {@code mu}.
154         * 
155         * @param mu
156         *            the number of parents
157         */
158        public void setParentsPerGeneration(int mu) {
159                this.parentsPerGeneration = mu;
160        }
161
162        /**
163         * Returns the type of crossover rate that is used.
164         * 
165         * @return the crossoverRateType
166         */
167        public CrossoverRateType getCrossoverRateType() {
168                return crossoverRateType;
169        }
170
171        /**
172         * Sets the type of crossover rate to use.
173         * 
174         * @param crossoverRateType
175         *            the crossoverRateType to set
176         */
177        public void setCrossoverRateType(CrossoverRateType crossoverRateType) {
178                this.crossoverRateType = crossoverRateType;
179        }
180
181        /**
182         * Returns the used crossover rate.
183         * 
184         * @return the crossoverRate
185         */
186        public double getCrossoverRate() {
187                return crossoverRate;
188        }
189
190        /**
191         * Sets the crossover rate.
192         * 
193         * @param crossoverRate
194         *            the crossoverRate to set
195         */
196        public void setCrossoverRate(double crossoverRate) {
197                this.crossoverRate = crossoverRate;
198        }
199
200        /*
201         * (non-Javadoc)
202         * 
203         * @see org.opt4j.start.Opt4JModule#config()
204         */
205        @Override
206        public void config() {
207                bindIterativeOptimizer(EvolutionaryAlgorithm.class);
208                bind(CrossoverRate.class).to(ConstantCrossoverRate.class).in(SINGLETON);
209        }
210}
