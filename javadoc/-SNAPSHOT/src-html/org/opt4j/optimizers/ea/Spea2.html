044/**
045 * <p>
046 * The {@link Spea2}-Selector is a Java implementation of the
047 * <a href="http://e-collection.ethbib.ethz.ch/ecol-pool/incoll/incoll_324.pdf"
048 * >SPEA2-MOEA</a>, see "SPEA2: Improving the Strength Pareto Evolutionary
049 * Algorithm For Multiobjective Optimization, Eckart Zitzler, Marco Laumanns,
050 * and Lothar Thiele, In Evolutionary Methods for Design, Optimisation, and
051 * Control, pages 19&ndash;26, 2002.".
052 * </p>
053 * <p>
054 * The {@link Spea2}-Selector will not work properly if the {@link Objectives}
055 * are not fixed, i.e., the {@link Objectives} of an {@link Individual} change
056 * during the optimization process. This caused by the internal caching of
057 * distances, etc. In this case, it is recommended to use the {@link Nsga2}.
058 * </p>
059 * 
060 * @author lukasiewycz
061 * 
062 */
063public class Spea2 implements Selector {
064
065        protected final Rand random;
066        protected final int tournament;
067
068        protected final Map<Individual, Spea2IndividualSet> map = new LinkedHashMap<>();
069        protected final Set<Spea2IndividualSet> individualSets = new LinkedHashSet<>();
070
071        protected final LinkedList<Integer> freeIDs = new LinkedList<>();
072        protected double[][] distance;
073        protected boolean fitnessDirty = true;
074
075        /**
076         * A wrapper for multiple equal (based on their objectives)
077         * {@link Individual}s.
078         * 
079         * @author lukasiewycz
080         * 
081         */
082        private static class Spea2IndividualSet extends LinkedHashSet<Individual>
083                        implements Comparable<Spea2IndividualSet> {
084                private static final long serialVersionUID = 1L;
085                protected final int id;
086                protected int fitness;
087                protected int strength;
088                protected final Objectives objectives;
089                protected double nextDistance = 0;
090
091                Spea2IndividualSet(Individual individual, int id) {
092                        this.id = id;
093                        this.add(individual);
094                        this.objectives = individual.getObjectives();
095                        this.fitness = 0;
096                        this.strength = 0;
097                }
098
099                public Objectives getObjectives() {
100                        return objectives;
101                }
102
103                public int getFitness() {
104                        return fitness;
105                }
106
107                public void setFitness(int fitness) {
108                        this.fitness = fitness;
109                }
110
111                public int getStrength() {
112                        return strength;
113                }
114
115                public void setStrength(int strength) {
116                        this.strength = strength;
117                }
118
119                public Individual first() {
120                        assert (size() != 0);
121                        return iterator().next();
122                }
123
124                public int getId() {
125                        return id;
126                }
127
128                public double getNextDistance() {
129                        return nextDistance;
130                }
131
132                public void setNextDistance(double nextDistance) {
133                        this.nextDistance = nextDistance;
134                }
135
136                @Override
137                public int hashCode() {
138                        return id;
139                }
140
141                @Override
142                public boolean equals(Object obj) {
143                        Spea2IndividualSet other = (Spea2IndividualSet) obj;
144                        return (id == other.id);
145                }
146
147                public boolean dominates(Spea2IndividualSet individualSet) {
148                        return this.getObjectives().dominates(individualSet.getObjectives());
149                }
150
151                @Override
152                public int compareTo(Spea2IndividualSet o) {
153                        return o.fitness - this.fitness;
154                }
155
156        }

158        /**
159         * Constructs a {@link Spea2}-{@link Selector}.
160         * 
161         * @param tournament
162         *            the number of individuals that fight against each other to
163         *            become a parent
164         * @param random
165         *            the random number generator
166         */
167        @Inject
168        public Spea2(@Constant(value = "tournament", namespace = Spea2.class) int tournament, Rand random) {
169                this.tournament = tournament;
170                this.random = random;
171        }

173        /*
174         * (non-Javadoc)
175         * 
176         * @see org.opt4j.optimizer.ea.Selector#init(int)
177         */
178        @Override
179        public void init(int maxsize) {
180                for (int i = 0; i < maxsize; i++) {
181                        freeIDs.add(i);
182                }
183
184                distance = new double[maxsize][maxsize];
185        }

187        /*
188         * (non-Javadoc)
189         * 
190         * @see org.opt4j.optimizer.ea.Selector#getParents(int,
191         * java.util.Collection)
192         */
193        @Override
194        public Collection<Individual> getParents(int mu, Collection<Individual> population) {
195                update(population);
196
197                List<Individual> parents = new ArrayList<>();
198
199                List<Individual> candidates = new ArrayList<>(population);
200                int size = candidates.size();
201
202                for (int i = 0; i < mu; i++) {
203                        Individual winner = candidates.get(random.nextInt(size));
204
205                        for (int j = 0; j < tournament; j++) {
206                                Individual opponent = candidates.get(random.nextInt(size));
207
208                                Spea2IndividualSet wWinner = map.get(winner);
209                                Spea2IndividualSet wOpponent = map.get(opponent);
210
211                                double oFitness = wWinner.getFitness();
212                                double wFitness = wOpponent.getFitness();
213
214                                if (oFitness > wFitness || (winner == opponent)) {
215                                        winner = opponent;
216                                } else if (oFitness == wFitness) {
217
218                                        double oDist = getMinDistance(wOpponent);
219                                        double wDist = getMinDistance(wWinner);
220
221                                        if (oDist > wDist) {
222                                                winner = opponent;
223                                        }
224                                }
225                        }
226                        parents.add(winner);
227                }
228
229                assert (parents.size() == mu);
230
231                return parents;
232        }

234        /*
235         * (non-Javadoc)
236         * 
237         * @see org.opt4j.optimizer.ea.Selector#getLames(int, java.util.Collection)
238         */
239        @Override
240        public Collection<Individual> getLames(int lambda, Collection<Individual> population) {
241                update(population);
242
243                assert (lambda <= population.size());
244
245                Set<Individual> lames = new LinkedHashSet<>();
246
247                if (lambda > 0) {
248                        List<Spea2IndividualSet> dominated = getDominated();
249
250                        if (countIndividuals(dominated) >= lambda) {
251
252                                Collections.sort(dominated);
253                                List<Individual> lameCandidates = new ArrayList<>();
254
255                                int i = 0;
256                                while (lameCandidates.size() < lambda) {
257                                        lameCandidates.addAll(dominated.get(i));
258                                        i++;
259                                }
260                                lames.addAll(lameCandidates.subList(0, lambda));
261
262                        } else {
263
264                                for (Spea2IndividualSet w0 : dominated) {
265                                        lames.addAll(w0);
266                                }
267
268                                for (Individual individual : getLamesFromNonDominated(lambda - lames.size())) {
269                                        lames.add(individual);
270                                }
271                        }
272
273                }
274
275                assert (lames.size() == lambda);
276
277                return lames;
278        }

280        /**
281         * Returns a specific number of lames from the non-dominated
282         * {@link Individual}s.
283         * 
284         * @param count
285         *            the specified number
286         * @return a collection of the lame individuals
287         */
288        public Collection<Individual> getLamesFromNonDominated(int count) {
289                Set<Individual> set = new LinkedHashSet<>();
290
291                while (set.size() < count) {
292                        int maxsize = 0;
293
294                        List<Spea2IndividualSet> candidates = new ArrayList<>();
295                        for (Spea2IndividualSet individualSet : getNonDominated()) {
296                                if (individualSet.size() > maxsize) {
297                                        maxsize = individualSet.size();
298                                        candidates.clear();
299                                }
300                                if (individualSet.size() == maxsize) {
301                                        candidates.add(individualSet);
302                                }
303                        }
304
305                        if (candidates.size() <= (count - set.size())) {
306                                for (Spea2IndividualSet individualSet : candidates) {
307                                        Individual individual = individualSet.first();
308                                        remove(individual);
309                                        set.add(individual);
310                                }
311                        } else { // candidates.size() > (count - set.size())
312                                for (Spea2IndividualSet individualSet : getNearest(count - set.size(), candidates)) {
313                                        Individual individual = individualSet.first();
314                                        remove(individual);
315                                        set.add(individual);
316                                }
317
318                        }
319
320                }
321
322                assert (set.size() == count);
323
324                return set;
325        }

327        /**
328         * Returns n with nearest neighbor based on distances.
329         * 
330         * @param n
331         *            the number of required IndividualSets
332         * @param candidates
333         *            the candidate IndividualSets
334         * @return the n nearest neighbors
335         */
336        protected List<Spea2IndividualSet> getNearest(int n, Collection<Spea2IndividualSet> candidates) {
337
338                assert (candidates.size() > n);
339
340                List<Spea2IndividualSet> lames = new ArrayList<>();
341
342                Map<Spea2IndividualSet, List<Spea2IndividualSet>> orderedLists = getNearestPreOrder(candidates);
343
344                while (lames.size() < n) {
345                        List<Spea2IndividualSet> lcandidates = new ArrayList<>(orderedLists.keySet());
346
347                        int size = lcandidates.size();
348
349                        for (int k = 0; k < size - 1; k++) {
350                                double min = Double.MAX_VALUE;
351
352                                for (Spea2IndividualSet candidate : lcandidates) {
353                                        double value = distance(candidate, orderedLists.get(candidate).get(k));
354                                        min = Math.min(min, value);
355                                }
356
357                                for (Iterator<Spea2IndividualSet> it = lcandidates.iterator(); it.hasNext();) {
358                                        Spea2IndividualSet candidate = it.next();
359                                        double value = distance(candidate, orderedLists.get(candidate).get(k));
360                                        if (value > min) {
361                                                it.remove();
362                                        }
363                                }
364
365                                if (lcandidates.size() == 1) {
366                                        break;
367                                }
368                        }
369
370                        Spea2IndividualSet lame = lcandidates.get(0);
371                        lames.add(lame);
372                        orderedLists.remove(lame);
373
374                        for (List<Spea2IndividualSet> list : orderedLists.values()) {
375                                list.remove(lame);
376                        }
377
378                }
379
380                assert (lames.size() == n);
381
382                return lames;
383        }

385        /**
386         * Helper function to preorder the candidates for getNearest() operation
387         * 
388         * @param candidates
389         *            the candidate IndividualSets
390         * @return the preordered IndividualSets
391         */
392        protected Map<Spea2IndividualSet, List<Spea2IndividualSet>> getNearestPreOrder(
393                        Collection<Spea2IndividualSet> candidates) {
394                Map<Spea2IndividualSet, List<Spea2IndividualSet>> orderedLists = new LinkedHashMap<>();
395
396                for (Spea2IndividualSet w0 : candidates) {
397                        List<Spea2IndividualSet> list = new ArrayList<>();
398                        for (Spea2IndividualSet w1 : candidates) {
399                                if (w0 != w1) {
400                                        w1.setNextDistance(distance(w0, w1));
401                                        list.add(w1);
402                                }
403                        }
404                        Collections.sort(list, new Comparator<Spea2IndividualSet>() {
405                                @Override
406                                public int compare(Spea2IndividualSet o1, Spea2IndividualSet o2) {
407                                        double v = o1.getNextDistance() - o2.getNextDistance();
408                                        if (v < 0) {
409                                                return -1;
410                                        } else if (v > 0) {
411                                                return 1;
412                                        } else {
413                                                return 0;
414                                        }
415                                }
416
417                        });
418                        orderedLists.put(w0, list);
419                }
420                return orderedLists;
421        }

423        protected double getMinDistance(Spea2IndividualSet w0) {
424                double min = Double.MAX_VALUE;
425                for (Spea2IndividualSet w1 : individualSets) {
426                        if (w0 != w1) {
427                                min = Math.min(min, distance(w0, w1));
428                        }
429                }
430                return min;
431        }

433        /**
434         * Update with current population.
435         * 
436         * @param population
437         *            the current population
438         */
439        protected void update(Collection<Individual> population) {
440                Set<Individual> popSet = new LinkedHashSet<>(population);
441
442                if (!popSet.equals(map.keySet())) {
443
444                        Set<Individual> adds = new LinkedHashSet<>(popSet);
445                        adds.removeAll(map.keySet());
446
447                        Set<Individual> removes = new LinkedHashSet<>(map.keySet());
448                        removes.removeAll(popSet);
449
450                        for (Individual individual : removes) {
451                                remove(individual);
452                        }
453
454                        for (Individual individual : adds) {
455                                add(individual);
456                        }
457                }
458
459                assert (population.size() == map.size());
460
461                if (fitnessDirty) {
462                        calculateFitness();
463                        fitnessDirty = false;
464                }
465
466        }

468        /**
469         * Return the distance of two {@code Spea2IndividualSet}s.
470         * 
471         * @param w0
472         *            first set
473         * @param w1
474         *            second set
475         * @return the distance
476         */
477        protected double distance(Spea2IndividualSet w0, Spea2IndividualSet w1) {
478                return distance[w0.getId()][w1.getId()];
479        }
480
481        /**
482         * Add a new {@link Individual}.
483         * 
484         * @param individual
485         *            the individual to add
486         */
487        protected void add(Individual individual) {
488                int id0 = freeIDs.removeFirst();
489                Spea2IndividualSet w0 = new Spea2IndividualSet(individual, id0);
490
491                Spea2IndividualSet eq = null;
492
493                for (Spea2IndividualSet w1 : individualSets) {
494                        int id1 = w1.getId();
495                        double dist = calculateDistance(w0, w1);
496                        if (dist == 0.0) {
497                                eq = w1;
498                                break;
499                        }
500                        distance[id0][id1] = dist;
501                        distance[id1][id0] = dist;
502                }
503                if (eq != null) {
504                        freeIDs.add(id0);
505                        w0 = eq;
506                        w0.add(individual);
507                } else {
508                        distance[id0][id0] = 0.0;
509                        individualSets.add(w0);
510                }
511
512                map.put(individual, w0);
513                fitnessDirty = true;
514        }
515
516        /**
517         * Remove an {@link Individual}.
518         * 
519         * @param individual
520         *            the individual to remove
521         */
522        protected void remove(Individual individual) {
523                Spea2IndividualSet individualSet = map.remove(individual);
524
525                if (individualSet.size() == 1) {
526                        individualSets.remove(individualSet);
527                        freeIDs.add(individualSet.getId());
528                } else {
529                        individualSet.remove(individual);
530                }
531
532                fitnessDirty = true;
533        }

535        /**
536         * Calculate the distance between two {@code Spea2IndividualSet}s.
537         * 
538         * @param w0
539         *            the first set
540         * @param w1
541         *            the second set
542         * @return the distance
543         */
544        protected double calculateDistance(Spea2IndividualSet w0, Spea2IndividualSet w1) {
545                return w0.getObjectives().distance(w1.getObjectives());
546        }
547
548        /**
549         * Calculate the fitness.
550         */
551        protected void calculateFitness() {
552                for (Spea2IndividualSet individual : individualSets) {
553                        int s = 0;
554                        for (Spea2IndividualSet other : individualSets) {
555                                if (individual != other && individual.dominates(other)) {
556                                        s += other.size();
557                                }
558                        }
559                        individual.setStrength(s);
560                }
561
562                for (Spea2IndividualSet individual : individualSets) {
563                        int f = 0;
564                        for (Spea2IndividualSet other : individualSets) {
565                                if (individual != other && other.dominates(individual)) {
566                                        f += other.getStrength() * other.size();
567                                }
568                        }
569                        individual.setFitness(f);
570                }
571        }
572
573        /**
574         * Returns all dominated {@code Spea2IndividualSet}s (fitness &gt; 0).
575         * 
576         * @return all dominated IndividualSets
577         */
578        protected List<Spea2IndividualSet> getDominated() {
579                List<Spea2IndividualSet> dominated = new ArrayList<>();
580
581                for (Spea2IndividualSet w0 : individualSets) {
582                        if (w0.getFitness() > 0.0) {
583                                dominated.add(w0);
584                        }
585                }
586
587                return dominated;
588        }
589
590        /**
591         * Returns all non-dominated {@code Spea2IndividualSet}s (fitness == 0).
592         * 
593         * @return all non-dominated individualSets
594         */
595        protected List<Spea2IndividualSet> getNonDominated() {
596                List<Spea2IndividualSet> dominated = new ArrayList<>();
597
598                for (Spea2IndividualSet w0 : individualSets) {
599                        if (w0.getFitness() == 0.0) {
600                                dominated.add(w0);
601                        }
602                }
603
604                return dominated;
605        }

607        /**
608         * Returns the total number of {@link Individual}s in a collection of
609         * {@code Spea2IndividualSet}s.
610         * 
611         * @param collection
612         *            the collection of IndividualSets
613         * @return the total number of Individuals
614         */
615        private int countIndividuals(Collection<Spea2IndividualSet> collection) {
616                int c = 0;
617                for (Spea2IndividualSet w : collection) {
618                        c += w.size();
619                }
620                return c;
621        }
