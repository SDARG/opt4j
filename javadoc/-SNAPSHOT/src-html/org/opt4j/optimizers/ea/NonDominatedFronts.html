001package org.opt4j.optimizers.ea;
002
003import java.util.ArrayList;
004import java.util.Collection;
005import java.util.HashMap;
006import java.util.List;
007import java.util.Map;
008
009import org.opt4j.core.Individual;
010import org.opt4j.core.Objectives;
011
012/**
013 * The {@link NonDominatedFronts} sorts each evaluated {@link Individual} into
014 * fronts based on the number of other individuals it is dominated by. The first
015 * front consists of points that are not dominated at all and so on.
016 * 
017 * @author Fedor Smirnov
018 *
019 */
020public class NonDominatedFronts {
021
022        protected final List<Collection<Individual>> fronts;
023
024        /**
025         * Creates the {@link NonDominatedFronts} for the given collection of
026         * {@link Individual}s.
027         * 
028         * @param individuals
029         *            the {@link Individual}s that are sorted into non dominated
030         *            fronts
031         */
032        public NonDominatedFronts(Collection<Individual> individuals) {
033                this.fronts = generateFronts(individuals);
034        }
035
036        /**
037         * Sorts the given {@link Individual}s into non-dominated fronts.
038         * 
039         * @param individuals
040         *            the collection of {@link Individual}s that shall be sorted
041         * @return the non-dominated fronts
042         */
043        protected List<Collection<Individual>> generateFronts(Collection<Individual> individuals) {
044                List<Collection<Individual>> fronts = new ArrayList<>();
045                // Assigns an id to each individual that corresponds to its index in an
046                // array.
047                Map<Individual, Integer> indexMap = new HashMap<>();
048                int index = 0;
049                for (Individual individual : individuals) {
050                        indexMap.put(individual, index++);
051                }
052                // Initializes a map where an individual is assigned to the individuals
053                // that it dominates.
054                Map<Individual, List<Individual>> dominatedIndividualsMap = new HashMap<>();
055                // Creates an array where for each individual, the number of individuals
056                // that dominate it is stored.
057                int[] dominatingIndividualNumber = new int[individuals.size()];
058                for (Individual e : individuals) {
059                        dominatedIndividualsMap.put(e, new ArrayList<>());
060                        dominatingIndividualNumber[indexMap.get(e)] = 0;
061                }
062                determineDomination(individuals, dominatedIndividualsMap, dominatingIndividualNumber, indexMap);
063                // The first front consists of individuals that are dominated by zero
064                // other individuals.
065                List<Individual> f1 = new ArrayList<>();
066                for (Individual i : individuals) {
067                        if (dominatingIndividualNumber[indexMap.get(i)] == 0) {
068                                f1.add(i);
069                        }
070                }
071                fronts.add(f1);
072                List<Individual> currentFront = f1;
073                // Creates the subsequent fronts. Front f_i is made up by individuals
074                // that
075                // are not dominated if all individuals from fronts f_j with j < i are
076                // removed.
077                while (!currentFront.isEmpty()) {
078                        List<Individual> nextFront = getNextFront(currentFront, dominatedIndividualsMap, dominatingIndividualNumber,
079                                        indexMap);
080                        if (!nextFront.isEmpty()) {
081                                fronts.add(nextFront);
082                        }
083                        currentFront = nextFront;
084                }
085                return fronts;
086        }
087
088        /**
089         * Returns the front at the specified index.
090         * 
091         * @param index
092         *            the specified index
093         * @return the front at the specified index
094         */
095        public Collection<Individual> getFrontAtIndex(int index) {
096                return fronts.get(index);
097        }
098
099        /**
100         * Returns the number of non-dominated fronts.
101         * 
102         * @return the number of non-dominated fronts
103         */
104        public int getFrontNumber() {
105                return fronts.size();
106        }
107
108        /**
109         * Finds the next non-dominated front by processing the current
110         * non-dominated front. The {@link Individual}s found therein are removed
111         * from consideration. The individuals that are then not dominated form the
112         * next non-dominated front.
113         * 
114         * @param currentFront
115         *            the list of individuals forming the current non-dominated
116         *            front
117         * @param dominatedIndividualsMap
118         *            map mapping an individual on the collection of individuals
119         *            that it dominates
120         * @param dominatingIndividualNumber
121         *            an array where the number of dominating individuals is stored
122         *            for each individual
123         * @param individual2IndexMap
124         *            a map storing the indices of the individuals used to access
125         *            the dominatingIndividualNumber
126         * @return the list of individuals forming the next non-dominated front
127         */
128        protected List<Individual> getNextFront(List<Individual> currentFront,
129                        Map<Individual, List<Individual>> dominatedIndividualsMap, int[] dominatingIndividualNumber,
130                        Map<Individual, Integer> individual2IndexMap) {
131                List<Individual> nextFront = new ArrayList<>();
132                for (Individual dominant : currentFront) {
133                        for (Individual dominated : dominatedIndividualsMap.get(dominant)) {
134                                dominatingIndividualNumber[individual2IndexMap.get(dominated)]--;
135                                if (dominatingIndividualNumber[individual2IndexMap.get(dominated)] == 0) {
136                                        nextFront.add(dominated);
137                                }
138                        }
139                }
140                return nextFront;
141        }
142
143        /**
144         * Compares all possible {@link Individual} pairs. For each individual,
145         * stores 1) the number of individuals it is dominated by and 2) the set of
146         * individuals it dominates.
147         * 
148         * @param individuals
149         *            a collection of individuals
150         * @param dominatedIndividualsMap
151         *            A map that is filled during the execution of the method. Each
152         *            individual is mapped onto the set of individuals that are
153         *            dominated by this individual.
154         * @param dominatingIndividualNumber
155         *            An integer array (initialized with zeros) that is filled
156         *            during the execution of this method. Each individual is
157         *            associated with an entry of this array. The integer therein is
158         *            the number of individuals this individual is dominated by.
159         * @param individual2IndexMap
160         *            a map mapping each individual onto its index in the
161         *            dominatingIndividualNumber - array
162         */
163        protected void determineDomination(Collection<Individual> individuals,
164                        Map<Individual, List<Individual>> dominatedIndividualsMap, int[] dominatingIndividualNumber,
165                        Map<Individual, Integer> individual2IndexMap) {
166                List<Individual> individualList = new ArrayList<>(individuals);
167                // compare each individual with each other individual
168                for (int i = 0; i < individualList.size(); i++) {
169                        for (int j = i + 1; j < individualList.size(); j++) {
170                                Individual p = individualList.get(i);
171                                Individual q = individualList.get(j);
172                                Objectives po = p.getObjectives();
173                                Objectives qo = q.getObjectives();
174                                if (po.dominates(qo)) {
175                                        dominatedIndividualsMap.get(p).add(q);
176                                        dominatingIndividualNumber[individual2IndexMap.get(q)]++;
177                                } else if (qo.dominates(po)) {
178                                        dominatedIndividualsMap.get(q).add(p);
179                                        dominatingIndividualNumber[individual2IndexMap.get(p)]++;
180                                }
181                                // Neither of the two points dominates the other one, so that
182                                // neither the array
183                                // keeping track of the domination number nor the map containing
184                                // the dominating
185                                // individuals has to be adjusted. Nothing is done in this case.
186                        }
187                }
188        }
189}
