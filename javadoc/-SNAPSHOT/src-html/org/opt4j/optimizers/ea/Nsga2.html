023package org.opt4j.optimizers.ea;
024
025import java.util.ArrayList;
026import java.util.Collection;
027import java.util.Collections;
028import java.util.HashMap;
029import java.util.List;
030import java.util.Map;
031import java.util.Random;
032
033import org.opt4j.core.Individual;
034import org.opt4j.core.common.archive.FrontDensityIndicator;
035import org.opt4j.core.common.random.Rand;
036import org.opt4j.core.start.Constant;
037
038import com.google.inject.Inject;
039
040/**
041 * The {@link Nsga2} {@link Selector}, see "A Fast Elitist Non-Dominated Sorting
042 * Genetic Algorithm for Multi-Objective Optimization: NSGA-II, K. Deb, Samir
043 * Agrawal, Amrit Pratap, and T. Meyarivan, Parallel MockProblem Solving from
044 * Nature, 2000".
045 * 
046 * @see Nsga2Module
047 * @author lukasiewycz, noorshams
048 * 
049 */
050public class Nsga2 implements Selector {
051
052        protected final Random random;
053        protected final int tournament;
054        protected final FrontDensityIndicator indicator;
055
056        /**
057         * Constructs a {@link Nsga2} {@link Selector}.
058         * 
059         * @param random
060         *            the random number generator
061         * @param tournament
062         *            the tournament value
063         */
064        @Inject
065        public Nsga2(Rand random, @Constant(value = "tournament", namespace = Nsga2.class) int tournament,
066                        FrontDensityIndicator indicator) {
067                this.random = random;
068                this.tournament = tournament;
069                this.indicator = indicator;
070        }
071
072        /*
073         * (non-Javadoc)
074         * 
075         * @see org.opt4j.optimizer.ea.Selector#init(int)
076         */
077        @Override
078        public void init(int maxsize) {
079                // do nothing
080        }
081
082        /*
083         * (non-Javadoc)
084         * 
085         * @see org.opt4j.optimizer.ea.Selector#getParents(int,
086         * java.util.Collection)
087         */
088        @Override
089        public Collection<Individual> getParents(int mu, Collection<Individual> population) {
090                List<Individual> all = new ArrayList<>(population);
091                List<Individual> parents = new ArrayList<>();
092
093                NonDominatedFronts fronts = new NonDominatedFronts(all);
094                Map<Individual, Integer> rank = getRank(fronts);
095                Map<Individual, Double> distance = new HashMap<>();
096
097                final int size = all.size();
098
099                for (int i = 0; i < mu; i++) {
100                        Individual winner = all.get(random.nextInt(size));
101
102                        for (int t = 0; t < tournament; t++) {
103                                Individual opponent = all.get(random.nextInt(size));
104                                if (rank.get(opponent) < rank.get(winner) || opponent == winner) {
105                                        winner = opponent;
106                                } else if (rank.get(opponent) == rank.get(winner)) {
107                                        // The winner is determined considering the crowding
108                                        // distance
109
110                                        if (!distance.containsKey(winner)) {
111                                                List<Individual> front = new ArrayList<>(fronts.getFrontAtIndex(rank.get(winner)));
112                                                distance.putAll(indicator.getDensityValues(front));
113                                        }
114
115                                        // Opponent wins, if it has a better crowding distance
116                                        if (distance.get(opponent) > distance.get(winner)) {
117                                                winner = opponent;
118                                        }
119
120                                }
121                        }
122
123                        parents.add(winner);
124                }
125
126                return parents;
127        }
128
129        /*
130         * (non-Javadoc)
131         * 
132         * @see org.opt4j.optimizer.ea.Selector#getLames(int, java.util.Collection)
133         */
134        @Override
135        public Collection<Individual> getLames(int size, Collection<Individual> population) {
136                List<Individual> lames = new ArrayList<>();
137
138                NonDominatedFronts fronts = new NonDominatedFronts(population);
139                for (int i = fronts.getFrontNumber() - 1; i >= 0; i--) {
140                        List<Individual> front = new ArrayList<>(fronts.getFrontAtIndex(i));
141                        if (lames.size() + front.size() < size) {
142                                lames.addAll(front);
143                        } else {
144                                final Map<Individual, Double> density = indicator.getDensityValues(front);
145                                Collections.sort(front, (Individual o1, Individual o2) -> density.get(o1).compareTo(density.get(o2)));
146                                lames.addAll(front.subList(0, size - lames.size()));
147                        }
148                }
149                return lames;
150        }
151
152        /**
153         * Determine the ranks of fronts.
154         * 
155         * @param fronts
156         *            the fronts
157         * @return the ranks
158         */
159        protected Map<Individual, Integer> getRank(NonDominatedFronts fronts) {
160                Map<Individual, Integer> ranks = new HashMap<>();
161                for (int i = 0; i < fronts.getFrontNumber(); i++) {
162                        for (Individual p : fronts.getFrontAtIndex(i)) {
163                                ranks.put(p, i);
164                        }
165                }
166                return ranks;
167        }
168}
