018package org.opt4j.optimizers.ea;
019
020import static org.opt4j.core.config.annotations.Citation.PublicationMonth.UNKNOWN;
021
022import org.opt4j.core.common.archive.FrontDensityIndicator;
023import org.opt4j.core.config.annotations.Citation;
024import org.opt4j.core.config.annotations.Info;
025import org.opt4j.core.start.Constant;
026
027/**
028 * Module for the S-Metric Selection ({@link Selector}) based on the {@link Hypervolume} contribution, see Emmerich et
029 * al. 2005.
030 * 
031 * @see Hypervolume
032 * @author lukasiewycz
033 * @author Ramin Etemaadi
034 * 
035 */
036@Info("SMS-EMOA: Multiobjective selection based on dominated hypervolume")
037@Citation(authors = "Michael Emmerich, Nicola Beume, and Boris Naujoks", title = "An EMO Algorithm Using the Hypervolume Measure as Selection Criterion", journal = "Evolutionary Multi-Criterion Optimization (EMO)", pageFirst = 62, pageLast = 76, year = 2005, month = UNKNOWN)
038public class SMSModule extends SelectorModule {
039
040        @Info("The offset value")
041        @Constant(value = "offset", namespace = Hypervolume.class)
042        protected double offset = 1.0;
043
044        @Info("The tournament value")
045        @Constant(value = "tournament", namespace = Nsga2.class)
046        protected int tournament = 0;
047
048        /**
049         * Returns the tournament value.
050         * 
051         * @see #setTournament
052         * @return the tournament value
053         */
054        public int getTournament() {
055                return tournament;
056        }
057
058        /**
059         * Sets the tournament value.
060         * 
061         * @see #getTournament
062         * @param tournament
063         *            the tournament to set
064         */
065        public void setTournament(int tournament) {
066                this.tournament = tournament;
067        }
068
069        /**
070         * Returns the offset value.
071         * 
072         * @return the offset value
073         */
074        public double getOffset() {
075                return offset;
076        }
077
078        /**
079         * Sets the offset value.
080         * 
081         * @param offset
082         *            the offset to set
083         */
084        public void setOffset(double offset) {
085                this.offset = offset;
086        }
087
088        /*
089         * (non-Javadoc)
090         * 
091         * @see org.opt4j.start.Opt4JModule#config()
092         */
093        @Override
094        public void config() {
095                bindSelector(Nsga2.class);
096                bind(FrontDensityIndicator.class).to(Hypervolume.class);
097        }
098
099}
