023package org.opt4j.viewer;
024
025import org.opt4j.core.config.Icons;
026import org.opt4j.core.config.annotations.Icon;
027import org.opt4j.core.config.annotations.Info;
028import org.opt4j.core.start.Constant;
029import org.opt4j.core.start.Opt4J;
030import org.opt4j.core.start.Progress;
031import org.opt4j.viewer.Viewer.CloseEvent;
032
033import ptolemy.plot.DefaultFonts;
034
035/**
036 * The {@link ViewerModule} configures the optimization process viewer.
037 * 
038 * @author lukasiewycz
039 * 
040 */
041@Icon(Icons.APPLICATION)
042@Info("A graphical viewer to observe the optimization process.")
043public class ViewerModule extends VisualizationModule {
044
045        static {
046                // initialize ptolemy plot once the viewer module is on the classpath
047                if (DefaultFonts.LABElFONT == null) {
048                        throw new IllegalStateException();
049                }
050        }
051
052        @Info("The title of the viewer frame.")
053        @Constant(value = "title", namespace = Viewer.class)
054        protected String title = "Opt4J " + Opt4J.getVersion() + " Viewer";
055
056        @Info("Event for the optimization process if the viewer is closed.")
057        @Constant(value = "closeEvent", namespace = Viewer.class)
058        protected CloseEvent closeEvent = CloseEvent.STOP;
059
060        @Info("Close viewer automatically when the optimization process stops.")
061        @Constant(value = "closeOnStop", namespace = Viewer.class)
062        protected boolean closeOnStop = false;
063
064        /**
065         * Returns {@code true} if the GUI is automatically closed if the
066         * optimization process has stopped.
067         * 
068         * @return {@code true} if the GUI is automatically closed on optimization
069         *         stop
070         */
071        public boolean isCloseOnStop() {
072                return closeOnStop;
073        }
074
075        /**
076         * Sets the option for automatically closing the GUI if the optimization
077         * stops.
078         * 
079         * @param closeOnStop
080         *            the closeOnStop to set
081         */
082        public void setCloseOnStop(boolean closeOnStop) {
083                this.closeOnStop = closeOnStop;
084        }
085
086        /**
087         * Returns the event when the GUI is closed.
088         * 
089         * @return the closeEvent
090         */
091        public CloseEvent getCloseEvent() {
092                return closeEvent;
093        }
094
095        /**
096         * Sets the event when the GUI is closed.
097         * 
098         * @param closeEvent
099         *            the closeEvent to set
100         */
101        public void setCloseEvent(CloseEvent closeEvent) {
102                this.closeEvent = closeEvent;
103        }
104
105        /**
106         * Returns the title of the GUI frame.
107         * 
108         * @return the title of the GUI frame
109         */
110        public String getTitle() {
111                return title;
112        }
113
114        /**
115         * Sets the title of the GUI frame.
116         * 
117         * @param title
118         *            the title of the GUI frame
119         */
120        public void setTitle(String title) {
121                this.title = title;
122        }
123
124        /*
125         * (non-Javadoc)
126         * 
127         * @see com.google.inject.AbstractModule#configure()
128         */
129        @Override
130        public void config() {
131                bind(Viewer.class).in(SINGLETON);
132                addOptimizerStateListener(Viewer.class);
133
134                bind(StatusBar.class).in(SINGLETON);
135                addOptimizerStateListener(StatusBar.class);
136                addOptimizerIterationListener(StatusBar.class);
137
138                bind(Viewport.class).in(SINGLETON);
139
140                addOptimizerIterationListener(Progress.class);
141                addOptimizerStateListener(ControlButtons.class);
142
143                addOptimizerIterationListener(ConvergencePlotData.class);
144
145                addToolBarService(ControlToolBarService.class);
146                addToolBarService(ViewsToolBarService.class);
147        }
148
149}
