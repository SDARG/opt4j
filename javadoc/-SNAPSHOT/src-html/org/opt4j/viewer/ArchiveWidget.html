105        /**
106         * The {@link TableMouseListener} that listens to right and double click of
107         * the table item.
108         * 
109         * @author lukasiewycz
110         * 
111         */
112        class TableMouseListener extends MouseAdapter {
113                private void reservTableShow(MouseEvent e, boolean released) {
114                        if (table.isEnabled()) {
115                                Point p = new Point(e.getX(), e.getY());
116                                int row = table.rowAtPoint(p);
117
118                                if (row >= 0 && row < table.getRowCount()) {
119                                        Individual individual = swtIndividuals.get(row);
120
121                                        if (e.isPopupTrigger()) {
122                                                table.getSelectionModel().setSelectionInterval(row, row);
123                                                JPopupMenu menu = new JPopupMenu();
124                                                for (IndividualMouseListener listener : individualMouseListeners) {
125                                                        listener.onPopup(individual, table, p, menu);
126                                                }
127                                                if (menu.getComponentCount() > 0) {
128                                                        menu.show(table, p.x, p.y);
129                                                }
130                                        } else if (e.getClickCount() == 2 && released) {
131                                                for (IndividualMouseListener listener : individualMouseListeners) {
132                                                        listener.onDoubleClick(individual, table, p);
133                                                }
134                                        }
135                                }
136                        }
137                }
138
139                @Override
140                public void mouseReleased(MouseEvent e) {
141                        reservTableShow(e, true);
142                }
143
144                @Override
145                public void mousePressed(MouseEvent e) {
146                        reservTableShow(e, false);
147                }
148        }

150        /**
151         * The model of the table.
152         * 
153         * @author lukasiewycz
154         * 
155         */
156        protected class Model extends AbstractTableModel {
157                private static final long serialVersionUID = 1L;
158
159                @Override
160                public String getColumnName(int col) {
161                        if (col == 0) {
162                                return INDEX;
163                        } else if (col == 1) {
164                                return "Individual";
165                        }
166                        final int index = col - OFFSET;
167                        final Objective objective = objectives.get(index);
168
169                        return objective.getName() + " (" + objective.getSign() + ")";
170                }
171
172                @Override
173                public int getColumnCount() {
174                        return OFFSET + objectives.size();
175                }
176
177                @Override
178                public int getRowCount() {
179                        return swtIndividuals.size();
180                }
181
182                @Override
183                public Object getValueAt(final int row, final int col) {
184
185                        try {
186                                final Individual individual = swtIndividuals.get(row);
187
188                                if (individual != null) {
189                                        if (col == 0) {
190                                                return row + 1;
191                                        } else if (col == 1) {
192                                                return individual.getPhenotype();
193                                        } else if (individual.getState() == Individual.State.EVALUATED) {
194
195                                                final int index = col - OFFSET;
196                                                final Objectives o = individual.getObjectives();
197                                                final Objective objective = objectives.get(index);
198
199                                                return o.get(objective).getValue();
200                                        }
201                                }
202                        } catch (IndexOutOfBoundsException e) {
203                        } catch (NullPointerException e) {
204
205                        }
206                        return null;
207                }
208        }

322        protected final JTable getTable() {
323                Model model = new Model();
324                QTable table = new QTable(model);

326
327                final TableColumnModel columnModel = table.getColumnModel();

333        protected void sortIndividiualMouseListeners() {
334                List<IndividualMouseListener> list = new ArrayList<>();
335                list.addAll(individualMouseListeners);
336                Collections.sort(list, new ToolBarOrderComparator<>());
337                individualMouseListeners.clear();
338                individualMouseListeners.addAll(list);
339        }

351        protected boolean updateOrder() {
352                if (table.getColumnCount() != objectives.size() + OFFSET) {
353                        return false;
354                }
355
356                List<Objective> list = new ArrayList<>();
357                TableColumnModel model = table.getColumnModel();
358
359                for (int i = 0; i < model.getColumnCount(); i++) {
360                        int index = model.getColumn(i).getModelIndex();
361                        if (index >= OFFSET) {
362                                Objective o = objectives.get(index - OFFSET);
363                                list.add(o);
364                        }
365                }
366
367                if (!list.equals(order)) {
368                        order.clear();
369                        order.addAll(list);
370                        return true;
371                }
372                return false;
373        }

375        /**
376         * Repaints the table.
377         */
378        protected void paint() {
379                task.execute(new Runnable() {
380                        @Override
381                        public void run() {
382                                final List<Individual> temp = new ArrayList<>();
383                                temp.addAll(archive);
384                                Collections.sort(temp, new Comp());
385
386                                SwingUtilities.invokeLater(new Runnable() {
387                                        @Override
388                                        public void run() {
389                                                swtIndividuals = temp;
390                                                sizeLabel.setText(Integer.toString(swtIndividuals.size()));
391
392                                                if (table != null) {
393                                                        if (table.getColumnCount() != table.getModel().getColumnCount()) {
394                                                                table.createDefaultColumnsFromModel();
395                                                        }
396                                                        table.revalidate();
397                                                        table.repaint();
398                                                }
399                                        }
400                                });
401                        }
402                });
403        }
404
405        protected class Comp implements Comparator<Individual> {
406
407                @Override
408                public int compare(Individual arg0, Individual arg1) {
409                        if (arg0 == null) {
410                                if (arg1 == null) {
411                                        return 0;
412                                }
413                                return 1;
414                        } else if (arg1 == null) {
415                                return -1;
416                        }
417
418                        Objectives obj0 = arg0.getObjectives();
419                        Objectives obj1 = arg1.getObjectives();
420
421                        for (Objective o : order) {
422                                assert obj0.get(o) != null;
423                                assert obj1.get(o) != null;
424
425                                Double v0 = obj0.get(o).getDouble();
426                                Double v1 = obj1.get(o).getDouble();
427
428                                int c = 0;
429
430                                if (v0 == null) {
431                                        if (v1 == null) {
432                                                continue;
433                                        }
434                                        c = 1;
435                                } else if (v1 == null) {
436                                        c = -1;
437                                } else {
438                                        c = v0.compareTo(v1);
439                                }
440
441                                if (c != 0) {
442                                        if (o.getSign() == Sign.MIN) {
443                                                return c;
444                                        }
445                                        return -c;
446                                }
447                        }
448                        return 0;
449                }
450
451        }
