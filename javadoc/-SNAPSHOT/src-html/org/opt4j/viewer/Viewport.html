023package org.opt4j.viewer;
024
025import java.awt.BorderLayout;
026import java.awt.Color;
027import java.beans.PropertyVetoException;
028import java.util.HashMap;
029import java.util.Map;
030
031import javax.swing.Icon;
032import javax.swing.JComponent;
033import javax.swing.JDesktopPane;
034import javax.swing.JInternalFrame;
035import javax.swing.JPanel;
036import javax.swing.SwingUtilities;
037import javax.swing.event.InternalFrameAdapter;
038import javax.swing.event.InternalFrameEvent;
039
040import org.opt4j.core.config.Icons;
041
042/**
043 * The {@link Viewport} is a desktop for {@link Widget}s.
044 * 
045 * @see Widget
046 * @author lukasiewycz
047 * 
048 */
049public class Viewport {
050
051        protected final JDesktopPane desktop = new JDesktopPane();
052
053        protected final JPanel panel = new JPanel();
054
055        protected final static int OFFSET = 30;
056
057        protected final Map<Widget, JInternalFrame> widgets = new HashMap<>();
058
059        /**
060         * Initialization. This method has to be called once after construction.
061         */
062        public void init() {
063                desktop.setDragMode(JDesktopPane.OUTLINE_DRAG_MODE);
064                panel.setLayout(new BorderLayout());
065                panel.add(desktop, BorderLayout.CENTER);
066
067                desktop.setBackground(Color.WHITE);
068        }
069
070        /**
071         * Adds a widget.
072         * 
073         * @param widget
074         *            the widget to be added
075         */
076        public void addWidget(final Widget widget) {
077                final JInternalFrame frame;
078
079                if (!widgets.keySet().contains(widget)) {
080                        frame = createInternalFrame(widget);
081                        addToDesktop(frame);
082                } else {
083                        frame = widgets.get(widget);
084                }
085                SwingUtilities.invokeLater(new Runnable() {
086                        @Override
087                        public void run() {
088                                if (frame.isIcon()) {
089                                        try {
090                                                frame.setIcon(false);
091                                        } catch (PropertyVetoException e) {
092                                                // ignore
093                                        }
094                                }
095                                frame.setVisible(true);
096                                frame.moveToFront();
097                        }
098                });
099        }
100
101        /**
102         * Creates a widget as a {@link JInternalFrame}.
103         * 
104         * @param widget
105         *            the widget
106         * @return the constructed JInternalFrame
107         */
108        protected JInternalFrame createInternalFrame(final Widget widget) {
109                final JPanel panel = widget.getPanel();
110
111                final String frameTitle;
112                final Icon frameIcon;
113
114                boolean resizable = true;
115                boolean closable = true;
116                boolean maximizable = true;
117                boolean iconifiable = true;
118
119                WidgetParameters parameters = widget.getClass().getAnnotation(WidgetParameters.class);
120                if (parameters != null) {
121                        frameTitle = parameters.title();
122                        if (!parameters.icon().equals("")) {
123                                frameIcon = Icons.getIcon(parameters.icon());
124                        } else {
125                                frameIcon = null;
126                        }
127                        resizable = parameters.resizable();
128                        closable = parameters.closable();
129                        maximizable = parameters.maximizable();
130                        iconifiable = parameters.iconifiable();
131                } else {
132                        frameTitle = null;
133                        frameIcon = null;
134                }
135
136                final JInternalFrame inFrame = new JInternalFrame(frameTitle, resizable, closable, maximizable, iconifiable);
137                widget.init(this);
138
139                SwingUtilities.invokeLater(new Runnable() {
140                        @Override
141                        public void run() {
142                                inFrame.addInternalFrameListener(new InternalFrameAdapter() {
143                                        @Override
144                                        public void internalFrameClosed(InternalFrameEvent arg0) {
145                                                widgets.remove(widget);
146                                        }
147                                });
148                                inFrame.setFrameIcon(frameIcon);
149                                inFrame.getContentPane().setLayout(new BorderLayout());
150                                inFrame.getContentPane().add(panel, BorderLayout.CENTER);
151
152                                int windows = desktop.getAllFrames().length;
153                                windows = windows % 6;
154                                inFrame.setLocation(OFFSET * windows, OFFSET * windows);
155                        }
156                });
157
158                widgets.put(widget, inFrame);
159
160                return inFrame;
161        }
162
163        /**
164         * Adds a {@link JInternalFrame}.
165         * 
166         * @param inFrame
167         *            the JInternalFrame to be added
168         */
169        public void addToDesktop(final JInternalFrame inFrame) {
170                SwingUtilities.invokeLater(new Runnable() {
171                        @Override
172                        public void run() {
173                                inFrame.pack();
174                                desktop.add(inFrame);
175                        }
176                });
177        }
178
179        /**
180         * Returns the {@link JInternalFrame} of a {@link Widget} if it exists and
181         * {@code null} otherwise.
182         * 
183         * @param widget
184         *            the widget
185         * @return the internal frame
186         */
187        public JInternalFrame getInternalFrame(Widget widget) {
188                return widgets.get(widget);
189        }
190
191        /**
192         * Returns the component.
193         * 
194         * @return the component
195         */
196        public JComponent get() {
197                return panel;
198        }
199
200}
