023package org.opt4j.optimizers.ea;
024
025import java.util.ArrayList;
026import java.util.Arrays;
027import java.util.Collection;
028import java.util.Random;
029
030import org.opt4j.core.Genotype;
031import org.opt4j.core.Individual;
032import org.opt4j.core.IndividualFactory;
033import org.opt4j.core.common.random.Rand;
034import org.opt4j.operators.copy.Copy;
035import org.opt4j.operators.crossover.Crossover;
036import org.opt4j.operators.crossover.Pair;
037import org.opt4j.operators.mutate.Mutate;
038import org.opt4j.operators.mutate.MutationRate;
039
040import com.google.inject.Inject;
041
042/**
043 * The {@link MatingCrossoverMutate} creates offspring from a given set of
044 * parents by using {@link Crossover} and {@code Mutate}.
045 * 
046 * @author glass, lukasiewycz
047 * 
048 */
049public class MatingCrossoverMutate implements Mating {
050
051        protected final Crossover<Genotype> crossover;
052        protected final Mutate<Genotype> mutate;
053        protected final Copy<Genotype> copy;
054        protected final Coupler coupler;
055        protected final CrossoverRate crossoverRate;
056        protected final MutationRate mutationRate;
057        protected final Random random;
058        protected final IndividualFactory individualFactory;
059
060        /**
061         * Constructs a {@link MatingCrossoverMutate} with a given {@link Crossover}
062         * , {@link Mutate}, {@link Copy}, {@link Coupler}, {@link CrossoverRate},
063         * {@link Rand}, and {@link IndividualFactory}.
064         * 
065         * @param crossover
066         *            the crossover operator
067         * @param mutate
068         *            the mutate operator
069         * @param copy
070         *            the copy operator
071         * @param coupler
072         *            the coupler
073         * @param crossoverRate
074         *            the used crossover rate
075         * @param mutationRate
076         *            the mutation rate
077         * @param random
078         *            the random number generator
079         * @param individualFactory
080         *            the individual factory
081         */
082        @Inject
083        public MatingCrossoverMutate(Crossover<Genotype> crossover, Mutate<Genotype> mutate, Copy<Genotype> copy,
084                        Coupler coupler, CrossoverRate crossoverRate, MutationRate mutationRate, Rand random,
085                        IndividualFactory individualFactory) {
086                super();
087                this.crossover = crossover;
088                this.mutate = mutate;
089                this.copy = copy;
090                this.coupler = coupler;
091                this.crossoverRate = crossoverRate;
092                this.mutationRate = mutationRate;
093                this.random = random;
094                this.individualFactory = individualFactory;
095        }
096
097        /*
098         * (non-Javadoc)
099         * 
100         * @see org.opt4j.optimizer.ea.Mating#getOffspring(int,
101         * org.opt4j.core.Individual[])
102         */
103        @Override
104        public Collection<Individual> getOffspring(int size, Individual... parents) {
105                return getOffspringInternal(size, Arrays.asList(parents));
106        }
107
108        /*
109         * (non-Javadoc)
110         * 
111         * @see org.opt4j.optimizer.ea.Mating#getOffspring(int,
112         * java.util.Collection)
113         */
114        @Override
115        public Collection<Individual> getOffspring(int size, Collection<Individual> parents) {
116                return getOffspringInternal(size, parents);
117        }
118
119        /**
120         * Creates offspring from a given set of parents.
121         * 
122         * The {@link Coupler} is used to create pairs of parents, which are mated
123         * using the {@link Mutate} and, depending on the {@link CrossoverRate}, the
124         * {@link Crossover} operator.
125         * 
126         * @param size
127         *            the number of individuals to create
128         * @param parents
129         *            the parents
130         * @return the offspring
131         */
132        protected Collection<Individual> getOffspringInternal(int size, Collection<Individual> parents) {
133                Collection<Individual> offspring = new ArrayList<>();
134                Collection<Pair<Individual>> couples = coupler.getCouples((int) Math.ceil(((double) size / 2)),
135                                new ArrayList<>(parents));
136
137                for (Pair<Individual> couple : couples) {
138                        boolean crossover = random.nextDouble() <= crossoverRate.get();
139                        Individual parent1 = couple.getFirst();
140                        Individual parent2 = couple.getSecond();
141                        Pair<Individual> i = mate(parent1, parent2, crossover);
142                        Individual i1 = i.getFirst();
143                        Individual i2 = i.getSecond();
144
145                        offspring.add(i1);
146                        if (offspring.size() < size) {
147                                offspring.add(i2);
148                        }
149                }
150
151                return offspring;
152        }
153
154        /**
155         * Performs the actual {@link Coupler} process of two parents.
156         * 
157         * @param parent1
158         *            parent one
159         * @param parent2
160         *            parent two
161         * @param doCrossover
162         *            indicates whether the coupler shall take place
163         * @return the two offspring individuals
164         */
165        protected Pair<Individual> mate(Individual parent1, Individual parent2, boolean doCrossover) {
166                Genotype p1 = parent1.getGenotype();
167                Genotype p2 = parent2.getGenotype();
168                Genotype o1;
169                Genotype o2;
170
171                if (doCrossover) {
172                        Pair<Genotype> offspring = crossover.crossover(p1, p2);
173                        o1 = offspring.getFirst();
174                        o2 = offspring.getSecond();
175                } else {
176                        o1 = copy.copy(p1);
177                        o2 = copy.copy(p2);
178                }
179
180                mutate.mutate(o1, mutationRate.get());
181                mutate.mutate(o2, mutationRate.get());
182
183                Individual i1 = individualFactory.create(o1);
184                Individual i2 = individualFactory.create(o2);
185
186                Pair<Individual> individuals = new Pair<>(i1, i2);
187                return individuals;
188        }
189
190}
