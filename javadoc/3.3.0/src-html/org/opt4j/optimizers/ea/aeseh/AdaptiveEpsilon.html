001package org.opt4j.optimizers.ea.aeseh;
002
003/**
004 * The {@link AdaptiveEpsilon} contains the information about an ε-value and the
005 * information that is used by the {@link EpsilonAdaptation} to adapt the
006 * ε-value.
007 * 
008 * @author Fedor Smirnov
009 *
010 */
011public class AdaptiveEpsilon {
012
013        protected double epsilon;
014        protected double epsilonDelta;
015        protected final double epsilonDeltaMax;
016        protected final double epsilonDeltaMin;
017
018        public AdaptiveEpsilon(double epsilon, double epsilonDelta, double epsilonDeltaMax, double epsilonDeltaMin) {
019                if (epsilonDelta < epsilonDeltaMin || epsilonDelta > epsilonDeltaMax) {
020                        throw new IllegalArgumentException("The ε-delta value exceeds the given bounds");
021                }
022                this.epsilon = epsilon;
023                this.epsilonDelta = epsilonDelta;
024                this.epsilonDeltaMax = epsilonDeltaMax;
025                this.epsilonDeltaMin = epsilonDeltaMin;
026        }
027
028        public double getEpsilon() {
029                return epsilon;
030        }
031
032        public void setEpsilon(double epsilon) {
033                this.epsilon = epsilon;
034        }
035
036        public double getEpsilonDelta() {
037                return epsilonDelta;
038        }
039
040        public void setEpsilonDelta(double epsilonDelta) {
041                double delta = Math.max(epsilonDeltaMin, epsilonDelta);
042                delta = Math.min(epsilonDeltaMax, delta);
043                this.epsilonDelta = delta;
044        }
045
046        public double getEpsilonDeltaMax() {
047                return epsilonDeltaMax;
048        }
049
050        public double getEpsilonDeltaMin() {
051                return epsilonDeltaMin;
052        }
053}
