024package org.opt4j.optimizers.de;
025
026import org.opt4j.core.config.annotations.Info;
027import org.opt4j.core.config.annotations.Order;
028import org.opt4j.core.optimizer.MaxIterations;
029import org.opt4j.core.optimizer.OptimizerModule;
030import org.opt4j.core.start.Constant;
031
032/**
033 * The {@link DifferentialEvolutionModule}.
034 * 
035 * @author lukasiewycz
036 * 
037 */
038@Info("A population based optimization heuristic using vector differences.")
039public class DifferentialEvolutionModule extends OptimizerModule {
040
041        @Info("The number of generations.")
042        @Order(0)
043        @MaxIterations
044        protected int generations = 1000;
045
046        @Info("The size of the population.")
047        @Order(1)
048        @Constant(value = "alpha", namespace = DifferentialEvolution.class)
049        protected int alpha = 100;
050
051        @Info("The scaling factor F (0 <= F <= 2.0).")
052        @Order(2)
053        @Constant(value = "scalingFactor", namespace = DifferentialEvolution.class)
054        protected double scalingFactor = 0.5;
055
056        /**
057         * Returns the alpha.
058         * 
059         * @see #setAlpha
060         * @return the alpha
061         */
062        public int getAlpha() {
063                return alpha;
064        }
065
066        /**
067         * Sets the alpha.
068         * 
069         * @see #getAlpha
070         * @param alpha
071         *            the alpha to set
072         */
073        public void setAlpha(int alpha) {
074                this.alpha = alpha;
075        }
076
077        /**
078         * Returns the generations.
079         * 
080         * @see #setGenerations
081         * @return the generations
082         */
083        public int getGenerations() {
084                return generations;
085        }
086
087        /**
088         * Sets the generations.
089         * 
090         * @see #getGenerations
091         * @param generations
092         *            the generations to set
093         */
094        public void setGenerations(int generations) {
095                this.generations = generations;
096        }
097
098        /**
099         * Returns the scaling factor.
100         * 
101         * @see #setScalingFactor
102         * @return the scalingFactor
103         */
104        public double getScalingFactor() {
105                return scalingFactor;
106        }
107
108        /**
109         * Sets the scaling factor.
110         * 
111         * @see #getScalingFactor
112         * @param scalingFactor
113         *            the scalingFactor to set
114         */
115        public void setScalingFactor(double scalingFactor) {
116                this.scalingFactor = scalingFactor;
117        }
118
119        /*
120         * (non-Javadoc)
121         * 
122         * @see org.opt4j.start.Opt4JModule#config()
123         */
124        @Override
125        public void config() {
126                bindIterativeOptimizer(DifferentialEvolution.class);
127        }
128}
