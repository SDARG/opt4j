023package org.opt4j.core.genotype;
024
025import java.lang.reflect.Constructor;
026import java.util.Collection;
027import java.util.Collections;
028import java.util.HashMap;
029import java.util.HashSet;
030import java.util.List;
031import java.util.Map;
032import java.util.Map.Entry;
033import java.util.Random;
034
035import org.opt4j.core.Genotype;
036
037/**
038 * <p>
039 * The {@link SelectMapGenotype} selects for each key an element from a given
040 * list.
041 * </p>
042 * <p>
043 * Example problem: Choose for each of five balls ball<sub>1</sub>,
044 * ball<sub>2</sub>, ball<sub>3</sub>, ball<sub>4</sub>, ball<sub>5</sub> a
045 * color. Available colors are blue, green, and red.<br/>
046 * Example usage: <blockquote>
047 * 
048 * <pre>
049 * SelectMapGenotype&lt;Ball, Color&gt; genotype = new SelectMapGenotype&lt;Ball, Color&gt;(
050 *              Arrays.asList(ball1, ball2, ball3, ball4, ball5), Arrays.asList(Color.BLUE, Color.GREEN, Color.RED));
051 * genotype.init(new Random());
052 * </pre>
053 * 
054 * </blockquote> Example instance: [ball1=green, ball2=red, ball3=red,
055 * ball4=blue, ball5=blue]<br/>
056 * Example search space size: 3<sup>5</sup>
057 * </p>
058 * 
059 * 
060 * @author lukasiewycz
061 * 
062 * @param <K>
063 *            the type of keys
064 * @param <V>
065 *            the type of elements
066 */
067public class SelectMapGenotype<K, V> extends IntegerGenotype implements MapGenotype<K, V> {
068
069        private static final long serialVersionUID = 1L;
070
071        protected final List<K> keys;
072        protected final Map<K, List<V>> values;
073
074        protected static class SelectBounds<O, P> implements Bounds<Integer> {
075                protected List<O> list;
076                protected Map<O, List<P>> map;
077
078                public SelectBounds(List<O> list, Map<O, List<P>> map) {
079                        this.list = list;
080                        this.map = map;
081                        for (Entry<O, List<P>> entry : map.entrySet()) {
082                                if (entry.getValue().isEmpty()) {
083                                        throw new IllegalArgumentException("Empty value map provided for key " + entry.getKey());
084                                }
085                        }
086                }
087
088                @Override
089                public Integer getLowerBound(int index) {
090                        return 0;
091                }
092
093                @Override
094                public Integer getUpperBound(int index) {
095                        return map.get(list.get(index)).size() - 1;
096                }
097        }
098
099        /**
100         * Constructs a {@link SelectMapGenotype}.
101         * 
102         * @param keys
103         *            the keys
104         * @param values
105         *            the values
106         */
107        public SelectMapGenotype(List<K> keys, Map<K, List<V>> values) {
108                super(new SelectBounds<>(keys, values));
109                if (!keys.containsAll(values.keySet()) || !values.keySet().containsAll(keys)) {
110                        throw new IllegalArgumentException("The provided list does not match the provided map");
111                }
112                if (new HashSet<>(keys).size() < keys.size()) {
113                        throw new IllegalArgumentException(MapGenotype.ERROR_MESSAGE_NON_UNIQUE_KEYS);
114                }
115                this.keys = keys;
116                this.values = values;
117        }
118
119        private static <K, V> Map<K, List<V>> toMap(List<K> keys, List<V> values) {
120                if (new HashSet<>(keys).size() < keys.size()) {
121                        throw new IllegalArgumentException(MapGenotype.ERROR_MESSAGE_NON_UNIQUE_KEYS);
122                }
123                Map<K, List<V>> map = new HashMap<>();
124                for (K key : keys) {
125                        map.put(key, values);
126                }
127                return map;
128        }
129
130        /**
131         * Constructs a {@link SelectMapGenotype}. Here, each key has the same
132         * target list of element values.
133         * 
134         * @param keys
135         *            the keys
136         * @param values
137         *            the values
138         */
139        public SelectMapGenotype(List<K> keys, List<V> values) {
140                super(new SelectBounds<>(keys, toMap(keys, values)));
141                this.keys = keys;
142                this.values = toMap(keys, values);
143        }
144
145        /**
146         * Initialize this genotype with random values based on the size of the
147         * {@code key} set.
148         * 
149         * @param random
150         *            the random number generator
151         */
152        public void init(Random random) {
153                super.init(random, keys.size());
154        }
155
156        /**
157         * Not supported. Use {@link SelectMapGenotype#init(Random)} instead.
158         * 
159         * @see org.opt4j.core.genotype.IntegerGenotype#init(java.util.Random, int)
160         */
161        @Override
162        public void init(Random random, int n) {
163                throw new UnsupportedOperationException(MapGenotype.ERROR_MESSAGE_UNSUPPORTED_INIT);
164        }
165
166        /*
167         * (non-Javadoc)
168         * 
169         * @see org.opt4j.genotype.MapGenotype#containsKey(java.lang.Object)
170         */
171        @Override
172        public boolean containsKey(K key) {
173                return keys.contains(key);
174        }
175
176        /*
177         * (non-Javadoc)
178         * 
179         * @see org.opt4j.genotype.MapGenotype#getIndexOf(java.lang.Object)
180         */
181        @Override
182        public int getIndexOf(K key) {
183                if (!containsKey(key)) {
184                        throw new IllegalArgumentException(MapGenotype.ERROR_MESSAGE_INVALID_KEY);
185                }
186                return keys.indexOf(key);
187        }
188
189        /*
190         * (non-Javadoc)
191         * 
192         * @see org.opt4j.genotype.MapGenotype#getValue(java.lang.Object)
193         */
194        @Override
195        public V getValue(K key) {
196                int i = getIndexOf(key);
197                int v = get(i);
198                List<V> valueList = values.get(key);
199                return valueList.get(v);
200        }
201
202        /*
203         * (non-Javadoc)
204         * 
205         * @see org.opt4j.genotype.MapGenotype#setValue(java.lang.Object,
206         * java.lang.Object)
207         */
208        @Override
209        public void setValue(K key, V value) {
210                int i = getIndexOf(key);
211                while (size() <= i) {
212                        add(bounds.getLowerBound(i));
213                }
214                List<V> valueList = values.get(key);
215                if (!valueList.contains(value)) {
216                        throw new IllegalArgumentException(
217                                        "The list provided for key " + key + " does not contain the value " + value);
218                }
219                int v = valueList.indexOf(value);
220                set(i, v);
221        }
222
223        /*
224         * (non-Javadoc)
225         * 
226         * @see org.opt4j.core.Genotype#newInstance()
227         */
228        @Override
229        @SuppressWarnings({ "unchecked", "rawtypes" })
230        public <G extends Genotype> G newInstance() {
231                try {
232                        Constructor<? extends SelectMapGenotype> cstr = this.getClass().getConstructor(List.class, Map.class);
233                        return (G) cstr.newInstance(keys, values);
234                } catch (Exception e) {
235                        throw new RuntimeException(e);
236                }
237        }
238
239        /*
240         * (non-Javadoc)
241         * 
242         * @see java.util.AbstractCollection#toString()
243         */
244        @Override
245        public String toString() {
246                String s = "[";
247                for (int i = 0; i < size(); i++) {
248                        K key = keys.get(i);
249                        int v = this.get(i);
250                        V value = values.get(key).get(v);
251                        s += key + "=" + value + ";";
252                }
253                return s + "]";
254        }
255
256        /*
257         * (non-Javadoc)
258         * 
259         * @see org.opt4j.genotype.MapGenotype#getKeys()
260         */
261        @Override
262        public Collection<K> getKeys() {
263                return Collections.unmodifiableList(keys);
264        }
265}
