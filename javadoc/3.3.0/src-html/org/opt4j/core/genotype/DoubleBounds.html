023package org.opt4j.core.genotype;
024
025import java.util.List;
026
027/**
028 * The {@link DoubleBounds} is an implementation of the {@link Bounds} for the
029 * {@link DoubleGenotype} that accepts arrays as well as lists as bounds.
030 * 
031 * @author lukasiewycz
032 * 
033 */
034public class DoubleBounds implements Bounds<Double> {
035
036        protected final double[] lower;
037        protected final double[] upper;
038
039        /**
040         * Constructs a {@link DoubleBounds} with arrays.
041         * 
042         * @param lower
043         *            the lower bounds
044         * @param upper
045         *            the upper bounds
046         */
047        public DoubleBounds(double[] lower, double[] upper) {
048                if (lower.length != upper.length) {
049                        throw new IllegalArgumentException("Lower and upper bounds arrays should have same length.");
050                }
051                this.lower = lower;
052                this.upper = upper;
053        }
054
055        /**
056         * Constructs a {@link DoubleBounds} with lists.
057         * 
058         * @param lower
059         *            the lower bounds
060         * @param upper
061         *            the upper bounds
062         */
063        public DoubleBounds(List<Double> lower, List<Double> upper) {
064                if (lower.size() != upper.size()) {
065                        throw new IllegalArgumentException("Lower and upper bounds lists should have the same size.");
066                }
067                this.lower = new double[lower.size()];
068                this.upper = new double[upper.size()];
069                for (int i = 0; i < lower.size(); i++) {
070                        this.lower[i] = lower.get(i);
071                }
072                for (int i = 0; i < upper.size(); i++) {
073                        this.upper[i] = upper.get(i);
074                }
075
076        }
077
078        /*
079         * (non-Javadoc)
080         * 
081         * @see org.opt4j.genotype.Bounds#getLowerBound(int)
082         */
083        @Override
084        public Double getLowerBound(int index) {
085                return lower[index];
086        }
087
088        /*
089         * (non-Javadoc)
090         * 
091         * @see org.opt4j.genotype.Bounds#getUpperBound(int)
092         */
093        @Override
094        public Double getUpperBound(int index) {
095                return upper[index];
096        }
097
098}
