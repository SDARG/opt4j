024package org.opt4j.core;
025
026/**
027 * The {@link DoubleValue} is a {@link Value} with a {@link Double} as value.
028 * The {@link Objectives} contains a convenience method that allows to add a
029 * {@link DoubleValue} without creating a new instance:
030 * {@link Objectives#add(Objective, double)}.
031 * 
032 * @see Objectives#add(Objective, double)
033 * @author lukasiewycz
034 * 
035 */
036public class DoubleValue implements Value<Double> {
037
038        protected Double value = null;
039
040        /**
041         * Constructs a {@link DoubleValue}.
042         * 
043         * @param value
044         *            the specified value
045         */
046        public DoubleValue(Double value) {
047                this.value = value;
048        }
049
050        /*
051         * (non-Javadoc)
052         * 
053         * @see org.opt4j.core.Value#getDouble()
054         */
055        @Override
056        public Double getDouble() {
057                return value;
058        }
059
060        /*
061         * (non-Javadoc)
062         * 
063         * @see org.opt4j.core.Value#getValue()
064         */
065        @Override
066        public Double getValue() {
067                return value;
068        }
069
070        /*
071         * (non-Javadoc)
072         * 
073         * @see org.opt4j.core.Value#setValue(java.lang.Object)
074         */
075        @Override
076        public void setValue(Double value) {
077                this.value = value;
078        }
079
080        /*
081         * (non-Javadoc)
082         * 
083         * @see java.lang.Comparable#compareTo(java.lang.Object)
084         */
085        @Override
086        public int compareTo(Value<Double> arg) {
087                if (arg == null) {
088                        return -1;
089                }
090                Double other = arg.getValue();
091                if (value == null) {
092                        if (other == null) {
093                                return 0;
094                        }
095                        return 1;
096                } else if (other == null) {
097                        return -1;
098                }
099                return value.compareTo(other);
100        }
101
102        /*
103         * (non-Javadoc)
104         * 
105         * @see java.lang.Object#hashCode()
106         */
107        @Override
108        public int hashCode() {
109                final int prime = 31;
110                int result = 1;
111                result = prime * result + ((value == null) ? 0 : value.hashCode());
112                return result;
113        }
114
115        /*
116         * (non-Javadoc)
117         * 
118         * @see java.lang.Object#equals(java.lang.Object)
119         */
120        @Override
121        public boolean equals(Object obj) {
122                if (this == obj) {
123                        return true;
124                }
125                if (obj == null) {
126                        return false;
127                }
128                if (!(obj instanceof DoubleValue)) {
129                        return false;
130                }
131                DoubleValue other = (DoubleValue) obj;
132                if (value == null) {
133                        if (other.value != null) {
134                                return false;
135                        }
136                } else if (!value.equals(other.value)) {
137                        return false;
138                }
139                return true;
140        }
141
142        /*
143         * (non-Javadoc)
144         * 
145         * @see java.lang.Object#toString()
146         */
147        @Override
148        public String toString() {
149                return value == null ? "null" : value.toString();
150        }
151
152}
