066/**
067 * The {@link PropertyPanel} is a panel for the configuration of one module.
068 * Properties and values are arranged in a table.
069 * 
070 * @author lukasiewycz
071 * 
072 */
073@SuppressWarnings("serial")
074public class PropertyPanel extends JPanel {
075
076        protected final PropertyModule module;
077
078        protected final FileChooser fileChooser;
079
080        protected final Format format;
081
082        protected final JPanel panel;
083
084        protected final Map<Property, Component> components = new HashMap<>();
085
086        /**
087         * Constructs a {@link PropertyPanel} for one {@link PropertyModule}
088         * instance.
089         * 
090         * @param module
091         *            the instance of the {@link PropertyModule}
092         * @param fileChooser
093         *            the FileChooser
094         * @param format
095         *            the format
096         */
097        public PropertyPanel(PropertyModule module, FileChooser fileChooser, Format format) {
098                this.module = module;
099                this.fileChooser = fileChooser;
100                this.format = format;
101
102                fillComponentsMap();
103
104                panel = new JPanel(new DialogLayout(20, 2));
105                panel.setBorder(BorderFactory.createEmptyBorder(5, 5, 5, 5));
106
107                setLayout(new BorderLayout());
108                add(BorderLayout.NORTH, panel);
109
110                update();
111        }
112
113        protected Component createComponent(final Property property) {
114                Component component = null;
115                Object value = property.getValue();
116                Class<?> type = property.getType();
117
118                if (type.isEnum()) {
119                        Object[] obj = new Object[type.getEnumConstants().length];
120                        for (int k = 0; k < obj.length; k++) {
121                                obj[k] = type.getEnumConstants()[k];
122                        }
123                        final JComboBox<Object> box = new JComboBox<>(obj);
124                        box.setSelectedItem(value);
125
126                        box.addActionListener(new ActionListener() {
127                                @Override
128                                public void actionPerformed(ActionEvent e) {
129                                        Object selected = box.getSelectedItem();
130                                        try {
131                                                property.setValue(selected);
132                                                update();
133                                        } catch (InvocationTargetException ex) {
134                                                System.err.println(ex.getMessage());
135                                        }
136                                }
137                        });
138
139                        component = box;
140                } else if (type.equals(Boolean.TYPE)) {
141                        final JCheckBox checkbox = new JCheckBox();
142
143                        boolean b = (Boolean) property.getValue();
144                        checkbox.setSelected(b);
145                        checkbox.addActionListener(new ActionListener() {
146                                @Override
147                                public void actionPerformed(ActionEvent e) {
148                                        boolean b = checkbox.isSelected();
149                                        try {
150                                                property.setValue(b);
151                                                update();
152                                        } catch (InvocationTargetException ex) {
153                                                System.err.println(ex.getMessage());
154                                        }
155
156                                }
157                        });
158
159                        component = checkbox;
160                } else {
161                        final JTextField field = new JTextField();
162
163                        if (property.isNumber()) {
164                                field.setHorizontalAlignment(SwingConstants.RIGHT);
165                        }
166                        if (value == null) {
167                                field.setText("");
168                        } else {
169                                field.setText(value.toString());
170                        }
171
172                        field.addFocusListener(new FocusAdapter() {
173                                @Override
174                                public void focusLost(FocusEvent e) {
175                                        String value = format(property, field.getText());
176                                        try {
177                                                property.setValue(value);
178                                        } catch (InvocationTargetException ex) {
179                                                System.err.println(ex.getMessage());
180                                        } finally {
181                                                field.setText(property.getValue().toString());
182                                                update();
183                                        }
184                                }
185                        });
186
187                        field.addKeyListener(new KeyAdapter() {
188
189                                @Override
190                                public void keyReleased(KeyEvent e) {
191                                        String value = format(property, field.getText());
192
193                                        try {
194                                                property.setValue(value);
195                                        } catch (InvocationTargetException ex) {
196                                                System.err.println(ex.getMessage());
197                                                field.setText(property.getValue().toString());
198                                        }
199                                }
200                        });
201
202                        component = field;
203                }
204                return component;
205        }
206
207        protected String format(Property property, String value) {
208                String result = value;
209                if (property.isNumber()) {
210                        result = result.trim();
211                        if ("".equals(result)) {
212                                result = "0";
213                        } else if ("-".equals(result)) {
214                                result = "0";
215                        } else {
216                                if (result.startsWith(".")) {
217                                        result = "0" + result;
218                                }
219                                if (result.endsWith(".")) {
220                                        result = result + "0";
221                                }
222                        }
223                        return result;
224                }
225                return result;
226        }
227
228        protected void fillComponentsMap() {
229                for (final Property property : module.getProperties()) {
230                        Component component = createComponent(property);
231                        components.put(property, component);
232                }
233        }
234
235        protected void updatePropertyPanel() {
236                panel.removeAll();
237
238                for (final Property property : module.getProperties()) {
239                        if (property.isActive()) {
240                                String name = property.getName();
241
242                                int i = getIndent(property);
243                                String s = "";
244                                for (int j = 0; j < i; j++) {
245                                        s += "     ";
246                                }
247                                if (i > 0) {
248                                        s = s.substring(2) + "\u21aa ";
249                                }
250
251                                JPanel labelPanel = new JPanel(new FlowLayout(FlowLayout.LEFT, 0, 0));
252                                JLabel label = new JLabel(s + name);
253                                label.setFocusable(false);
254                                String tooltip = format.getTooltip(property);
255                                if (tooltip != null) {
256                                        label.setToolTipText(tooltip);
257                                }
258                                labelPanel.add(label);
259
260                                File file = property.getAnnotation(File.class);
261                                if (file != null) {
262                                        JButton browse = new JButton(Icons.getIcon(Icons.FOLDER));
263                                        browse.setFocusable(false);
264                                        browse.setBorderPainted(false);
265                                        browse.setContentAreaFilled(false);
266                                        browse.setBorder(BorderFactory.createEmptyBorder(0, 4, 0, 0));
267
268                                        browse.addActionListener((ActionEvent e) -> selectFile(property));
269                                        browse.setCursor(new Cursor(Cursor.HAND_CURSOR));
270                                        browse.setToolTipText("Browse ...");
271                                        labelPanel.add(browse);
272                                }
273                                panel.add(labelPanel);
274
275                                Component component = components.get(property);
276                                panel.add(component);
277                        }
278                }
279
280                if (module.getModule().getClass().isAnnotationPresent(Citation.class)) {
281                        Citation citation = module.getModule().getClass().getAnnotation(Citation.class);
282                        addReferenceRow(citation);
283                }
284        }
285
286        /**
287         * Adds a row showing the {@link Citation}. On the left hand side,
288         * "reference" is printed while on the right hand side, the {@link Citation}
289         * is added in a static {@link TextArea}.
290         * 
291         * @param citation
292         *            the citation to add
293         */
294        protected void addReferenceRow(Citation citation) {
295                JPanel labelPanel = new JPanel(new FlowLayout(FlowLayout.LEFT, 0, 0));
296                JLabel label = new JLabel("reference");
297                label.setFocusable(false);
298                labelPanel.add(label);
299
300                panel.add(labelPanel);
301
302                final JTextArea field = new JTextArea();
303                field.setLineWrap(true);
304                field.setWrapStyleWord(true);
305                field.setFont(label.getFont());
306                field.setBackground(label.getBackground());
307                field.setEditable(false);
308                field.setText(Format.formatJava(citation));
309
310                if (!citation.doi().isEmpty()) {
311                        final String doi = citation.doi();
312                        field.setCursor(Cursor.getPredefinedCursor(Cursor.HAND_CURSOR));
313                        field.addMouseListener(new MouseListener() {
314
315                                @Override
316                                public void mouseReleased(MouseEvent e) {
317                                        // nothing to be done
318                                }
319
320                                @Override
321                                public void mousePressed(MouseEvent e) {
322                                        // nothing to be done
323                                }
324
325                                @Override
326                                public void mouseExited(MouseEvent e) {
327                                        // nothing to be done
328                                }
329
330                                @Override
331                                public void mouseEntered(MouseEvent e) {
332                                        // nothing to be done
333                                }
334
335                                @Override
336                                public void mouseClicked(MouseEvent e) {
337                                        try {
338                                                Desktop.getDesktop().browse(new URI("https://doi.org/" + doi));
339                                        } catch (IOException e1) {
340                                                e1.printStackTrace();
341                                        } catch (URISyntaxException e1) {
342                                                e1.printStackTrace();
343                                        }
344                                }
345                        });
346                }
347
348                JPanel rightPanel = new JPanel(new BorderLayout());
349                rightPanel.add(BorderLayout.NORTH, field);
350                rightPanel.setPreferredSize(
351                                new Dimension((int) (panel.getPreferredSize().width * 0.70), panel.getPreferredSize().height));
352                panel.add(rightPanel);
353        }
354
355        protected void update() {
356                updatePropertyPanel();
357                revalidate();
358                repaint();
359        }
360
361        /**
362         * Returns the instance of the {@link PropertyModule}.
363         * 
364         * @return the instance of the property module
365         */
366        public PropertyModule getModule() {
367                return module;
368        }
369
370        protected int getIndent(Property property) {
371                int i = 0;
372                for (Requirement requirement : property.getRequirements()) {
373                        i = Math.max(i, getIndent(requirement.getProperty()));
374                }
375                if (!property.getRequirements().isEmpty()) {
376                        i++;
377                }
378                return i;
379        }
380
381        private void selectFile(Property property) {
382                final File file = property.getAnnotation(File.class);
383                final JFileChooser fileChooser = PropertyPanel.this.fileChooser.get();
384
385                java.io.File dir = null;
386                try {
387                        dir = new java.io.File(property.getValue().toString());
388                } catch (Exception ex) {
389                }
390
391                fileChooser.setCurrentDirectory(dir);
392
393                if (file != null && !file.value().equals("")) {
394                        FileFilter filter = new FileFilter() {
395
396                                @Override
397                                public boolean accept(java.io.File pathname) {
398                                        if (pathname.isDirectory()) {
399                                                return true;
400                                        }
401                                        String f = pathname.getName().toLowerCase();
402                                        String ext = file.value().toLowerCase();
403
404                                        if (f.endsWith(ext)) {
405                                                return true;
406                                        }
407                                        return false;
408                                }
409
410                                @Override
411                                public String getDescription() {
412                                        String ext = file.value().toLowerCase();
413                                        return "(*" + ext + ")";
414                                }
415                        };
416
417                        fileChooser.setFileFilter(filter);
418                } else {
419                        fileChooser.setFileFilter(null);
420                }
421
422                if (file.folder()) {
423                        if (file.file()) {
424                                fileChooser.setFileSelectionMode(JFileChooser.FILES_AND_DIRECTORIES);
425                        } else {
426                                fileChooser.setFileSelectionMode(JFileChooser.DIRECTORIES_ONLY);
427                        }
428                } else {
429                        fileChooser.setFileSelectionMode(JFileChooser.FILES_ONLY);
430                }
431                fileChooser.setVisible(true);
432
433                int status = fileChooser.showOpenDialog(null);
434                if (status == JFileChooser.APPROVE_OPTION) {
435                        java.io.File f = fileChooser.getSelectedFile();
436                        try {
437                                property.setValue(f.getAbsolutePath());
438                                JTextField field = (JTextField) components.get(property);
439                                field.setText("" + property.getValue());
440                                update();
441                        } catch (InvocationTargetException ex) {
442                                System.err.println(ex.getMessage());
443                        }
444                }
445        }
