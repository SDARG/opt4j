023package org.opt4j.core;
024
025import static org.opt4j.core.Objective.Sign.MAX;
026
027import java.util.Collection;
028import java.util.Iterator;
029import java.util.Map.Entry;
030import java.util.SortedMap;
031import java.util.TreeMap;
032
033import org.opt4j.core.Objective.Sign;
034
035/**
036 * The {@link Objectives} contain the {@link Objective}-{@link Value} pairs of
037 * an {@link Individual}.
038 * 
039 * @see Value
040 * @see Objective
041 * @author lukasiewycz
042 * 
043 */
044public class Objectives implements Iterable<Entry<Objective, Value<?>>> {
045
046        protected final SortedMap<Objective, Value<?>> map = new TreeMap<>();
047
048        protected double[] array = null;
049
050        /*
051         * (non-Javadoc)
052         * 
053         * @see java.lang.Iterable#iterator()
054         */
055        @Override
056        public Iterator<Entry<Objective, Value<?>>> iterator() {
057                return map.entrySet().iterator();
058        }
059
060        /**
061         * Returns an array of all values which all have to be minimized. Do not
062         * call this method before all objectives were added!
063         * 
064         * @see Value#getDouble()
065         * @return an array containing values which have to be minimized
066         */
067        public double[] array() {
068                if (array == null) {
069                        submit();
070                }
071
072                return array;
073        }
074
075        /**
076         * Calculates the array.
077         */
078        protected synchronized void submit() {
079                if (array == null) {
080                        array = new double[size()];
081                        int i = 0;
082                        for (Entry<Objective, Value<?>> entry : this) {
083                                Objective objective = entry.getKey();
084                                Value<?> value = entry.getValue();
085
086                                Double v = value.getDouble();
087
088                                if (v == null) {
089                                        array[i] = Double.MAX_VALUE;
090                                } else if (objective.getSign() == MAX) {
091                                        array[i] = -v;
092                                } else {
093                                        array[i] = v;
094                                }
095
096                                i++;
097                        }
098                }
099        }
100
101        /**
102         * Returns the number of set {@link Objective}s.
103         * 
104         * @return the number of set objectives
105         */
106        public int size() {
107                return map.size();
108        }
109
110        /**
111         * Returns all objectives.
112         * 
113         * @return all objectives
114         */
115        public Collection<Objective> getKeys() {
116                return map.keySet();
117        }
118
119        /**
120         * Returns all values.
121         * 
122         * @return all values
123         */
124        public Collection<Value<?>> getValues() {
125                return map.values();
126        }
127
128        /**
129         * Returns the value that is assigned to the given objective. Returns
130         * {@code null} if the objective does not exist.
131         * 
132         * @param objective
133         *            the given objective
134         * @return the value
135         */
136        public Value<?> get(Objective objective) {
137                return map.get(objective);
138        }
139
140        /**
141         * Returns the objective that is assigned to the given value. Returns
142         * {@code null} if the value does not exist.
143         * 
144         * @param value
145         *            the given value
146         * @return the objective
147         */
148        public Objective get(Value<?> value) {
149                Objective objective = null;
150                for (Entry<Objective, Value<?>> entry : this) {
151                        Value<?> v = entry.getValue();
152                        if (value.equals(v)) {
153                                Objective o = entry.getKey();
154                                objective = o;
155                                break;
156                        }
157                }
158                return objective;
159        }
160
161        /**
162         * Adds the objective with the specified value.
163         * 
164         * @param objective
165         *            the objective
166         * @param value
167         *            the value
168         */
169        public void add(Objective objective, Value<?> value) {
170                if (value == null) {
171                        map.put(objective, new DoubleValue(null));
172                } else {
173                        map.put(objective, value);
174                }
175                array = null;
176        }
177
178        /**
179         * Adds the objective with the specified value.
180         * 
181         * @param name
182         *            the name of the objective
183         * @param sign
184         *            the optimization direction
185         * @param value
186         *            the value
187         */
188        public void add(String name, Sign sign, Value<?> value) {
189                add(new Objective(name, sign), value);
190        }
191
192        /**
193         * Adds the objective with the specified double value.
194         * 
195         * @param objective
196         *            the objective
197         * @param value
198         *            the value
199         */
200        public void add(Objective objective, double value) {
201                add(objective, new DoubleValue(value));
202        }
203
204        /**
205         * Adds the objective with the specified value.
206         * 
207         * @param name
208         *            the name of the objective
209         * @param sign
210         *            the optimization direction
211         * @param value
212         *            the value
213         */
214        public void add(String name, Sign sign, double value) {
215                add(new Objective(name, sign), value);
216        }
217
218        /**
219         * Adds the objective with the specified integer value.
220         * 
221         * @param objective
222         *            the objective
223         * @param value
224         *            the value
225         */
226        public void add(Objective objective, int value) {
227                add(objective, new IntegerValue(value));
228        }
229
230        /**
231         * Adds the objective with the specified value.
232         * 
233         * @param name
234         *            the name of the objective
235         * @param sign
236         *            the optimization direction
237         * @param value
238         *            the value
239         */
240        public void add(String name, Sign sign, int value) {
241                add(new Objective(name, sign), value);
242        }
243
244        /**
245         * Adds all objective with the specified value specified in
246         * {@link Objectives}.
247         * 
248         * @param objectives
249         *            the objectives
250         */
251        public void addAll(Objectives objectives) {
252                map.putAll(objectives.map);
253                array = null;
254        }
255
256        /**
257         * Returns {@code true} if this objectives weakly dominates the specified
258         * objectives. This comparison is based on the {@link #array()} values.
259         * 
260         * @param opponent
261         *            other objectives
262         * @return {@code true} if this objectives weakly dominate the
263         *         {@code opponent}
264         */
265        public boolean weaklyDominates(Objectives opponent) {
266                double[] va = this.array();
267                double[] vb = opponent.array();
268                for (int i = 0; i < va.length; i++) {
269                        if (vb[i] < va[i]) {
270                                return false;
271                        }
272                }
273                return true;
274        }
275
276        /**
277         * Returns {@code true} if this objectives dominate the specified
278         * objectives. This comparison is based on the {@link #array()} values.
279         * 
280         * @param opponent
281         *            other objectives
282         * @return {@code true} if these objectives dominate the {@code opponent}
283         */
284        public boolean dominates(Objectives opponent) {
285                double[] va = this.array();
286                double[] vb = opponent.array();
287
288                boolean equal = true;
289                for (int i = 0; i < va.length; i++) {
290                        if (va[i] > vb[i]) {
291                                return false;
292                        } else if (va[i] < vb[i]) {
293                                equal = false;
294                        }
295                }
296                return !equal;
297        }
298
299        /**
300         * Returns {@code true} if this objectives are equal to the specified
301         * objectives. This comparison is based on the {@link #array()} values.
302         * 
303         * @param opponent
304         *            other objectives
305         * @return {@code true} if these objectives dominate the {@code opponent}
306         */
307        public boolean isEqual(Objectives opponent) {
308                double[] va = this.array();
309                double[] vb = opponent.array();
310
311                for (int i = 0; i < va.length; i++) {
312                        if (va[i] != vb[i]) {
313                                return false;
314                        }
315                }
316                return true;
317        }
318
319        /**
320         * Calculates the euclidean distance of two {@link Objectives}. This
321         * calculation is based on the {@link #array()} values.
322         * 
323         * @param other
324         *            the second objectives
325         * @return the euclidean distance
326         */
327        public double distance(Objectives other) {
328                double[] va = this.array();
329                double[] vb = other.array();
330
331                double s = 0;
332                for (int i = 0; i < va.length; i++) {
333                        s += (va[i] - vb[i]) * (va[i] - vb[i]);
334                }
335
336                return Math.sqrt(s);
337        }
338
339        /*
340         * (non-Javadoc)
341         * 
342         * @see java.lang.Object#toString()
343         */
344        @Override
345        public String toString() {
346                StringBuilder s = new StringBuilder("");
347                for (Entry<Objective, Value<?>> entry : map.entrySet()) {
348                        Objective o = entry.getKey();
349                        Value<?> v = entry.getValue();
350                        s.append(o).append("=").append(v).append(" ");
351                }
352                return s.toString();
353        }
