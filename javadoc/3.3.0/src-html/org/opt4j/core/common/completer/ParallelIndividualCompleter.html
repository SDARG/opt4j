023package org.opt4j.core.common.completer;
024
025import java.util.ArrayList;
026import java.util.List;
027import java.util.concurrent.Callable;
028import java.util.concurrent.ExecutionException;
029import java.util.concurrent.ExecutorService;
030import java.util.concurrent.Executors;
031import java.util.concurrent.Future;
032
033import org.opt4j.core.Genotype;
034import org.opt4j.core.Individual;
035import org.opt4j.core.optimizer.Control;
036import org.opt4j.core.optimizer.Optimizer;
037import org.opt4j.core.optimizer.OptimizerStateListener;
038import org.opt4j.core.optimizer.TerminationException;
039import org.opt4j.core.problem.Decoder;
040import org.opt4j.core.problem.Evaluator;
041import org.opt4j.core.start.Constant;
042
043import com.google.inject.Inject;
044
045/**
046 * The {@link ParallelIndividualCompleter} completes {@link Individual}s with
047 * multiple threads.
048 * 
049 * @author lukasiewycz
050 * 
051 */
052public class ParallelIndividualCompleter extends SequentialIndividualCompleter implements OptimizerStateListener {
053        protected final ExecutorService executor;
054
055        /**
056         * The {@link Complete} class completes a single {@link Individual}.
057         * 
058         * @author lukasiewycz
059         * 
060         */
061        protected class Complete implements Callable<Void> {
062
063                protected final Individual individual;
064
065                protected final Control control;
066
067                /**
068                 * Constructs {@link Complete} with an {@link Individual}.
069                 * 
070                 * @param individual
071                 *            the individual to complete
072                 * @param control
073                 *            the control
074                 */
075                public Complete(final Individual individual, final Control control) {
076                        this.individual = individual;
077                        this.control = control;
078                }
079
080                /*
081                 * (non-Javadoc)
082                 * 
083                 * @see java.util.concurrent.Callable#call()
084                 */
085                @Override
086                public Void call() throws TerminationException {
087                        if (!individual.isEvaluated()) {
088                                control.checkpoint();
089                                ParallelIndividualCompleter.this.decode(individual);
090                                control.checkpoint();
091                                ParallelIndividualCompleter.this.evaluate(individual);
092                                control.checkpoint();
093                        }
094                        return null;
095                }
096        }
097
098        /**
099         * Constructs a {@link ParallelIndividualCompleter} with a specified maximal
100         * number of concurrent threads.
101         * 
102         * @param control
103         *            the control
104         * @param decoder
105         *            the decoder
106         * @param evaluator
107         *            the evaluator
108         * @param maxThreads
109         *            the maximal number of parallel threads (using namespace
110         *            {@link ParallelIndividualCompleter})
111         */
112        @Inject
113        public ParallelIndividualCompleter(Control control, Decoder<Genotype, Object> decoder, Evaluator<Object> evaluator,
114                        @Constant(value = "maxThreads", namespace = ParallelIndividualCompleter.class) int maxThreads) {
115                super(control, decoder, evaluator);
116
117                if (maxThreads < 1) {
118                        throw new IllegalArgumentException("Invalid number of threads: " + maxThreads);
119                }
120                this.executor = Executors.newFixedThreadPool(maxThreads);
121        }
122
123        /*
124         * (non-Javadoc)
125         * 
126         * @see org.opt4j.common.completer.SequentialCompleter#complete(java.lang.
127         * Iterable )
128         */
129        @Override
130        public void complete(Iterable<? extends Individual> iterable) throws TerminationException {
131
132                try {
133
134                        List<Future<Void>> returns = new ArrayList<>();
135
136                        for (Individual individual : iterable) {
137                                if (individual.getState() != Individual.State.EVALUATED) {
138                                        returns.add(executor.submit(new Complete(individual, control)));
139                                }
140                        }
141
142                        for (Future<Void> future : returns) {
143                                try {
144                                        future.get();
145                                } catch (InterruptedException e) {
146                                        e.printStackTrace();
147                                }
148                        }
149                } catch (ExecutionException ex) {
150                        executor.shutdownNow();
151                        if (ex.getCause() instanceof TerminationException) {
152                                throw (TerminationException) ex.getCause();
153                        }
154                        throw new RuntimeException(ex);
155                }
156        }
157
158        /*
159         * (non-Javadoc)
160         * 
161         * @see java.lang.Object#finalize()
162         */
163        @Override
164        protected void finalize() throws Throwable {
165                shutdownExecutorService();
166                super.finalize();
167        }
168
169        /**
170         * Shutdown the {@link ExecutorService}.
171         */
172        protected synchronized void shutdownExecutorService() {
173                if (!executor.isShutdown()) {
174                        executor.shutdown();
175                }
176        }
177
178        /*
179         * (non-Javadoc)
180         * 
181         * @see
182         * org.opt4j.core.optimizer.OptimizerStateListener#optimizationStarted(org
183         * .opt4j.core.optimizer.Optimizer)
184         */
185        @Override
186        public void optimizationStarted(Optimizer optimizer) {
187                // do nothing
188        }
189
190        /*
191         * (non-Javadoc)
192         * 
193         * @see
194         * org.opt4j.core.optimizer.OptimizerStateListener#optimizationStopped(org
195         * .opt4j.core.optimizer.Optimizer)
196         */
197        @Override
198        public void optimizationStopped(Optimizer optimizer) {
199                shutdownExecutorService();
200        }
201}
