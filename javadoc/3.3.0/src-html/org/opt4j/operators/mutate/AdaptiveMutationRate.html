024package org.opt4j.operators.mutate;
025
026import static org.opt4j.core.Individual.State.EMPTY;
027
028import org.opt4j.core.Individual;
029import org.opt4j.core.IndividualFactory;
030import org.opt4j.core.IndividualStateListener;
031
032import com.google.inject.Inject;
033import com.google.inject.Singleton;
034
035/**
036 * Self adaptive {@link MutationRate} that uses the size of the genotype (
037 * {@code size}) to estimate a rate. The calculated rate is 1/{@code size}.
038 * 
039 * @author lukasiewycz
040 * 
041 */
042@Singleton
043public class AdaptiveMutationRate implements MutationRate, IndividualStateListener {
044
045        protected boolean isInit = false;
046
047        protected double rate = 0;
048
049        protected final IndividualFactory individualFactory;
050
051        /**
052         * Constructs an {@link AdaptiveMutationRate}.
053         * 
054         * @param individualFactory
055         *            the individual creator
056         */
057        @Inject
058        public AdaptiveMutationRate(IndividualFactory individualFactory) {
059                this.individualFactory = individualFactory;
060        }
061
062        /**
063         * Initializes the listeners.
064         */
065        @Inject
066        public void init() {
067                individualFactory.addIndividualStateListener(this);
068        }
069
070        /*
071         * (non-Javadoc)
072         * 
073         * @see org.opt4j.operator.mutate.MutationRate#get()
074         */
075        @Override
076        public double get() {
077                return rate;
078        }
079
080        /*
081         * (non-Javadoc)
082         * 
083         * @see org.opt4j.operator.mutate.MutationRate#set(double)
084         */
085        @Override
086        public void set(double value) {
087                this.rate = value;
088        }
089
090        /*
091         * (non-Javadoc)
092         * 
093         * @see
094         * org.opt4j.core.IndividualStateListener#inidividualStateChanged(org.opt4j
095         * .core.Individual)
096         */
097        @Override
098        public synchronized void inidividualStateChanged(Individual individual) {
099                if (!isInit && individual.getState() != EMPTY) {
100                        final int size = individual.getGenotype().size();
101                        if (size > 0) {
102                                set(1.0 / individual.getGenotype().size());
103                        }
104                        individualFactory.removeIndividualStateListener(this);
105                        isInit = true;
106                }
107        }
108}
