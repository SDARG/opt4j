023package org.opt4j.benchmarks.queens;
024
025import static org.opt4j.core.Objective.Sign.MIN;
026
027import org.opt4j.core.Objective;
028import org.opt4j.core.Objectives;
029import org.opt4j.core.problem.Evaluator;
030import org.opt4j.core.problem.Priority;
031
032import com.google.inject.Inject;
033
034/**
035 * The evaluator for the {@link QueensProblem}.
036 * 
037 * @author lukasiewycz
038 * 
039 */
040@Priority(-10)
041public class QueensErrorEvaluator implements Evaluator<QueensBoard> {
042
043        protected final QueensProblem problem;
044
045        protected final Objective error = new Objective("error", MIN);
046
047        /**
048         * Constructs the evaluator for the {@link QueensProblem}.
049         * 
050         * @param problem
051         *            the problem to set
052         * 
053         */
054        @Inject
055        public QueensErrorEvaluator(QueensProblem problem) {
056                this.problem = problem;
057        }
058
059        /*
060         * (non-Javadoc)
061         * 
062         * @see org.opt4j.core.problem.Evaluator#evaluate(org.opt4j.core.Phenotype)
063         */
064        @Override
065        public Objectives evaluate(QueensBoard queensBoard) {
066                int errors = countErrors(queensBoard);
067
068                Objectives objectives = new Objectives();
069                objectives.add(error, errors);
070                queensBoard.setFeasible(errors == 0);
071                return objectives;
072        }
073
074        /**
075         * Counts the number of errors for a given {@link QueensBoard}. That means
076         * the number of attacking queens.
077         * 
078         * @param queensBoard
079         *            the board
080         * @return the error count
081         */
082        private int countErrors(QueensBoard queensBoard) {
083                int errorcount = 0;
084
085                errorcount += countErrorsRowsColumns(queensBoard);
086                errorcount += countErrorsDiagonals(queensBoard);
087
088                return errorcount;
089        }
090
091        /**
092         * Helper function to count the errors in rows and columns.
093         * 
094         * @param queensBoard
095         *            the board
096         * @return the error count in rows and columns
097         */
098        protected int countErrorsRowsColumns(QueensBoard queensBoard) {
099                int errorcount = 0;
100                int size = problem.size();
101
102                for (int i = 0; i < size; i++) {
103                        // horizontal
104                        int sum = 0;
105                        for (int j = 0; j < size; j++) {
106                                if (queensBoard.isQueen(i, j)) {
107                                        sum++;
108                                }
109                        }
110                        sum = Math.abs(1 - sum);
111                        errorcount += sum;
112
113                        // vertikal
114                        sum = 0;
115                        for (int j = 0; j < size; j++) {
116                                if (queensBoard.isQueen(j, i)) {
117                                        sum++;
118                                }
119                        }
120                        sum = Math.abs(1 - sum);
121                        errorcount += sum;
122                }
123
124                return errorcount;
125        }
126
127        /**
128         * Helper function to count the errors in diagonals.
129         * 
130         * @param queensBoard
131         *            the board
132         * @return the error count in diagonals
133         */
134        protected int countErrorsDiagonals(QueensBoard queensBoard) {
135                int errorcount = 0;
136                int size = problem.size();
137
138                for (int k = -size + 1; k < size; k++) {
139                        // diagonal 1
140                        int sum = 0;
141                        for (int j = 0; j < size; j++) {
142                                int i = k + j;
143                                if (0 <= i && i < size && queensBoard.isQueen(j, i)) {
144                                        sum++;
145                                }
146                        }
147                        sum = Math.max(0, sum - 1);
148                        errorcount += sum;
149                }
150
151                for (int k = 0; k < 2 * size - 1; k++) {
152                        // diagonal 2
153                        int sum = 0;
154                        for (int j = 0; j < size; j++) {
155                                int i = k - j;
156                                if (0 <= i && i < size && queensBoard.isQueen(j, i)) {
157                                        sum++;
158                                }
159                        }
160                        sum = Math.max(0, sum - 1);
161                        errorcount += sum;
162                }
163
164                return errorcount;
165        }
166
167}
