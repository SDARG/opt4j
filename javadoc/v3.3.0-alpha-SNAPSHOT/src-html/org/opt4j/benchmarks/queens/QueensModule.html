024package org.opt4j.benchmarks.queens;
025
026import org.opt4j.core.config.Icons;
027import org.opt4j.core.config.annotations.Icon;
028import org.opt4j.core.config.annotations.Info;
029import org.opt4j.core.config.annotations.Order;
030import org.opt4j.core.problem.Creator;
031import org.opt4j.core.problem.Decoder;
032import org.opt4j.core.problem.ProblemModule;
033import org.opt4j.core.start.Constant;
034
035/**
036 * The {@link QueensModule} is used for the configuration of the queens problem.
037 * Containing the size and dimensions of the problem as well as the
038 * {@link Decoder} strategy.
039 * 
040 * @author lukasiewycz
041 * 
042 */
043@Icon(Icons.PROBLEM)
044@Info("The n-Queens problem as optimization problem with linear objective functions.")
045public class QueensModule extends ProblemModule {
046
047        @Info("The size of the board (size*size).")
048        @Order(0)
049        @Constant(value = "size", namespace = QueensProblem.class)
050        protected int size = 40;
051
052        @Info("The number of objective functions.")
053        @Order(1)
054        @Constant(value = "dim", namespace = QueensProblem.class)
055        protected int dimensions = 2;
056
057        @Info("The seed of the problem generator (for the objective function).")
058        @Order(2)
059        @Constant(value = "seed", namespace = QueensProblem.class)
060        protected int seed = 0;
061
062        @Info("the used decoder")
063        @Order(3)
064        protected Dec decoder = Dec.SAT;
065
066        /**
067         * The {@link Decoder} strategy for the queens problem.
068         * 
069         * @author lukasiewycz
070         * 
071         */
072        public enum Dec {
073                /**
074                 * Use the {@link QueensCopyDecoder}.
075                 */
076                COPY,
077                /**
078                 * Use the {@link QueensPermutationDecoder}.
079                 */
080                PERMUTATION,
081                /**
082                 * Use the {@link QueensSATDecoder}.
083                 */
084                SAT;
085        }
086
087        /*
088         * (non-Javadoc)
089         * 
090         * @see org.opt4j.start.Opt4JModule#config()
091         */
092        @Override
093        public void config() {
094
095                bind(QueensProblem.class).in(SINGLETON);
096
097                Class<? extends Creator<?>> creatorClass = null;
098                Class<? extends Decoder<?, ?>> decoderClass = null;
099
100                switch (decoder) {
101                default: // SAT
102                        creatorClass = QueensSATDecoder.class;
103                        decoderClass = QueensSATDecoder.class;
104                        break;
105                case PERMUTATION:
106                        creatorClass = QueensPermutationDecoder.class;
107                        decoderClass = QueensPermutationDecoder.class;
108                        break;
109                case COPY:
110                        creatorClass = QueensCopyDecoder.class;
111                        decoderClass = QueensCopyDecoder.class;
112                        break;
113                }
114
115                bindProblem(creatorClass, decoderClass, QueensErrorEvaluator.class);
116                addEvaluator(QueensEvaluator.class);
117        }
118
119        /**
120         * Returns the decoder strategy.
121         * 
122         * @return the decoder strategy
123         */
124        public Dec getDecoder() {
125                return decoder;
126        }
127
128        /**
129         * Sets the decoder strategy.
130         * 
131         * @param decoder
132         *            the decoder strategy
133         */
134        public void setDecoder(Dec decoder) {
135                this.decoder = decoder;
136        }
137
138        /**
139         * Returns the number of dimensions of the queens problem.
140         * 
141         * @return the number of dimensions
142         */
143        public int getDimensions() {
144                return dimensions;
145        }
146
147        /**
148         * Sets the number of dimensions of the queens problem.
149         * 
150         * @param dimensions
151         *            the number of dimensions
152         */
153        public void setDimensions(int dimensions) {
154                this.dimensions = dimensions;
155        }
156
157        /**
158         * Sets the size of the board of the queens problem.
159         * 
160         * @param size
161         *            the size of the board
162         */
163        public void setSize(int size) {
164                this.size = size;
165        }
166
167        /**
168         * Returns the size of the board of the queens problem.
169         * 
170         * @return the size of the board
171         */
172        public int getSize() {
173                return size;
174        }
175
176        /**
177         * Return the seed of the problem generator.
178         * 
179         * @return the seed of the problem generator
180         */
181        public int getSeed() {
182                return seed;
183        }
184
185        /**
186         * Sets the seed of the problem generator.
187         * 
188         * @param seed
189         *            the seed of the problem generator
190         */
191        public void setSeed(int seed) {
192                this.seed = seed;
193        }
194
195}
