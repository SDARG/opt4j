023package org.opt4j.benchmarks.queens;
024
025import static org.opt4j.core.Objective.INFEASIBLE;
026import static org.opt4j.core.Objective.Sign.MIN;
027
028import java.util.ArrayList;
029import java.util.List;
030
031import org.opt4j.core.Objective;
032import org.opt4j.core.Objectives;
033import org.opt4j.core.problem.Evaluator;
034
035import com.google.inject.Inject;
036
037/**
038 * The evaluator for the {@link QueensProblem}.
039 * 
040 * @author lukasiewycz
041 * 
042 */
043public class QueensEvaluator implements Evaluator<QueensBoard> {
044
045        protected final QueensProblem problem;
046
047        protected final List<Objective> objectives = new ArrayList<>();
048
049        /**
050         * Constructs the evaluator for the {@link QueensProblem}.
051         * 
052         * @param problem
053         *            the problem to set
054         * 
055         */
056        @Inject
057        public QueensEvaluator(QueensProblem problem) {
058                this.problem = problem;
059        }
060
061        /**
062         * Initializes the {@link QueensEvaluator}.
063         */
064        @Inject
065        public void init() {
066                for (int d = 0; d < problem.dim(); d++) {
067                        objectives.add(new Objective("sum" + d, MIN));
068                }
069        }
070
071        /*
072         * (non-Javadoc)
073         * 
074         * @see org.opt4j.core.problem.Evaluator#evaluate(org.opt4j.core.Phenotype)
075         */
076        @Override
077        public Objectives evaluate(QueensBoard queensBoard) {
078                Objectives obj = new Objectives();
079                for (int d = 0; d < problem.dim(); d++) {
080                        Objective objective = objectives.get(d);
081
082                        if (queensBoard.isFeasible()) {
083                                int sum = sum(queensBoard, d);
084                                obj.add(objective, sum);
085                        } else {
086                                obj.add(objective, INFEASIBLE);
087                        }
088                }
089                return obj;
090        }
091
092        /**
093         * Sums the costs of a {@link QueensBoard} in a specific dimension {@code d}
094         * .
095         * 
096         * @param board
097         *            the board
098         * @param d
099         *            the dimension
100         * @return the sum of the costs
101         */
102        private int sum(QueensBoard board, int d) {
103                int sum = 0;
104
105                for (int i = 0; i < problem.size(); i++) {
106                        for (int j = 0; j < problem.size(); j++) {
107                                if (board.isQueen(i, j)) {
108                                        sum += problem.costs(d, i, j);
109                                }
110                        }
111                }
112
113                return sum;
114        }
115}
