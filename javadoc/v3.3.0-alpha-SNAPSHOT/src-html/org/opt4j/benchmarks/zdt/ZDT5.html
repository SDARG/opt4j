024package org.opt4j.benchmarks.zdt;
025
026import java.util.List;
027
028import org.opt4j.benchmarks.BinaryString;
029
030/**
031 * Function ZDT 5.
032 * 
033 * @author lukasiewycz
034 * 
035 */
036public class ZDT5 extends ZDTEvaluator<BinaryString> {
037
038        /*
039         * (non-Javadoc)
040         * 
041         * @see org.opt4j.benchmark.zdt.ZDTEvaluator#f1(org.opt4j.core.Phenotype)
042         */
043        @Override
044        protected double f1(BinaryString list) {
045                double f1 = 1.0 + u(list, 0, 30);
046                return f1;
047        }
048
049        /*
050         * (non-Javadoc)
051         * 
052         * @see org.opt4j.benchmark.zdt.ZDTEvaluator#g(org.opt4j.core.Phenotype)
053         */
054        @Override
055        protected double g(BinaryString list) {
056                double sum = 0;
057                int begin = 30;
058                while (begin + 5 <= list.size()) {
059                        int end = begin + 5;
060                        sum += v(u(list, begin, end));
061                        begin = end;
062                }
063                return sum;
064        }
065
066        /*
067         * (non-Javadoc)
068         * 
069         * @see org.opt4j.benchmark.zdt.ZDTEvaluator#h(double, double)
070         */
071        @Override
072        protected double h(double f1, double g) {
073                return 1.0 / f1;
074        }
075
076        /**
077         * Counts the number of {@code true}s in the array.
078         * 
079         * @param x
080         *            list of booleans
081         * @param begin
082         *            start of the array
083         * @param end
084         *            end of the array
085         * @return the number of {@code true}s in the array
086         */
087        protected int u(List<Boolean> x, int begin, int end) {
088                int u = 0;
089                for (int i = begin; i < end; i++) {
090                        if (x.get(i)) {
091                                u++;
092                        }
093                }
094                return u;
095        }
096
097        /**
098         * Returns {@code 2+u} if {@code u<5} else {@code 1}.
099         * 
100         * @param u
101         *            the value {@code u <= 5}
102         * @return {@code 2+u} if {@code u<5} else {@code 1}
103         */
104        protected int v(int u) {
105                if (u < 5) {
106                        return 2 + u;
107                } else if (u == 5) {
108                        return 1;
109                }
110                throw new IllegalArgumentException("Wrong argument u = " + u);
111        }
112
113}
