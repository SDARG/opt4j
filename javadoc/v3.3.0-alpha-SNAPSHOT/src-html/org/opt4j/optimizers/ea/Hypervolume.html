023package org.opt4j.optimizers.ea;
024
025import java.util.ArrayList;
026import java.util.Arrays;
027import java.util.Collection;
028import java.util.Collections;
029import java.util.Comparator;
030import java.util.HashMap;
031import java.util.Iterator;
032import java.util.List;
033import java.util.Map;
034
035import org.opt4j.core.Individual;
036import org.opt4j.core.common.archive.FrontDensityIndicator;
037import org.opt4j.core.start.Constant;

039import com.google.inject.Inject;

041/**
042 * The {@link Hypervolume}, see "Zitzler, E., and Thiele, L. (1998):
043 * Multiobjective Optimization Using Evolutionary Algorithms - A Comparative
044 * Case Study. Parallel Problem Solving from Nature (PPSN-V), 292-301." is a
045 * {@link FrontDensityIndicator} based on determination of the hypervolume
046 * contribution. The calculation is based on a normalization between 0 and 1 in
047 * each dimension and a transformation to a maximization problem. Additionally
048 * an offset value (default 1) is added to each dimension.
049 * 
050 * 
051 * @see SMSModule
052 * @author Ramin Etemaadi
053 * @author Johannes Kruisselbrink
054 * @author Rui Li
055 * @author lukasiewycz
056 * 
057 */
058public class Hypervolume implements FrontDensityIndicator {
059
060        protected final double offset;
061
062        /**
063         * Constructs a {@link Hypervolume}.
064         * 
065         * @param offset
066         *            the offset that is added to each dimension before the
067         *            hypervolume is calculated
068         */
069        @Inject
070        public Hypervolume(@Constant(value = "offset", namespace = Hypervolume.class) double offset) {
071                this.offset = offset;
072        }
073
074        /*
075         * (non-Javadoc)
076         * 
077         * @see
078         * org.opt4j.optimizer.ea.FrontDensityIndicator#getDensityValues(java.util
079         * .Collection)
080         */
081        @Override
082        public Map<Individual, Double> getDensityValues(Collection<Individual> individuals) {
083                return getDensityValues(individuals, this.offset);
084        }
085
086        /**
087         * Calculates the density values for a front of non-dominated individuals
088         * based on the contribution of the {@link Hypervolume}.
089         * 
090         * A special approach for two dimension exists as well as a general approach
091         * for n dimensions.
092         * 
093         * @param individuals
094         *            the individuals
095         * @param offset
096         *            the offset
097         * @return the map of density values
098         */
099        protected Map<Individual, Double> getDensityValues(Collection<Individual> individuals, double offset) {
100                if (individuals.isEmpty()) {
101                        throw new IllegalArgumentException("Individuals is empty.");
102                }
103
104                List<Individual> orderIndividuals = new ArrayList<>(individuals);
105                int m = individuals.iterator().next().getObjectives().size();
106
107                if (m < 2) {
108                        Map<Individual, Double> result = new HashMap<>();
109                        for (Individual individual : individuals) {
110                                result.put(individual, 0.0);
111                        }
112                        return result;
113                } else if (m == 2) {
114                        return calculateHypervolumeContribution2D(orderIndividuals, offset);
115                } else {
116                        return calculateHypervolumeContributionN(orderIndividuals, offset);
117                }
118        }
119
120        /**
121         * Calculates the {@link Hypervolume} contribution for n dimensions.
122         * 
123         * @param individuals
124         *            the individuals
125         * @param offset
126         *            the offset
127         * @return the map of density values
128         */
129        protected Map<Individual, Double> calculateHypervolumeContributionN(List<Individual> individuals, double offset) {
130                Map<Individual, Double> result = new HashMap<>();
131                List<double[]> front = invert(normalize(getMinValues(individuals)), offset);
132
133                int m = front.get(0).length;
134
135                double hvAll = calculateHypervolume(front, m);
136
137                for (int i = 0; i < front.size(); i++) {
138                        List<double[]> iFront = new ArrayList<>(front);
139                        iFront.remove(i);
140                        double iHv = calculateHypervolume(iFront, m);
141                        result.put(individuals.get(i), hvAll - iHv);
142                }
143
144                return result;
145        }
146
147        /**
148         * Calculates the {@link Hypervolume} contribution for two dimensions.
149         * 
150         * @param individuals
151         *            the individuals
152         * @param offset
153         *            the offset
154         * @return the map of density values
155         */
156        protected Map<Individual, Double> calculateHypervolumeContribution2D(List<Individual> individuals, double offset) {
157                Map<Individual, Double> result = new HashMap<>();
158                List<double[]> front = invert(normalize(getMinValues(individuals)), offset);
159                List<double[]> sorted = new ArrayList<>(front);
160
161                Collections.sort(sorted, new Comparator<double[]>() {
162                        @Override
163                        public int compare(double[] o1, double[] o2) {
164                                Double v1 = o1[0];
165                                Double v2 = o2[0];
166                                return v1.compareTo(v2);
167                        }
168                });
169
170                final int size = sorted.size();
171
172                for (int i = 0; i < size; i++) {
173                        double diffX = sorted.get(i)[0] - (i > 0 ? sorted.get(i - 1)[0] : 0);
174                        double diffY = sorted.get(i)[1] - (i < size - 1 ? sorted.get(i + 1)[1] : 0);
175                        double contribution = diffX * diffY;
176
177                        result.put(individuals.get(front.indexOf(sorted.get(i))), contribution);
178                }
179
180                return result;
181        }
182
183        /**
184         * Transforms the non-dominated {@link Individual}s to a front where each
185         * objective is to be minimized.
186         * 
187         * @param individuals
188         *            the individuals
189         * @return the front of vectors that is minimized
190         */
191        protected List<double[]> getMinValues(List<Individual> individuals) {
192                List<double[]> minValues = new ArrayList<>();
193                for (Individual individual : individuals) {
194                        minValues.add(individual.getObjectives().array());
195                }
196                return minValues;
197        }
198
199        /**
200         * Normalizes a front of non-dominated solutions to values between 0 and 1.
201         * 
202         * @param front
203         *            the front of non-dominated solutions
204         * @return the normalized front
205         */
206        protected List<double[]> normalize(List<double[]> front) {
207                int m = front.get(0).length;
208
209                double[] min = new double[m];
210                double[] max = new double[m];
211
212                Arrays.fill(min, +Double.MAX_VALUE);
213                Arrays.fill(max, -Double.MAX_VALUE);
214
215                for (double[] p : front) {
216                        for (int i = 0; i < m; i++) {
217                                min[i] = Math.min(min[i], p[i]);
218                                max[i] = Math.max(max[i], p[i]);
219                        }
220                }
221
222                for (int i = 0; i < m; i++) {
223                        if (min[i] == max[i]) {
224                                max[i]++;
225                        }
226                }
227
228                List<double[]> normalized = new ArrayList<>();
229                for (double[] p : front) {
230                        double[] pn = new double[m];
231                        for (int i = 0; i < m; i++) {
232                                pn[i] = (p[i] - min[i]) / (max[i] - min[i]);
233                        }
234                        normalized.add(pn);
235                }
236                return normalized;
237        }
238
239        /**
240         * Inverts (from a minimization to a maximization problem) a front of
241         * solutions and adds an offset value to each dimension.
242         * 
243         * @param front
244         *            the front of non-dominated solutions
245         * @param offset
246         *            the offset
247         * @return the inverted front
248         */
249        protected List<double[]> invert(List<double[]> front, double offset) {
250                int m = front.get(0).length;
251
252                double[] nadir = new double[m];
253                Arrays.fill(nadir, 1.0 + offset);
254
255                List<double[]> inverted = new ArrayList<>();
256                for (double[] element : front) {
257                        double[] in = new double[element.length];
258                        for (int i = 0; i < element.length; i++) {
259                                in[i] = nadir[i] - element[i];
260                        }
261                        inverted.add(in);
262                }
263
264                return inverted;
265        }
266
267        /**
268         * Implements the {@link Hypervolume} calculations as proposed by Zitzler,
269         * E., and Thiele, L. (1998). All points have positive values in all
270         * dimensions and the hypervolume is calculated from 0.
271         * 
272         * @param front
273         *            the front of non-dominated solutions
274         * @param nObjectives
275         *            the number of objectives
276         * @return the hypervolume
277         */
278        protected double calculateHypervolume(List<double[]> front, int nObjectives) {
279                double volume = 0.0;
280                double distance = 0.0;
281
282                while (!front.isEmpty()) {
283                        List<double[]> nondominatedPoints = filterNondominatedSet(front, nObjectives - 1);
284                        double tempVolume;
285                        if (nObjectives < 3) {
286                                assert nondominatedPoints.size() > 0;
287                                tempVolume = nondominatedPoints.get(0)[0];
288                        } else {
289                                tempVolume = calculateHypervolume(nondominatedPoints, nObjectives - 1);
290                        }
291                        double tempDistance = surfaceUnchangedTo(front, nObjectives - 1);
292                        volume += tempVolume * (tempDistance - distance);
293                        distance = tempDistance;
294                        reduceNondominatedSet(front, nObjectives - 1, distance);
295                }
296
297                return volume;
298        }
299
300        protected List<double[]> filterNondominatedSet(List<double[]> front, int nObjectives) {
301                List<double[]> nondominated = new ArrayList<>();
302
303                for (double[] p1 : front) {
304                        boolean dominated = false;
305                        for (double[] p2 : nondominated) {
306                                if (dominates(p2, p1, nObjectives)) {
307                                        dominated = true;
308                                        break;
309                                }
310                        }
311
312                        if (!dominated) {
313                                for (Iterator<double[]> it = nondominated.iterator(); it.hasNext();) {
314                                        double[] p2 = it.next();
315                                        if (dominates(p1, p2, nObjectives)) {
316                                                it.remove();
317                                        }
318                                }
319                                nondominated.add(p1);
320                        }
321                }
322
323                return nondominated;
324        }
325
326        protected boolean dominates(double[] p1, double[] p2, int nObjectives) {
327                boolean strong = false;
328                for (int i = 0; i < nObjectives; i++) {
329                        if (p1[i] > p2[i]) {
330                                strong = true;
331                        } else if (p1[i] < p2[i]) {
332                                return false;
333                        }
334                }
335                return strong;
336        }
337
338        protected double surfaceUnchangedTo(List<double[]> front, int objective) {
339                assert front.size() > 0;
340                double value = Double.MAX_VALUE;
341                for (double[] p : front) {
342                        value = Math.min(value, p[objective]);
343                }
344                return value;
345        }
346
347        protected void reduceNondominatedSet(List<double[]> front, int objective, double threshold) {
348                List<double[]> lames = new ArrayList<>();
349
350                for (double[] p : front) {
351                        if (p[objective] <= threshold) {
352                                lames.add(p);
353                        }
354                }
355
356                front.removeAll(lames);
357        }
