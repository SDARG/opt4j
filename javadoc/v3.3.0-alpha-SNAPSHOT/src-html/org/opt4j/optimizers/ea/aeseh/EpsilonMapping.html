001package org.opt4j.optimizers.ea.aeseh;
002
003import java.util.Map;
004import java.util.Set;
005
006import org.opt4j.core.Individual;
007import org.opt4j.core.Objective;
008import org.opt4j.core.Objectives;
009
010import com.google.inject.ImplementedBy;
011
012/**
013 * The {@link EpsilonMapping} implements the ε mapping used by the
014 * {@link EpsilonSamplingSelector}.
015 * 
016 * @author Fedor Smirnov
017 *
018 */
019@ImplementedBy(EpsilonMappingAdditive.class)
020public interface EpsilonMapping {
021
022        /**
023         * Maps the given {@link Objectives} on the objectives used for the check of the ε
024         * dominance.
025         * 
026         * @param original
027         *            the actual objectives of the individual
028         * @param epsilon
029         *            the ε value
030         * @param objectiveAmplitudes
031         *            a map containing the amplitude values of the objectives
032         * @return {@link Objectives} enhanced by the ε value
033         */
034        public Objectives mapObjectives(final Objectives original, double epsilon,
035                        Map<Objective, Double> objectiveAmplitudes);
036
037        /**
038         * Creates a map mapping the {@link Objective}s to their amplitudes (difference between
039         * maximal and minimal value). These extreme values are used during ε mapping to
040         * scale the delta according to the objective values of the {@link Individual} group
041         * under consideration.
042         * 
043         * @param individuals
044         * @return map mapping each {@link Objective} onto its amplitude
045         */
046        public Map<Objective, Double> findObjectiveAmplitudes(Set<Individual> individuals);
047
048}
