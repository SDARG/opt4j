001package org.opt4j.optimizers.ea;
002
003import org.opt4j.core.config.Icons;
004import org.opt4j.core.config.annotations.Category;
005import org.opt4j.core.config.annotations.Icon;
006import org.opt4j.core.config.annotations.Parent;
007import org.opt4j.core.start.Opt4JModule;
008
009/**
010 * Abstract module class for the {@link Coupler}.
011 * 
012 * @author Fedor Smirnov
013 *
014 */
015@Icon(Icons.SELECTOR)
016@Category
017@Parent(EvolutionaryAlgorithmModule.class)
018public abstract class CouplerModule extends Opt4JModule {
019
020        /**
021         * Binds the given {@link Coupler}.
022         * 
023         * @param coupler
024         *            the {@link Coupler} to bind
025         */
026        protected void bindCoupler(Class<? extends Coupler> coupler) {
027                bind(Coupler.class).to(coupler).in(SINGLETON);
028        }
029}
