023package org.opt4j.optimizers.ea;
024
025import java.util.ArrayList;
026import java.util.Collection;
027import java.util.Collections;
028import java.util.Comparator;
029import java.util.HashMap;
030import java.util.List;
031import java.util.Map;
032import java.util.Random;
033
034import org.opt4j.core.Individual;
035import org.opt4j.core.common.random.Rand;
036
037import com.google.inject.Inject;
038
039/**
040 * The {@link ElitismSelector} is a single objective elitism select. If a
041 * multi-objective problem is optimized, the objectives are summed up to a
042 * single value.
043 * 
044 * @author lukasiewycz
045 * 
046 */
047public class ElitismSelector implements Selector {
048
049        protected final Random random;
050
051        protected final Map<Individual, Double> fitness = new HashMap<>();
052
053        /**
054         * Comparator that sorts the {@link Individual}s based on their fitness
055         * values.
056         * 
057         * @author lukasiewycz
058         * 
059         */
060        protected class FitnessComparator implements Comparator<Individual> {
061                @Override
062                public int compare(Individual o1, Individual o2) {
063                        final double f1 = fitness.get(o1);
064                        final double f2 = fitness.get(o2);
065                        if (f1 < f2) {
066                                return -1;
067                        } else if (f2 < f1) {
068                                return 1;
069                        } else {
070                                return 0;
071                        }
072                }
073        }
074
075        /**
076         * Constructs an {@link ElitismSelector}.
077         * 
078         * @param random
079         *            the random number generator
080         */
081        @Inject
082        public ElitismSelector(Rand random) {
083                this.random = random;
084        }
085
086        /*
087         * (non-Javadoc)
088         * 
089         * @see org.opt4j.optimizer.ea.Selector#getLames(int, java.util.Collection)
090         */
091        @Override
092        public Collection<Individual> getLames(int lambda, Collection<Individual> population) {
093                List<Individual> list = new ArrayList<>(population);
094                calculateFitness(list);
095                Collections.sort(list, new FitnessComparator());
096                Collections.reverse(list);
097
098                List<Individual> lames = new ArrayList<>();
099                for (int i = 0; i < lambda; i++) {
100                        lames.add(list.get(i));
101                }
102                return lames;
103        }
104
105        /*
106         * (non-Javadoc)
107         * 
108         * @see org.opt4j.optimizer.ea.Selector#getParents(int,
109         * java.util.Collection)
110         */
111        @Override
112        public Collection<Individual> getParents(int mu, Collection<Individual> population) {
113                List<Individual> parents = new ArrayList<>();
114                List<Individual> individuals = new ArrayList<>(population);
115
116                for (int i = 0; i < mu; i++) {
117                        int r = random.nextInt(individuals.size());
118                        parents.add(individuals.get(r));
119                }
120                return parents;
121        }
122
123        /**
124         * Calculates the fitness of the {@link Individual}s: the sum of all double
125         * values (these always have to be minimized) of the objectives.
126         * 
127         * @param individuals
128         *            the individuals to process
129         */
130        protected void calculateFitness(Collection<Individual> individuals) {
131                fitness.clear();
132                for (Individual individual : individuals) {
133                        double f = 0;
134                        for (double v : individual.getObjectives().array()) {
135                                f += v;
136                        }
137                        fitness.put(individual, f);
138                }
139        }
140
141        /*
142         * (non-Javadoc)
143         * 
144         * @see org.opt4j.optimizer.ea.Selector#init(int)
145         */
146        @Override
147        public void init(int maxsize) {
148                // do nothing
149        }
150
151}
