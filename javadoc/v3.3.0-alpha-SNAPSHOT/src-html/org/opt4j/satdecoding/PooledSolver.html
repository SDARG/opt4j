023package org.opt4j.satdecoding;
024
025import java.util.HashSet;
026import java.util.Set;
027import java.util.concurrent.BlockingQueue;
028import java.util.concurrent.LinkedBlockingQueue;
029
030import org.opt4j.core.start.Constant;
031
032import com.google.inject.Inject;
033import com.google.inject.Provider;
034
035/**
036 * The {@link PooledSolver} is a decorator that enables pooling of
037 * {@link Solver} instances and, therefore, a parallel execution of the
038 * {@link Solver}s.
039 * 
040 * @author lukasiewycz
041 * 
042 */
043public class PooledSolver implements Solver {
044
045        protected final Set<Solver> solvers = new HashSet<>();
046
047        protected final BlockingQueue<Solver> queue = new LinkedBlockingQueue<>();
048
049        /**
050         * Constructs a {@link PooledSolver}.
051         * 
052         * @param solverProvider
053         *            a solver provider
054         * @param instances
055         *            the number of pooled instances
056         */
057        @Inject
058        public PooledSolver(@Constant(value = "solver", namespace = PooledSolver.class) Provider<Solver> solverProvider,
059                        @Constant(value = "instances", namespace = PooledSolver.class) int instances) {
060                for (int i = 0; i < instances; i++) {
061                        Solver solver = solverProvider.get();
062                        solvers.add(solver);
063                        queue.add(solver);
064                }
065        }
066
067        /*
068         * (non-Javadoc)
069         * 
070         * @see org.opt4j.sat.Solver#addConstraint(org.opt4j.sat.Constraint)
071         */
072        @Override
073        public void addConstraint(Constraint constraint) {
074                for (Solver solver : solvers) {
075                        solver.addConstraint(constraint);
076                }
077        }
078
079        /*
080         * (non-Javadoc)
081         * 
082         * @see org.opt4j.sat.Solver#solve(org.opt4j.sat.Order)
083         */
084        @Override
085        public Model solve(Order order) throws TimeoutException {
086                Solver solver = null;
087                Model model = null;
088                try {
089                        solver = queue.take();
090                        model = solver.solve(order);
091                } catch (InterruptedException e) {
092                        e.printStackTrace();
093                } finally {
094                        if (solver != null) {
095                                queue.offer(solver);
096                        }
097                }
098                return model;
099        }
100
101}
