023package org.opt4j.core;
024
025import java.util.Set;
026import java.util.concurrent.CopyOnWriteArraySet;
027
028import org.opt4j.core.problem.Creator;
029
030import com.google.inject.Inject;
031import com.google.inject.Provider;
032
033/**
034 * The {@link AbstractIndividualFactory} creates {@link Individual}s using a
035 * given {@link Provider} and sets the registered
036 * {@link IndividualStateListener}s. The {@link Creator} is used to create the
037 * problem specific {@link Genotype} with which the created {@link Individual}
038 * is initialized.
039 * 
040 * @author lukasiewycz
041 * 
042 * @param <I>
043 *            the type of Individual
044 */
045public class AbstractIndividualFactory<I extends Individual> implements IndividualFactory {
046
047        protected final Creator<Genotype> creator;
048        protected final Provider<I> individualProvider;
049
050        protected final Set<IndividualStateListener> individualStateListeners = new CopyOnWriteArraySet<>();
051
052        /**
053         * Constructs an {@link AbstractIndividualFactory} with a {@link Provider}
054         * for {@link Individual}s.
055         * 
056         * @param individualProvider
057         *            the provider that creates new individuals
058         * @param creator
059         *            the creator that creates random genotypes
060         */
061        public AbstractIndividualFactory(Provider<I> individualProvider, Creator<Genotype> creator) {
062                this.individualProvider = individualProvider;
063                this.creator = creator;
064        }
065
066        /**
067         * The {@link IndividualStateListener}s will be transmitted to each
068         * {@link Individual} that is created by this class. The listeners are
069         * invoked if the state of the {@link Individual} changes.
070         * 
071         * @param listeners
072         *            the listeners
073         */
074        @Inject
075        protected void injectListeners(Set<IndividualStateListener> listeners) {
076                individualStateListeners.addAll(listeners);
077        }
078
079        /*
080         * (non-Javadoc)
081         * 
082         * @see
083         * org.opt4j.core.IndividualFactory#addIndividualStateListener(org.opt4j
084         * .core.IndividualStateListener)
085         */
086        @Override
087        public void addIndividualStateListener(IndividualStateListener listener) {
088                individualStateListeners.add(listener);
089        }
090
091        /*
092         * (non-Javadoc)
093         * 
094         * @see org.opt4j.core.IndividualFactory#create()
095         */
096        @Override
097        public Individual create() {
098                Individual individual = individualProvider.get();
099                individual.setIndividualStatusListeners(individualStateListeners);
100                Genotype genotype = creator.create();
101                individual.setGenotype(genotype);
102
103                return individual;
104        }
105
106        /*
107         * (non-Javadoc)
108         * 
109         * @see
110         * org.opt4j.core.IndividualFactory#create(org.opt4j.core.problem.Genotype)
111         */
112        @Override
113        public Individual create(Genotype genotype) {
114                Individual individual = individualProvider.get();
115                individual.setIndividualStatusListeners(individualStateListeners);
116                individual.setGenotype(genotype);
117                return individual;
118        }
119
120        /*
121         * (non-Javadoc)
122         * 
123         * @see
124         * org.opt4j.core.IndividualFactory#removeIndividualStateListener(org.opt4j
125         * .core.IndividualStateListener)
126         */
127        @Override
128        public void removeIndividualStateListener(IndividualStateListener listener) {
129                individualStateListeners.remove(listener);
130        }
131}
