024package org.opt4j.core.config.visualization;
025
026import java.awt.BorderLayout;
027import java.awt.HeadlessException;
028import java.awt.Window;
029import java.awt.event.WindowAdapter;
030import java.awt.event.WindowEvent;
031import java.awt.event.WindowListener;
032import java.io.File;
033import java.io.IOException;
034
035import javax.swing.JPanel;
036import javax.swing.JSeparator;
037import javax.swing.SwingUtilities;
038
039import org.opt4j.core.config.Icons;
040import org.opt4j.core.start.Opt4J;
041
042import com.google.inject.Inject;
043import com.google.inject.Singleton;
044
045/**
046 * The {@link DefaultApplicationFrame}.
047 * 
048 * @author lukasiewycz
049 * 
050 */
051@SuppressWarnings("serial")
052@Singleton
053public class DefaultApplicationFrame extends ApplicationFrame {
054
055        protected final ContentPanel contentPanel;
056        protected final About about;
057        protected final ToolBar toolBar;
058        protected final Menu menu;
059        protected final FileOperations fileOperations;
060
061        protected final String title = "Opt4J " + Opt4J.getVersion() + " Configurator";
062
063        /**
064         * Constructs a {@link DefaultApplicationFrame} .
065         * 
066         * @param panel
067         *            the content panel
068         * @param about
069         *            the about information
070         * @param menu
071         *            the menu
072         * @param toolBar
073         *            the tool bar
074         * @param fileOperations
075         *            the file operations
076         * @throws HeadlessException
077         */
078        @Inject
079        public DefaultApplicationFrame(ContentPanel panel, About about, Menu menu, ToolBar toolBar,
080                        FileOperations fileOperations) throws HeadlessException {
081                this.contentPanel = panel;
082                this.about = about;
083                this.menu = menu;
084                this.toolBar = toolBar;
085                this.fileOperations = fileOperations;
086        }
087
088        /**
089         * Registers the listeners.
090         */
091        @Inject
092        public void init() {
093                fileOperations.addListener(this);
094        }
095
096        /*
097         * (non-Javadoc)
098         * 
099         * @see org.opt4j.config.visualization.Startupable#startup()
100         */
101        @Override
102        public void startup() {
103                WindowListener exitListener = new WindowAdapter() {
104                        @Override
105                        public void windowClosing(WindowEvent e) {
106                                Window window = e.getWindow();
107                                window.setVisible(false);
108                                window.dispose();
109                                try {
110                                        Thread.sleep(1000);
111                                } catch (InterruptedException ignore) {
112                                } finally {
113                                        System.exit(0);
114                                }
115                        }
116                };
117                addWindowListener(exitListener);
118
119                setIconImage(Icons.getIcon(Icons.OPT4J).getImage());
120                setTitle(title);
121                setLayout(new BorderLayout());
122                JSeparator jSeparator = new JSeparator();
123
124                JPanel toolBarPanel = new JPanel(new BorderLayout());
125                toolBarPanel.add(toolBar, BorderLayout.CENTER);
126                toolBarPanel.add(jSeparator, BorderLayout.SOUTH);
127
128                add(toolBarPanel, BorderLayout.NORTH);
129                add(contentPanel, BorderLayout.CENTER);
130
131                setJMenuBar(menu);
132
133                menu.startup();
134                toolBar.startup();
135                contentPanel.startup();
136
137                pack();
138                setVisible(true);
139        }
140
141        /*
142         * (non-Javadoc)
143         * 
144         * @see
145         * org.opt4j.conf.gui.FileOperations.FileOperationsListener#setCurrentFile
146         * (java.io.File)
147         */
148        @Override
149        public void setCurrentFile(final File file) {
150                SwingUtilities.invokeLater(new Runnable() {
151                        @Override
152                        public void run() {
153                                try {
154                                        setTitle(title + " [" + file.getCanonicalPath() + "]");
155                                } catch (IOException e) {
156                                        e.printStackTrace();
157                                }
158                        }
159                });
160        }
161
162}
