023package org.opt4j.core.config.visualization;
024
025import java.awt.event.ActionEvent;
026import java.awt.event.KeyEvent;
027import java.io.File;
028import java.util.HashSet;
029import java.util.Set;
030
031import javax.swing.JButton;
032import javax.swing.JToolBar;
033
034import org.opt4j.core.config.ExecutionEnvironment;
035import org.opt4j.core.config.Icons;
036import org.opt4j.core.config.ModuleSaver;
037import org.opt4j.core.config.visualization.FileOperations.FileOperationsListener;
038
039import com.google.inject.Inject;
040import com.google.inject.Module;
041import com.google.inject.Singleton;
042
043/**
044 * The {@link ToolBar}.
045 * 
046 * @author lukasiewycz
047 * 
048 */
049@SuppressWarnings("serial")
050@Singleton
051public class ToolBar extends JToolBar implements FileOperationsListener, Startupable {
052
053        protected JButton execute;
054        protected JButton save;
055        protected JButton saveAs;
056        protected JButton load;
057        protected JButton showXML;
058
059        protected final ExecutionEnvironment executionEnvironment;
060        protected final FileOperations fileOperations;
061        protected final SelectedModules selectedModules;
062
063        /**
064         * Constructs a {@link ToolBar}.
065         * 
066         * @param executionEnvironment
067         *            the execution environment
068         * @param selectedModules
069         *            the selected modules
070         * @param fileOperations
071         *            the file operations
072         */
073        @Inject
074        public ToolBar(final ExecutionEnvironment executionEnvironment, final SelectedModules selectedModules,
075                        FileOperations fileOperations) {
076                super();
077                this.executionEnvironment = executionEnvironment;
078                this.selectedModules = selectedModules;
079                this.fileOperations = fileOperations;
080        }
081
082        /**
083         * Initializes the listeners.
084         */
085        @Inject
086        public void init() {
087                fileOperations.addListener(this);
088                setFloatable(false);
089        }
090
091        /*
092         * (non-Javadoc)
093         * 
094         * @see org.opt4j.conf.gui.Startupable#startup()
095         */
096        @Override
097        public void startup() {
098                execute = new JButton("Run ", Icons.getIcon(Icons.PLAY)) {
099                        {
100                                setFocusable(false);
101                                setMnemonic(KeyEvent.VK_R);
102                                addActionListener((ActionEvent e) -> {
103                                        Set<Module> modules = new HashSet<>();
104                                        modules.addAll(selectedModules);
105                                        executionEnvironment.execute(modules);
106                                });
107                        }
108                };
109
110                load = new JButton("Load ... ", Icons.getIcon(Icons.FOLDER)) {
111                        {
112                                setFocusable(false);
113                                setMnemonic(KeyEvent.VK_L);
114                                addActionListener((ActionEvent e) -> fileOperations.load());
115                        }
116                };
117                save = new JButton("Save ", Icons.getIcon(Icons.DISK)) {
118                        {
119                                setFocusable(false);
120                                setMnemonic(KeyEvent.VK_S);
121                                setEnabled(false);
122                                addActionListener((ActionEvent e) -> fileOperations.save());
123                        }
124                };
125                saveAs = new JButton("Save As ... ", Icons.getIcon(Icons.DISK)) {
126                        {
127                                setFocusable(false);
128                                setMnemonic(KeyEvent.VK_A);
129                                addActionListener((ActionEvent e) -> fileOperations.saveAs());
130                        }
131                };
132
133                showXML = new JButton("Show configuration", Icons.getIcon(Icons.XML)) {
134                        {
135                                setFocusable(false);
136                                setMnemonic(KeyEvent.VK_X);
137                                addActionListener((ActionEvent e) -> {
138                                        ModuleSaver saver = new ModuleSaver();
139                                        ClipboardFrame clipboardFrame = new ClipboardFrame(saver.toXMLString(selectedModules));
140                                        clipboardFrame.pack();
141                                        clipboardFrame.setVisible(true);
142                                });
143                        }
144                };
145
146                add(execute);
147                addSeparator();
148                add(load);
149                add(save);
150                add(saveAs);
151                addSeparator();
152                add(showXML);
153        }
154
155        /*
156         * (non-Javadoc)
157         * 
158         * @see
159         * org.opt4j.conf.gui.FileOperations.FileOperationsListener#setCurrentFile
160         * (java.io.File)
161         */
162        @Override
163        public void setCurrentFile(File file) {
164                save.setEnabled(file == null ? false : true);
165        }
166}
