023package org.opt4j.core.config.visualization;
024
025import java.io.File;
026
027import javax.swing.SwingUtilities;
028import javax.swing.UIManager;
029
030import org.opt4j.core.config.Task;
031
032import com.google.inject.Binder;
033import com.google.inject.Guice;
034import com.google.inject.Injector;
035import com.google.inject.Module;
036import com.google.inject.Singleton;
037
038/**
039 * The {@link Configurator} configures the modules and starts the specific
040 * {@link ApplicationFrame}.
041 * 
042 * @author lukasiewycz
043 * @see ApplicationFrame
044 * 
045 */
046@Singleton
047public class Configurator {
048
049        /**
050         * Starts the {@link Configurator}.
051         * 
052         * @param args
053         *            specify one or more input configuration files to be loaded
054         * @throws Exception
055         */
056        public static void main(String[] args) throws Exception {
057                Configurator configurator = new Configurator();
058                configurator.start(args);
059        }
060
061        /**
062         * Starts the {@link Configurator}.
063         * 
064         * @param args
065         *            the input configuration files to be loaded
066         * @throws ClassNotFoundException
067         */
068        public void start(String[] args) throws ClassNotFoundException {
069                String filename = null;
070                Class<? extends Task> taskClass = null;
071                if (args.length < 1) {
072                        throw new IllegalArgumentException("Specify the task class");
073                }
074
075                String taskname = args[0];
076                taskClass = Class.forName(taskname).asSubclass(Task.class);
077                if (args.length > 1) {
078                        filename = args[1];
079                }
080
081                main(taskClass, filename);
082        }
083
084        /**
085         * Returns the module for the {@link Configurator}.
086         * 
087         * @param taskClass
088         *            the task class
089         * @return the module
090         */
091        public Module getModule(final Class<? extends Task> taskClass) {
092                Module module = new Module() {
093                        @Override
094                        public void configure(Binder b) {
095                                b.bind(Task.class).to(taskClass);
096                        }
097                };
098                return module;
099        }
100
101        /**
102         * Starts the {@link Configurator} with the specified task class and the
103         * file to be loaded.
104         * 
105         * @param taskClass
106         *            the task class
107         * @param filename
108         *            the filename of the configuration file to be loaded
109         */
110        public void main(final Class<? extends Task> taskClass, final String filename) {
111
112                try {
113                        UIManager.setLookAndFeel(UIManager.getSystemLookAndFeelClassName());
114                } catch (Exception e) {
115                        e.printStackTrace();
116                }
117
118                Module module = getModule(taskClass);
119
120                final Injector injector = Guice.createInjector(module);
121
122                SwingUtilities.invokeLater(new Runnable() {
123                        @Override
124                        public void run() {
125                                ApplicationFrame frame = injector.getInstance(ApplicationFrame.class);
126                                frame.startup();
127                                if (filename != null) {
128                                        File file = new File(filename);
129                                        FileOperations fo = injector.getInstance(FileOperations.class);
130                                        fo.load(file);
131                                }
132                        }
133                });
134        }
135
136}
