037/**
038 * The {@link Format} contains several format rules.
039 * 
040 * @author lukasiewycz
041 * 
042 */
043public class Format {
044        protected static final String xmlBreak = "<br/>";
045
046        /**
047         * Returns the name of a {@link Class}.
048         * 
049         * @param clazz
050         *            the class
051         * @return the name
052         */
053        public String getName(Class<?> clazz) {
054                Name name = clazz.getAnnotation(Name.class);
055                Category category = clazz.getAnnotation(Category.class);
056                if (name != null && !"".equals(name.value())) {
057                        return name.value();
058                } else if (category != null && !"".equals(category.value())) {
059                        return category.value();
060                } else {
061                        String n = clazz.getName();
062                        int c;
063                        while ((c = n.indexOf('.')) != -1) {
064                                n = n.substring(c + 1);
065                        }
066
067                        // Trim suffix "Module".
068                        if (n.endsWith("Module")) {
069                                n = n.substring(0, n.length() - "Module".length());
070                        }
071                        return n;
072                }
073        }
074
075        /**
076         * Returns the info of a {@link Class}.
077         * 
078         * @param c
079         *            the class
080         * @return the info
081         */
082        public String getInfo(Class<?> c) {
083                Info info = c.getAnnotation(Info.class);
084                String text = null;
085                if (info != null && !"".equals(info.value())) {
086                        text = info.value();
087                }
088                return text;
089        }
090
091        /**
092         * Returns the tooltip of a {@link Class}.
093         * 
094         * @param c
095         *            the class
096         * @return the tooltip
097         */
098        public String getTooltip(Class<?> c) {
099
100                String name = getName(c);
101                String info = getInfo(c);
102
103                String text = "<html><b>" + name + "</b>";
104                if (info != null) {
105                        text += xmlBreak + info;
106                }
107
108                Citation citation = c.getAnnotation(Citation.class);
109                if (citation != null)
110                        text += xmlBreak + xmlBreak + "<span style='color: gray;'>[" + formatIEEE(citation) + "]</span>";
111
112                text += "</html>";
113
114                return text;
115        }
116
117        /**
118         * Returns the icon of a {@link Class}.
119         * 
120         * @param clazz
121         *            the class
122         * @return the icon
123         */
124        public ImageIcon getIcon(Class<?> clazz) {
125                Icon icon = clazz.getAnnotation(Icon.class);
126
127                if (icon == null) {
128                        return null;
129                }
130
131                return Icons.getIcon(icon.value());
132        }

142                Class<?> c = module.getModule().getClass();
143
144                return getName(c);

154        public String getTooltip(PropertyModule module) {
155                Class<?> c = module.getModule().getClass();
156
157                return getTooltip(c);

163         * @param module

167        public ImageIcon getIcon(PropertyModule module) {
168                Class<?> c = module.getModule().getClass();
169
170                return getIcon(c);
171        }
172
173        /**
174         * Converts the given {@link ImageIcon} into a folder icon.
175         * 
176         * @param icon
177         *            the icon
178         * @return the converted folder icon
179         */
180        public ImageIcon asFolder(ImageIcon icon) {
181                ImageIcon i = Icons.getIcon(Icons.FOLDER);
182                if (icon != null) {
183                        ImageIcon background = icon;
184                        ImageIcon foreground = Icons.getIcon(Icons.SFOLDER);
185                        i = Icons.merge(background, foreground);
186                }
187                return i;
188        }
189
190        /**
191         * Returns the formatted name of the property.
192         * 
193         * @param property
194         *            the property
195         * @return the formatted name
196         */
197        public String getName(Property property) {
198                String name;
199                Name n = property.getAnnotation(Name.class);
200                if (n == null) {
201                        name = property.getName();
202                } else {
203                        name = n.value();
204                }
205                return name;
206        }
207
208        /**
209         * Returns the formatted (html) tooltip of a {@link Property}.
210         * 
211         * @param property
212         *            the property
213         * @return the tooltip
214         */
215        public String getTooltip(Property property) {
216                String text = "<html><b>" + getName(property) + "</b>";
217                String info = property.getInfo();
218                if (info != null) {
219                        text += xmlBreak + info;
220                }
221                if (property.getType().isEnum()) {
222                        Class<?> type = property.getType();
223                        text += xmlBreak;
224
225                        Field[] fields = type.getDeclaredFields();
226                        for (Field field : fields) {
227                                if (field.isEnumConstant()) {
228                                        String name = field.getName();
229                                        Icon icon = field.getAnnotation(Icon.class);
230                                        Info i = field.getAnnotation(Info.class);
231                                        text += "&nbsp;" + name;
232                                        if (icon != null || i != null) {
233                                                text += " - ";
234                                        }
235                                        if (icon != null) {
236                                                text += "<img src=\"" + Icons.getURL(icon.value()) + "\">";
237
238                                                System.out.println(text + " " + icon.value());
239                                        }
240                                        if (i != null) {
241                                                text += i.value();
242                                        }
243
244                                        Citation c = field.getAnnotation(Citation.class);
245                                        if (c != null)
246                                                text += "&nbsp;&nbsp;<span style='color: gray;'>[" + formatIEEE(c) + "]</span>";
247                                        text += xmlBreak;
248                                }
249                        }
250                }
251
252                text += "</html>";
253
254                return text;
255        }
256
257        /**
258         * Formats a text for tooltip (inserts break-lines).
259         * 
260         * @param text
261         *            the input text
262         * @return the formatted tooltip text
263         */
264        public String formatTooltip(String text) {
265                return "<html>" + text.replaceAll("\n", xmlBreak) + "</html>";
266        }

268        public static String formatIEEE(Citation citation) {
269                StringBuilder builder = new StringBuilder(citation.authors());
270                builder.append(": <em>").append(citation.title()).append(".</em> ");
271                if (!citation.journal().isEmpty()) {
272                        builder.append("In: <em>").append(citation.journal()).append(".</em> ");
273                }
274                if (citation.volume() >= 0) {
275                        builder.append("vol. ").append(citation.volume()).append(", ");
276                }
277                if (citation.number() >= 0) {
278                        builder.append("no. ").append(citation.number()).append(", ");
279                }
280                if (citation.pageFirst() >= 0) {
281                        if (citation.pageLast() > 0 && citation.pageFirst() != citation.pageLast()) {
282                                builder.append("pp. ").append(citation.pageFirst()).append("&ndash;").append(citation.pageLast())
283                                                .append(", ");
284                        } else {
285                                builder.append("p. ").append(citation.pageFirst()).append(", ");
286                        }
287                }
288                if (citation.month() != UNKNOWN) {
289                        builder.append(citation.month().toMonth().getDisplayName(FULL, ENGLISH)).append(" ");
290                }
291                builder.append(citation.year()).append(".");
292
293                if (!citation.doi().isEmpty()) {
294                        builder.append(" <a href=\"https://doi.org/").append(citation.doi()).append("\">").append(citation.doi())
295                                        .append("</a>");
296                }
297                return builder.toString();
298        }
299
300        public static String formatJava(Citation citation) {
301                StringBuilder builder = new StringBuilder();
302                if (!citation.authors().isEmpty())
303                        builder.append(citation.authors()).append(": ");
304                builder.append("\"").append(citation.title()).append("\". ");
305                if (!citation.journal().isEmpty())
306                        builder.append("In: ").append(citation.journal()).append(". ");
307                if (citation.volume() >= 0)
308                        builder.append("vol. ").append(citation.volume()).append(", ");
309                if (citation.number() >= 0)
310                        builder.append("no. ").append(citation.number()).append(", ");
311                if (citation.pageFirst() >= 0) {
312                        if (citation.pageLast() > 0 && citation.pageFirst() != citation.pageLast()) {
313                                builder.append("pp. ").append(citation.pageFirst()).append("–").append(citation.pageLast())
314                                                .append(", ");
315                        } else {
316                                builder.append("p. ").append(citation.pageFirst()).append(", ");
317                        }
318                }
319                if (citation.month() != UNKNOWN) {
320                        builder.append(citation.month().toMonth().getDisplayName(FULL, ENGLISH)).append(" ");
321                }
322                builder.append(citation.year()).append(".");
323                if (!citation.doi().isEmpty()) {
324                        builder.append(" https://doi.org/").append(citation.doi());
325                }
326                return builder.toString();
327        }
