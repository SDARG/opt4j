023package org.opt4j.core.config;
024
025import java.util.ArrayList;
026import java.util.Collection;
027import java.util.List;
028import java.util.Set;
029import java.util.concurrent.CopyOnWriteArraySet;
030import java.util.concurrent.ExecutionException;
031import java.util.concurrent.ExecutorService;
032import java.util.concurrent.Executors;
033import java.util.concurrent.Future;
034
035import com.google.inject.Inject;
036import com.google.inject.Module;
037import com.google.inject.Provider;
038import com.google.inject.Singleton;
039
040/**
041 * The {@link ExecutionEnvironment} for {@link Task}s. The environment starts
042 * and monitors optimization tasks. Tasks are not submitted directly, instead a
043 * collection of {@link Module}s is submitted and a {@link Task} is created. The
044 * {@link ExecutionEnvironment} is processing the {@link Task}s sequentially.
045 * 
046 * @author lukasiewycz
047 * @see Task
048 * 
049 */
050@Singleton
051public class ExecutionEnvironment implements TaskStateListener {
052
053        protected final Provider<Task> taskProvider;
054
055        protected final ExecutorService executor;
056
057        protected final List<Task> tasks = new ArrayList<>();
058
059        protected final Set<TaskListener> listeners = new CopyOnWriteArraySet<>();
060
061        /**
062         * Constructs a {@link ExecutionEnvironment}.
063         * 
064         * @param taskProvider
065         *            the task provider
066         */
067        @Inject
068        public ExecutionEnvironment(Provider<Task> taskProvider) {
069                this.taskProvider = taskProvider;
070                this.executor = Executors.newFixedThreadPool(1);
071        }
072
073        /**
074         * Executes the specified modules: A {@link Task} is created and submitted.
075         * 
076         * @param modules
077         *            the collection of modules for a {@link Task}
078         */
079        public void execute(Collection<Module> modules) {
080
081                final Task task = taskProvider.get();
082                task.addStateListener(this);
083                task.init(modules);
084
085                addTask(task);
086
087                Thread thread = new Thread() {
088                        @Override
089                        public void run() {
090                                Future<Void> future = executor.submit(task);
091                                try {
092                                        future.get();
093                                } catch (InterruptedException e) {
094                                } catch (ExecutionException e) {
095                                }
096                        }
097                };
098                thread.start();
099        }
100
101        /**
102         * Adds a {@link Task} and calls the listeners.
103         * 
104         * @param task
105         *            the task to add
106         */
107        protected void addTask(final Task task) {
108                tasks.add(task);
109                for (TaskListener listener : listeners) {
110                        listener.added(task);
111                }
112        }
113
114        /**
115         * Adds a {@link TaskListener}.
116         * 
117         * @param listener
118         *            the listener to be added
119         */
120        public void addListener(TaskListener listener) {
121                listeners.add(listener);
122        }
123
124        /**
125         * Removes a {@link TaskListener}.
126         * 
127         * @see #addListener
128         * @param listener
129         *            the listener to be removed
130         */
131        public void removeListener(TaskListener listener) {
132                listeners.remove(listener);
133        }
134
135        /**
136         * Returns the list of all {@link Task}s.
137         * 
138         * @return the list of all tasks
139         */
140        public List<Task> getTasks() {
141                return tasks;
142        }
143
144        /*
145         * (non-Javadoc)
146         * 
147         * @see
148         * org.opt4j.config.TaskStateListener#stateChanged(org.opt4j.config.Task)
149         */
150        @Override
151        public void stateChanged(Task task) {
152                for (TaskListener listener : listeners) {
153                        listener.stateChanged(task);
154                }
155        }
156
157}
