023package org.opt4j.core.config;
024
025import java.util.Collections;
026import java.util.HashMap;
027import java.util.HashSet;
028import java.util.Iterator;
029import java.util.Map;
030import java.util.Set;
031
032import org.opt4j.core.config.annotations.Multi;
033import org.opt4j.core.config.visualization.Configurator;
034
035import com.google.inject.Inject;
036import com.google.inject.Module;
037import com.google.inject.Singleton;
038
039/**
040 * The {@link ModuleRegister} contains all found modules for the
041 * {@link Configurator}.
042 * 
043 * @author lukasiewycz.
044 * 
045 */
046@Singleton
047public class ModuleRegister implements Iterable<Class<? extends Module>> {
048
049        protected final Map<Class<? extends Module>, PropertyModule> map = Collections.synchronizedMap(new HashMap<>());
050        protected final Set<Class<? extends Module>> set = Collections.synchronizedSet(new HashSet<>());
051
052        protected final ModuleList finder;
053
054        protected boolean isInit = false;
055
056        /**
057         * Constructs and populates the {@link ModuleRegister}.
058         * 
059         * @param finder
060         *            the module finder to be used
061         */
062        @Inject
063        public ModuleRegister(final ModuleList finder) {
064                super();
065                this.finder = finder;
066        }
067
068        /*
069         * (non-Javadoc)
070         * 
071         * @see java.lang.Iterable#iterator()
072         */
073        @Override
074        public synchronized Iterator<Class<? extends Module>> iterator() {
075                checkInit();
076
077                return set.iterator();
078        }
079
080        /**
081         * Returns the number of found modules.
082         * 
083         * @see java.util.HashSet#size()
084         * @return the number of found modules
085         */
086        public synchronized int size() {
087                checkInit();
088                return set.size();
089        }
090
091        /**
092         * Returns the {@link PropertyModule} of the specified class and creates a
093         * new instance if necessary.
094         * 
095         * If the property module can not be created, {@code null} is returned.
096         * 
097         * @param clazz
098         *            the class of the module
099         * @return the property module or null
100         */
101        public PropertyModule get(Class<? extends Module> clazz) {
102                PropertyModule propertyModule = null;
103
104                if (clazz.getAnnotation(Multi.class) == null && map.containsKey(clazz)) {
105                        propertyModule = map.get(clazz);
106                } else {
107
108                        try {
109                                Module module = clazz.newInstance();
110                                propertyModule = new PropertyModule(module);
111                                map.put(clazz, propertyModule);
112                        } catch (InstantiationException e) {
113                                System.err.println("Failed to create an instance of module " + clazz + ".");
114                        } catch (IllegalAccessException e) {
115                                System.err.println("Failed to create an instance of module " + clazz + ".");
116                        }
117                }
118
119                return propertyModule;
120        }
121
122        /**
123         * Initialize the modules once.
124         */
125        private synchronized void checkInit() {
126                while (!isInit) {
127                        System.out.print("Searching Modules ... ");
128                        for (Class<? extends Module> clazz : finder.getModules()) {
129                                set.add(clazz);
130                                try {
131                                        // try to initialize static code of the module classes while
132                                        // searching modules and not later
133                                        clazz.newInstance();
134                                } catch (InstantiationException e) {
135                                } catch (IllegalAccessException e) {
136                                }
137                        }
138                        System.out.println("Done");
139
140                        isInit = true;
141                }
142        }
143
144}
