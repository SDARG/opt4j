024package org.opt4j.core.common.logger;
025
026import org.opt4j.core.config.Icons;
027import org.opt4j.core.config.annotations.File;
028import org.opt4j.core.config.annotations.Icon;
029import org.opt4j.core.config.annotations.Info;
030import org.opt4j.core.config.annotations.Order;
031import org.opt4j.core.config.annotations.Required;
032import org.opt4j.core.start.Constant;
033
034/**
035 * Module for logging.
036 * 
037 * @see Logger
038 * @author reimann, lukasiewycz
039 * 
040 */
041@Icon(Icons.TEXT)
042@Info("Logs the contents of the archive to a file.")
043public class LoggerModule extends OutputModule {
044
045        @Info("The name of the output file.")
046        @Order(0)
047        @File
048        @Constant(value = "filename", namespace = TsvLogger.class)
049        protected String filename = "output.tsv";
050
051        @Info("Log per evaluation activated.")
052        protected boolean loggingPerEvaluation = false;
053
054        @Info("Log per iteration activated.")
055        protected boolean loggingPerIteration = true;
056
057        @Info("Number of evaluations after which the archive is logged.")
058        @Required(property = "loggingPerEvaluation", elements = { "TRUE" })
059        protected int evaluationStep = 100;
060
061        @Info("Number of iterations after which the archive is logged.")
062        @Required(property = "loggingPerIteration", elements = { "TRUE" })
063        protected int iterationStep = 1;
064
065        /*
066         * (non-Javadoc)
067         * 
068         * @see org.opt4j.start.Opt4JModule#config()
069         */
070        @Override
071        public void config() {
072
073                bind(TsvLogger.class).in(SINGLETON);
074
075                addIndividualStateListener(TsvLogger.class);
076                addOptimizerIterationListener(TsvLogger.class);
077                addOptimizerStateListener(TsvLogger.class);
078
079                int evaluationStep = this.evaluationStep;
080                int iterationStep = this.iterationStep;
081
082                if (!loggingPerEvaluation) {
083                        evaluationStep = -1;
084                }
085                if (!loggingPerIteration) {
086                        iterationStep = -1;
087                }
088
089                bindConstant("evaluationStep", TsvLogger.class).to(evaluationStep);
090                bindConstant("iterationStep", TsvLogger.class).to(iterationStep);
091        }
092
093        /**
094         * Returns the filename.
095         * 
096         * @see #setFilename
097         * @return the filename
098         */
099        public String getFilename() {
100                return filename;
101        }
102
103        /**
104         * Sets the filename.
105         * 
106         * @see #getFilename
107         * @param filename
108         *            the filename
109         */
110        public void setFilename(String filename) {
111                this.filename = filename;
112        }
113
114        /**
115         * Returns the step size of the evaluations.
116         * 
117         * @see #setEvaluationStep
118         * @return the step size of the evaluations
119         */
120        public int getEvaluationStep() {
121                return evaluationStep;
122        }
123
124        /**
125         * Sets the step size of the evaluations.
126         * 
127         * @see #getEvaluationStep
128         * @param evaluationStep
129         *            the step size of the evaluations
130         */
131        public void setEvaluationStep(int evaluationStep) {
132                this.evaluationStep = evaluationStep;
133        }
134
135        /**
136         * Returns the step size of the iterations.
137         * 
138         * @see #setIterationStep
139         * @return the step size of the iterations
140         */
141        public int getIterationStep() {
142                return iterationStep;
143        }
144
145        /**
146         * Sets the step size of the iterations.
147         * 
148         * @see #getIterationStep
149         * @param iterationStep
150         *            the step size of the iterations
151         */
152        public void setIterationStep(int iterationStep) {
153                this.iterationStep = iterationStep;
154        }
155
156        /**
157         * Returns {@code true} if the logger observes the number of evaluations.
158         * 
159         * @return {@code true} if the logger observes the number of evaluations
160         */
161        public boolean isLoggingPerEvaluation() {
162                return loggingPerEvaluation;
163        }
164
165        /**
166         * Sets the observation for the number of evaluations.
167         * 
168         * @param loggingPerEvaluation
169         *            {@code true} if the logger observes the number of evaluations
170         */
171        public void setLoggingPerEvaluation(boolean loggingPerEvaluation) {
172                this.loggingPerEvaluation = loggingPerEvaluation;
173        }
174
175        /**
176         * Returns {@code true} if the logger observes the number of iterations.
177         * 
178         * @return {@code true} if the logger observes the number of iterations
179         */
180        public boolean isLoggingPerIteration() {
181                return loggingPerIteration;
182        }
183
184        /**
185         * Sets the observation for the number of iterations.
186         * 
187         * @param loggingPerIteration
188         *            {@code true} if the logger observes the number of iterations
189         */
190        public void setLoggingPerIteration(boolean loggingPerIteration) {
191                this.loggingPerIteration = loggingPerIteration;
192        }
193}
