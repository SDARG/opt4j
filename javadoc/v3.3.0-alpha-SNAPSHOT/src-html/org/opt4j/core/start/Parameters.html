023package org.opt4j.core.start;
024
025import java.lang.reflect.ParameterizedType;
026import java.lang.reflect.Type;
027import java.lang.reflect.TypeVariable;
028import java.util.Collection;
029import java.util.HashMap;
030import java.util.HashSet;
031import java.util.Map;
032
033/**
034 * The {@link Parameters} is a class for the identification of types/classes of
035 * a generic parameter.
036 * 
037 * @author lukasiewycz
038 * 
039 */
040public class Parameters {
041
042        /**
043         * Returns the class of a type.
044         * 
045         * @param type
046         *            the type
047         * @return the class
048         */
049        public static Class<?> getClass(Type type) {
050                if (type instanceof Class<?>) {
051                        return (Class<?>) type;
052                } else if (type instanceof ParameterizedType) {
053                        return (Class<?>) ((ParameterizedType) type).getRawType();
054                } else {
055                        return null;
056                }
057        }
058
059        /**
060         * Returns the type of the generic of the interface or class of an class for
061         * an object.
062         * 
063         * @param clazz
064         *            the interface with the undefined generic
065         * @param object
066         *            the current implementation of the interface
067         * @param variable
068         *            the identifier
069         * @return the implemented type of the generic
070         */
071        public static Type getType(Class<?> clazz, Object object, String variable) {
072                return getType(clazz, variable, getTypes(object.getClass()));
073        }
074
075        protected static Type getType(Class<?> clazz, String variable, Map<Type, ParameterizedType> map) {
076
077                // get first
078                ParameterizedType ptype = map.get(clazz);
079                if (ptype == null) {
080                        return null;
081                }
082
083                Type type = null;
084                for (int j = 0; j < ptype.getActualTypeArguments().length; j++) {
085                        TypeVariable<?> v = clazz.getTypeParameters()[j];
086                        if (v.getName().equals(variable)) {
087                                type = ptype.getActualTypeArguments()[j];
088                        }
089                }
090
091                while (type instanceof TypeVariable<?>) {
092                        TypeVariable<?> var = (TypeVariable<?>) type;
093                        ParameterizedType paramType = map.get(var.getGenericDeclaration());
094                        Class<?> freeType = (Class<?>) paramType.getRawType();
095
096                        int size = freeType.getTypeParameters().length;
097
098                        for (int j = 0; j < size; j++) {
099                                TypeVariable<?> f = freeType.getTypeParameters()[j];
100                                if (f.equals(var)) {
101                                        type = paramType.getActualTypeArguments()[j];
102                                        break;
103                                }
104                        }
105                }
106
107                return type;
108        }
109
110        protected static Map<Type, ParameterizedType> getTypes(Class<?> clazz) {
111                Collection<Class<?>> classes = getAllClasses(clazz);
112                Collection<ParameterizedType> paramTypes = getParameterizedTypes(classes);
113
114                Map<Type, ParameterizedType> map = new HashMap<>();
115
116                for (ParameterizedType p : paramTypes) {
117                        map.put(p.getRawType(), p);
118                }
119
120                return map;
121        }
122
123        protected static Collection<Class<?>> getAllClasses(final Class<?> clazz) {
124                Collection<Class<?>> set = new HashSet<>();
125                if (clazz != null && !clazz.equals(Object.class)) {
126                        set.add(clazz);
127
128                        Class<?> superclass = clazz.getSuperclass();
129                        set.addAll(getAllClasses(superclass));
130
131                        for (Class<?> inf : clazz.getInterfaces()) {
132                                set.addAll(getAllClasses(inf));
133                        }
134                }
135                return set;
136        }
137
138        protected static Collection<ParameterizedType> getParameterizedTypes(Collection<Class<?>> classes) {
139                Collection<ParameterizedType> paramTypes = new HashSet<>();
140
141                for (Class<?> clazz : classes) {
142
143                        Type type = clazz.getGenericSuperclass();
144
145                        if (type instanceof ParameterizedType) {
146                                ParameterizedType p = (ParameterizedType) type;
147                                paramTypes.add(p);
148                        }
149
150                        for (Type t : clazz.getGenericInterfaces()) {
151                                if (t instanceof ParameterizedType) {
152                                        ParameterizedType p = (ParameterizedType) t;
153                                        paramTypes.add(p);
154                                }
155                        }
156                }
157
158                return paramTypes;
159        }
160
161}
