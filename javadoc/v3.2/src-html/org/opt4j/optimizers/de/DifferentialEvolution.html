058/**
059 * The {@link DifferentialEvolution}.
060 * 
061 * @author lukasiewycz
062 * 
063 */
064public class DifferentialEvolution implements IterativeOptimizer {
065
066        protected final double scalingFactor;
067
068        protected final int alpha;
069
070        protected final Algebra<Genotype> algebra;
071
072        protected final Crossover<Genotype> crossover;
073
074        protected final Selector selector;
075
076        protected final Random random;
077
078        private final IndividualFactory individualFactory;
079
080        private final Population population;
081
082        private final IndividualCompleter completer;
083
084        private Term term;
085
086        /**
087         * Constructs a {@link DifferentialEvolution}.
088         * 
089         * @param population
090         *            the population
091         * @param individualFactory
092         *            the individual factory
093         * @param completer
094         *            the completer
095         * @param algebra
096         *            the algebra operator
097         * @param selector
098         *            the selector
099         * @param random
100         *            the random number generator
101         * @param crossover
102         *            the crossover operator
103         * @param alpha
104         *            the population size
105         * @param scalingFactor
106         *            the scaling factor F
107         */
108        @Inject
109        public DifferentialEvolution(
110                        Population population,
111                        IndividualFactory individualFactory,
112                        IndividualCompleter completer,
113                        Algebra<Genotype> algebra,
114                        Selector selector,
115                        Rand random,
116                        Crossover<Genotype> crossover,
117                        @Constant(value = "alpha", namespace = DifferentialEvolution.class) int alpha,
118                        @Constant(value = "scalingFactor", namespace = DifferentialEvolution.class) double scalingFactor) {
119                this.algebra = algebra;
120                this.selector = selector;
121                this.random = random;
122                this.crossover = crossover;
123                this.alpha = alpha;
124                this.scalingFactor = scalingFactor;
125                this.individualFactory = individualFactory;
126                this.population = population;
127                this.completer = completer;
128        }
129
130        /*
131         * (non-Javadoc)
132         * 
133         * @see org.opt4j.core.optimizer.IterativeOptimizer#initialize()
134         */
135        @Override
136        public void initialize() {
137                Index i0 = new Index(0);
138                Index i1 = new Index(1);
139                Index i2 = new Index(2);
140                Var c = new Var(scalingFactor);
141                term = new Add(i0, new Mult(c, new Sub(i1, i2)));
142                selector.init(2 * alpha);
143        }
144
145        /*
146         * (non-Javadoc)
147         * 
148         * @see org.opt4j.core.optimizer.IterativeOptimizer#next()
149         */
150        @Override
151        public void next() throws TerminationException {
152                if (population.isEmpty()) {
153                        firstIterationNext();
154                } else {
155                        followingIterationsNext();
156                }
157        }
158
159        private void firstIterationNext() {
160                while (population.size() < alpha) {
161                        Individual individual = individualFactory.create();
162
163                        Genotype genotype = individual.getGenotype();
164                        if (!(genotype instanceof DoubleGenotype)) {
165                                throw new IncompatibilityException(
166                                                "DifferentialEvolution is restricted to "
167                                                                + DoubleGenotype.class
168                                                                + ", current Genotype is: "
169                                                                + genotype.getClass());
170                        }
171
172                        population.add(individual);
173                }
174        }
175
176        private void followingIterationsNext() throws TerminationException {
177                /*
178                 * Map from each parent to its offspring
179                 */
180                Map<Individual, Individual> relation = new HashMap<Individual, Individual>();
181
182                /*
183                 * Create an offspring for each parent individual in the population and
184                 * add the offspring to the population and evaluate their objectives
185                 */
186                List<Individual> list = new ArrayList<Individual>(population);
187                for (Individual parent : population) {
188                        Individual offspring = createOffspring(parent, list, term);
189                        relation.put(parent, offspring);
190                }
191                population.addAll(relation.values());
192                completer.complete(population);
193
194                /*
195                 * Remove those offspring individuals from the population that are
196                 * dominated by their corresponding parent
197                 */
198                for (Entry<Individual, Individual> entry : relation.entrySet()) {
199                        Individual parent = entry.getKey();
200                        Individual offspring = entry.getValue();
201
202                        if (parent.getObjectives().weaklyDominates(
203                                        offspring.getObjectives())) {
204                                population.remove(offspring);
205                        }
206                }
207
208                /*
209                 * Truncate the population size to alpha
210                 */
211                Collection<Individual> lames = selector.getLames(population.size()
212                                - alpha, population);
213                population.removeAll(lames);
214        }
215
216        protected Individual createOffspring(Individual parent,
217                        List<Individual> individuals, Term term) {
218                Triple triple = getTriple(parent, individuals);
219
220                Genotype g0 = triple.getFirst().getGenotype();
221                Genotype g1 = triple.getSecond().getGenotype();
222                Genotype g2 = triple.getThird().getGenotype();
223
224                Genotype result = algebra.algebra(term, g0, g1, g2);
225                Pair<Genotype> g = crossover.crossover(result, parent.getGenotype());
226
227                Individual i;
228                if (random.nextBoolean()) {
229                        i = individualFactory.create(g.getFirst());
230                } else {
231                        i = individualFactory.create(g.getSecond());
232                }
233                return i;
234        }
235
236        /**
237         * The {@link Triple} is a container for three individuals.
238         * 
239         * @author lukasiewycz
240         * 
241         */
242        protected static class Triple {
243                protected final Individual first;
244
245                protected final Individual second;
246
247                protected final Individual third;
248
249                public Triple(final Individual first, final Individual second,
250                                final Individual third) {
251                        super();
252                        this.first = first;
253                        this.second = second;
254                        this.third = third;
255                }
256
257                public Individual getFirst() {
258                        return first;
259                }
260
261                public Individual getSecond() {
262                        return second;
263                }
264
265                public Individual getThird() {
266                        return third;
267                }
268        }
269
270        /**
271         * Returns three different {@link Individual}s from the {@code individuals}
272         * list. Each {@link Individual} is not equal to the parent.
273         * 
274         * @param parent
275         *            the parent Individual
276         * @param individuals
277         *            the population
278         * @return the three individuals as a Triple
279         */
280        protected Triple getTriple(Individual parent, List<Individual> individuals) {
281                individuals.remove(parent);
282                Individual ind0 = individuals
283                                .remove(random.nextInt(individuals.size()));
284                Individual ind1 = individuals
285                                .remove(random.nextInt(individuals.size()));
286                Individual ind2 = individuals
287                                .remove(random.nextInt(individuals.size()));
288
289                Triple triple = new Triple(ind0, ind1, ind2);
290
291                individuals.add(parent);
292                individuals.add(ind0);
293                individuals.add(ind1);
294                individuals.add(ind2);
295
296                return triple;
297
298        }
