024package org.opt4j.optimizers.sa;
025
026import java.util.Random;
027
028import org.opt4j.core.Genotype;
029import org.opt4j.core.Individual;
030import org.opt4j.core.IndividualFactory;
031import org.opt4j.core.Objectives;
032import org.opt4j.core.common.random.Rand;
033import org.opt4j.core.optimizer.Archive;
034import org.opt4j.core.optimizer.IndividualCompleter;
035import org.opt4j.core.optimizer.Iteration;
036import org.opt4j.core.optimizer.IterativeOptimizer;
037import org.opt4j.core.optimizer.Population;
038import org.opt4j.core.optimizer.TerminationException;
039import org.opt4j.operators.copy.Copy;
040import org.opt4j.operators.neighbor.Neighbor;
041
042import com.google.inject.Inject;
043
044/**
045 * The {@link SimulatedAnnealing} is a standard implementation of the
046 * optimization heuristic. This heuristic optimizes a single objective
047 * (multi-objective problems are optimized by the sum of all objectives).
048 * 
049 * @author lukasiewycz
050 * 
051 */
052public class SimulatedAnnealing implements IterativeOptimizer {
053
054        protected final Random random;
055
056        protected final Neighbor<Genotype> neighbor;
057
058        protected final Copy<Genotype> copy;
059
060        protected final CoolingSchedule coolingSchedule;
061
062        private final IndividualFactory individualFactory;
063
064        private final IndividualCompleter completer;
065
066        private final Population population;
067
068        private final Archive archive;
069
070        private final Iteration iteration;
071
072        /**
073         * Constructs a new {@link SimulatedAnnealing}.
074         * 
075         * @param population
076         *            the population
077         * @param individualFactory
078         *            the individual factory
079         * @param completer
080         *            the completer
081         * @param random
082         *            the random number generator
083         * @param neighbor
084         *            the neighbor operator
085         * @param copy
086         *            the copy operator
087         * @param iteration
088         *            the iteration counter
089         * @param coolingSchedule
090         *            the cooling schedule
091         */
092        @Inject
093        public SimulatedAnnealing(Population population, Archive archive,
094                        IndividualFactory individualFactory, IndividualCompleter completer,
095                        Rand random, Neighbor<Genotype> neighbor, Copy<Genotype> copy,
096                        CoolingSchedule coolingSchedule, Iteration iteration) {
097                this.random = random;
098                this.neighbor = neighbor;
099                this.copy = copy;
100                this.coolingSchedule = coolingSchedule;
101                this.individualFactory = individualFactory;
102                this.population = population;
103                this.completer = completer;
104                this.archive = archive;
105                this.iteration = iteration;
106        }
107
108        /*
109         * (non-Javadoc)
110         * 
111         * @see org.opt4j.core.optimizer.IterativeOptimizer#initialize()
112         */
113        @Override
114        public void initialize() throws TerminationException {
115
116        }
117
118        private double fold;
119        private Individual old;
120
121        /*
122         * (non-Javadoc)
123         * 
124         * @see org.opt4j.core.optimizer.IterativeOptimizer#next()
125         */
126        @Override
127        public void next() throws TerminationException {
128
129                if (population.isEmpty()) {
130                        // the first iteration
131                        
132                        old = individualFactory.create();
133
134                        population.add(old);
135                        completer.complete(population);
136
137                        fold = f(old);
138                } else {
139                        // all iterations > 1
140                        
141                        Genotype g = copy.copy(old.getGenotype());
142                        neighbor.neighbor(g);
143
144                        Individual y = individualFactory.create(g);
145
146                        completer.complete(y);
147                        archive.update(y);
148
149                        double fy = f(y);
150                        // boolean value that indicates a switch of the individuals
151                        boolean sw = false;
152
153                        if (fy <= fold) {
154                                sw = true;
155                        } else {
156                                double a = (fold - fy)
157                                                / coolingSchedule.getTemperature(iteration.value(),
158                                                                iteration.max());
159                                double e = Math.exp(a);
160                                if (random.nextDouble() < e) {
161                                        sw = true;
162                                }
163                        }
164
165                        if (sw) {
166                                population.remove(old);
167                                population.add(y);
168                                fold = fy;
169                                old = y;
170                        }
171
172                }
173        }
174
175        /**
176         * Calculates the sum of the {@link Objectives} of one {@link Individual}.
177         * 
178         * @param individual
179         *            the individual
180         * @return the sum of the objective values
181         */
182        protected double f(Individual individual) {
183                Objectives objectives = individual.getObjectives();
184
185                double sum = 0;
186
187                for (double d : objectives.array()) {
188                        sum += d;
189                }
190
191                return sum;
192        }
193
194}
