018package org.opt4j.benchmarks.knapsack;
019
020import static org.opt4j.core.config.annotations.Citation.PublicationMonth.NOVEMBER;
021
022import org.opt4j.core.config.annotations.Citation;
023import org.opt4j.core.config.annotations.File;
024import org.opt4j.core.config.annotations.Info;
025import org.opt4j.core.config.annotations.Required;
026import org.opt4j.core.problem.ProblemModule;
027import org.opt4j.core.start.Constant;
028
029/**
030 * The multiobjective 0/1 ILP knapsack problem as proposed in Zitzler and Thiele 1999. Either one of the nine benchmark
031 * problems from Zitzler and Thiele 1999 can be selected or the number of knapsacks and items can be set manually.
032 * 
033 * @see <a href=
034 *      "http://www.tik.ee.ethz.ch/sop/download/supplementary/testProblemSuite/">http://www.tik.ee.ethz.ch/sop/download/supplementary/testProblemSuite/</a>
035 * 
036 * @author reimann, lukasiewycz
037 * 
038 */
039@Citation(authors = "Eckart Zitzler and Lothar Thiele", title = "Multiobjective evolutionary algorithms: A comparative case study and the strength Pareto approach", journal = "IEEE Transactions on Evolutionary Computation", volume = 3, number = 4, pageFirst = 257, pageLast = 271, month = NOVEMBER, year = 1999)
040public class KnapsackModule extends ProblemModule {
041
042        protected int knapsacks = 5;
043
044        protected int items = 250;
045
046        @Required(property = "testCase", elements = { "MANUAL", "ZT1", "ZT2", "ZT3", "ZT4", "ZT5", "ZT6", "ZT7", "ZT8",
047                        "ZT9" })
048        @Constant(value = "seed", namespace = KnapsackProblemRandom.class)
049        protected int seed = 0;
050
051        @File
052        @Required(property = "testCase", elements = { "FILE" })
053        @Constant(value = "filename", namespace = KnapsackProblemFile.class)
054        protected String filename = "";
055
056        protected Representation representation = Representation.BITSTRING;
057
058        protected TestCase testCase = TestCase.MANUAL;
059
060        public enum TestCase {
061                FILE, @Info("use test data from an input file")
062                MANUAL, @Info("2 knapsacks, 250 items")
063                ZT1, @Info("3 knapsacks, 250 items")
064                ZT2, @Info("5 knapsacks, 250 items")
065                ZT3, @Info("2 knapsacks, 500 items")
066                ZT4, @Info("3 knapsacks, 500 items")
067                ZT5, @Info("5 knapsacks, 500 items")
068                ZT6, @Info("2 knapsacks, 750 items")
069                ZT7, @Info("3 knapsacks, 750 items")
070                ZT8, @Info("5 knapsacks, 750 items")
071                ZT9;
072        }
073
074        public enum Representation {
075                @Info("Bitstring genotype")
076                BITSTRING, @Info("SAT decoding")
077                SAT;
078        }
079
080        @Override
081        protected void config() {
082                switch (testCase) {
083                case ZT1:
084                        knapsacks = 2;
085                        items = 250;
086                        break;
087                case ZT2:
088                        knapsacks = 3;
089                        items = 250;
090                        break;
091                case ZT3:
092                        knapsacks = 5;
093                        items = 250;
094                        break;
095                case ZT4:
096                        knapsacks = 2;
097                        items = 500;
098                        break;
099                case ZT5:
100                        knapsacks = 3;
101                        items = 500;
102                        break;
103                case ZT6:
104                        knapsacks = 5;
105                        items = 500;
106                        break;
107                case ZT7:
108                        knapsacks = 2;
109                        items = 750;
110                        break;
111                case ZT8:
112                        knapsacks = 3;
113                        items = 750;
114                        break;
115                case ZT9:
116                        knapsacks = 5;
117                        items = 750;
118                        break;
119                default:
120                        break;
121                }
122
123                bindConstant("items", KnapsackProblemRandom.class).to(items);
124                bindConstant("knapsacks", KnapsackProblemRandom.class).to(knapsacks);
125
126                if (testCase == TestCase.FILE) {
127                        bind(KnapsackProblem.class).to(KnapsackProblemFile.class).in(SINGLETON);
128                } else {
129                        bind(KnapsackProblem.class).to(KnapsackProblemRandom.class).in(SINGLETON);
130                }
131
132                switch (representation) {
133                case BITSTRING:
134                        bindProblem(KnapsackBinaryCreatorDecoder.class, KnapsackBinaryCreatorDecoder.class,
135                                        KnapsackProfitEvaluator.class);
136                        break;
137                case SAT:
138                        bindProblem(KnapsackSATCreatorDecoder.class, KnapsackSATCreatorDecoder.class,
139                                        KnapsackProfitEvaluator.class);
140                        break;
141                }
142                addEvaluator(KnapsackOverloadEvaluator.class);
143        }
144
145        @Required(property = "testCase", elements = { "MANUAL" })
146        public int getKnapsacks() {
147                return knapsacks;
148        }
149
150        public void setKnapsacks(int knapsacks) {
151                this.knapsacks = knapsacks;
152        }
153
154        @Required(property = "testCase", elements = { "MANUAL" })
155        public int getItems() {
156                return items;
157        }
158
159        public void setItems(int items) {
160                this.items = items;
161        }
162
163        public Representation getRepresentation() {
164                return representation;
165        }
166
167        public void setRepresentation(Representation representation) {
168                this.representation = representation;
169        }
170
171        public TestCase getTestCase() {
172                return testCase;
173        }
174
175        public void setTestCase(TestCase testCase) {
176                this.testCase = testCase;
177        }
178
179        public int getSeed() {
180                return seed;
181        }
182
183        public void setSeed(int seed) {
184                this.seed = seed;
185        }
186
187        public String getFilename() {
188                return filename;
189        }
190
191        public void setFilename(String filename) {
192                this.filename = filename;
193        }
194
195}
