024package org.opt4j.core;
025
026/**
027 * The {@link Value} represents the result for an {@link Objective}. Each
028 * {@link Value} has to be {@link Comparable} and have a {@link Double}
029 * representation which is used by some optimization algorithms. Thus, an
030 * appropriate {@link Double} representation is not necessary if the
031 * optimization algorithm does not require it.
032 * 
033 * @see DoubleValue
034 * @see IntegerValue
035 * @see Objectives#add(Objective, Value)
036 * @author lukasiewycz
037 * 
038 * @param <V>
039 *            the type of the value
040 */
041public interface Value<V> extends Comparable<Value<V>> {
042
043        /**
044         * Returns the value.
045         * 
046         * @see #setValue
047         * @return the value
048         */
049        public V getValue();
050
051        /**
052         * Sets the value.
053         * 
054         * @see #getValue
055         * @param value
056         *            the value to set
057         */
058        public void setValue(V value);
059
060        /**
061         * Returns a double value.
062         * 
063         * @return double value
064         */
065        public Double getDouble();
066
067}
