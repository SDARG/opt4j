018package org.opt4j.core.common.archive;
019
020import java.util.ArrayList;
021import java.util.Collection;
022import java.util.LinkedHashMap;
023import java.util.LinkedHashSet;
024import java.util.List;
025import java.util.Map;
026import java.util.Set;
027
028import org.opt4j.core.Individual;
029import org.opt4j.core.common.random.Rand;
030import org.opt4j.core.optimizer.Archive;
031import org.opt4j.core.start.Constant;
032
033import com.google.inject.Inject;
034
035/**
036 * The {@link AdaptiveGridArchive} uses an adaptive grid in order to bound the size of the {@link Archive}, see
037 * "Properties of an Adaptive Archiving Algorithm for Storing Nondominated Vectors, J. Knowles, D. Corne, Transactions
038 * of Evolutionary Computation, Vol. 7, No. 2, April 2003".
039 * 
040 * @author lukasiewycz, helwig
041 * 
042 */
043public class AdaptiveGridArchive extends BoundedArchive {
044
045        protected boolean isInit = false;
046
047        protected final int div;
048
049        protected final Rand random;
050
051        protected final Map<Long, Cell> cells = new LinkedHashMap<Long, Cell>();
052
053        protected final List<Individual> extrema = new ArrayList<Individual>();
054
055        protected double[] lb;
056
057        protected double[] ub;
058
059        protected int dim;
060
061        /**
062         * A single {@link Cell} of the grid.
063         * 
064         * @author lukasiewycz
065         * 
066         */
067        protected static class Cell extends LinkedHashSet<Individual> {
068
069                private static final long serialVersionUID = 1L;
070
071                protected final long identifier;
072
073                /**
074                 * Constructs the {@link Cell} with the {@code identifier}.
075                 * 
076                 * @param identifier
077                 *            the indentifier
078                 */
079                public Cell(long identifier) {
080                        super();
081                        this.identifier = identifier;
082                }
083
084                /**
085                 * Returns the {@code identifier}.
086                 * 
087                 * @return the identifier
088                 */
089                public long getIdentifier() {
090                        return identifier;
091                }
092        }
093
094        /**
095         * Constructs an {@link AdaptiveGridArchive}.
096         * 
097         * @param capacity
098         *            the capacity
099         * @param div
100         *            the divisions
101         * @param random
102         *            the random number generator
103         */
104        @Inject
105        public AdaptiveGridArchive(@Constant(value = "capacity", namespace = BoundedArchive.class) int capacity,
106                        @Constant(value = "div", namespace = AdaptiveGridArchive.class) int div, Rand random) {
107                super(capacity);
108                this.random = random;
109                this.div = div;
110
111        }
112
113        /*
114         * (non-Javadoc)
115         * 
116         * @see org.opt4j.common.archive.AbstractArchive#updateWithNondominated(java. util.Collection)
117         */
118        @Override
119        protected boolean updateWithNondominated(Collection<Individual> candidates) {
120
121                if (!isInit && candidates.size() > 0) {
122                        init(candidates.iterator().next());
123                }
124
125                if (determineBounds(candidates)) {
126                        cells.clear();
127                        addToCell(this);
128                }
129                addToCell(candidates);
130
131                boolean changed = false;
132                if (this.size() + candidates.size() <= capacity) {
133                        changed = addCheckedIndividuals(candidates);
134                } else {
135                        candidates.addAll(this);
136
137                        while (candidates.size() > capacity) {
138                                Cell cell = getMostCrowdedCell();
139
140                                List<Individual> list = new ArrayList<Individual>(cell);
141                                list.removeAll(extrema);
142
143                                Individual individual = list.get(random.nextInt(list.size()));
144                                candidates.remove(individual);
145                                removeFromCell(individual);
146                        }
147
148                        this.retainAll(candidates);
149                        for (Individual individual : candidates) {
150                                if (!this.contains(individual)) {
151                                        changed |= addCheckedIndividual(individual);
152                                }
153                        }
154                }
155
156                return changed;
157        }
158
159        /**
160         * Initialize the {@link org.opt4j.core.optimizer.Archive}.
161         * 
162         * @param individual
163         *            one random individual
164         */
165        protected void init(Individual individual) {
166                if (!isInit) {
167                        dim = individual.getObjectives().size();
168
169                        lb = new double[dim];
170                        ub = new double[dim];
171
172                        for (int i = 0; i < dim; i++) {
173                                lb[i] = Double.MAX_VALUE;
174                                ub[i] = Double.MIN_VALUE;
175                                extrema.add(null);
176                                extrema.add(null);
177                        }
178
179                        isInit = true;
180                }
181        }
182
183        /**
184         * Returns the appropriate {@link Cell} for the given {@link Individual}.
185         * 
186         * @param individual
187         *            the individual
188         * @return the cell
189         */
190        protected Cell getCell(Individual individual) {
191                final double[] obj = individual.getObjectives().array();
192
193                int scale = 1;
194                long identifier = 0;
195                for (int i = 0; i < obj.length; i++) {
196                        double half = (1.0 / (2.0 * div)) * (ub[i] - lb[i]);
197                        double lbi = lb[i] - half;
198                        double ubi = ub[i] + half;
199
200                        double step = (ubi - lbi) / div;
201                        int position = (int) Math.floor((obj[i] - lbi) / step);
202
203                        identifier += position * scale;
204                        scale *= div;
205                }
206
207                Cell cell = cells.get(identifier);
208                if (cell != null) {
209                        return cell;
210                } else {
211                        return new Cell(identifier);
212                }
213        }
214
215        /**
216         * Returns the most crowded {@link Cell}.
217         * 
218         * @return the most crowded cell
219         */
220        protected Cell getMostCrowdedCell() {
221                Cell mc = null;
222                int size = 0;
223
224                Set<Individual> extrema = new LinkedHashSet<Individual>(this.extrema);
225
226                for (Cell cell : cells.values()) {
227                        int s = 0;
228                        assert (!cell.isEmpty());
229                        for (Individual individual : cell) {
230                                if (!extrema.contains(individual)) {
231                                        s++;
232                                }
233                        }
234                        if (s > size) {
235                                size = s;
236                                mc = cell;
237                        }
238                }
239
240                return mc;
241        }
242
243        /*
244         * (non-Javadoc)
245         * 
246         * @see org.opt4j.core.IndividualCollection#remove(java.lang.Object)
247         */
248        @Override
249        public boolean remove(Object o) {
250                Individual individual = (Individual) o;
251
252                removeFromCell(individual);
253
254                if (extrema.contains(individual)) {
255                        for (int i = 0; i < extrema.size(); i++) {
256                                if (individual.equals(extrema.get(i))) {
257                                        extrema.set(i, null);
258                                        int j = (int) Math.floor(i / 2.0);
259                                        if (i % 2 == 0) {
260                                                lb[j] = Double.MAX_VALUE;
261                                        } else {
262                                                ub[j] = Double.MIN_VALUE;
263                                        }
264                                }
265                        }
266                }
267
268                return super.remove(individual);
269        }
270
271        /**
272         * Adds all {@link Individual}s to theirs {@link Cell}s and the {@link Cell} s to the {@link Archive}.
273         * 
274         * @param individuals
275         *            the individuals to be added
276         */
277        protected void addToCell(Iterable<Individual> individuals) {
278                for (Individual individual : individuals) {
279                        Cell cell = getCell(individual);
280                        cell.add(individual);
281                        cells.put(cell.identifier, cell);
282                }
283        }
284
285        /**
286         * Removes this {@link Individual} from its {@link Cell} and removes the {@link Cell} from the
287         * {@link org.opt4j.core.optimizer.Archive} if it is empty.
288         * 
289         * @param individual
290         *            the individual to be removed
291         */
292        protected void removeFromCell(Individual individual) {
293                Cell cell = getCell(individual);
294                cell.remove(individual);
295                if (cell.isEmpty()) {
296                        cells.remove(cell.getIdentifier());
297                }
298        }
299
300        /**
301         * Determines the new bounds for the individuals and returns {@code true} if the bounds have changed.
302         * 
303         * @param individuals
304         *            the individuals
305         * @return {@code true} if the bounds have changed
306         */
307        protected boolean determineBounds(Collection<Individual> individuals) {
308                boolean boundsChanged = false;
309
310                for (Individual individual : individuals) {
311                        final double[] obj = individual.getObjectives().array();
312
313                        for (int i = 0; i < dim; i++) {
314                                double value = obj[i];
315
316                                if (value < lb[i]) {
317                                        boundsChanged = true;
318                                        extrema.set(2 * i, individual);
319                                        lb[i] = obj[i];
320                                } else if (value > ub[i]) {
321                                        boundsChanged = true;
322                                        extrema.set(2 * i + 1, individual);
323                                        ub[i] = obj[i];
324                                }
325                        }
326                }
327
328                return boundsChanged;
329        }
