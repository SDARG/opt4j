023package org.opt4j.core.genotype;
024
025import java.lang.reflect.Constructor;
026import java.util.Collection;
027import java.util.Collections;
028import java.util.HashMap;
029import java.util.HashSet;
030import java.util.List;
031import java.util.Map;
032import java.util.Map.Entry;
033import java.util.Random;
034
035import org.opt4j.core.Genotype;
036
037/**
038 * <p>
039 * The {@link SelectMapGenotype} selects for each key an element from a given
040 * list.
041 * </p>
042 * <p>
043 * Example problem: Choose for each of five balls ball<sub>1</sub>,
044 * ball<sub>2</sub>, ball<sub>3</sub>, ball<sub>4</sub>, ball<sub>5</sub> a
045 * color. Available colors are blue, green, and red.<br/>
046 * Example usage: <blockquote>
047 * 
048 * <pre>
049 * SelectMapGenotype&lt;Ball, Color&gt; genotype = new SelectMapGenotype&lt;Ball, Color&gt;(
050 *              Arrays.asList(ball1, ball2, ball3, ball4, ball5), Arrays.asList(Color.BLUE, Color.GREEN, Color.RED));
051 * genotype.init(new Random());
052 * </pre>
053 * 
054 * </blockquote> Example instance: [ball1=green, ball2=red, ball3=red,
055 * ball4=blue, ball5=blue]<br/>
056 * Example search space size: 3<sup>5</sup>
057 * </p>
058 * 
059 * 
060 * @author lukasiewycz
061 * 
062 * @param <K>
063 *            the type of keys
064 * @param <V>
065 *            the type of elements
066 */
067public class SelectMapGenotype<K, V> extends IntegerGenotype implements MapGenotype<K, V> {
068
069        protected final List<K> keys;
070        protected final Map<K, List<V>> values;
071
072        protected static class SelectBounds<O, P> implements Bounds<Integer> {
073                protected List<O> list;
074                protected Map<O, List<P>> map;
075
076                public SelectBounds(List<O> list, Map<O, List<P>> map) {
077                        this.list = list;
078                        this.map = map;
079                        for (Entry<O, List<P>> entry : map.entrySet()) {
080                                if (entry.getValue().isEmpty()) {
081                                        throw new IllegalArgumentException("Empty value map provided for key " + entry.getKey());
082                                }
083                        }
084                }
085
086                @Override
087                public Integer getLowerBound(int index) {
088                        return 0;
089                }
090
091                @Override
092                public Integer getUpperBound(int index) {
093                        return map.get(list.get(index)).size() - 1;
094                }
095        }
096
097        /**
098         * Constructs a {@link SelectMapGenotype}.
099         * 
100         * @param keys
101         *            the keys
102         * @param values
103         *            the values
104         */
105        public SelectMapGenotype(List<K> keys, Map<K, List<V>> values) {
106                super(new SelectBounds<K, V>(keys, values));
107                if (!keys.containsAll(values.keySet()) || !values.keySet().containsAll(keys)) {
108                        throw new IllegalArgumentException("The provided list does not match the provided map");
109                }
110                if (new HashSet<K>(keys).size() < keys.size()) {
111                        throw new IllegalArgumentException(MapGenotype.ERROR_MESSAGE_NON_UNIQUE_KEYS);
112                }
113                this.keys = keys;
114                this.values = values;
115        }
116
117        private static <K, V> Map<K, List<V>> toMap(List<K> keys, List<V> values) {
118                if (new HashSet<K>(keys).size() < keys.size()) {
119                        throw new IllegalArgumentException(MapGenotype.ERROR_MESSAGE_NON_UNIQUE_KEYS);
120                }
121                Map<K, List<V>> map = new HashMap<K, List<V>>();
122                for (K key : keys) {
123                        map.put(key, values);
124                }
125                return map;
126        }
127
128        /**
129         * Constructs a {@link SelectMapGenotype}. Here, each key has the same
130         * target list of element values.
131         * 
132         * @param keys
133         *            the keys
134         * @param values
135         *            the values
136         */
137        public SelectMapGenotype(List<K> keys, List<V> values) {
138                super(new SelectBounds<K, V>(keys, toMap(keys, values)));
139                this.keys = keys;
140                this.values = toMap(keys, values);
141        }
142
143        /**
144         * Initialize this genotype with random values based on the size of the
145         * {@code key} set.
146         * 
147         * @param random
148         *            the random number generator
149         */
150        public void init(Random random) {
151                super.init(random, keys.size());
152        }
153
154        /**
155         * Not supported. Use {@link SelectMapGenotype#init(Random)} instead.
156         * 
157         * @see org.opt4j.core.genotype.IntegerGenotype#init(java.util.Random, int)
158         */
159        @Override
160        public void init(Random random, int n) {
161                throw new UnsupportedOperationException(MapGenotype.ERROR_MESSAGE_UNSUPPORTED_INIT);
162        }
163
164        /*
165         * (non-Javadoc)
166         * 
167         * @see org.opt4j.genotype.MapGenotype#containsKey(java.lang.Object)
168         */
169        @Override
170        public boolean containsKey(K key) {
171                return keys.contains(key);
172        }
173
174        /*
175         * (non-Javadoc)
176         * 
177         * @see org.opt4j.genotype.MapGenotype#getIndexOf(java.lang.Object)
178         */
179        @Override
180        public int getIndexOf(K key) {
181                if (!containsKey(key)) {
182                        throw new IllegalArgumentException(MapGenotype.ERROR_MESSAGE_INVALID_KEY);
183                }
184                return keys.indexOf(key);
185        }
186
187        /*
188         * (non-Javadoc)
189         * 
190         * @see org.opt4j.genotype.MapGenotype#getValue(java.lang.Object)
191         */
192        @Override
193        public V getValue(K key) {
194                int i = getIndexOf(key);
195                int v = get(i);
196                List<V> valueList = values.get(key);
197                return valueList.get(v);
198        }
199
200        /*
201         * (non-Javadoc)
202         * 
203         * @see org.opt4j.genotype.MapGenotype#setValue(java.lang.Object,
204         * java.lang.Object)
205         */
206        @Override
207        public void setValue(K key, V value) {
208                int i = getIndexOf(key);
209                while (size() <= i) {
210                        add(bounds.getLowerBound(i));
211                }
212                List<V> valueList = values.get(key);
213                if (!valueList.contains(value)) {
214                        throw new IllegalArgumentException(
215                                        "The list provided for key " + key + " does not contain the value " + value);
216                }
217                int v = valueList.indexOf(value);
218                set(i, v);
219        }
220
221        /*
222         * (non-Javadoc)
223         * 
224         * @see org.opt4j.core.Genotype#newInstance()
225         */
226        @Override
227        @SuppressWarnings({ "unchecked", "rawtypes" })
228        public <G extends Genotype> G newInstance() {
229                try {
230                        Constructor<? extends SelectMapGenotype> cstr = this.getClass().getConstructor(List.class, Map.class);
231                        return (G) cstr.newInstance(keys, values);
232                } catch (Exception e) {
233                        throw new RuntimeException(e);
234                }
235        }
236
237        /*
238         * (non-Javadoc)
239         * 
240         * @see java.util.AbstractCollection#toString()
241         */
242        @Override
243        public String toString() {
244                String s = "[";
245                for (int i = 0; i < size(); i++) {
246                        K key = keys.get(i);
247                        int v = this.get(i);
248                        V value = values.get(key).get(v);
249                        s += key + "=" + value + ";";
250                }
251                return s + "]";
252        }
253
254        private static final long serialVersionUID = 1L;
255
256        /*
257         * (non-Javadoc)
258         * 
259         * @see org.opt4j.genotype.MapGenotype#getKeys()
260         */
261        @Override
262        public Collection<K> getKeys() {
263                return Collections.unmodifiableList(keys);
264        }
265}
