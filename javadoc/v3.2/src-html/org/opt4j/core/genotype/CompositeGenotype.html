024package org.opt4j.core.genotype;
025
026import java.util.Collection;
027import java.util.HashMap;
028import java.util.Iterator;
029import java.util.Map;
030import java.util.Map.Entry;
031import java.util.Set;
032
033import org.opt4j.core.Genotype;
034
035/**
036 * <p>
037 * The {@link CompositeGenotype} is a base class for {@link Genotype} classes
038 * that consist of multiple {@link Genotype}s. The method
039 * {@link CompositeGenotype#size()} returns the sum of the sizes of the
040 * contained {@link Genotype}s.
041 * </p>
042 * A specific {@link CompositeGenotype} has to add each contained
043 * {@link Genotype} by calling the method {@link #put(Object, Genotype)} where
044 * {@link Object} is an arbitrary identifier.
045 * 
046 * <p>
047 * Example:
048 * 
049 * <pre>
050 *      SpecificGenotype extends CompositeGenotype&lt;Integer, Genotype&gt; {
051 *              public void setDoubleVector(DoubleGenotype genotype){
052 *                      put(0, genotype);
053 *      }
054 *              public DoubleGenotype getDoubleVector(){ 
055 *                      return get(0);
056 *      }
057 * 
058 *              public void setBinaryVector(BooleanGenotype genotype){
059 *                      put(1, genotype);
060 *              }
061 *              public BooleanGenotype getDoubleVector(){
062 *                      return get(1); 
063 *              }
064 * }
065 * </pre>
066 * 
067 * </p>
068 * 
069 * @param <K>
070 *            the type of key for the mapping
071 * @param <V>
072 *            the type of {@link Genotype}
073 * 
074 * @author lukasiewycz
075 */
076public class CompositeGenotype<K, V extends Genotype> implements Genotype, Iterable<Entry<K, V>> {
077
078        protected final Map<K, V> map = new HashMap<K, V>();
079
080        /**
081         * Constructs a {@link CompositeGenotype}.
082         */
083        public CompositeGenotype() {
084                super();
085        }
086
087        /**
088         * Constructs a {@link CompositeGenotype} with values from a given map.
089         * 
090         * @param map
091         *            initial values
092         */
093        public CompositeGenotype(Map<K, V> map) {
094                this();
095                this.map.putAll(map);
096        }
097
098        /*
099         * (non-Javadoc)
100         * 
101         * @see org.opt4j.core.Genotype#size()
102         */
103        @Override
104        public int size() {
105                int sum = 0;
106                for (Genotype element : map.values()) {
107                        sum += element.size();
108                }
109                return sum;
110        }
111
112        /**
113         * Returns the {@link Genotype} for the given key with an implicit cast to
114         * the specific {@link Genotype} type.
115         * 
116         * @param <G>
117         *            the type of genotype
118         * @param key
119         *            the key
120         * @return the addressed genotype
121         */
122        @SuppressWarnings("unchecked")
123        public <G> G get(Object key) {
124                V value = map.get(key);
125                return (G) value;
126        }
127
128        /**
129         * Adds a {@code key}, {@code value} pair.
130         * 
131         * @param key
132         *            the key
133         * @param value
134         *            the value ({@code Genotype})
135         */
136        public void put(K key, V value) {
137                map.put(key, value);
138        }
139
140        /**
141         * Removes all {@code key}, {@code value} pairs.
142         */
143        public void clear() {
144                map.clear();
145        }
146
147        /**
148         * Returns all {@code keys}.
149         * 
150         * @return all {@code keys}
151         */
152        public Set<K> keySet() {
153                return map.keySet();
154        }
155
156        /**
157         * Returns all {@code values} which are the contained {@link Genotype}
158         * objects.
159         * 
160         * @return all {@code values}
161         */
162        public Collection<V> values() {
163                return map.values();
164        }
165
166        /**
167         * Returns the {@link Iterator} over the {@link Entry} pairs.
168         * 
169         * @return the iterator over the entry pairs
170         */
171        @Override
172        public Iterator<Entry<K, V>> iterator() {
173                return map.entrySet().iterator();
174        }
175
176        /*
177         * (non-Javadoc)
178         * 
179         * @see org.opt4j.core.Genotype#newInstance()
180         */
181        @Override
182        @SuppressWarnings("unchecked")
183        public <G extends Genotype> G newInstance() {
184                try {
185                        return (G) this.getClass().newInstance();
186                } catch (InstantiationException e) {
187                        throw new RuntimeException(e);
188                } catch (IllegalAccessException e) {
189                        throw new RuntimeException(e);
190                }
191        }
192
193        /*
194         * (non-Javadoc)
195         * 
196         * @see java.lang.Object#toString()
197         */
198        @Override
199        public String toString() {
200                String s = "[";
201                for (Entry<K, V> entry : this) {
202                        K key = entry.getKey();
203                        V value = entry.getValue();
204                        s += key + "=" + value + ";";
205                }
206                return s + "]";
207        }
208}
