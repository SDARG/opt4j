024package org.opt4j.core.genotype;
025
026import java.util.List;
027
028/**
029 * The {@link IntegerBounds} is an implementation of the {@link Bounds} for the
030 * {@link IntegerGenotype} that accepts arrays as well as lists for as bounds.
031 * 
032 * @author lukasiewycz
033 * 
034 */
035public class IntegerBounds implements Bounds<Integer> {
036
037        protected final int[] lower;
038        protected final int[] upper;
039
040        /**
041         * Constructs a {@link IntegerBounds} with arrays.
042         * 
043         * @param lower
044         *            the lower bounds
045         * @param upper
046         *            the upper bounds
047         */
048        public IntegerBounds(int[] lower, int[] upper) {
049                if (lower.length != upper.length) {
050                        throw new IllegalArgumentException("Lower and upper bounds arrays should have same length.");
051                }
052                this.lower = lower;
053                this.upper = upper;
054        }
055
056        /**
057         * Constructs a {@link IntegerBounds} with lists.
058         * 
059         * @param lower
060         *            the lower bounds
061         * @param upper
062         *            the upper bounds
063         */
064        public IntegerBounds(List<Integer> lower, List<Integer> upper) {
065                if (lower.size() != upper.size()) {
066                        throw new IllegalArgumentException("Lower and upper bounds lists should have the same size.");
067                }
068                this.lower = new int[lower.size()];
069                this.upper = new int[upper.size()];
070                for (int i = 0; i < lower.size(); i++) {
071                        this.lower[i] = lower.get(i);
072                }
073                for (int i = 0; i < upper.size(); i++) {
074                        this.upper[i] = upper.get(i);
075                }
076
077        }
078
079        /*
080         * (non-Javadoc)
081         * 
082         * @see org.opt4j.genotype.Bounds#getLowerBound(int)
083         */
084        @Override
085        public Integer getLowerBound(int index) {
086                return lower[index];
087        }
088
089        /*
090         * (non-Javadoc)
091         * 
092         * @see org.opt4j.genotype.Bounds#getUpperBound(int)
093         */
094        @Override
095        public Integer getUpperBound(int index) {
096                return upper[index];
097        }
098
099}
