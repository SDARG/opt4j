023package org.opt4j.core.genotype;
024
025import java.lang.reflect.Constructor;
026import java.util.Collection;
027import java.util.Collections;
028import java.util.HashSet;
029import java.util.List;
030import java.util.Random;
031import java.util.Set;
032
033import org.opt4j.core.Genotype;
034
035/**
036 * <p>
037 * The {@link DoubleMapGenotype} is a {@link DoubleGenotype} with the
038 * {@link MapGenotype} functionality.
039 * </p>
040 * <p>
041 * Example problem: Select filling level of five bottles bottle<sub>1</sub>,
042 * bottle<sub>2</sub>, bottle<sub>3</sub>, bottle<sub>4</sub>,
043 * bottle<sub>5</sub><br/>
044 * Example usage: <blockquote>
045 * 
046 * <pre>
047 * DoubleMapGenotype&lt;Bottle&gt; genotype = new DoubleMapGenotype&lt;Bottle&gt;(
048 *              Arrays.asList(bottle1, bottle2, bottle3, bottle4, bottle5));
049 * genotype.init(new Random());
050 * </pre>
051 * 
052 * </blockquote> Example instance:
053 * [bottle1=0.5035947840006195,bottle2=0.9693492473483428
054 * ,bottle3=0.12786372316728167
055 * ,bottle4=0.5299369900029843,bottle5=0.8055193291478467]<br/>
056 * Example search space size: [0;1]<sup>5</sup>
057 * </p>
058 * 
059 * @author lukasiewycz
060 * @param <K>
061 *            the type of keys
062 */
063public class DoubleMapGenotype<K> extends DoubleGenotype implements MapGenotype<K, Double> {
064
065        protected final List<K> keys;
066
067        /**
068         * Constructs a {@link DoubleMapGenotype}.
069         * 
070         * @param keys
071         *            the list of keys
072         * @param bounds
073         *            the lower and upper bounds
074         */
075        public DoubleMapGenotype(List<K> keys, Bounds<Double> bounds) {
076                super(bounds);
077                Set<K> uniqueKeys = new HashSet<K>(keys);
078                if (uniqueKeys.size() < keys.size()) {
079                        throw new IllegalArgumentException(MapGenotype.ERROR_MESSAGE_NON_UNIQUE_KEYS);
080                }
081                this.keys = keys;
082        }
083
084        /**
085         * Initialize this genotype with random values based on the size of the
086         * {@code keys}.
087         * 
088         * @param random
089         *            the random number generator
090         */
091        public void init(Random random) {
092                super.init(random, keys.size());
093        }
094
095        /**
096         * Not supported. Use {@link DoubleMapGenotype#init(Random)} instead.
097         * 
098         * @see org.opt4j.core.genotype.DoubleGenotype#init(java.util.Random, int)
099         */
100        @Override
101        public void init(Random random, int n) {
102                throw new UnsupportedOperationException(MapGenotype.ERROR_MESSAGE_UNSUPPORTED_INIT);
103        }
104
105        /*
106         * (non-Javadoc)
107         * 
108         * @see org.opt4j.genotype.MapGenotype#containsKey(java.lang.Object)
109         */
110        @Override
111        public boolean containsKey(K key) {
112                return keys.contains(key);
113        }
114
115        /*
116         * (non-Javadoc)
117         * 
118         * @see org.opt4j.genotype.MapGenotype#getValue(java.lang.Object)
119         */
120        @Override
121        public Double getValue(K key) {
122                int i = keys.indexOf(key);
123                return get(i);
124        }
125
126        /*
127         * (non-Javadoc)
128         * 
129         * @see org.opt4j.genotype.MapGenotype#setValue(java.lang.Object,
130         * java.lang.Object)
131         */
132        @Override
133        public void setValue(K key, Double value) {
134                if (!containsKey(key)) {
135                        throw new IllegalArgumentException(MapGenotype.ERROR_MESSAGE_INVALID_KEY);
136                }
137                int i = keys.indexOf(key);
138                if (value < bounds.getLowerBound(i) || value > bounds.getUpperBound(i))
139                        throw new IllegalArgumentException(MapGenotype.ERROR_MESSAGE_OUT_OF_BOUNDS);
140                while (size() <= i) {
141                        add(bounds.getLowerBound(i));
142                }
143                set(i, value);
144        }
145
146        /*
147         * (non-Javadoc)
148         * 
149         * @see org.opt4j.core.Genotype#newInstance()
150         */
151        @Override
152        @SuppressWarnings({ "unchecked", "rawtypes" })
153        public <G extends Genotype> G newInstance() {
154                try {
155                        Constructor<? extends DoubleMapGenotype> cstr = this.getClass().getConstructor(List.class, Bounds.class);
156                        return (G) cstr.newInstance(keys, bounds);
157                } catch (Exception e) {
158                        throw new RuntimeException(e);
159                }
160        }
161
162        /*
163         * (non-Javadoc)
164         * 
165         * @see java.util.AbstractCollection#toString()
166         */
167        @Override
168        public String toString() {
169                StringBuilder stringBuilder = new StringBuilder();
170                stringBuilder.append("[");
171                for (int i = 0; i < size(); i++) {
172                        K key = keys.get(i);
173                        double value = this.get(i);
174                        stringBuilder.append(key + "=" + value + ";");
175                }
176                stringBuilder.append("]");
177                return stringBuilder.toString();
178        }
179
180        /*
181         * (non-Javadoc)
182         * 
183         * @see org.opt4j.genotype.MapGenotype#getIndexOf(java.lang.Object)
184         */
185        @Override
186        public int getIndexOf(K key) {
187                if (!containsKey(key)) {
188                        throw new IllegalArgumentException(MapGenotype.ERROR_MESSAGE_INVALID_KEY);
189                }
190                return keys.indexOf(key);
191        }
192
193        private static final long serialVersionUID = 1L;
194
195        /*
196         * (non-Javadoc)
197         * 
198         * @see org.opt4j.genotype.MapGenotype#getKeys()
199         */
200        @Override
201        public Collection<K> getKeys() {
202                return Collections.unmodifiableList(keys);
203        }
204
