073        static class ProgressRenderer extends DefaultTableCellRenderer {
074                private final JProgressBar b = new JProgressBar(0, 100);
075
076                public ProgressRenderer() {
077                        super();
078                        setOpaque(true);
079                        b.setBorder(BorderFactory.createEmptyBorder(1, 1, 1, 1));
080                        b.setStringPainted(true);
081                        b.setBackground(Color.WHITE);
082                }
083
084                @Override
085                public Component getTableCellRendererComponent(JTable table, Object value, boolean isSelected,
086                                boolean hasFocus, int row, int column) {
087                        if (value instanceof Progress) {
088                                Progress progress = (Progress) value;
089                                Double v = progress.get();
090                                if (v == null) {
091                                        v = 0.0;
092                                }
093                                b.setValue((int) (v * 100));
094                                b.setString("" + progress.getCurrentIteration() + "/" + progress.getMaxIterations());
095                                return b;
096                        }
097                        return super.getTableCellRendererComponent(table, value, isSelected, hasFocus, row, column);
098                }
099        }

101        class Table extends JTable {
102
103                /*
104                 * (non-Javadoc)
105                 * 
106                 * @see
107                 * javax.swing.JTable#prepareRenderer(javax.swing.table.TableCellRenderer
108                 * , int, int)
109                 */
110                @Override
111                public Component prepareRenderer(TableCellRenderer renderer, int row, int column) {
112                        Component c = super.prepareRenderer(renderer, row, column);
113
114                        if (!(renderer instanceof ProgressRenderer)) {
115                                Task task = executionEnvironment.getTasks().get(row);
116                                if (task.getException() == null) {
117                                        c.setForeground(Color.BLACK);
118                                } else {
119                                        c.setForeground(Color.RED);
120                                }
121
122                                if (c instanceof JComponent) {
123                                        JComponent jc = (JComponent) c;
124
125                                        jc.setBorder(BorderFactory.createEmptyBorder(0, 3, 0, 0));
126
127                                        Object value = this.getValueAt(row, column);
128                                        if (value != null) {
129                                                char[] chars = value.toString().toCharArray();
130                                                int length = jc.getFontMetrics(jc.getFont()).charsWidth(chars, 0, chars.length);
131                                                if (this.getColumnModel().getColumn(column).getWidth() < length) {
132                                                        jc.setToolTipText(format.formatTooltip(value.toString()));
133                                                } else {
134                                                        jc.setToolTipText(null);
135                                                }
136                                        }
137                                }
138                        }
139
140                        return c;
141                }
142
143                @Override
144                public TableCellRenderer getCellRenderer(final int row, final int column) {
145                        if (column == 2) {
146                                return new ProgressRenderer();
147                        }
148                        return super.getCellRenderer(row, column);
149                }

153        /**
154         * The {@link Model} of the table.
155         */
156        protected class Model extends AbstractTableModel {
157                protected final String[] columnNames = { "Task", "State", "Progress" };
158
159                @Override
160                public int getColumnCount() {
161                        return columnNames.length;
162                }
163
164                @Override
165                public int getRowCount() {
166                        return executionEnvironment.getTasks().size();
167                }
168
169                @Override
170                public Object getValueAt(int row, int column) {
171                        Opt4JTask task = (Opt4JTask) executionEnvironment.getTasks().get(row);
172                        if (column == 0) {
173                                return row + " " + task.toString();
174                        } else if (column == 1) {
175                                if (task.getException() != null) {
176                                        return "EXCEPTION: \n" + task.getException();
177                                }
178                                try {
179                                        Control control = task.getInstance(Control.class);
180                                        return "" + control.getState();
181                                } catch (RuntimeException e) {
182                                        return "";
183                                }
184                        } else if (column == 2) {
185                                switch (task.getState()) {
186                                case EXECUTING:
187                                        Progress progress = Opt4JTasksPanel.this.progessMap.get(task);
188                                        if (progress != null) {
189                                                return progress;
190                                        }
191                                        Optimizer optimizer = task.getInstance(Optimizer.class);
192                                        if (optimizer != null) {
193                                                progress = task.getInstance(Progress.class);
194                                                optimizer.addOptimizerIterationListener(progress);
195                                                progessMap.put(task, progress);
196                                                return progress;
197                                        }
198                                        return " " + task.getState();
199                                default:
200                                        return "  " + task.getState();
201                                }
202                        }
203                        return null;
204                }
205
206                @Override
207                public String getColumnName(int i) {
208                        return columnNames[i];
209                }
210        }

265        /**
266         * The {@link PopupListener} for the table elements.
267         */
268        protected class PopupListener extends MouseAdapter {
269
270                /*
271                 * (non-Javadoc)
272                 * 
273                 * @see
274                 * java.awt.event.MouseAdapter#mousePressed(java.awt.event.MouseEvent)
275                 */
276                @Override
277                public void mousePressed(MouseEvent e) {
278                        if (e.isPopupTrigger()) {
279                                display(e);
280                        }
281                }
282
283                /*
284                 * (non-Javadoc)
285                 * 
286                 * @see
287                 * java.awt.event.MouseAdapter#mouseReleased(java.awt.event.MouseEvent)
288                 */
289                @Override
290                public void mouseReleased(MouseEvent e) {
291                        if (e.isPopupTrigger()) {
292                                display(e);
293                        }
294                }
295
296                /**
297                 * Displays a popup menu with controls for running tasks.
298                 * 
299                 * @param event
300                 *            the triggered event
301                 */
302                void display(MouseEvent event) {
303
304                        if (event.isPopupTrigger()) {
305                                try {
306                                        Point p = event.getPoint();
307                                        int row = table.rowAtPoint(p);
308
309                                        Opt4JTask task = (Opt4JTask) executionEnvironment.getTasks().get(row);
310
311                                        JPopupMenu menu = new JPopupMenu();
312                                        addConfigLoadMenuItems(menu, task);
313                                        if (task.getState() == State.EXECUTING) {
314                                                menu.addSeparator();
315                                                addControlMenuItems(menu, task);
316                                        }
317                                        menu.show(event.getComponent(), event.getX(), event.getY());
318                                } catch (IndexOutOfBoundsException e) {
319                                }
320                        }
321                }
322
323                /**
324                 * Adds an item to the given menu which allows to restore the
325                 * configuration of a task.
326                 * 
327                 * @param menu
328                 *            the menu
329                 * @param task
330                 *            the task which configuration is restored
331                 */
332                private void addConfigLoadMenuItems(JPopupMenu menu, final Opt4JTask task) {
333                        JMenuItem loadConfig = new JMenuItem("Restore Configuration", Icons.getIcon(Icons.FOLDER_ADD));
334
335                        loadConfig.addActionListener(new ActionListener() {
336                                @Override
337                                public void actionPerformed(ActionEvent e) {
338                                        selectedModules.clear();
339                                        for (Module module : task.getModules()) {
340                                                PropertyModule pModule;
341                                                if (module instanceof PropertyModule) {
342                                                        pModule = (PropertyModule) module;
343                                                } else {
344                                                        pModule = new PropertyModule(module);
345                                                }
346
347                                                selectedModules.add(pModule);
348                                        }
349                                }
350                        });
351
352                        menu.add(loadConfig);
353                }
354
355                /**
356                 * Adds the control menu items to start, pause, stop and terminate the
357                 * given task.
358                 * 
359                 * @param menu
360                 *            the menu to enhance
361                 * @param task
362                 *            the optimization task to control
363                 */
364                private void addControlMenuItems(JPopupMenu menu, Opt4JTask task) {
365                        final Control control;
366                        try {
367                                control = task.getInstance(Control.class);
368                        } catch (RuntimeException e) {
369                                return;
370                        }
371                        assert control != null : "Task should be in state EXECUTING but is in state " + task.getState();
372
373                        JMenuItem start = new JMenuItem("Start", Icons.getIcon(Icons.CONTROL_START));
374                        JMenuItem pause = new JMenuItem("Pause", Icons.getIcon(Icons.CONTROL_PAUSE));
375                        JMenuItem stop = new JMenuItem("STOP", Icons.getIcon(Icons.CONTROL_STOP));
376                        JMenuItem terminate = new JMenuItem("Terminate", Icons.getIcon(Icons.CONTROL_TERM));
377                        terminate.addActionListener(new ActionListener() {
378                                @Override
379                                public void actionPerformed(ActionEvent e) {
380                                        control.doTerminate();
381                                }
382                        });
383                        start.addActionListener(new ActionListener() {
384                                @Override
385                                public void actionPerformed(ActionEvent e) {
386                                        control.doStart();
387                                }
388                        });
389                        pause.addActionListener(new ActionListener() {
390                                @Override
391                                public void actionPerformed(ActionEvent e) {
392                                        control.doPause();
393                                }
394                        });
395                        stop.addActionListener(new ActionListener() {
396                                @Override
397                                public void actionPerformed(ActionEvent e) {
398                                        control.doStop();
399                                }
400                        });
401
402                        if (control != null) {
403                                switch (control.getState()) {
404                                case RUNNING:
405                                        start.setEnabled(false);
406                                        break;
407                                case PAUSED:
408                                        pause.setEnabled(false);
409                                        break;
410                                case TERMINATED:
411                                        terminate.setEnabled(false);
412                                        start.setEnabled(false);
413                                        pause.setEnabled(false);
414                                        stop.setEnabled(false);
415                                        break;
416                                case STOPPED:
417                                        start.setEnabled(false);
418                                        pause.setEnabled(false);
419                                        stop.setEnabled(false);
420                                        break;
421                                }
422                        } else {
423                                terminate.setEnabled(false);
424                                start.setEnabled(false);
425                                pause.setEnabled(false);
426                                stop.setEnabled(false);
427                        }
428
429                        menu.add(start);
430                        menu.add(pause);
431                        menu.add(stop);
432                        menu.add(terminate);
433                }
434        }
