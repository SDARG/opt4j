024package org.opt4j.core.optimizer;
025
026import java.util.Set;
027import java.util.concurrent.CopyOnWriteArraySet;
028
029import com.google.inject.Singleton;
030
031/**
032 * <p>
033 * The {@link Control} allows to pause, stop, and terminate, the optimization
034 * process.
035 * </p>
036 * <p>
037 * The {@link Optimizer} calls the methods
038 * <ul>
039 * <li>{@link Control#checkpoint()} and</li>
040 * <li>{@link Control#checkpointStop()}</li>
041 * </ul>
042 * in which the process might get paused, terminated (by the
043 * {@link TerminationException}) or stopped (by the {@link StopException}). By
044 * definition, a {@link StopException} might get thrown only between iterations,
045 * a {@link TerminationException} also within an iteration of the optimization
046 * algorithm. In this context, the optimization algorithm calls
047 * {@link #checkpointStop()} between the iterations and {@link #checkpoint()}
048 * within the iteration.
049 * </p>
050 * 
051 * @author lukasiewycz
052 * 
053 */
054@Singleton
055public class Control {
056
057        protected State state = State.RUNNING;
058
059        protected final Set<ControlListener> listeners = new CopyOnWriteArraySet<ControlListener>();
060
061        /**
062         * The {@link State} of the control.
063         * 
064         * @author lukasiewycz
065         * 
066         */
067        public enum State {
068                /**
069                 * The running state.
070                 */
071                RUNNING,
072                /**
073                 * The paused state.
074                 */
075                PAUSED,
076                /**
077                 * The stopped state.
078                 */
079                STOPPED,
080                /**
081                 * The terminated state.
082                 */
083                TERMINATED;
084
085                /**
086                 * Returns {@code true} if the current state allows to start the
087                 * optimization.
088                 * 
089                 * @return {@code true} if start is available
090                 */
091                public boolean isStartable() {
092                        return (this == State.PAUSED);
093                }
094
095                /**
096                 * Returns {@code true} if the current state allows to pause the
097                 * optimization.
098                 * 
099                 * @return {@code true} if pause is available
100                 */
101                public boolean isPausable() {
102                        return (this == State.RUNNING);
103                }
104
105                /**
106                 * Returns {@code true} if the current state allows to stop the
107                 * optimization.
108                 * 
109                 * @return {@code true} if stop is available
110                 */
111                public boolean isStoppable() {
112                        return (this != State.TERMINATED);
113                }
114        }
115
116        /**
117         * Starts the optimization.
118         */
119        public synchronized void doStart() {
120                if (state.isStartable()) {
121                        setState(State.RUNNING);
122                }
123        }
124
125        /**
126         * Pauses the optimization.
127         */
128        public synchronized void doPause() {
129                if (state.isPausable()) {
130                        setState(State.PAUSED);
131                }
132        }
133
134        /**
135         * Stops the optimization.
136         */
137        public synchronized void doStop() {
138                if (state.isStoppable()) {
139                        setState(State.STOPPED);
140                }
141        }
142
143        /**
144         * Terminates the optimization.
145         */
146        public synchronized void doTerminate() {
147                setState(State.TERMINATED);
148        }
149
150        /**
151         * Returns {@code true} if the optimization is running.
152         * 
153         * @return {@code true} if the optimization is running
154         */
155        public synchronized boolean isRunning() {
156                return (state == State.RUNNING);
157        }
158
159        /**
160         * Returns {@code true} if the optimization is paused.
161         * 
162         * @return {@code true} if the optimization is paused
163         */
164        public synchronized boolean isPaused() {
165                return (state == State.PAUSED);
166        }
167
168        /**
169         * Returns {@code true} if the optimization is stopped.
170         * 
171         * @return {@code true} if the optimization is stopped
172         */
173        public synchronized boolean isStopped() {
174                return (state == State.STOPPED);
175        }
176
177        /**
178         * Returns {@code true} if the optimization is terminated.
179         * 
180         * @return {@code true} if the optimization is terminated
181         */
182        public synchronized boolean isTerminated() {
183                return (state == State.TERMINATED);
184        }
185
186        /**
187         * A checkpoint that checks for termination.
188         * 
189         * @throws TerminationException
190         *             if the optimization is terminated
191         */
192        public synchronized void checkpoint() throws TerminationException {
193                while (state == State.PAUSED) {
194                        try {
195                                wait();
196                        } catch (InterruptedException e) {
197                        }
198                }
199
200                if (state == State.TERMINATED) {
201                        throw new TerminationException();
202                }
203        }
204
205        /**
206         * A checkpoint that checks for termination and stop.
207         * 
208         * @throws TerminationException
209         *             if the optimization is terminated
210         * @throws StopException
211         *             if the optimization is stopped
212         */
213        public synchronized void checkpointStop() throws TerminationException, StopException {
214                checkpoint();
215                if (state == State.STOPPED) {
216                        throw new StopException();
217                }
218        }
219
220        /**
221         * Returns the state.
222         * 
223         * @return the state
224         */
225        public State getState() {
226                return state;
227        }
228
229        /**
230         * Sets the state and invokes listeners if the state has changed.
231         * 
232         * @param state
233         *            the desired state
234         */
235        protected synchronized void setState(State state) {
236                boolean changed = (this.state != state);
237
238                if (changed) {
239                        this.state = state;
240                        // wake up all potentially waiting (paused) processes
241                        notifyAll();
242
243                        for (ControlListener listener : listeners) {
244                                listener.stateChanged(state);
245                        }
246                }
247        }
248
249        /**
250         * Adds a {@link ControlListener}.
251         * 
252         * @see #removeListener
253         * @param listener
254         *            the listener to add
255         */
256        public void addListener(ControlListener listener) {
257                listeners.add(listener);
258        }
259
260        /**
261         * Removes a {@link ControlListener}.
262         * 
263         * @see #addListener
264         * @param listener
265         *            the listener to remove
266         */
267        public void removeListener(ControlListener listener) {
268                listeners.remove(listener);
269        }
270}
