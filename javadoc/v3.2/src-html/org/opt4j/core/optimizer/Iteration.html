023package org.opt4j.core.optimizer;
024
025import com.google.inject.Inject;
026import com.google.inject.Singleton;
027
028/**
029 * The {@link Iteration} object is used as iteration counter for the
030 * optimization.
031 * 
032 * @author lukasiewycz
033 * 
034 */
035@Singleton
036public class Iteration {
037
038        protected int value = 0;
039        protected final int maxIterations;
040
041        /**
042         * Constructs a {@link Iteration} object.
043         * 
044         * @param maxIterations
045         *            the maximal number of iterations
046         */
047        @Inject
048        public Iteration(@MaxIterations int maxIterations) {
049                this.maxIterations = maxIterations;
050        }
051
052        /**
053         * Returns the current iteration value.
054         * 
055         * @return the current iteration value
056         */
057        public int value() {
058                return value;
059        }
060
061        /**
062         * Increases the iteration value by 1.
063         */
064        public void next() {
065                value++;
066        }
067
068        /**
069         * Returns the maximal number of iterations.
070         * 
071         * @return the maximal number of iterations
072         */
073        public int max() {
074                return maxIterations;
075        }
076
077}
