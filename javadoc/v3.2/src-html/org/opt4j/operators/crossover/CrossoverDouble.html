024package org.opt4j.operators.crossover;
025
026import java.util.List;
027import java.util.Random;
028
029import org.opt4j.core.common.random.Rand;
030import org.opt4j.core.genotype.DoubleGenotype;
031import org.opt4j.operators.normalize.NormalizeDouble;
032
033import com.google.inject.ImplementedBy;
034import com.google.inject.Inject;
035
036/**
037 * Crossover for the {@link DoubleGenotype}.
038 * 
039 * @author lukasiewycz, glass
040 * 
041 */
042@ImplementedBy(CrossoverDoubleDefault.class)
043public abstract class CrossoverDouble implements Crossover<DoubleGenotype> {
044
045        protected final Random random;
046
047        protected final NormalizeDouble normalize;
048
049        /**
050         * Constructs a new crossover for the {@link DoubleGenotype}.
051         * 
052         * @param normalize
053         *            a normalize operator
054         * @param random
055         *            the random number generator
056         */
057        @Inject
058        public CrossoverDouble(NormalizeDouble normalize, Rand random) {
059                this.normalize = normalize;
060                this.random = random;
061        }
062
063        /*
064         * (non-Javadoc)
065         * 
066         * @see
067         * org.opt4j.operator.crossover.Crossover#crossover(org.opt4j.core.Genotype,
068         * org.opt4j.core.Genotype)
069         */
070        @Override
071        public Pair<DoubleGenotype> crossover(DoubleGenotype p1, DoubleGenotype p2) {
072                DoubleGenotype o1 = p1.newInstance();
073                DoubleGenotype o2 = p2.newInstance();
074
075                crossover(p1, p2, o1, o2);
076
077                normalize.normalize(o1);
078                normalize.normalize(o2);
079
080                Pair<DoubleGenotype> offspring = new Pair<DoubleGenotype>(o1, o2);
081                return offspring;
082        }
083
084        /**
085         * Performs a crossover of two parent {@link org.opt4j.core.Genotype}s that
086         * consist of double vectors.
087         * 
088         * @param p1
089         *            the first parent
090         * @param p2
091         *            the second parent
092         * @param o1
093         *            the first offspring
094         * @param o2
095         *            the second offspring
096         */
097        protected abstract void crossover(List<Double> p1, List<Double> p2, List<Double> o1, List<Double> o2);
098
099}
