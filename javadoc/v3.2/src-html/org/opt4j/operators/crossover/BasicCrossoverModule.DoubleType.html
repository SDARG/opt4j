024package org.opt4j.operators.crossover;
025
026import org.opt4j.core.config.annotations.Info;
027import org.opt4j.core.config.annotations.Required;
028import org.opt4j.core.genotype.BooleanGenotype;
029import org.opt4j.core.genotype.DoubleGenotype;
030import org.opt4j.core.genotype.IntegerGenotype;
031import org.opt4j.core.genotype.PermutationGenotype;
032import org.opt4j.core.optimizer.Operator;
033import org.opt4j.core.start.Constant;
034import org.opt4j.operators.crossover.CrossoverDoubleSBX.Nu;
035
036/**
037 * The {@link BasicCrossoverModule}.
038 * 
039 * @author lukasiewycz
040 * 
041 */
042@Info("Setting for the basic crossover classOperators for genotype variation.")
043public class BasicCrossoverModule extends CrossoverModule {
044
045        @Info("The type of the crossover operator for the Boolean genotype.")
046        protected BooleanType booleanType = BooleanType.RATE;
047
048        @Required(property = "booleanType", elements = { "RATE" })
049        @Info("The probability for a crossover point.")
050        @Constant(value = "rate", namespace = CrossoverBooleanRate.class)
051        protected double booleanRate = 0.5;
052
053        @Required(property = "booleanType", elements = { "XPOINT" })
054        @Info("The number of crossover points.")
055        @Constant(value = "x", namespace = CrossoverBooleanXPoint.class)
056        protected int booleanXPoints = 1;
057
058        @Info("The type of the crossover operator for the Double genotype.")
059        protected DoubleType doubleType = DoubleType.SBX;
060
061        @Required(property = "doubleType", elements = { "BLX", "UNFAIR_AVERAGE" })
062        protected double alpha = 0.5;
063
064        @Required(property = "doubleType", elements = { "SBX" })
065        @Constant(value = "nu", namespace = CrossoverDoubleSBX.class)
066        protected double nu = 15;
067
068        @Info("The type of the crossover operator for the Permutation genotype.")
069        protected PermutationType permutationType = PermutationType.ONEPOINT;
070
071        @Required(property = "permutationType", elements = { "ONEPOINT" })
072        @Constant(value = "rotation", namespace = CrossoverPermutationOnePoint.class)
073        protected boolean rotation = false;
074
075        @Info("The type of the crossover operator for the Integer genotype.")
076        protected IntegerType integerType = IntegerType.RATE;
077
078        @Required(property = "integerType", elements = { "RATE" })
079        @Info("The probability for a crossover point.")
080        @Constant(value = "rate", namespace = CrossoverIntegerRate.class)
081        protected double integerRate = 0.5;
082
083        @Required(property = "integerType", elements = { "XPOINT" })
084        @Info("The number of crossover points.")
085        @Constant(value = "x", namespace = CrossoverIntegerXPoint.class)
086        protected int integerXPoints = 1;
087
088        /**
089         * Type of {@link Crossover} operator for the {@link BooleanGenotype}.
090         * 
091         * @author lukasiewycz
092         */
093        public enum BooleanType {
094                /**
095                 * Use the {@link CrossoverBooleanRate}.
096                 */
097                RATE,
098                /**
099                 * Use the {@link CrossoverBooleanXPoint}.
100                 */
101                XPOINT;
102        }
103
104        /**
105         * Type of {@link Crossover} operator for the {@link DoubleGenotype}.
106         * 
107         * @author glass
108         */
109        public enum DoubleType {
110                /**
111                 * Use the {@link CrossoverDoubleBLX}.
112                 */
113                BLX,
114                /**
115                 * Use the {@link CrossoverDoubleSBX}.
116                 */
117                SBX,
118                /**
119                 * Use the {@link CrossoverDoubleUnfairAverage}.
120                 */
121                UNFAIR_AVERAGE;
122        }
123
124        /**
125         * Type of {@link Crossover} operator for the {@link PermutationGenotype}.
126         * 
127         * @author lukasiewycz
128         * 
129         */
130        public enum PermutationType {
131                /**
132                 * Use the {@link CrossoverPermutationOnePoint}.
133                 */
134                ONEPOINT,
135                /**
136                 * Use the {@link CrossoverPermutationBucket}.
137                 */
138                BUCKET;
139        }
140
141        /**
142         * Type of {@link Crossover} operator for the {@link IntegerGenotype}.
143         * 
144         * @author lukasiewycz
145         */
146        public enum IntegerType {
147                /**
148                 * Use the {@link CrossoverBooleanRate}.
149                 */
150                RATE,
151                /**
152                 * Use the {@link CrossoverBooleanXPoint}.
153                 */
154                XPOINT;
155        }
156
157        /**
158         * Constructs a {@link BasicCrossoverModule}.
159         */
160        public BasicCrossoverModule() {
161                super();
162        }
163
164        /**
165         * Returns the rate of the {@link CrossoverIntegerRate}.
166         * 
167         * @return the rate
168         */
169        public double getIntegerRate() {
170                return integerRate;
171        }
172
173        /**
174         * Sets the rate of the {@link CrossoverIntegerRate}.
175         * 
176         * @param integerRate
177         *            the rate
178         */
179        public void setIntegerRate(double integerRate) {
180                this.integerRate = integerRate;
181        }
182
183        /**
184         * Returns the number of crossover points of the
185         * {@link CrossoverIntegerXPoint}.
186         * 
187         * @return the number of crossover points
188         */
189        public int getIntegerXPoints() {
190                return integerXPoints;
191        }
192
193        /**
194         * Sets the number of crossover points of the {@link CrossoverIntegerXPoint}
195         * .
196         * 
197         * @param integerXPoint
198         *            the number of crossover points
199         */
200        public void setIntegerXPoints(int integerXPoint) {
201                this.integerXPoints = integerXPoint;
202        }
203
204        /**
205         * Return the {@link Operator} for {@link IntegerGenotype}.
206         * 
207         * @return the operator
208         */
209        public IntegerType getIntegerType() {
210                return integerType;
211        }
212
213        /**
214         * Sets the {@link Operator} for {@link IntegerGenotype}.
215         * 
216         * @param integerType
217         *            the operator
218         */
219        public void setIntegerType(IntegerType integerType) {
220                this.integerType = integerType;
221        }
222
223        /**
224         * Returns the rate of the {@link CrossoverBooleanRate}.
225         * 
226         * @return the rate
227         */
228        public double getBooleanRate() {
229                return booleanRate;
230        }
231
232        /**
233         * Sets the rate of the {@link CrossoverBooleanRate}.
234         * 
235         * @param booleanRate
236         *            the rate
237         */
238        public void setBooleanRate(double booleanRate) {
239                this.booleanRate = booleanRate;
240        }
241
242        /**
243         * Returns the number of crossover points of the
244         * {@link CrossoverBooleanXPoint}.
245         * 
246         * @return the number of crossover points
247         */
248        public int getBooleanXPoints() {
249                return booleanXPoints;
250        }
251
252        /**
253         * Sets the number of crossover points of the {@link CrossoverBooleanXPoint}
254         * .
255         * 
256         * @param booleanXPoints
257         *            the number of crossover points
258         */
259        public void setBooleanXPoints(int booleanXPoints) {
260                this.booleanXPoints = booleanXPoints;
261        }
262
263        /**
264         * Return the {@link Operator} for {@link BooleanGenotype}.
265         * 
266         * @return the operator
267         */
268        public BooleanType getBooleanType() {
269                return booleanType;
270        }
271
272        /**
273         * Sets the {@link Operator} for {@link BooleanGenotype}.
274         * 
275         * @param booleanType
276         *            the operator
277         */
278        public void setBooleanType(BooleanType booleanType) {
279                this.booleanType = booleanType;
280        }
281
282        /**
283         * Returns the alpha value of the {@link CrossoverDoubleBLX} or
284         * {@link CrossoverDoubleUnfairAverage}, respectively.
285         * 
286         * @return the alpha value
287         */
288        public double getAlpha() {
289                return alpha;
290        }
291
292        /**
293         * Sets the alpha value of the {@link CrossoverDoubleBLX} or
294         * {@link CrossoverDoubleUnfairAverage}, respectively.
295         * 
296         * @param alpha
297         *            the alpha value
298         */
299        public void setAlpha(double alpha) {
300                this.alpha = alpha;
301        }
302
303        /**
304         * Returns the nu value of the {@link CrossoverDoubleSBX}.
305         * 
306         * @return the nu value
307         */
308        public double getNu() {
309                return nu;
310        }
311
312        /**
313         * Sets the nu value of the {@link CrossoverDoubleSBX}.
314         * 
315         * @param nu
316         *            the nu value
317         */
318        public void setNu(double nu) {
319                this.nu = nu;
320        }
321
322        /**
323         * Return the {@link Operator} for {@link DoubleGenotype}.
324         * 
325         * @return the operator
326         */
327        public DoubleType getDoubleType() {
328                return doubleType;
329        }
330
331        /**
332         * Sets the {@link Operator} for {@link DoubleGenotype}.
333         * 
334         * @param doubleType
335         *            the operator
336         */
337        public void setDoubleType(DoubleType doubleType) {
338                this.doubleType = doubleType;
339        }
340
341        /**
342         * Returns the {@link PermutationType}.
343         * 
344         * @return the permutation type
345         */
346        public PermutationType getPermutationType() {
347                return permutationType;
348        }
349
350        /**
351         * Sets the {@link PermutationType}.
352         * 
353         * @param permutationType
354         *            the permutation type
355         */
356        public void setPermutationType(PermutationType permutationType) {
357                this.permutationType = permutationType;
358        }
359
360        /**
361         * Returns {@code true} if rotation is used for the {@link Crossover}
362         * operator for the {@link PermutationGenotype}.
363         * 
364         * @return {@code true} if rotation is used
365         */
366        public boolean isRotation() {
367                return rotation;
368        }
369
370        /**
371         * Sets the rotation value for the {@link Crossover} operator for the
372         * {@link PermutationGenotype}.
373         * 
374         * @param rotation
375         *            the rotation value
376         */
377        public void setRotation(boolean rotation) {
378                this.rotation = rotation;
379        }
380
381        /*
382         * (non-Javadoc)
383         * 
384         * @see org.opt4j.start.Opt4JModule#config()
385         */
386        @Override
387        public void config() {
388                switch (booleanType) {
389                case XPOINT:
390                        bind(CrossoverBoolean.class).to(CrossoverBooleanXPoint.class).in(SINGLETON);
391                        break;
392                default: // RATE
393                        bind(CrossoverBoolean.class).to(CrossoverBooleanRate.class).in(SINGLETON);
394                        break;
395                }
396
397                switch (integerType) {
398                case XPOINT:
399                        bind(CrossoverInteger.class).to(CrossoverIntegerXPoint.class).in(SINGLETON);
400                        break;
401                default: // RATE
402                        bind(CrossoverInteger.class).to(CrossoverIntegerRate.class).in(SINGLETON);
403                        break;
404                }
405
406                switch (doubleType) {
407                case BLX:
408                        bind(CrossoverDouble.class).to(CrossoverDoubleBLX.class).in(SINGLETON);
409                        break;
410                case UNFAIR_AVERAGE:
411                        bind(CrossoverDouble.class).to(CrossoverDoubleUnfairAverage.class).in(SINGLETON);
412
413                        break;
414                default: // SBX
415                        bind(CrossoverDouble.class).to(CrossoverDoubleSBX.class).in(SINGLETON);
416                        break;
417                }
418
419                switch (permutationType) {
420                case BUCKET:
421                        bind(CrossoverPermutation.class).to(CrossoverPermutationBucket.class).in(SINGLETON);
422                        break;
423                default: // ONEPOINT
424                        bind(CrossoverPermutation.class).to(CrossoverPermutationOnePoint.class).in(SINGLETON);
425                        break;
426                }
427
428                bindConstant("alpha", CrossoverDoubleBLX.class).to(alpha);
429                bindConstant("alpha", CrossoverDoubleUnfairAverage.class).to(alpha);
430                bindConstant(Nu.class).to(nu);
431        }
