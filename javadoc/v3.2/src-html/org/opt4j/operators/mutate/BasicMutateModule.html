024package org.opt4j.operators.mutate;
025
026import org.opt4j.core.config.annotations.Info;
027import org.opt4j.core.config.annotations.Required;
028import org.opt4j.core.genotype.DoubleGenotype;
029import org.opt4j.core.genotype.PermutationGenotype;
030import org.opt4j.core.start.Constant;
031import org.opt4j.operators.copy.CopyModule;
032
033/**
034 * The basic {@link CopyModule}.
035 * 
036 * @author lukasiewycz
037 * 
038 */
039@Info("Setting for the basic mutate classOperators for genotype variation.")
040public class BasicMutateModule extends MutateModule {
041
042        @Info("The type of mutation rate.")
043        protected MutationRateType mutationRateType = MutationRateType.ADAPTIVE;
044
045        @Info("The type of the mutate operator for the Permutation genotype.")
046        protected PermutationType permutationType = PermutationType.MIXED;
047
048        @Info("The type of the mutate operator for the Double genotype.")
049        protected DoubleType doubleType = DoubleType.POLYNOMIAL;
050
051        @Required(property = "doubleType", elements = { "POLYNOMIAL" })
052        @Constant(value = "eta", namespace = MutateDoublePolynomial.class)
053        protected double eta = 20;
054
055        @Required(property = "doubleType", elements = { "GAUSS" })
056        @Constant(value = "sigma", namespace = MutateDoubleGauss.class)
057        protected double sigma = 0.1;
058
059        @Required(property = "mutationRateType", elements = { "CONSTANT" })
060        @Constant(value = "rate", namespace = ConstantMutationRate.class)
061        protected double mutationRate = 0.01;
062
063        /**
064         * The type of the used mutation rate.
065         * 
066         * @author lukasiewycz
067         * 
068         */
069        public enum MutationRateType {
070                /**
071                 * Use a constant mutation rate.
072                 */
073                CONSTANT,
074                /**
075                 * Use a variable (adaptive) mutation rate.
076                 */
077                ADAPTIVE;
078        }
079
080        /**
081         * Type of {@link Mutate} operator for the {@link DoubleGenotype}.
082         * 
083         * @author lukasiewycz
084         * 
085         */
086        public enum DoubleType {
087                /**
088                 * Use a constant mutation rate.
089                 */
090                GAUSS,
091                /**
092                 * Use a variable (adaptive) mutation rate.
093                 */
094                POLYNOMIAL;
095        }
096
097        /**
098         * Type of {@link Mutate} operator for the {@link PermutationGenotype}.
099         * 
100         * @author lukasiewycz
101         * 
102         */
103        public enum PermutationType {
104                /**
105                 * Use the {@link MutatePermutationMixed}.
106                 */
107                @Info("Use randomly SWAP,INSERT, or REVERT")
108                MIXED,
109                /**
110                 * Use the {@link MutatePermutationSwap}.
111                 */
112                @Info("Swaps two elements")
113                SWAP,
114                /**
115                 * Use the {@link MutatePermutationInsert}.
116                 */
117                @Info("Moves one element to another position")
118                INSERT,
119                /**
120                 * Use the {@link MutatePermutationRevert}.
121                 */
122                @Info("Reverts a part for the genotype")
123                REVERT;
124        }
125
126        /**
127         * Constructs a {@link BasicMutateModule}.
128         */
129        public BasicMutateModule() {
130                super();
131        }
132
133        /**
134         * Returns the {@link MutationRateType}.
135         * 
136         * @return the type of mutation rate
137         */
138        public MutationRateType getMutationRateType() {
139                return mutationRateType;
140        }
141
142        /**
143         * Sets the {@link MutationRateType}.
144         * 
145         * @param mutationRateType
146         *            the type of mutation rate
147         */
148        public void setMutationRateType(MutationRateType mutationRateType) {
149                this.mutationRateType = mutationRateType;
150        }
151
152        /**
153         * Returns the {@link MutationRate} as a double value.
154         * 
155         * @return the mutation rate
156         */
157        public double getMutationRate() {
158                return mutationRate;
159        }
160
161        /**
162         * Sets the {@link MutationRate}.
163         * 
164         * @param mutationRate
165         *            the mutation rate
166         */
167        public void setMutationRate(double mutationRate) {
168                this.mutationRate = mutationRate;
169        }
170
171        /**
172         * Returns the permutation mode.
173         * 
174         * @return the permutation mode
175         */
176        public PermutationType getPermutationType() {
177                return permutationType;
178        }
179
180        /**
181         * Sets the permutation mode.
182         * 
183         * @param permutationMode
184         *            the permutation mode
185         */
186        public void setPermutationType(PermutationType permutationMode) {
187                this.permutationType = permutationMode;
188        }
189
190        /**
191         * Returns the double type.
192         * 
193         * @return the doubleType
194         */
195        public DoubleType getDoubleType() {
196                return doubleType;
197        }
198
199        /**
200         * Sets the double type.
201         * 
202         * @param doubleType
203         *            the doubleType to set
204         */
205        public void setDoubleType(DoubleType doubleType) {
206                this.doubleType = doubleType;
207        }
208
209        /**
210         * Returns the eta.
211         * 
212         * @return the eta
213         */
214        public double getEta() {
215                return eta;
216        }
217
218        /**
219         * Sets the eta.
220         * 
221         * @param eta
222         *            the eta to set
223         */
224        public void setEta(double eta) {
225                this.eta = eta;
226        }
227
228        /**
229         * Returns the sigma value.
230         * 
231         * @return the sigma
232         */
233
234        public double getSigma() {
235                return sigma;
236        }
237
238        /**
239         * Sets the sigma value.
240         * 
241         * @param sigma
242         *            the sigma to set
243         */
244        public void setSigma(double sigma) {
245                this.sigma = sigma;
246        }
247
248        /*
249         * (non-Javadoc)
250         * 
251         * @see org.opt4j.start.Opt4JModule#config()
252         */
253        @Override
254        public void config() {
255                switch (mutationRateType) {
256                case ADAPTIVE:
257                        bind(MutationRate.class).to(AdaptiveMutationRate.class).in(SINGLETON);
258                        break;
259                case CONSTANT:
260                        bind(MutationRate.class).to(ConstantMutationRate.class).in(SINGLETON);
261                }
262
263                Class<? extends MutatePermutation> permutation = MutatePermutationMixed.class;
264
265                switch (permutationType) {
266                case MIXED:
267                        permutation = MutatePermutationMixed.class;
268                        break;
269                case SWAP:
270                        permutation = MutatePermutationSwap.class;
271                        break;
272                case INSERT:
273                        permutation = MutatePermutationInsert.class;
274                        break;
275                case REVERT:
276                        permutation = MutatePermutationRevert.class;
277                        break;
278                }
279                bind(MutatePermutation.class).to(permutation).in(SINGLETON);
280
281                switch (doubleType) {
282                case GAUSS:
283                        bind(MutateDouble.class).to(MutateDoubleGauss.class).in(SINGLETON);
284                        break;
285                case POLYNOMIAL:
286                        bind(MutateDouble.class).to(MutateDoublePolynomial.class).in(SINGLETON);
287                        break;
288                }
289        }
290}
