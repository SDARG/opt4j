023package org.opt4j.viewer;
024
025import java.awt.event.ActionEvent;
026import java.awt.event.ActionListener;
027
028import javax.swing.JButton;
029import javax.swing.SwingUtilities;
030import javax.swing.UIManager;
031
032import org.opt4j.core.config.Icons;
033import org.opt4j.core.optimizer.Control;
034import org.opt4j.core.optimizer.Control.State;
035import org.opt4j.core.optimizer.ControlListener;
036import org.opt4j.core.optimizer.Optimizer;
037import org.opt4j.core.optimizer.OptimizerStateListener;
038
039import com.google.inject.Inject;
040import com.google.inject.Singleton;
041
042/**
043 * <p>
044 * The {@link ControlButtons} contains {@link JButton}s for the controls:
045 * </p>
046 * <ul>
047 * <li>start</li>
048 * <li>pause</li>
049 * <li>stop</li>
050 * <li>terminate</li>
051 * </ul>
052 * 
053 * @see Control
054 * @author lukasiewycz
055 * 
056 */
057@Singleton
058public class ControlButtons implements OptimizerStateListener, ControlListener {
059
060        protected final Control control;
061
062        protected JButton start;
063        protected JButton pause;
064        protected JButton stop;
065        protected JButton terminate;
066
067        /**
068         * Constructs a {@link ControlButtons}.
069         * 
070         * @param control
071         *            the control
072         */
073        @Inject
074        public ControlButtons(Control control) {
075                this.control = control;
076
077                try {
078                        UIManager.setLookAndFeel(UIManager.getSystemLookAndFeelClassName());
079                } catch (Exception e) {
080                        e.printStackTrace();
081                }
082
083                construct();
084        }
085
086        /**
087         * Constructs the buttons.
088         */
089        protected final void construct() {
090                start = new JButton("", Icons.getIcon(Icons.CONTROL_START));
091                pause = new JButton("", Icons.getIcon(Icons.CONTROL_PAUSE));
092                stop = new JButton("", Icons.getIcon(Icons.CONTROL_STOP));
093                terminate = new JButton("", Icons.getIcon(Icons.CONTROL_TERM));
094
095                start.setToolTipText("Start");
096                pause.setToolTipText("Pause");
097                stop.setToolTipText("Stop");
098                terminate.setToolTipText("Terminate");
099
100                start.setFocusable(false);
101                pause.setFocusable(false);
102                stop.setFocusable(false);
103                terminate.setFocusable(false);
104
105                start.addActionListener(new ActionListener() {
106                        @Override
107                        public void actionPerformed(ActionEvent e) {
108                                control.doStart();
109                                update(null);
110                        }
111                });
112
113                pause.addActionListener(new ActionListener() {
114                        @Override
115                        public void actionPerformed(ActionEvent e) {
116                                control.doPause();
117                                update(null);
118                        }
119                });
120
121                stop.addActionListener(new ActionListener() {
122                        @Override
123                        public void actionPerformed(ActionEvent e) {
124                                control.doStop();
125                                update(null);
126                        }
127                });
128
129                terminate.addActionListener(new ActionListener() {
130                        @Override
131                        public void actionPerformed(ActionEvent e) {
132                                control.doTerminate();
133                                update(null);
134                        }
135                });
136        }
137
138        /**
139         * Updates the view.
140         */
141        protected void update(Optimizer optimizer) {
142
143                final State state;
144
145                if (optimizer != null && !optimizer.isRunning()) {
146                        state = State.TERMINATED;
147                } else {
148                        state = control.getState();
149                }
150
151                final boolean bStart = (state == State.PAUSED);
152                final boolean bPause = (state == State.RUNNING);
153                final boolean bStop = (state == State.RUNNING || state == State.PAUSED);
154                final boolean bTerminate = (state != State.TERMINATED);
155
156                SwingUtilities.invokeLater(new Runnable() {
157                        @Override
158                        public void run() {
159                                start.setEnabled(bStart);
160                                pause.setEnabled(bPause);
161                                stop.setEnabled(bStop);
162                                terminate.setEnabled(bTerminate);
163                        }
164                });
165        }
166
167        /*
168         * (non-Javadoc)
169         * 
170         * @see
171         * org.opt4j.core.optimizer.OptimizationStartListener#optimizationStarted
172         * (org.opt4j.core.optimizer.Optimizer)
173         */
174        @Override
175        public void optimizationStarted(Optimizer optimizer) {
176                update(optimizer);
177        }
178
179        /*
180         * (non-Javadoc)
181         * 
182         * @see
183         * org.opt4j.core.optimizer.OptimizationStopListener#optimizationStopped
184         * (org.opt4j.core.optimizer.Optimizer)
185         */
186        @Override
187        public void optimizationStopped(Optimizer optimizer) {
188                update(optimizer);
189        }
190
191        /**
192         * Returns the start button.
193         * 
194         * @return the start button
195         */
196        public JButton getStart() {
197                return start;
198        }
199
200        /**
201         * Returns the pause button.
202         * 
203         * @return the pause button
204         */
205        public JButton getPause() {
206                return pause;
207        }
208
209        /**
210         * Returns the stop button.
211         * 
212         * @return the stop button
213         */
214        public JButton getStop() {
215                return stop;
216        }
217
218        /**
219         * Returns the terminate button.
220         * 
221         * @return the terminate button
222         */
223        public JButton getTerminate() {
224                return terminate;
225        }
226
227        /*
228         * (non-Javadoc)
229         * 
230         * @see
231         * org.opt4j.core.optimizer.ControlListener#stateChanged(org.opt4j.core.
232         * optimizer.Control.State)
233         */
234        @Override
235        public void stateChanged(State state) {
236                update(null);
237        }
238
239}
