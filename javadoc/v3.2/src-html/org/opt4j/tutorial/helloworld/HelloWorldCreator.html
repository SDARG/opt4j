001package org.opt4j.tutorial.helloworld;
002
003import java.util.Random;
004
005import org.opt4j.core.genotype.SelectGenotype;
006import org.opt4j.core.problem.Creator;
007
008public class HelloWorldCreator implements Creator<SelectGenotype<Character>> {
009
010        Character[] ALPHABET = { 'A', 'B', 'C', 'D', 'E', 'F', 'G', 'H', 'I', 'J', 'K', 'L',
011                        'M', 'N', 'O', 'P', 'Q', 'R', 'S', 'T', 'U', 'V', 'W', 'X', 'Y', 'Z', ' ' };
012        Random random = new Random();
013
014        public SelectGenotype<Character> create() {
015                SelectGenotype<Character> genotype = new SelectGenotype<Character>(ALPHABET);
016                genotype.init(random, 11);
017                return genotype;
018        }
019
020}
