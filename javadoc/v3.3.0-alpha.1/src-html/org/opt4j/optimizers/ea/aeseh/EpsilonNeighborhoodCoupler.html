001package org.opt4j.optimizers.ea.aeseh;
002
003import java.util.ArrayList;
004import java.util.Collection;
005import java.util.HashSet;
006import java.util.List;
007import java.util.Map;
008import java.util.Random;
009import java.util.Set;
010
011import org.opt4j.core.Individual;
012import org.opt4j.core.Objective;
013import org.opt4j.core.Objectives;
014import org.opt4j.core.start.Constant;
015import org.opt4j.operators.crossover.Pair;
016import org.opt4j.optimizers.ea.Coupler;
017
018import com.google.inject.Inject;
019
020/**
021 * The {@link EpsilonNeighborhoodCoupler} implements a parent selection process
022 * based on by the ε-neighborhood.
023 * 
024 * @author Fedor Smirnov
025 *
026 */
027public class EpsilonNeighborhoodCoupler implements Coupler {
028
029        protected final EpsilonAdaptation epsilonAdaption;
030        protected final EpsilonMapping epsilonMapping;
031        protected final Random random;
032        protected final int plannedNeighborhoodNumber;
033        protected final AdaptiveEpsilon adaptiveEpsilonNeighborhood;
034
035        /**
036         * Basic constructor.
037         * 
038         * @param epsilonMapping
039         *            an {@link EpsilonMapping} that is used to enhance the
040         *            {@link Objectives} during the creation of the neighborhoods
041         * @param epsilonAdaption
042         *            an {@link EpsilonAdaptation} that adjusts the ε valued used
043         *            for the creation of the neighborhoods
044         * @param random
045         *            a {@link Random}
046         * @param plannedNeighborhoodNumber
047         *            A value provided by the user. The ε used for the creation of
048         *            the neighborhoods is adjusted in order to create a number of
049         *            neighborhoods similar to this value.
050         */
051        @Inject
052        public EpsilonNeighborhoodCoupler(EpsilonMapping epsilonMapping, EpsilonAdaptation epsilonAdaption, Random random,
053                        @Constant(value = "neighborhoodNumber", namespace = EpsilonNeighborhoodCoupler.class) int plannedNeighborhoodNumber,
054                        @Constant(value = "epsilonNeighborhood", namespace = EpsilonNeighborhoodCoupler.class) double epsilonNeighborhood,
055                        @Constant(value = "epsilonNeighborhoodDelta", namespace = EpsilonNeighborhoodCoupler.class) double epsilonNeighborhoodDelta,
056                        @Constant(value = "epsilonNeighborhoodDeltaMax", namespace = EpsilonNeighborhoodCoupler.class) double epsilonNeighborhoodDeltaMax,
057                        @Constant(value = "epsilonNeighborhoodDeltaMin", namespace = EpsilonNeighborhoodCoupler.class) double epsilonNeighborhoodDeltaMin) {
058                this.epsilonMapping = epsilonMapping;
059                this.epsilonAdaption = epsilonAdaption;
060                this.random = random;
061                this.plannedNeighborhoodNumber = plannedNeighborhoodNumber;
062                this.adaptiveEpsilonNeighborhood = new AdaptiveEpsilon(epsilonNeighborhood, epsilonNeighborhoodDelta,
063                                epsilonNeighborhoodDeltaMax, epsilonNeighborhoodDeltaMin);
064        }
065
066        /**
067         * Generates parent couples. Distributes the parent {@link Individual}s onto
068         * neighborhoods. Both parents of a couple are picked from the same
069         * neighborhood. Uses a {@link NeighborhoodSchedulerRoundRobin} to arbitrate
070         * the neighborhoods from where the parent couples are picked.
071         * 
072         * @param size
073         *            the number of couples that is generated
074         * @param survivors
075         *            the {@link Individual}s that can be used as parents
076         * @return a collection of {@link Individual} pairs, that will be used to
077         *         generate the next generation of individuals
078         * 
079         */
080        @Override
081        public Collection<Pair<Individual>> getCouples(int size, List<Individual> survivors) {
082                Collection<Pair<Individual>> result = new HashSet<>();
083                List<Set<Individual>> neighborhoods = createNeighborhoods(survivors);
084                NeighborhoodSchedulerRoundRobin scheduler = new NeighborhoodSchedulerRoundRobin(neighborhoods);
085                for (int i = 0; i < size; i++) {
086                        result.add(pickCouple(scheduler.next()));
087                }
088                return result;
089        }
090
091        /**
092         * Picks a couple of parents from the given neighborhood. Here, we just pick
093         * two random individuals.
094         * 
095         * @param neighborhood
096         *            the set of similar {@link Individual}s from where the parents
097         *            are picked
098         * @return the pair that was picked as parents for a crossover
099         */
100        protected Pair<Individual> pickCouple(Set<Individual> neighborhood) {
101                if (neighborhood.size() == 1) {
102                        Individual hermit = neighborhood.iterator().next();
103                        return new Pair<>(hermit, hermit);
104                }
105                List<Individual> individualList = new ArrayList<>(neighborhood);
106                Individual first = individualList.remove(random.nextInt(individualList.size()));
107                Individual second = individualList.remove(random.nextInt(individualList.size()));
108                return new Pair<>(first, second);
109        }
110
111        /**
112         * Applies the epsilon neighborhood creation.
113         * 
114         * @param survivorPool
115         *            a list of {@link Individual}s that can be used as parents
116         * @return a list of individual sets. Each set is considered as a
117         *         neighborhood.
118         */
119        protected List<Set<Individual>> createNeighborhoods(List<Individual> survivorPool) {
120                List<Set<Individual>> neighborhoods = new ArrayList<>();
121                Map<Objective, Double> objectiveAmplitudes = epsilonMapping
122                                .findObjectiveAmplitudes(new HashSet<>(survivorPool));
123                List<Individual> survivors = new ArrayList<>(survivorPool);
124                while (!survivors.isEmpty()) {
125                        // pick a random individual
126                        int idx = random.nextInt(survivors.size());
127                        Individual reference = survivors.remove(idx);
128                        Set<Individual> neighborhood = new HashSet<>();
129                        Objectives epsilonEnhancedObjectives = epsilonMapping.mapObjectives(reference.getObjectives(),
130                                        adaptiveEpsilonNeighborhood.getEpsilon(), objectiveAmplitudes);
131                        // put the individuals epsilon-dominated by the reference into its
132                        // neighborhood
133                        for (Individual candidate : survivors) {
134                                if (epsilonEnhancedObjectives.dominates(candidate.getObjectives())) {
135                                        neighborhood.add(candidate);
136                                }
137                        }
138                        survivors.removeAll(neighborhood);
139                        neighborhood.add(reference);
140                        neighborhoods.add(neighborhood);
141                }
142                // adapt the epsilon
143                boolean epsilonTooBig = neighborhoods.size() < plannedNeighborhoodNumber;
144                epsilonAdaption.adaptEpsilon(adaptiveEpsilonNeighborhood, epsilonTooBig);
145                return neighborhoods;
146        }
147}
