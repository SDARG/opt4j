001package org.opt4j.optimizers.ea.aeseh;
002
003import com.google.inject.ImplementedBy;
004
005/**
006 * The {@link EpsilonAdaptation} manages the adaptation of the ε-value stored in
007 * the {@link AdaptiveEpsilon}.
008 * 
009 * @author Fedor Smirnov
010 *
011 */
012@ImplementedBy(EpsilonAdaptationDelta.class)
013public interface EpsilonAdaptation {
014
015        /**
016         * Adjusts the ε-value according to the given {@link AdaptiveEpsilon}.
017         * 
018         * @param adaptiveEpsilon
019         *            the {@link AdaptiveEpsilon} contains the ε-value and the
020         *            information about its adaptation
021         * @param epsilonTooBig
022         *            {@code true} if the current epsilon is too big
023         */
024        public void adaptEpsilon(AdaptiveEpsilon adaptiveEpsilon, boolean epsilonTooBig);
025}
