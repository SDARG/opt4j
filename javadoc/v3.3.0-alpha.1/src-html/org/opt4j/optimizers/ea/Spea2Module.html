018package org.opt4j.optimizers.ea;
019
020import static org.opt4j.core.config.annotations.Citation.PublicationMonth.UNKNOWN;
021
022import org.opt4j.core.config.annotations.Citation;
023import org.opt4j.core.config.annotations.Info;
024import org.opt4j.core.start.Constant;
025
026/**
027 * The {@link Spea2Module} configures the {@link Spea2} selector.
028 * 
029 * @author lukasiewycz
030 * 
031 */
032@Info("The Strength Pareto Evolutionary Algorithm (SPEA II).")
033@Citation(title = "SPEA2: Improving the Strength Pareto Evolutionary Algorithm For Multiobjective Optimization", authors = "Eckart Zitzler, Marco Laumanns, and Lothar Thiele", journal = "Evolutionary Methods for Design, Optimisation, and Control", pageFirst = 19, pageLast = 26, year = 2002, month = UNKNOWN)
034public class Spea2Module extends SelectorModule {
035
036        @Info("The tournament value (number of opponents for the mating selection).")
037        @Constant(value = "tournament", namespace = Spea2.class)
038        protected int tournament = 0;
039
040        /**
041         * Returns the number of tournaments.
042         * 
043         * @see #setTournament
044         * @return the number of tournaments
045         */
046        public int getTournament() {
047                return tournament;
048        }
049
050        /**
051         * Sets the number of tournaments.
052         * 
053         * @see #getTournament
054         * @param tournament
055         *            the number of tournaments
056         */
057        public void setTournament(int tournament) {
058                this.tournament = tournament;
059        }
060
061        /*
062         * (non-Javadoc)
063         * 
064         * @see org.opt4j.start.Opt4JModule#config()
065         */
066        @Override
067        public void config() {
068                bindSelector(Spea2.class);
069        }
070}
