024package org.opt4j.optimizers.mopso;
025
026import org.opt4j.core.common.random.Rand;
027import org.opt4j.core.genotype.DoubleGenotype;
028import org.opt4j.core.optimizer.Iteration;
029import org.opt4j.core.start.Constant;
030import org.opt4j.operators.normalize.NormalizeDouble;
031
032import com.google.inject.Inject;
033
034/**
035 * The {@link MutateDoubleNonUniform} non-uniformly mutates a
036 * {@link DoubleGenotype} elementwise.
037 * 
038 * @author lukasiewycz
039 * 
040 */
041public class MutateDoubleNonUniform extends MutateDoubleUniform {
042
043        protected final Iteration iteration;
044
045        /**
046         * Constructs a {@link MutateDoubleNonUniform}.
047         * 
048         * @param random
049         *            the random number generator
050         * @param normalize
051         *            the normalize operator
052         * @param iteration
053         *            the iteration counter
054         * @param perturbation
055         *            the perturbation index
056         */
057        @Inject
058        public MutateDoubleNonUniform(Rand random, NormalizeDouble normalize, Iteration iteration,
059                        @Constant(value = "perturbation", namespace = MutateDoubleNonUniform.class) double perturbation) {
060                super(random, normalize, perturbation);
061                this.iteration = iteration;
062        }
063
064        /*
065         * (non-Javadoc)
066         * 
067         * @see
068         * org.opt4j.operator.mutate.MutateDoubleElementwise#mutateElement(double,
069         * double, double, double)
070         */
071        @Override
072        protected double mutateElement(double x, double lb, double ub, double p) {
073                double element = x;
074                if (random.nextDouble() < p) {
075                        double tau = random.nextBoolean() ? 1 : -1;
076                        element += tau * delta(perturbation);
077                }
078                return element;
079        }
080
081        /**
082         * Calculates the delta value for the non-uniform mutation.
083         * 
084         * @param perturbation
085         *            the perturbation value
086         * @return the delta value
087         */
088        protected double delta(double perturbation) {
089                final double diff = 1.0 - 0.0;
090                double r = random.nextDouble();
091
092                double actIter = iteration.value();
093                double maxIter = iteration.max();
094                double ratio = actIter / maxIter;
095                ratio = Math.min(ratio, 1.0);
096
097                double y = Math.pow(1.0 - ratio, perturbation);
098                return diff * (1.0 - Math.pow(r, y));
099        }
100
101}
