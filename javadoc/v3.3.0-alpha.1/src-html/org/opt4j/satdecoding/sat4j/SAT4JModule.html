024package org.opt4j.satdecoding.sat4j;
025
026import org.opt4j.core.config.Icons;
027import org.opt4j.core.config.annotations.Icon;
028import org.opt4j.core.config.annotations.Info;
029import org.opt4j.core.config.annotations.Order;
030import org.opt4j.core.config.annotations.Required;
031import org.opt4j.core.start.Constant;
032import org.opt4j.satdecoding.SATModule;
033import org.opt4j.satdecoding.sat4j.SAT4JSolver.Learning;
034import org.opt4j.satdecoding.sat4j.SAT4JSolver.Restarts;
035
036/**
037 * The {@link SAT4JModule} enables and configures the {@link SAT4JSolver}.
038 * 
039 * @author lukasiewycz
040 * 
041 */
042@Icon(Icons.PUZZLE_GREEN)
043@Info("The SAT/PB-Solver configuration for <a href=\"http://www.sat4j.org\">Sat4J.org</a>.")
044public class SAT4JModule extends SATModule {
045
046        @Info("Timeout in seconds.")
047        @Order(0)
048        @Constant(value = "timeout", namespace = SAT4JSolver.class)
049        protected int timeout = 3600;
050
051        @Info("Learned clauses up to this size are kept.")
052        @Required(property = "learning", elements = { "FIXEDLENGTH" })
053        @Order(2)
054        @Constant(value = "clauseLearningLength", namespace = SAT4JSolver.class)
055        protected int learnSize = 10;
056
057        @Info("Learning strategy.")
058        @Order(1)
059        @Constant(value = "learning", namespace = SAT4JSolver.class)
060        protected Learning learning = Learning.FIXEDLENGTH;
061
062        @Info("Restart strategy.")
063        @Order(3)
064        @Constant(value = "restarts", namespace = SAT4JSolver.class)
065        protected Restarts restarts = Restarts.MINISAT;
066
067        @Info("The number of instances.")
068        @Order(4)
069        protected int instances = 1;
070
071        /**
072         * Returns the number of instances.
073         * 
074         * @see #setInstances
075         * @return the number of instances
076         */
077        public int getInstances() {
078                return instances;
079        }
080
081        /**
082         * Sets the number of instances.
083         * 
084         * @see #getInstances
085         * @param instances
086         *            the number of instances
087         */
088        public void setInstances(int instances) {
089                if (instances <= 0) {
090                        throw new IllegalArgumentException("The number of instances must be positive: " + instances);
091                }
092                this.instances = instances;
093        }
094
095        /**
096         * Returns the learning strategy.
097         * 
098         * @see #setLearning
099         * @return the learning strategy
100         */
101        public Learning getLearning() {
102                return learning;
103        }
104
105        /**
106         * Sets the learning strategy.
107         * 
108         * @see #getLearning
109         * @param learning
110         *            the learning strategy to set
111         */
112        public void setLearning(Learning learning) {
113                this.learning = learning;
114        }
115
116        /**
117         * Returns the restart strategy.
118         * 
119         * @see #setRestarts
120         * @return the restart strategy
121         */
122        public Restarts getRestarts() {
123                return restarts;
124        }
125
126        /**
127         * Sets the restart strategy.
128         * 
129         * @see #getRestarts
130         * @param restarts
131         *            the restart strategy to set
132         */
133        public void setRestarts(Restarts restarts) {
134                this.restarts = restarts;
135        }
136
137        /**
138         * Returns the number of learned clauses that are kept.
139         * 
140         * @see #setLearnSize
141         * @return the number of learned clauses that are kept
142         */
143        public int getLearnSize() {
144                return learnSize;
145        }
146
147        /**
148         * Sets the number of learned clauses that are kept.
149         * 
150         * @see #getLearnSize
151         * @param learnSize
152         *            the number of learned clauses
153         */
154        public void setLearnSize(int learnSize) {
155                this.learnSize = learnSize;
156        }
157
158        /**
159         * Returns the timeout (in seconds) of the {@link SAT4JSolver}.
160         * 
161         * @see #setTimeout
162         * @return the timeout (in seconds) of the solver
163         */
164        public int getTimeout() {
165                return timeout;
166        }
167
168        /**
169         * Sets the timeout (in seconds) for the {@link SAT4JSolver}.
170         * 
171         * @see #getTimeout
172         * @param timeout
173         *            the timeout for the solver
174         */
175        public void setTimeout(int timeout) {
176                this.timeout = timeout;
177        }
178
179        /*
180         * (non-Javadoc)
181         * 
182         * @see org.opt4j.start.Opt4JModule#config()
183         */
184        @Override
185        public void config() {
186                bindSolver(SAT4JSolver.class, instances);
187        }
188
189}
