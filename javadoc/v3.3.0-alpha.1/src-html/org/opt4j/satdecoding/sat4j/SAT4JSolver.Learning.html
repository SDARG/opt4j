067/**
068 * The {@link SAT4JSolver} implements a {@link Solver}. It is based on the Java
069 * SAT/PB-Solver from <a href="http://www.sat4j.org">Sat4J.org</a>.
070 * 
071 * @author lukasiewycz
072 * 
073 */
074@Singleton
075public class SAT4JSolver implements Solver {
076
077        public enum Learning {
078                @Info("Learning only constraints with less or equal number of variables.")
079                FIXEDLENGTH, @Info("Use the MiniSAT learning scheme.")
080                MINISAT, @Info("Learn only clauses.")
081                CLAUSEONLY;
082        }
083
084        public enum Restarts {
085                @Info("Use MiniSAT restarts.")
086                MINISAT, @Info("Use Luby restarts.")
087                LUBY, @Info("Use Rapid restarts.")
088                RAPID;
089        }
090
091        protected final Queue<Constraint> constraints = new ConcurrentLinkedQueue<>();
092
093        protected final PBSolverResolution solver;
094
095        protected final Map<Object, Integer> variables = new HashMap<>();
096
097        protected int nextVariable = 1;
098
099        private boolean solverValid = false;
100
101        /**
102         * Constructs a new {@link SAT4JSolver} with a timeout and the number of
103         * kept learning clauses. Additionally, this constructor allows the
104         * specification of the learning and restart strategy.
105         * 
106         * @param timeout
107         *            timeout in seconds
108         * @param clauseLearningLength
109         *            clauses are learned if they have a smaller or equal number of
110         *            literals per clause
111         * @param learning
112         *            the learning strategy
113         * @param restarts
114         *            the restart strategy
115         */
116        @Inject
117        public SAT4JSolver(@Constant(value = "timeout", namespace = SAT4JSolver.class) int timeout,
118                        @Constant(value = "clauseLearningLength", namespace = SAT4JSolver.class) int clauseLearningLength,
119                        @Constant(value = "learning", namespace = SAT4JSolver.class) Learning learning,
120                        @Constant(value = "restarts", namespace = SAT4JSolver.class) Restarts restarts) {
121                LearningStrategy<PBDataStructureFactory> l = null;
122                switch (learning) {
123                case FIXEDLENGTH:
124                        if (clauseLearningLength < 0) {
125                                throw new IllegalArgumentException("learning length must not be less than 0.");
126                        }
127                        l = new FixedLengthLearning<>(clauseLearningLength);
128                        break;
129                case MINISAT:
130                        l = new MiniSATLearning<>();
131                        break;
132                case CLAUSEONLY:
133                        l = new ClauseOnlyLearning<>();
134                        break;
135                default:
136                        throw new IllegalArgumentException("Learning strategy not supported: " + learning);
137                }
138
139                RestartStrategy r = null;
140                switch (restarts) {
141                case MINISAT:
142                        r = new MiniSATRestarts();
143                        break;
144                case LUBY:
145                        r = new LubyRestarts();
146                        break;
147                case RAPID:
148                        r = new ArminRestarts();
149                        break;
150                default:
151                        throw new IllegalArgumentException("Restart strategy not supported: " + restarts);
152                }
153                solver = new PBSolverResolution(l, new CompetResolutionPBMixedHTClauseCardConstrDataStructure(),
154                                new VarOrderHeap(new PositiveLiteralSelectionStrategy()), r);
155                l.setSolver(solver);
156                l.setVarActivityListener(solver);
157                if (l instanceof MiniSATLearning) {
158                        ((MiniSATLearning<PBDataStructureFactory>) l).setDataStructureFactory(solver.getDSFactory());
159                }
160
161                if (timeout <= 0) {
162                        throw new IllegalArgumentException("Invalid timeout: " + timeout);
163                }
164                solver.setTimeout(timeout);
165                solver.setVerbose(false);
166
167                setNVars(100);
168                solverValid = true;
169        }
170
171        /*
172         * (non-Javadoc)
173         * 
174         * @see org.opt4j.sat.Solver#addConstraint(org.opt4j.sat.Constraint)
175         */
176        @Override
177        public void addConstraint(Constraint constraint) {
178                constraints.add(constraint);
179        }
180
181        /*
182         * (non-Javadoc)
183         * 
184         * @see org.opt4j.sat.Solver#solve(org.opt4j.sat.Order)
185         */
186        @Override
187        public synchronized Model solve(Order order) throws TimeoutException, ContradictionException {
188                if (!solverValid) {
189                        throw new ContradictionException();
190                }
191
192                Constraint c = null;
193                while ((c = constraints.poll()) != null) {
194                        addConstraintToSolver(c);
195                }
196
197                if (order instanceof VarOrder) {
198
199                        VarOrder varOrder = (VarOrder) order;
200                        VariableOrder o = new VariableOrder();
201                        solver.setOrder(o);
202
203                        for (Entry<Object, Double> entry : varOrder.getActivityEntrySet()) {
204                                if (variables.containsKey(entry.getKey())) {
205                                        int var = variables.get(entry.getKey());
206                                        o.setVarActivity(var, entry.getValue());
207                                }
208                        }
209
210                        for (Entry<Object, Boolean> entry : varOrder.getPhaseEntrySet()) {
211                                if (variables.containsKey(entry.getKey())) {
212                                        int var = variables.get(entry.getKey());
213                                        o.setVarPhase(var, entry.getValue());
214                                }
215                        }
216
217                        o.setVarInc(varOrder.getVarInc());
218                        o.setVarDecay(varOrder.getVarDecay());
219
220                }
221
222                try {
223                        boolean success = solver.isSatisfiable();
224
225                        if (success) {
226                                Model model = new Model();
227
228                                for (Entry<Object, Integer> entry : variables.entrySet()) {
229                                        model.set(entry.getKey(), solver.model(entry.getValue()));
230                                }
231                                return model;
232                        } else {
233                                throw new ContradictionException();
234                        }
235                } catch (org.sat4j.specs.TimeoutException e) {
236                        throw new TimeoutException();
237                }
238        }
239
240        protected void addConstraintToSolver(Constraint constraint) {
241
242                VecInt lits = toVecInt(constraint.getLiterals());
243
244                IVec<BigInteger> coeffs = new Vec<>();
245
246                for (Integer value : constraint.getCoefficients()) {
247                        coeffs.push(BigInteger.valueOf(value));
248                }
249
250                BigInteger d = BigInteger.valueOf(constraint.getRhs());
251                Operator operator = constraint.getOperator();
252
253                try {
254                        if (operator == LE || operator == EQ) {
255                                solver.addPseudoBoolean(lits, coeffs, false, d);
256                        }
257                        if (operator == GE || operator == EQ) {
258                                solver.addPseudoBoolean(lits, coeffs, true, d);
259                        }
260                } catch (org.sat4j.specs.ContradictionException e) {
261                        solverValid = false;
262                        throw new ContradictionException(e);
263                }
264        }
265
266        protected VecInt toVecInt(Iterable<Literal> list) {
267                VecInt vector = new VecInt();
268
269                for (Literal literal : list) {
270                        Object var = literal.variable();
271                        if (!variables.containsKey(var)) {
272                                variables.put(var, nextVariable++);
273                                if (variables.size() > solver.nVars()) {
274                                        setNVars(variables.size() * 2);
275                                }
276                        }
277                        boolean phase = literal.phase();
278
279                        vector.push(variables.get(var) * (phase ? 1 : -1));
280                }
281
282                return vector;
283        }
284
285        /**
286         * Sets the instance to n vars.
287         * 
288         * @param n
289         *            the number of vars.
290         */
291        protected void setNVars(int n) {
292                solver.newVar(n);
293        }
