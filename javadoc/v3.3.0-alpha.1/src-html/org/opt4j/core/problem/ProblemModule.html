023package org.opt4j.core.problem;
024
025import java.util.HashSet;
026import java.util.Set;
027
028import org.opt4j.core.Genotype;
029import org.opt4j.core.config.Icons;
030import org.opt4j.core.config.annotations.Category;
031import org.opt4j.core.config.annotations.Icon;
032import org.opt4j.core.start.Opt4JModule;
033
034import com.google.inject.Binder;
035import com.google.inject.TypeLiteral;
036import com.google.inject.multibindings.Multibinder;
037
038/**
039 * The {@link ProblemModule} is an abstract module class for the binding of the
040 * {@link Creator}, {@link Decoder}, and {@link Evaluator}.
041 * 
042 * @see Creator
043 * @see Decoder
044 * @see Evaluator
045 * @author lukasiewycz
046 * 
047 */
048@Icon(Icons.PROBLEM)
049@Category
050public abstract class ProblemModule extends Opt4JModule {
051
052        /**
053         * <p>
054         * Binds a problem. A value {@code null} is allowed. In this case, the
055         * corresponding interface is not bound. Therefore, the binding for the
056         * omitted interfaces has to be done in other modules.
057         * </p>
058         * <p>
059         * Additional {@link Evaluator}s can be bound using
060         * {@link #addEvaluator(Class)}.
061         * </p>
062         * 
063         * 
064         * @param creator
065         *            the creator
066         * @param decoder
067         *            the decoder
068         * @param evaluator
069         *            the evaluator
070         */
071        protected void bindProblem(final Class<? extends Creator<? extends Genotype>> creator,
072                        final Class<? extends Decoder<? extends Genotype, ? extends Object>> decoder,
073                        final Class<? extends Evaluator<? extends Object>> evaluator) {
074                bindProblem(binder(), creator, decoder, evaluator);
075        }
076
077        /**
078         * Binds an additional {@link Evaluator}.
079         * 
080         * @param evaluator
081         *            the evaluator to use
082         */
083        protected void addEvaluator(final Class<? extends Evaluator<? extends Object>> evaluator) {
084                addEvaluator(binder(), evaluator);
085        }
086
087        /**
088         * Binds an additional {@link Evaluator} to a given {@link Binder}.
089         * 
090         * @param binder
091         *            the guice binder to use
092         * @param evaluator
093         *            the evaluator to use
094         * @see #binder()
095         */
096        @SuppressWarnings({ "unchecked" })
097        public static void addEvaluator(Binder binder, final Class<? extends Evaluator<? extends Object>> evaluator) {
098                Multibinder<Evaluator<Object>> multibinder = Multibinder.newSetBinder(binder,
099                                new TypeLiteral<Evaluator<Object>>() {
100                                });
101                multibinder.addBinding().to((Class<Evaluator<Object>>) evaluator);
102        }
103
104        /**
105         * <p>
106         * Binds a problem. A value {@code null} is allowed. In this case, the
107         * corresponding interface is not bound. Therefore, the binding for the
108         * omitted interfaces has to be done in other modules.
109         * </p>
110         * <p>
111         * Additional {@link Evaluator}s can be bound using
112         * {@link #addEvaluator(Class)}.
113         * </p>
114         * 
115         * @param binder
116         *            the guice binder
117         * @param creator
118         *            the creator
119         * @param decoder
120         *            the decoder
121         * @param evaluator
122         *            the evaluator
123         * @see #binder()
124         */
125        @SuppressWarnings("unchecked")
126        public static void bindProblem(Binder binder, final Class<? extends Creator<? extends Genotype>> creator,
127                        final Class<? extends Decoder<? extends Genotype, ? extends Object>> decoder,
128                        final Class<? extends Evaluator<? extends Object>> evaluator) {
129
130                Set<Class<?>> classes = new HashSet<>();
131                if (creator != null) {
132                        classes.add(creator);
133                }
134                if (decoder != null) {
135                        classes.add(decoder);
136                }
137
138                for (Class<?> clazz : classes) {
139                        binder.bind(clazz).in(SINGLETON);
140                }
141
142                if (creator != null) {
143                        binder.bind(new TypeLiteral<Creator<Genotype>>() {
144                        }).to((Class<? extends Creator<Genotype>>) creator);
145                }
146                if (decoder != null) {
147                        binder.bind(new TypeLiteral<Decoder<Genotype, Object>>() {
148                        }).to((Class<? extends Decoder<Genotype, Object>>) decoder);
149                }
150                if (evaluator != null) {
151                        addEvaluator(binder, evaluator);
152                }
153        }
154}
