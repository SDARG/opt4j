023package org.opt4j.core.common.completer;
024
025import java.util.Arrays;
026import java.util.Collection;
027import java.util.HashSet;
028import java.util.List;
029import java.util.Set;
030
031import org.opt4j.core.Genotype;
032import org.opt4j.core.Individual;
033import org.opt4j.core.Individual.State;
034import org.opt4j.core.Objective;
035import org.opt4j.core.Objectives;
036import org.opt4j.core.optimizer.Control;
037import org.opt4j.core.optimizer.IndividualCompleter;
038import org.opt4j.core.optimizer.TerminationException;
039import org.opt4j.core.problem.Decoder;
040import org.opt4j.core.problem.Evaluator;
041
042import com.google.inject.Inject;
043
044/**
045 * <p>
046 * The {@link SequentialIndividualCompleter} completes the {@link Individual}s
047 * sequentially.
048 * </p>
049 * <p>
050 * It updates the {@link State} of the {@link Individual} according to the state
051 * of the completion process. It uses {@link Control} between the different
052 * (possibly time consuming) completion steps to allow the user to control the
053 * completion process.
054 * </p>
055 * 
056 * @author lukasiewycz
057 * 
058 */
059public class SequentialIndividualCompleter implements IndividualCompleter {
060
061        protected final Decoder<Genotype, Object> decoder;
062        protected final Evaluator<Object> evaluator;
063        protected final Control control;
064
065        /**
066         * Constructs a {@link SequentialIndividualCompleter}.
067         * 
068         * @param control
069         *            the optimization control
070         * @param decoder
071         *            the decoder
072         * @param evaluator
073         *            the evaluator
074         */
075        @Inject
076        public SequentialIndividualCompleter(Control control, Decoder<Genotype, Object> decoder,
077                        Evaluator<Object> evaluator) {
078                super();
079                this.control = control;
080                this.decoder = decoder;
081                this.evaluator = evaluator;
082        }
083
084        /*
085         * (non-Javadoc)
086         * 
087         * @see org.opt4j.core.optimizer.Completer#complete(java.lang.Iterable)
088         */
089        @Override
090        public void complete(Iterable<? extends Individual> iterable) throws TerminationException {
091                for (Individual individual : iterable) {
092                        if (!individual.isEvaluated()) {
093                                control.checkpoint();
094                                decode(individual);
095                                control.checkpoint();
096                                evaluate(individual);
097                                control.checkpoint();
098                        }
099                }
100        }
101
102        /*
103         * (non-Javadoc)
104         * 
105         * @see
106         * org.opt4j.core.optimizer.Completer#complete(org.opt4j.core.Individual[])
107         */
108        @Override
109        public void complete(Individual... individuals) throws TerminationException {
110                List<Individual> list = Arrays.asList(individuals);
111                complete(list);
112        }
113
114        /**
115         * Evaluates the phenotype of the {@link Individual}. After this operation,
116         * the {@link Individual} is in {@link State} {@link State#EVALUATED} and
117         * {@link Individual#getObjectives()} returns the {@link Objectives}.
118         * 
119         * @param individual
120         */
121        protected void evaluate(Individual individual) {
122                State state = individual.getState();
123
124                if (state == State.PHENOTYPED) {
125                        individual.setState(State.EVALUATING);
126                        Object phenotype = individual.getPhenotype();
127
128                        Objectives objectives = evaluator.evaluate(phenotype);
129                        assert isSameLength(objectives.getKeys()) : "Objectives changed: " + objectives.getKeys();
130
131                        individual.setObjectives(objectives);
132                } else {
133                        throw new IllegalStateException("Cannot evaluate Individual, current state: " + state);
134                }
135        }
136
137        /**
138         * Decodes the {@link Genotype} of the {@link Individual}. After this
139         * operation, the {@link Individual} is in {@link State}
140         * {@link State#PHENOTYPED} and {@link Individual#getPhenotype()} returns
141         * the phenotype.
142         * 
143         * @param individual
144         */
145        protected void decode(Individual individual) {
146                State state = individual.getState();
147
148                if (state == State.GENOTYPED) {
149                        individual.setState(State.DECODING);
150                        Genotype genotype = individual.getGenotype();
151                        Object phenotype = decoder.decode(genotype);
152                        individual.setPhenotype(phenotype);
153                } else {
154                        throw new IllegalStateException("Cannot decode Individual, current state: " + state);
155                }
156        }
157
158        private Set<Objective> objectives = null;
159
160        /**
161         * Check if the given {@link Objectives} have the same length as prior
162         * evaluated ones.
163         * 
164         * @param objectives
165         *            the objectives to check
166         * @return true if the number of objectives is constant
167         */
168        private boolean isSameLength(Collection<Objective> objectives) {
169                Set<Objective> set = new HashSet<>(objectives);
170                if (this.objectives == null) {
171                        this.objectives = set;
172                        return true;
173                }
174                return this.objectives.equals(set);
175        }
176}
