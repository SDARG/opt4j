024package org.opt4j.core.common.logger;
025
026import java.util.Collection;
027import java.util.concurrent.atomic.AtomicInteger;
028
029import org.opt4j.core.Individual;
030import org.opt4j.core.Individual.State;
031import org.opt4j.core.IndividualStateListener;
032import org.opt4j.core.Objective;
033import org.opt4j.core.optimizer.Optimizer;
034import org.opt4j.core.optimizer.OptimizerIterationListener;
035import org.opt4j.core.optimizer.OptimizerStateListener;
036import org.opt4j.core.start.Opt4JModule;
037
038/**
039 * The {@link AbstractLogger} triggers its refinement on specific events
040 * dependent on the iteration and evaluation count. Each {@link AbstractLogger}
041 * has to listen to the be registered as listener of the {@link Optimizer} (
042 * {@link Opt4JModule#addOptimizerStateListener(Class)} and
043 * {@link Opt4JModule#addOptimizerIterationListener(Class)}) and the state of
044 * the {@link Individual} ({@link Opt4JModule#addIndividualStateListener(Class)}
045 * .
046 * 
047 * @author lukasiewycz
048 * 
049 */
050public abstract class AbstractLogger implements OptimizerStateListener, OptimizerIterationListener,
051                IndividualStateListener {
052
053        protected final int iterationStep;
054        protected final int evaluationStep;
055        protected final AtomicInteger evaluationCount = new AtomicInteger(0);
056
057        protected boolean isFirst = true;
058        protected int evaluationCountLast = 0;
059
060        /**
061         * Constructs an {@link AbstractLogger}.
062         * 
063         * @param iterationStep
064         *            the number of iterations between two logging events
065         * @param evaluationStep
066         *            the number of evaluations between two logging events
067         */
068        public AbstractLogger(int iterationStep, int evaluationStep) {
069                super();
070                this.iterationStep = iterationStep;
071                this.evaluationStep = evaluationStep;
072        }
073
074        /**
075         * Callback method called if the specific number of iterations or
076         * evaluations is reached.
077         * 
078         * @param iteration
079         *            the current iteration number
080         * @param evaluation
081         *            the current evaluation number
082         */
083        public abstract void logEvent(int iteration, int evaluation);
084
085        /**
086         * Writes the header.
087         * 
088         * @param objectives
089         *            a collection of objectives
090         */
091        public abstract void logHeader(Collection<Objective> objectives);
092
093        /**
094         * Callback method invoked once the optimization starts.
095         * 
096         */
097        public abstract void optimizationStarted();
098
099        /**
100         * Callback method invoked once the optimization stops.
101         * 
102         */
103        public abstract void optimizationStopped();
104
105        /*
106         * (non-Javadoc)
107         * 
108         * @see
109         * org.opt4j.core.optimizer.OptimizerIterationListener#iterationComplete
110         * (org.opt4j.core.optimizer.Optimizer, int)
111         */
112        @Override
113        public void iterationComplete(int iteration) {
114
115                boolean logIteration = false;
116                if (iterationStep > 0) {
117                        logIteration = iteration % iterationStep == 0;
118                }
119
120                boolean logEvaluation = false;
121                if (evaluationStep > 0) {
122                        logEvaluation = ((evaluationCount.intValue() - evaluationCountLast) >= evaluationStep);
123                }
124
125                if (logEvaluation) {
126                        do {
127                                evaluationCountLast += evaluationStep;
128                        } while ((evaluationCount.intValue() - evaluationCountLast) >= evaluationStep);
129                }
130
131                if (logIteration || logEvaluation) {
132                        logEvent(iteration, evaluationCount.intValue());
133                }
134        }
135
136        /*
137         * (non-Javadoc)
138         * 
139         * @see
140         * org.opt4j.core.optimizer.OptimizerStateListener#optimizationStarted(org
141         * .opt4j.core.optimizer.Optimizer)
142         */
143        @Override
144        public void optimizationStarted(Optimizer optimizer) {
145                optimizationStarted();
146        }
147
148        /*
149         * (non-Javadoc)
150         * 
151         * @see
152         * org.opt4j.core.optimizer.OptimizerStateListener#optimizationStopped(org
153         * .opt4j.core.optimizer.Optimizer)
154         */
155        @Override
156        public void optimizationStopped(Optimizer optimizer) {
157                optimizationStopped();
158        }
159
160        /*
161         * (non-Javadoc)
162         * 
163         * @see
164         * org.opt4j.core.IndividualStateListener#inidividualStateChanged(org.opt4j
165         * .core.Individual)
166         */
167        @Override
168        public void inidividualStateChanged(Individual individual) {
169                if (individual.getState() == State.EVALUATED) {
170                        if (isFirst) {
171                                synchronized (this) {
172                                        if (isFirst) {
173                                                logHeader(individual.getObjectives().getKeys());
174                                                isFirst = false;
175                                        }
176                                }
177                        }
178                        evaluationCount.incrementAndGet();
179                }
180        }
181
182}
