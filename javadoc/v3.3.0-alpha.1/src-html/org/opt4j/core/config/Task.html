023package org.opt4j.core.config;
024
025import java.util.ArrayList;
026import java.util.Arrays;
027import java.util.Collection;
028import java.util.List;
029import java.util.Set;
030import java.util.concurrent.Callable;
031import java.util.concurrent.CopyOnWriteArraySet;
032
033import com.google.inject.Module;
034
035/**
036 * The {@link Task} is an abstract class for optimization tasks.
037 * 
038 * @author lukasiewycz
039 * @see ExecutionEnvironment
040 */
041public abstract class Task implements Callable<Void> {
042
043        protected State state = State.WAITING;
044
045        protected Exception e = null;
046
047        protected final List<Module> modules = new ArrayList<>();
048
049        protected boolean isInit = false;
050
051        protected final Set<TaskStateListener> listeners = new CopyOnWriteArraySet<>();
052
053        /**
054         * The {@link State} of a task.
055         * 
056         * @author lukasiewycz
057         * 
058         */
059        public enum State {
060                /**
061                 * The task is waiting for its execution.
062                 */
063                WAITING,
064                /**
065                 * The task is currently executed.
066                 */
067                EXECUTING,
068                /**
069                 * The task is done.
070                 */
071                DONE;
072        }
073
074        /**
075         * Initializes the {@link Task} with modules.
076         * 
077         * @see Task#init(Collection)
078         * @param modules
079         *            the modules for the task
080         */
081        public void init(Module... modules) {
082                init(Arrays.asList(modules));
083        }
084
085        /**
086         * Initializes the {@link Task} with modules. Note that if a module is a
087         * {@link PropertyModule}, a copy is made with the
088         * {@link PropertyModule#clone()} method. If the {@link Module} is not a
089         * {@link PropertyModule}, the module is copied as a reference, i.e., it can
090         * be changed until the {@link Task#execute()} is called.
091         * 
092         * @param modules
093         *            the modules for the task
094         */
095        public void init(Collection<Module> modules) {
096                for (Module module : modules) {
097                        if (module instanceof PropertyModule) {
098                                PropertyModule pModule = (PropertyModule) module;
099                                this.modules.add(pModule.clone());
100                        } else {
101                                this.modules.add(module);
102                        }
103                }
104                isInit = true;
105        }
106
107        /**
108         * Executes the {@link Task}.
109         * 
110         * @throws Exception
111         */
112        public abstract void execute() throws Exception;
113
114        /*
115         * (non-Javadoc)
116         * 
117         * @see java.util.concurrent.Callable#call()
118         */
119        @Override
120        public Void call() throws Exception {
121                setState(State.EXECUTING);
122                try {
123                        execute();
124                } catch (Error e) {
125                        this.e = new RuntimeException(e);
126                        e.printStackTrace();
127                        throw e;
128                } catch (Exception e) {
129                        this.e = e;
130                        e.printStackTrace();
131                        throw e;
132                } finally {
133                        setState(State.DONE);
134                }
135
136                return null;
137        }
138
139        /**
140         * Returns the {@link Exception} that was thrown during the {@link Task}
141         * execution if an error occurred.
142         * 
143         * @return the occurred exception
144         */
145        public Exception getException() {
146                return e;
147        }
148
149        /**
150         * Returns the state of the {@link Task}.
151         * 
152         * @return the state of the task
153         */
154        public State getState() {
155                return state;
156        }
157
158        /**
159         * Sets the state of the {@link Task}.
160         * 
161         * @param state
162         *            the state of the task to be set
163         */
164        protected void setState(State state) {
165                this.state = state;
166                for (TaskStateListener listener : listeners) {
167                        listener.stateChanged(this);
168                }
169        }
170
171        /*
172         * (non-Javadoc)
173         * 
174         * @see java.lang.Object#toString()
175         */
176        @Override
177        public String toString() {
178                return "";
179        }
180
181        /**
182         * Adds a {@link TaskStateListener}.
183         * 
184         * @see #removeStateListener
185         * @param listener
186         *            the listener to be added
187         */
188        public void addStateListener(TaskStateListener listener) {
189                listeners.add(listener);
190        }
191
192        /**
193         * Removes a {@link TaskStateListener}.
194         * 
195         * @see #addStateListener
196         * @param listener
197         *            the listener to be removed
198         */
199        public void removeStateListener(TaskStateListener listener) {
200                listeners.remove(listener);
201        }
202
203        /**
204         * Returns a copy of the {@link Module}s of this {@link Task}.
205         * 
206         * @return a set of modules
207         */
208        public Collection<Module> getModules() {
209                List<Module> clones = new ArrayList<>();
210                for (Module module : modules) {
211                        PropertyModule pModule;
212                        if (module instanceof PropertyModule) {
213                                pModule = (PropertyModule) module;
214                        } else {
215                                pModule = new PropertyModule(module);
216                        }
217
218                        clones.add(pModule.clone());
219                }
220                return clones;
221        }
222}
