023package org.opt4j.core.optimizer;
024
025/**
026 * The {@link IncompatibilityException}. This exception is thrown if a problem,
027 * optimizer, operator, etc. are not compatible.
028 * 
029 * @author lukasiewycz
030 * 
031 */
032@SuppressWarnings("serial")
033public class IncompatibilityException extends RuntimeException {
034
035        /**
036         * Constructs a {@link IncompatibilityException}.
037         * 
038         */
039        public IncompatibilityException() {
040                super();
041        }
042
043        /**
044         * Constructs a {@link IncompatibilityException}.
045         * 
046         * @param message
047         *            the message
048         * @param cause
049         *            the cause
050         */
051        public IncompatibilityException(String message, Throwable cause) {
052                super(message, cause);
053        }
054
055        /**
056         * Constructs a {@link IncompatibilityException}.
057         * 
058         * @param message
059         *            the message
060         */
061        public IncompatibilityException(String message) {
062                super(message);
063        }
064
065        /**
066         * Constructs a {@link IncompatibilityException}.
067         * 
068         * @param cause
069         *            the cause
070         */
071        public IncompatibilityException(Throwable cause) {
072                super(cause);
073        }
074
075}
