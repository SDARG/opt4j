023package org.opt4j.core.optimizer;
024
025import java.util.Set;
026import java.util.concurrent.CopyOnWriteArraySet;
027
028import org.opt4j.core.IndividualFactory;
029
030import com.google.inject.Inject;
031
032/**
033 * The {@link AbstractOptimizer} is an abstract implementation of a
034 * population-based {@link Optimizer}. It provides the necessary coupling to the
035 * framework by informing all {@link OptimizerStateListener}s and
036 * {@link OptimizerIterationListener}s, calling the {@link IndividualCompleter}
037 * to decode and evaluate new {@link org.opt4j.core.Individual}s and adding them
038 * to the {@link Archive}. The {@link Control} is checked regularly to allow
039 * user interaction.
040 * 
041 * To implement an iteration-based {@link Optimizer}, the
042 * {@link IterativeOptimizer} interface is available.
043 * 
044 * @author glass, lukasiewycz
045 * 
046 */
047public abstract class AbstractOptimizer implements Optimizer {
048
049        protected final Iteration iteration;
050
051        protected final Population population;
052
053        protected final Archive archive;
054
055        protected final IndividualCompleter completer;
056
057        protected final Control control;
058
059        protected final Set<OptimizerStateListener> stateListeners = new CopyOnWriteArraySet<>();
060
061        protected final Set<OptimizerIterationListener> iterationListeners = new CopyOnWriteArraySet<>();
062
063        protected boolean optimizing = false;
064
065        /**
066         * Constructs an {@link AbstractOptimizer} with a {@link Population}, an
067         * {@link Archive}, an {@link IndividualFactory}, and a
068         * {@link IndividualCompleter}.
069         * 
070         * @param population
071         *            the specified population
072         * @param archive
073         *            the specified archive
074         * @param completer
075         *            the specified completer
076         * @param control
077         *            the control
078         * @param iteration
079         *            the iteration counter
080         */
081        public AbstractOptimizer(Population population, Archive archive, IndividualCompleter completer, Control control,
082                        Iteration iteration) {
083                this.population = population;
084                this.archive = archive;
085                this.completer = completer;
086                this.control = control;
087                this.iteration = iteration;
088        }
089
090        @Inject
091        protected void injectListeners(Set<OptimizerStateListener> stateListeners,
092                        Set<OptimizerIterationListener> iterationListeners) {
093                this.stateListeners.addAll(stateListeners);
094                this.iterationListeners.addAll(iterationListeners);
095        }
096
097        /*
098         * (non-Javadoc)
099         * 
100         * @see org.opt4j.core.optimizer.Optimizer#getIteration()
101         */
102        @Override
103        public int getIteration() {
104                return iteration.value;
105        }
106
107        /*
108         * (non-Javadoc)
109         * 
110         * @see org.opt4j.core.optimizer.Optimizer#isRunning()
111         */
112        @Override
113        public boolean isRunning() {
114                return optimizing;
115        }
116
117        /**
118         * Call this method if a new iteration started.
119         * 
120         * @throws StopException
121         *             if the optimization is stopped
122         * @throws TerminationException
123         *             if the optimization is terminated
124         */
125        protected void nextIteration() throws TerminationException, StopException {
126                completer.complete(population);
127                archive.update(population);
128                iteration.next();
129                for (OptimizerIterationListener listener : iterationListeners) {
130                        listener.iterationComplete(iteration.value());
131                }
132                control.checkpointStop();
133        }
134
135        /*
136         * (non-Javadoc)
137         * 
138         * @see org.opt4j.core.optimizer.Optimizer#startOptimization()
139         */
140        @Override
141        public void startOptimization() {
142                optimizing = true;
143                for (OptimizerStateListener listener : stateListeners) {
144                        listener.optimizationStarted(this);
145                }
146                try {
147                        optimize();
148                } catch (StopException e) {
149                        System.out.println("Optimization stopped.");
150                } catch (TerminationException e) {
151                        System.err.println("Optimization terminated.");
152                } finally {
153                        stopOptimization();
154                }
155        }
156
157        /*
158         * (non-Javadoc)
159         * 
160         * @see org.opt4j.core.optimizer.Optimizer#stopOptimization()
161         */
162        @Override
163        public void stopOptimization() {
164                optimizing = false;
165                for (OptimizerStateListener listener : stateListeners) {
166                        listener.optimizationStopped(this);
167                }
168        }
169
170        /*
171         * (non-Javadoc)
172         * 
173         * @see
174         * org.opt4j.core.optimizer.Optimizer#addOptimizerIterationListener(org.
175         * opt4j.core.optimizer.OptimizerIterationListener)
176         */
177        @Override
178        public void addOptimizerIterationListener(OptimizerIterationListener listener) {
179                iterationListeners.add(listener);
180        }
181
182        /*
183         * (non-Javadoc)
184         * 
185         * @see
186         * org.opt4j.core.optimizer.Optimizer#addOptimizerStateListener(org.opt4j
187         * .core.optimizer.OptimizerStateListener)
188         */
189        @Override
190        public void addOptimizerStateListener(OptimizerStateListener listener) {
191                stateListeners.add(listener);
192        }
193
194        /*
195         * (non-Javadoc)
196         * 
197         * @see
198         * org.opt4j.core.optimizer.Optimizer#removeOptimizerIterationListener(org
199         * .opt4j.core.optimizer.OptimizerIterationListener)
200         */
201        @Override
202        public void removeOptimizerIterationListener(OptimizerIterationListener listener) {
203                iterationListeners.remove(listener);
204        }
205
206        /*
207         * (non-Javadoc)
208         * 
209         * @see
210         * org.opt4j.core.optimizer.Optimizer#removeOptimizerStateListener(org.opt4j
211         * .core.optimizer.OptimizerStateListener)
212         */
213        @Override
214        public void removeOptimizerStateListener(OptimizerStateListener listener) {
215                stateListeners.remove(listener);
216        }
217
218}
