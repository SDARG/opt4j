001package org.opt4j.tutorial.operator;
002
003import org.opt4j.core.genotype.BooleanGenotype;
004import org.opt4j.operators.crossover.Crossover;
005import org.opt4j.operators.crossover.Pair;
006
007// By implementing CrossoverBoolean an operator is created which performs a
008// crossover
009// operation (as used by, e.g., EAs) on the BooleanGenotype.
010public class MyOperator implements Crossover<BooleanGenotype> {
011
012        // method defined by CrossoverBoolean
013        @Override
014        public Pair<BooleanGenotype> crossover(BooleanGenotype parent1, BooleanGenotype parent2) {
015                // Instantiate genotypes of the same class as the parents. Each
016                // Genotype has the newInstance() method.
017                BooleanGenotype g1 = parent1.newInstance();
018                BooleanGenotype g2 = parent1.newInstance();
019
020                assert parent1.size() == parent2.size() : "problem size differs";
021
022                for (int i = 0; i < parent1.size(); i++) {
023                        // replace this with your operator logic
024                        g1.add(i, parent1.get(i) || parent2.get(i));
025                        g2.add(i, parent1.get(i) && parent2.get(i));
026                }
027
028                // The crossover operator returns two new genotypes based on the two
029                // parents.
030                return new Pair<>(g1, g2);
031        }
032}
