001package org.opt4j.tutorial.minones;
002
003import java.util.HashSet;
004import java.util.Random;
005import java.util.Set;
006
007import org.opt4j.core.Genotype;
008import org.opt4j.core.common.random.Rand;
009import org.opt4j.satdecoding.AbstractSATDecoder;
010import org.opt4j.satdecoding.Constraint;
011import org.opt4j.satdecoding.Literal;
012import org.opt4j.satdecoding.Model;
013import org.opt4j.satdecoding.SATManager;
014
015import com.google.inject.Inject;
016
017public class MinOnesDecoder extends AbstractSATDecoder<Genotype, MinOnesResult> {
018
019        @Inject
020        public MinOnesDecoder(SATManager satManager, Rand random) {
021                super(satManager, random);
022        }
023
024        // Here you can set the constraints of your problem. In our case, we will
025        // randomly generate a problem as a 3SAT problem (3 literals per clause)
026        // with 1000 variables and 1000 clauses. This problem is known to be
027        // NP-complete. However, we hope that there exists at least one feasible
028        // solution (and with the seed 0 of random it does ;) ).
029        @Override
030        public Set<Constraint> createConstraints() {
031                Set<Constraint> constraints = new HashSet<>();
032                Random random = new Random(0);
033
034                for (int i = 0; i < 1000; i++) {
035                        Constraint clause = new Constraint(">=", 1);
036                        HashSet<Integer> vars = new HashSet<>();
037                        do {
038                                vars.add(random.nextInt(1000));
039                        } while (vars.size() < 3);
040
041                        for (int n : vars) {
042                                clause.add(new Literal(n, random.nextBoolean()));
043                        }
044
045                        constraints.add(clause);
046                }
047
048                return constraints;
049        }
050
051        @Override
052        public MinOnesResult convertModel(Model model) {
053
054                MinOnesResult minOnesResult = new MinOnesResult();
055
056                for (int i = 0; i < 1000; i++) {
057                        minOnesResult.add(model.get(i));
058                }
059
060                return minOnesResult;
061        }
062
063}
