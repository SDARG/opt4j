001package org.opt4j.tutorial.helloworld;
002
003import org.opt4j.core.Objective.Sign;
004import org.opt4j.core.Objectives;
005import org.opt4j.core.problem.Evaluator;
006
007public class HelloWorldEvaluator implements Evaluator<String> {
008        @Override
009        public Objectives evaluate(String phenotype) {
010                int value = 0;
011                for (int i = 0; i < phenotype.length(); i++) {
012                        value += (phenotype.charAt(i) == "HELLO WORLD".charAt(i)) ? 1 : 0;
013                }
014
015                Objectives objectives = new Objectives();
016                objectives.add("objective", Sign.MAX, value);
017                return objectives;
018        }
019}
