023package org.opt4j.benchmarks.queens;
024
025import java.util.HashSet;
026import java.util.Set;
027
028import org.opt4j.core.Genotype;
029import org.opt4j.core.common.random.Rand;
030import org.opt4j.satdecoding.AbstractSATDecoder;
031import org.opt4j.satdecoding.Constraint;
032import org.opt4j.satdecoding.Literal;
033import org.opt4j.satdecoding.Model;
034import org.opt4j.satdecoding.SATManager;
035
036import com.google.inject.Inject;
037
038/**
039 * A SAT decoder strategy. The decoder is based on the
040 * {@link AbstractSATDecoder}.
041 * 
042 * @author lukasiewycz
043 * 
044 */
045public class QueensSATDecoder extends AbstractSATDecoder<Genotype, QueensBoard> {
046
047        protected final QueensProblem problem;
048
049        /**
050         * Constructs a {@link QueensSATDecoder}.
051         * 
052         * @param satManager
053         *            the SATManager
054         * @param problem
055         *            the problem instance
056         * @param random
057         *            the random number generator
058         */
059        @Inject
060        public QueensSATDecoder(SATManager satManager, QueensProblem problem, Rand random) {
061                super(satManager, random);
062                this.problem = problem;
063        }
064
065        /*
066         * (non-Javadoc)
067         * 
068         * @see org.opt4j.sat.AbstractSATDecoder#convertModel(org.opt4j.sat.Model)
069         */
070        @Override
071        public QueensBoard convertModel(Model model) {
072                int size = problem.size();
073
074                QueensBoard queensBoard = new QueensBoard(size);
075                for (int i = 0; i < size; i++) {
076                        for (int j = 0; j < size; j++) {
077                                queensBoard.setQueen(i, j, model.get(i * size + j));
078                        }
079                }
080
081                return queensBoard;
082        }
083
084        /*
085         * (non-Javadoc)
086         * 
087         * @see org.opt4j.sat.AbstractSATDecoder#createConstraints()
088         */
089        @Override
090        public Set<Constraint> createConstraints() {
091                Set<Constraint> constraints = new HashSet<>();
092
093                constraints.addAll(createConstraintsRowsColumns());
094                constraints.addAll(createConstraintsDiagonal());
095                return constraints;
096        }
097
098        /**
099         * Helper function for createConstraints() that creates the constraints
100         * regarding rows and columns.
101         * 
102         * @return constraints regarding rows and columns
103         */
104        protected Set<Constraint> createConstraintsRowsColumns() {
105                int size = problem.size();
106                Set<Constraint> constraints = new HashSet<>();
107
108                for (int i = 0; i < size; i++) {
109
110                        Constraint constraint = new Constraint("=", 1);
111                        for (int j = 0; j < size; j++) {
112                                constraint.add(new Literal(i * size + j, true));
113                        }
114
115                        constraints.add(constraint);
116                }
117
118                for (int j = 0; j < size; j++) {
119                        Constraint constraint = new Constraint("=", 1);
120
121                        for (int i = 0; i < size; i++) {
122                                constraint.add(new Literal(i * size + j, true));
123                        }
124
125                        constraints.add(constraint);
126                }
127
128                return constraints;
129
130        }
131
132        /**
133         * Helper function for createConstraints() that creates the constraints
134         * regarding diagonals.
135         * 
136         * @return constraints regarding diagonals
137         */
138        protected Set<Constraint> createConstraintsDiagonal() {
139                int size = problem.size();
140                Set<Constraint> constraints = new HashSet<>();
141
142                for (int k = -size + 1; k < size; k++) {
143                        // diagonal 1
144                        Constraint constraint = new Constraint("<=", 1);
145                        for (int j = 0; j < size; j++) {
146                                int i = k + j;
147                                if (0 <= i && i < size) {
148                                        constraint.add(new Literal(i * size + j, true));
149                                }
150                        }
151                        constraints.add(constraint);
152                }
153
154                for (int k = 0; k < 2 * size - 1; k++) {
155                        // diagonal 2
156                        Constraint constraint = new Constraint("<=", 1);
157                        for (int j = 0; j < size; j++) {
158                                int i = k - j;
159                                if (0 <= i && i < size) {
160                                        constraint.add(new Literal(i * size + j, true));
161                                }
162                        }
163                        constraints.add(constraint);
164                }
165
166                return constraints;
167
168        }
169}
