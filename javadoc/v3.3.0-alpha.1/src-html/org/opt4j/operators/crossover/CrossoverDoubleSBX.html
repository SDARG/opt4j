023package org.opt4j.operators.crossover;
024
025import static java.lang.annotation.RetentionPolicy.RUNTIME;
026
027import java.lang.annotation.Retention;
028
029import org.opt4j.core.common.random.Rand;
030import org.opt4j.core.start.Constant;
031import org.opt4j.operators.normalize.NormalizeDouble;
032
033import com.google.inject.BindingAnnotation;
034import com.google.inject.Inject;
035
036/**
037 * The {@link CrossoverDoubleSBX} is an implementation of the simulated binary
038 * crossover operator proposed by Deb and Kumar, 1995.
039 * 
040 * <p>
041 * The SBX crossover is applied bitwise. In the original work, two offspring are
042 * created using
043 * </p>
044 * <p>
045 * {@code z_1 = 0.5 * [(1 + beta)x + (1 - beta)y]}<br>
046 * {@code z_2 = 0.5 * [(1 - beta)x + (1 + beta)y]}.
047 * </p>
048 * <p>
049 * In this implementation, one offspring is created while the actual bit is set
050 * to {@code z_1} with a probability of {@code 0.5} and to {@code z_2} with a
051 * probability of {@code 0.5}, respectively.<br>
052 * The value {@code beta} is defined as<br>
053 * </p>
054 * <p>
055 * {@code beta = (2u)^(1 / (nu + 1))} if {@code u <= 0.5} and<br>
056 * {@code beta = (1 / (2 (1 - u)))^(1 / (nu + 1))} otherwise
057 * </p>
058 * <p>
059 * At this juncture, {@code u} is a random number between {@code (0, 1)}. The
060 * variable {@code nu} influences the crossover with a high value leading to a
061 * higher probability of 'near-parent' solution while a small value leads to a
062 * higher probability of more distant solutions, respectively.
063 * </p>
064 * 
065 * @author lukasiewycz
066 * 
067 */
068public class CrossoverDoubleSBX extends CrossoverDoubleElementwise {
069
070        protected double nu;
071
072        @Retention(RUNTIME)
073        @BindingAnnotation
074        protected @interface Nu {
075        }
076
077        /**
078         * Constructs a {@link CrossoverDoubleSBX} with a nu value and a random
079         * generator.
080         * 
081         * @param nu
082         *            the {@code nu} value
083         * @param normalize
084         *            the normalize operator
085         * @param random
086         *            the random number generator
087         */
088        @Inject
089        public CrossoverDoubleSBX(@Constant(value = "nu", namespace = CrossoverDoubleSBX.class) double nu,
090                        NormalizeDouble normalize, Rand random) {
091                super(normalize, random);
092                this.nu = nu;
093        }
094
095        /*
096         * (non-Javadoc)
097         * 
098         * @see
099         * org.opt4j.operator.crossover.CrossoverDoubleElementwise#crossover(double,
100         * double)
101         */
102        @Override
103        public Pair<Double> crossover(double x, double y) {
104                double u = random.nextDouble(); // [0,1)
105                double beta;
106                if (0.5 <= u) {
107                        beta = Math.pow(2 * u, 1 / (nu + 1));
108                } else {
109                        beta = Math.pow(1 / (2 - 2 * u), 1 / (nu + 1));
110                }
111
112                double o1 = 0.5 * ((1 + beta) * x + (1 - beta) * y);
113                double o2 = 0.5 * ((1 - beta) * x + (1 + beta) * y);
114
115                if (random.nextDouble() < 0.5) {
116                        double tmp = o1;
117                        o1 = o2;
118                        o2 = tmp;
119                }
120
121                return new Pair<>(o1, o2);
122        }
123
124}
