024package org.opt4j.benchmarks.lotz;
025
026import static org.opt4j.core.Objective.Sign.MAX;
027
028import org.opt4j.benchmarks.BinaryString;
029import org.opt4j.core.Objective;
030import org.opt4j.core.Objectives;
031import org.opt4j.core.problem.Evaluator;
032
033/**
034 * The {@link LOTZEvaluator}.
035 * 
036 * @author lukasiewycz
037 * 
038 */
039public class LOTZEvaluator implements Evaluator<BinaryString> {
040
041        protected final Objective onesObj = new Objective("LeadingOnes", MAX);
042        protected final Objective zerosObj = new Objective("TrailingZeros", MAX);
043
044        /*
045         * (non-Javadoc)
046         * 
047         * @see org.opt4j.core.problem.Evaluator#evaluate(org.opt4j.core.Phenotype)
048         */
049        @Override
050        public Objectives evaluate(BinaryString phenotype) {
051                int ones = 0;
052                int size = phenotype.size();
053                for (; ones < size; ones++) {
054                        boolean value = phenotype.get(ones);
055                        if (!value) {
056                                break;
057                        }
058                }
059                int zeros = 0;
060                for (; zeros < size; zeros++) {
061                        boolean value = phenotype.get(size - zeros - 1);
062                        if (value) {
063                                break;
064                        }
065                }
066
067                Objectives objectives = new Objectives();
068                objectives.add(onesObj, ones);
069                objectives.add(zerosObj, zeros);
070                return objectives;
071        }
072}
