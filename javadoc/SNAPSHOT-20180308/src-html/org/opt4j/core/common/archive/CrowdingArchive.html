024package org.opt4j.core.common.archive;
025
026import java.util.Collection;
027import java.util.List;
028
029import org.opt4j.core.Individual;
030import org.opt4j.core.start.Constant;
031
032import com.google.inject.Inject;
033
034/**
035 * The {@link CrowdingArchive} is based on the {@link Crowding} distance.
036 * 
037 * @author lukasiewycz
038 * @see Crowding
039 * 
040 */
041public class CrowdingArchive extends BoundedArchive {
042
043        /**
044         * Constructs a {@link CrowdingArchive}.
045         * 
046         * @param capacity
047         *            capacity of this archive (using namespace
048         *            {@link BoundedArchive})
049         */
050        @Inject
051        public CrowdingArchive(@Constant(value = "capacity", namespace = BoundedArchive.class) int capacity) {
052                super(capacity);
053        }
054
055        /*
056         * (non-Javadoc)
057         * 
058         * @see
059         * org.opt4j.common.archive.AbstractArchive#updateWithNondominated(java.
060         * util.Collection)
061         */
062        @Override
063        protected boolean updateWithNondominated(Collection<Individual> candidates) {
064                boolean changed = false;
065                if (this.size() + candidates.size() <= capacity) {
066                        changed = addCheckedIndividuals(candidates);
067                } else {
068                        candidates.addAll(this);
069                        Crowding crowding = new Crowding();
070                        List<Individual> list = crowding.order(crowding.getDensityValues(candidates));
071                        List<Individual> worst = list.subList(capacity, list.size());
072                        candidates.removeAll(worst);
073
074                        this.retainAll(candidates);
075                        for (Individual i : candidates) {
076                                if (!this.contains(i)) {
077                                        changed |= addCheckedIndividual(i);
078                                }
079                        }
080                }
081                return changed;
082        }
083
084}
