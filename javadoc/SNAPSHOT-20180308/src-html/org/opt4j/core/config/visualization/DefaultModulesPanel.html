099        /**
100         * The {@link UserNode} is a basic node of the tree.
101         */
102        protected abstract class UserNode extends DefaultMutableTreeNode {
103
104                public UserNode(Object object) {
105                        super(object);
106                }
107
108                @Override
109                public String toString() {
110                        return format.getName(getType());
111                }
112
113                public ImageIcon getIcon() {
114                        return format.getIcon(getType());
115                }
116
117                public String getTooltip() {
118                        return format.getTooltip(getType());
119                }
120
121                public abstract Class<?> getType();
122        }

124        /**
125         * The {@link RootTreeNode} is the root node of the tree.
126         */
127        protected class RootTreeNode extends UserNode {
128
129                public RootTreeNode(boolean b) {
130                        super(b);
131                }
132
133                public void setLoading(boolean b) {
134                        userObject = b;
135                }
136
137                public boolean isLoading() {
138                        return (Boolean) userObject;
139                }
140
141                @Override
142                public ImageIcon getIcon() {
143                        if (isLoading()) {
144                                ImageIcon icon = Icons.getIcon(Icons.LOADING);
145                                icon.setImageObserver(tree);
146                                return icon;
147                        }
148                        return format.asFolder(null);
149                }
150
151                @Override
152                public String toString() {
153                        if (isLoading()) {
154                                return "Loading";
155                        }
156                        return "Modules";
157                }
158
159                @Override
160                public String getTooltip() {
161                        return null;
162                }
163
164                @Override
165                public Class<?> getType() {
166                        return null;
167                }
168        }

225        /**
226         * The {@link TreeCellRenderer} for the modules tree.
227         * 
228         */
229        protected static class TreeCellRenderer extends DefaultTreeCellRenderer {
230                @Override
231                public Component getTreeCellRendererComponent(JTree tree, Object value, boolean sel, boolean expanded,
232                                boolean leaf, int row, boolean hasFocus) {
233                        Component comp = super.getTreeCellRendererComponent(tree, value, sel, expanded, leaf, row, hasFocus);
234
235                        ImageIcon icon = null;
236                        String tooltip = null;
237
238                        if (value instanceof UserNode) {
239                                UserNode userNode = (UserNode) value;
240                                icon = userNode.getIcon();
241                                tooltip = userNode.getTooltip();
242                        }
243                        if (icon == null) {
244                                icon = Icons.getDefault();
245                        }
246
247                        setIcon(icon);
248                        setToolTipText(tooltip);
249
250                        return comp;
251                }
252        }

254        /**
255         * The {@link MyTree} that extends a {@link JTree} by a drag method.
256         * 
257         */
258        protected static class MyTree extends JTree implements DragGestureListener {
259
260                DragSource dragSource = null;
261
262                public MyTree(TreeNode root) {
263                        super(root);
264
265                        dragSource = new DragSource();
266                        dragSource.createDefaultDragGestureRecognizer(this, DnDConstants.ACTION_MOVE, this);
267                }
268
269                protected DragSourceListener dragSourceListener = new DragSourceAdapter() {
270                };
271
272                @Override
273                public void dragGestureRecognized(DragGestureEvent dge) {
274                        TreeNode node = (TreeNode) this.getLastSelectedPathComponent();
275
276                        if (node instanceof ModuleTreeNode) {
277                                ModuleTreeNode moduleNode = (ModuleTreeNode) node;
278                                PropertyModule module = moduleNode.getModule();
279
280                                Transferable transferable = new ModuleTransferable(module);
281
282                                dragSource.startDrag(dge, DragSource.DefaultMoveDrop, transferable, dragSourceListener);
283                        }
284                }
285

341        private Class<?> getCategory(Class<?> clazz, boolean include) {
342                if (include && isCategory(clazz)) {
343                        return clazz;
344                }
345                if (clazz.equals(Object.class)) {
346                        return null;
347                } else if (clazz.getAnnotation(Parent.class) != null) {
348                        Class<?> pClass = clazz.getAnnotation(Parent.class).value();
349                        return pClass;
350                } else {
351                        assert (clazz.getSuperclass() != null) : clazz;
352                        return getCategory(clazz.getSuperclass(), true);
353                }
354        }
355
356        private boolean isCategory(Class<?> clazz) {
357                Category category = clazz.getAnnotation(Category.class);
358                return (category != null);
359        }

361        /**
362         * Build and order the tree
363         */
364        protected void populateTree() {
365                // get all relevant categories
366
367                // map from the modules (categories) to their categories
368                // (supercategories)
369                Map<Class<?>, Class<?>> map = new HashMap<Class<?>, Class<?>>();
370
371                for (Class<? extends Module> module : allModules) {
372                        Class<?> category = getCategory(module, false);
373                        assert (module != category) : module;
374                        map.put(module, category);
375                }
376
377                int size;
378                do {
379                        size = map.size();
380                        Set<Class<?>> values = new HashSet<Class<?>>(map.values());
381                        for (Class<?> cat : values) {
382                                if (cat != null) {
383                                        Class<?> category = getCategory(cat, false);
384                                        map.put(cat, category);
385                                }
386                        }
387                } while (map.size() != size);
388
389                // create all nodes
390                Map<Class<?>, CategoryTreeNode> ctn = new HashMap<Class<?>, CategoryTreeNode>();
391                Map<Class<?>, ModuleTreeNode> mtn = new HashMap<Class<?>, ModuleTreeNode>();
392                Map<Class<?>, DefaultMutableTreeNode> atn = new HashMap<Class<?>, DefaultMutableTreeNode>();
393
394                for (Class<?> clazz : map.keySet()) {
395                        if (isCategory(clazz)) {
396                                CategoryTreeNode node = new CategoryTreeNode(clazz);
397                                ctn.put(clazz, node);
398                        } else {
399                                @SuppressWarnings("unchecked")
400                                ModuleTreeNode node = new ModuleTreeNode((Class<? extends Module>) clazz);
401                                mtn.put(clazz, node);
402                        }
403                }
404                atn.putAll(ctn);
405                atn.putAll(mtn);
406
407                // create hierarchy by parent annotation
408                CategoryTreeNode def = new CategoryTreeNode(Default.class);
409                root.add(def);
410                for (Entry<Class<?>, DefaultMutableTreeNode> entry : atn.entrySet()) {
411                        Class<?> clazz = entry.getKey();
412                        DefaultMutableTreeNode node = entry.getValue();
413
414                        Class<?> category = map.get(clazz);
415
416                        if (category == null) {
417                                if (isCategory(clazz)) {
418                                        root.add(node);
419                                } else {
420                                        def.add(node);
421                                }
422                        } else {
423                                DefaultMutableTreeNode n = atn.get(category);
424                                assert (n != null && n != node) : clazz + " " + category;
425                                n.add(node);
426                        }
427
428                }
429
430                // clean empty categories
431                boolean removed;
432                do {
433                        removed = removeEmptyCategories(root);
434                } while (removed);
435
436                SwingUtilities.invokeLater(new Runnable() {
437                        @Override
438                        public void run() {
439                                tree.setSelectionModel(new DefaultTreeSelectionModel());
440                                tree.getSelectionModel().setSelectionMode(TreeSelectionModel.SINGLE_TREE_SELECTION);
441                                root.setLoading(false);
442                                sort(tree);
443                                tree.expandRow(0);
444                        }
445                });
446                revalidate();
447                repaint();
448        }

450        /**
451         * Removes empty categories. This method can create some new empty
452         * categories!
453         * 
454         * @param current
455         *            the current node
456         * @return true if an empty category was removed
457         */
458        private boolean removeEmptyCategories(DefaultMutableTreeNode current) {
459                if (!current.children().hasMoreElements()) {
460                        return false;
461                }
462
463                TreeNode node = current.getFirstChild();
464                boolean removed = false;
465                while (node != null) {
466                        DefaultMutableTreeNode n = (DefaultMutableTreeNode) node;
467                        node = current.getChildAfter(n);
468
469                        if (n.isLeaf()) {
470                                if (n instanceof CategoryTreeNode && n.getChildCount() == 0) {
471                                        DefaultMutableTreeNode parent = (DefaultMutableTreeNode) n.getParent();
472                                        parent.remove(n);
473                                        removed = true;
474                                }
475                        } else {
476                                removed = removed || removeEmptyCategories(n);
477                        }
478                }
479                return removed;
480        }

482        protected MouseListener mouseListener = new MouseAdapter() {
483
484                @Override
485                public void mousePressed(MouseEvent e) {
486                        if (e.isPopupTrigger()) {
487                                displayMenu(e);
488                        }
489                }
490
491                @Override
492                public void mouseReleased(MouseEvent e) {
493                        int count = e.getClickCount();
494
495                        if (e.isPopupTrigger()) {
496                                displayMenu(e);
497                        } else if (count > 1) {
498                                Object object = tree.getLastSelectedPathComponent();
499                                if (object instanceof ModuleTreeNode) {
500                                        ModuleTreeNode node = (ModuleTreeNode) object;
501                                        PropertyModule pmodule = node.getModule();
502                                        if (selectedModules.contains(pmodule)) {
503                                                remove(pmodule);
504                                        } else {
505                                                add(pmodule);
506                                        }
507                                }
508                        }
509                }
510        };

512        /**
513         * Adds a new {@link PropertyModule}.
514         * 
515         * @param pm
516         *            the property module to be added
517         */
518        protected void add(PropertyModule pm) {
519                Module module = pm.getModule();
520                if (module.getClass().getAnnotation(Multi.class) != null) {
521                        try {
522                                Module m2 = module.getClass().newInstance();
523                                PropertyModule pm2 = new PropertyModule(m2);
524                                selectedModules.add(pm2);
525                        } catch (InstantiationException e1) {
526                                e1.printStackTrace();
527                        } catch (IllegalAccessException e1) {
528                                e1.printStackTrace();
529                        }
530                } else {
531                        selectedModules.add(pm);
532                }
533        }

545        /**
546         * Displays a popup menu for the tree elements.
547         * 
548         * @param e
549         *            the mouse event
550         */
551        protected void displayMenu(MouseEvent e) {
552                if (e.isPopupTrigger()) {
553                        Point pt = e.getPoint();
554                        tree.setSelectionPath(tree.getPathForLocation(pt.x, pt.y));
555
556                        Object object = tree.getLastSelectedPathComponent();
557                        if (object instanceof ModuleTreeNode) {
558
559                                ModuleTreeNode node = (ModuleTreeNode) object;
560                                final PropertyModule module = node.getModule();
561
562                                moduleMenu.removeAll();
563
564                                JMenuItem add = new JMenuItem("Add Module", Icons.getIcon(Icons.ADD));
565                                add.addActionListener(new ActionListener() {
566                                        @Override
567                                        public void actionPerformed(ActionEvent e) {
568                                                add(module);
569                                        }
570                                });
571
572                                JMenuItem remove = new JMenuItem("Remove Module", Icons.getIcon(Icons.DELETE));
573                                remove.addActionListener(new ActionListener() {
574                                        @Override
575                                        public void actionPerformed(ActionEvent e) {
576                                                remove(module);
577                                        }
578                                });
579
580                                if (selectedModules.contains(module)) {
581                                        add.setEnabled(false);
582                                } else {
583                                        remove.setEnabled(false);
584                                }
585
586                                moduleMenu.add(add);
587                                moduleMenu.add(remove);
588                                moduleMenu.show(e.getComponent(), e.getX(), e.getY());
589
590                        }
591
592                }
593        }

595        /**
596         * Sorts the {@link JTree} alphabetically.
597         * 
598         * @param tree
599         *            the tree to be sorted
600         */
601        protected void sort(JTree tree) {
602                DefaultMutableTreeNode root = (DefaultMutableTreeNode) tree.getModel().getRoot();
603                sort(root);
604
605        }

607        /**
608         * Sorts the child elements of one tree node alphabetically.
609         * 
610         * @param node
611         *            the node which children are sorted
612         */
613        protected void sort(DefaultMutableTreeNode node) {
614                if (!node.children().hasMoreElements()) {
615                        return;
616                }
617
618                List<UserNode> nodes = new ArrayList<UserNode>();
619
620                Enumeration<?> children = node.children();
621
622                while (children.hasMoreElements()) {
623                        UserNode n = (UserNode) children.nextElement();
624                        nodes.add(n);
625                }
626
627                Collections.sort(nodes, new Comparator<UserNode>() {
628                        @Override
629                        public int compare(UserNode o1, UserNode o2) {
630                                boolean cat1 = o1 instanceof CategoryTreeNode;
631                                boolean cat2 = o2 instanceof CategoryTreeNode;
632
633                                if (cat1 && !cat2) {
634                                        return -1;
635                                } else if (!cat1 && cat2) {
636                                        return 1;
637                                } else {
638                                        return o1.toString().compareTo(o2.toString());
639                                }
640                        }
641                });
642
643                node.removeAllChildren();
644
645                for (UserNode n : nodes) {
646                        node.add(n);
647                        sort(n);
648                }
649        }
