001package org.opt4j.tutorial.salesman;
002
003import java.awt.BasicStroke;
004import java.awt.Color;
005import java.awt.Component;
006import java.awt.Dimension;
007import java.awt.Graphics;
008import java.awt.Graphics2D;
009import java.awt.Point;
010import java.awt.RenderingHints;
011import java.awt.event.ActionEvent;
012import java.awt.event.ActionListener;
013
014import javax.swing.JMenuItem;
015import javax.swing.JPanel;
016import javax.swing.JPopupMenu;
017
018import org.opt4j.core.Individual;
019import org.opt4j.tutorial.salesman.SalesmanProblem.City;
020import org.opt4j.viewer.IndividualMouseListener;
021import org.opt4j.viewer.Viewport;
022import org.opt4j.viewer.Widget;
023import org.opt4j.viewer.WidgetParameters;
024
025import com.google.inject.Inject;
026
027// The SalesmanWidgetService is an additional feature of this tutorial. It
028// enables the visualization
029// of a single route in the viewer.
030public class SalesmanWidgetService implements IndividualMouseListener {
031
032        // Panel that paints a single SalesmanRoute (which is the phenotype of an
033        // Individual)
034        public static class MyPanel extends JPanel {
035
036                private static final long serialVersionUID = 1L;
037
038                protected final Individual individual;
039
040                public MyPanel(Individual individual) {
041                        super();
042                        this.individual = individual;
043                        setPreferredSize(new Dimension(208, 208));
044                }
045
046                @Override
047                protected void paintComponent(Graphics g) {
048                        Graphics2D g2d = (Graphics2D) g;
049                        g2d.setBackground(Color.WHITE);
050                        g2d.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
051                        g2d.setStroke(new BasicStroke(2f));
052                        g2d.clearRect(0, 0, 208, 212);
053
054                        SalesmanRoute salesmanRoute = (SalesmanRoute) individual.getPhenotype();
055
056                        for (int i = 0; i < salesmanRoute.size(); i++) {
057                                final int j = (i + 1) % salesmanRoute.size();
058                                City one = salesmanRoute.get(i);
059                                City two = salesmanRoute.get(j);
060
061                                int x1 = (int) (one.getX() * 2) + 4;
062                                int y1 = (int) (one.getY() * 2) + 4;
063                                int x2 = (int) (two.getX() * 2) + 4;
064                                int y2 = (int) (two.getY() * 2) + 4;
065
066                                g2d.drawLine(x1, y1, x2, y2);
067                                g2d.drawOval(x1 - 2, y1 - 2, 4, 4);
068
069                        }
070                }
071        }
072
073        // Use a custom widget
074        @WidgetParameters(title = "Route", resizable = false, maximizable = false)
075        protected static class SalesmanWidget implements Widget {
076
077                final Individual individual;
078
079                public SalesmanWidget(Individual individual) {
080                        super();
081                        this.individual = individual;
082                }
083
084                @Override
085                public JPanel getPanel() {
086                        JPanel panel = new MyPanel(individual);
087                        return panel;
088                }
089
090                @Override
091                public void init(Viewport viewport) {
092                }
093
094        }
095
096        protected final Viewport viewport;
097
098        // The route is shown by a double click of a individual in the archive
099        // monitor panel. Thus we need the ArchiveMonitorPanel and the main
100        // GUIFrame.
101        @Inject
102        public SalesmanWidgetService(Viewport viewport) {
103                this.viewport = viewport;
104        }
105
106        // If an individual is double clicked, paint the route.
107        @Override
108        public void onDoubleClick(Individual individual, Component table, Point p) {
109                paintRoute(individual);
110        }
111
112        // If an individual is clicked with the right mouse button, open a popup
113        // menu that contains the option to paint the route.
114        @Override
115        public void onPopup(final Individual individual, Component table, Point p, JPopupMenu menu) {
116                JMenuItem paint = new JMenuItem("show route");
117                menu.add(paint);
118
119                paint.addActionListener(new ActionListener() {
120                        @Override
121                        public void actionPerformed(ActionEvent e) {
122                                paintRoute(individual);
123                        }
124                });
125
126        }
127
128        // Paint the route: Construct a JInternalFrame, add the MyPanel and add the
129        // frame to the desktop of the main GUIFrame.
130        protected void paintRoute(Individual individual) {
131                Widget widget = new SalesmanWidget(individual);
132                viewport.addWidget(widget);
133        }
134
135}
