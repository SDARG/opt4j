001package org.opt4j.tutorial.helloworld;
002
003import org.opt4j.core.genotype.SelectGenotype;
004import org.opt4j.core.problem.Decoder;
005
006public class HelloWorldDecoder implements Decoder<SelectGenotype<Character>, String> {
007
008        public String decode(SelectGenotype<Character> genotype) {
009                String phenotype = "";
010                for (int i = 0; i < genotype.size(); i++) {
011                        phenotype += genotype.getValue(i);
012                }
013                return phenotype;
014        }
015
016}
