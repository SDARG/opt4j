001package org.opt4j.tutorial.minones;
002
003import org.opt4j.core.Objective;
004import org.opt4j.core.Objective.Sign;
005import org.opt4j.core.Objectives;
006import org.opt4j.core.problem.Evaluator;
007
008public class MinOnesEvaluator implements Evaluator<MinOnesResult> {
009
010        Objective ones = new Objective("ones", Sign.MIN);
011
012        @Override
013        public Objectives evaluate(MinOnesResult minOnesResult) {
014
015                int value = 0;
016                for (Boolean v : minOnesResult) {
017                        if (v != null && v) {
018                                value++;
019                        }
020                }
021
022                Objectives objectives = new Objectives();
023                objectives.add(ones, value);
024                return objectives;
025        }
026}
