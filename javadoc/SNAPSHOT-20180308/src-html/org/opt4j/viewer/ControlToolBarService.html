023package org.opt4j.viewer;
024
025import javax.swing.JToolBar;
026
027import org.opt4j.core.optimizer.Control;
028import org.opt4j.core.optimizer.Optimizer;
029
030import com.google.inject.Inject;
031
032/**
033 * The {@link ControlToolBarService} is a {@link ToolBarService} that allows to
034 * {@link Control} the {@link Optimizer}.This service has an
035 * {@link ToolBarOrder} with {@code -100}.
036 * 
037 * @see Control
038 * @author lukasiewycz
039 * 
040 */
041@ToolBarOrder(-100)
042public class ControlToolBarService implements ToolBarService {
043
044        protected final ControlButtons controlButtons;
045
046        /**
047         * Constructs a {@link ControlToolBarService}.
048         * 
049         * @param controlButtons
050         *            the control buttons
051         */
052        @Inject
053        public ControlToolBarService(ControlButtons controlButtons) {
054                this.controlButtons = controlButtons;
055        }
056
057        /*
058         * (non-Javadoc)
059         * 
060         * @see org.opt4j.viewer.ToolBarService#getToolBar()
061         */
062        @Override
063        public JToolBar getToolBar() {
064                JToolBar toolbar = new JToolBar("Control");
065
066                toolbar.add(controlButtons.getStart());
067                toolbar.add(controlButtons.getPause());
068                toolbar.add(controlButtons.getStop());
069                toolbar.add(controlButtons.getTerminate());
070
071                return toolbar;
072        }
073
074}
