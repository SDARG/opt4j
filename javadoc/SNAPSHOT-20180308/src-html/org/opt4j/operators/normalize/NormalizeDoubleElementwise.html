023package org.opt4j.operators.normalize;
024
025import org.opt4j.core.genotype.DoubleGenotype;
026
027/**
028 * <p>
029 * The {@link NormalizeDoubleElementwise} normalizes {@link DoubleGenotype}s
030 * elementwise.
031 * </p>
032 * 
033 * @author lukasiewycz
034 * 
035 */
036public abstract class NormalizeDoubleElementwise implements NormalizeDouble {
037
038        /*
039         * (non-Javadoc)
040         * 
041         * @see
042         * org.opt4j.operator.normalize.Normalize#normalize(org.opt4j.core.problem
043         * .Genotype)
044         */
045        @Override
046        public void normalize(DoubleGenotype genotype) {
047                int size = genotype.size();
048
049                for (int i = 0; i < size; i++) {
050                        double value = genotype.get(i);
051                        double lb = genotype.getLowerBound(i);
052                        double ub = genotype.getUpperBound(i);
053
054                        if (value < lb || ub < value) {
055                                value = normalize(value, lb, ub);
056                                assert (lb <= value && value <= ub);
057                                genotype.set(i, value);
058                        }
059                }
060
061        }
062
063        /**
064         * Normalize a double value.
065         * 
066         * @param value
067         *            the value to be normalized
068         * @param lb
069         *            the lower bound
070         * @param ub
071         *            the upper bound
072         * @return a normalize value in the bounds
073         */
074        public abstract double normalize(double value, double lb, double ub);
075
076}
