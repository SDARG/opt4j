024package org.opt4j.operators.crossover;
025
026import java.util.HashSet;
027import java.util.Random;
028import java.util.Set;
029
030import org.opt4j.core.common.random.Rand;
031import org.opt4j.core.genotype.PermutationGenotype;
032
033import com.google.inject.Inject;
034
035/**
036 * <p>
037 * Crossover for the {@link PermutationGenotype}.
038 * </p>
039 * 
040 * <p>
041 * The bucket operator randomly runs through to permutation from the beginning
042 * to the end, and fill the current element into the new permutation if not
043 * existent.
044 * </p>
045 * 
046 * <p>
047 * Given two permutations
048 * </p>
049 * <p>
050 * {@code 1 2 3 4 5 6 7 8} and {@code 8 7 6 5 4 3 2 1}
051 * </p>
052 * <p>
053 * this results, for instance, in
054 * </p>
055 * <p>
056 * {@code 1 8 2 7 3 6 4 5} or {@code 8 7 1 2 3 6 5 4}.
057 * </p>
058 * 
059 * @author lukasiewycz
060 * 
061 */
062public class CrossoverPermutationBucket implements CrossoverPermutation {
063
064        protected final Random random;
065
066        /**
067         * Constructs a new {@link CrossoverPermutationBucket}.
068         * 
069         * @param random
070         *            the random number generator
071         */
072        @Inject
073        public CrossoverPermutationBucket(Rand random) {
074                this.random = random;
075        }
076
077        /*
078         * (non-Javadoc)
079         * 
080         * @see
081         * org.opt4j.operator.crossover.Crossover#crossover(org.opt4j.core.Genotype,
082         * org.opt4j.core.Genotype)
083         */
084        @Override
085        public Pair<PermutationGenotype<?>> crossover(PermutationGenotype<?> p1, PermutationGenotype<?> p2) {
086
087                PermutationGenotype<Object> o1 = p1.newInstance();
088                PermutationGenotype<Object> o2 = p1.newInstance();
089
090                int size = p1.size();
091
092                assert (size == p2.size()) : "Permutation is undefined for genotypes with different lengths.";
093                assert p1.containsAll(p2) : "Permutation is undefined for different domains.";
094
095                Set<Object> elements = new HashSet<Object>();
096
097                int i = 0;
098                int j = 0;
099
100                while (o1.size() != size || o2.size() != size) {
101                        if (j == size || (random.nextBoolean() && i < size)) {
102                                Object e = p1.get(i);
103                                i++;
104                                if (elements.add(e)) {
105                                        o1.add(e);
106                                } else {
107                                        o2.add(e);
108                                }
109                        } else {
110                                Object e = p2.get(j);
111                                j++;
112
113                                if (elements.add(e)) {
114                                        o1.add(e);
115                                } else {
116                                        o2.add(e);
117                                }
118                        }
119                }
120
121                Pair<PermutationGenotype<?>> offspring = new Pair<PermutationGenotype<?>>(o1, o2);
122                return offspring;
123        }
124}
