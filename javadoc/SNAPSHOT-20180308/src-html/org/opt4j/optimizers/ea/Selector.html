024package org.opt4j.optimizers.ea;
025
026import java.util.Collection;
027
028import org.opt4j.core.Individual;
029import org.opt4j.core.optimizer.Population;
030
031import com.google.inject.ImplementedBy;
032
033/**
034 * The interface {@link Selector} is used to select a certain subset of
035 * {@link Individual}s from a {@link Population} by respecting certain metrics
036 * like their fitness. The core methods are {@link #getParents}, that returns a
037 * set of {@link Individual}s that can be used as parents for the next
038 * generation and {@link #getLames}, that returns a list of {@link Individual}s
039 * that can be removed from the given individual list in order to form the next
040 * generation.
041 * 
042 * @author glass, lukasiewycz
043 * 
044 */
045@ImplementedBy(SelectorDefault.class)
046public interface Selector {
047
048        /**
049         * Selects a subset of {@link Individual}s and returns it as a new
050         * {@link Collection}. These so called parents can be used to form the next
051         * generation.
052         * 
053         * @param mu
054         *            the number of parents to select
055         * @param population
056         *            the list of individuals
057         * @return the parents
058         */
059        public Collection<Individual> getParents(int mu, Collection<Individual> population);
060
061        /**
062         * Selects a subset of {@code lambda} {@link Individual}s and returns it as
063         * a new {@link Collection}. These individuals can be erased in the next
064         * generation.
065         * 
066         * @param lambda
067         *            the number of lames to select
068         * @param population
069         *            the list of individuals
070         * @return the worst {@code lambda} individuals
071         */
072        public Collection<Individual> getLames(int lambda, Collection<Individual> population);
073
074        /**
075         * Sets the maximal number of {@link Individual}s.
076         * 
077         * @param maxsize
078         *            the number of individuals
079         */
080        public void init(int maxsize);
081
082}
