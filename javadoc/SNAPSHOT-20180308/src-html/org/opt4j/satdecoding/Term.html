023package org.opt4j.satdecoding;
024
025/**
026 * The {@link Term} is an element of the linear {@link Constraint}. The
027 * {@link Term} is an integer coefficient times a {@link Literal}.
028 * 
029 * @author lukasiewycz
030 * 
031 */
032public class Term {
033
034        protected final int coeff;
035        protected final Literal lit;
036
037        /**
038         * Constructs a {@link Term}.
039         * 
040         * @param coeff
041         *            the integer coefficient
042         * @param lit
043         *            the literal must not be {@code null}
044         */
045        public Term(int coeff, Literal lit) {
046                this.coeff = coeff;
047                this.lit = lit;
048                if (lit == null) {
049                        throw new NullPointerException();
050                }
051        }
052
053        /**
054         * Returns the coefficient.
055         * 
056         * @return the coefficient
057         */
058        public int getCoefficient() {
059                return coeff;
060        }
061
062        /**
063         * Returns the literal.
064         * 
065         * @return the literal
066         */
067        public Literal getLiteral() {
068                return lit;
069        }
070
071        /**
072         * Constructs a copy to this term.
073         * 
074         * @return a copy of this term
075         */
076        public Term copy() {
077                return new Term(coeff, lit);
078        }
079
080        /*
081         * (non-Javadoc)
082         * 
083         * @see java.lang.Object#toString()
084         */
085        @Override
086        public String toString() {
087                return coeff + " " + lit;
088        }
089
090        /*
091         * (non-Javadoc)
092         * 
093         * @see java.lang.Object#hashCode()
094         */
095        @Override
096        public int hashCode() {
097                final int prime = 31;
098                int result = 1;
099                result = prime * result + coeff;
100                result = prime * result + ((lit == null) ? 0 : lit.hashCode());
101                return result;
102        }
103
104        /**
105         * A {@link Term} equals another {@link Term} if they have equal
106         * {@link Literal}s and coefficients. This method checks, if this
107         * {@link Term} equals the given {@code other} {@link Object}.
108         * 
109         * @param other
110         *            the other object
111         * @return {@code true} if both are equal; false otherwise
112         */
113        @Override
114        public boolean equals(Object other) {
115                if (this == other) {
116                        return true;
117                }
118                if (other == null) {
119                        return false;
120                }
121                if (getClass() != other.getClass()) {
122                        return false;
123                }
124                Term otherTerm = (Term) other;
125                if (coeff != otherTerm.coeff) {
126                        return false;
127                }
128                if (lit == null) {
129                        if (otherTerm.lit != null) {
130                                return false;
131                        }
132                } else if (!lit.equals(otherTerm.lit)) {
133                        return false;
134                }
135                return true;
136        }
137}
